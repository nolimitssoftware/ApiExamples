CREATE TABLE [job].[Processing]
(
[Id] [uniqueidentifier] NOT NULL CONSTRAINT [DF_Processing_Id] DEFAULT (newsequentialid()),
[Context] [int] NOT NULL,
[ContextId] [uniqueidentifier] NOT NULL,
[LastProcessed] [datetimeoffset] NULL,
[LastJobStatus] [tinyint] NULL,
[JobTypeId] [int] NOT NULL
) ON [PRIMARY]
GO
CREATE TRIGGER job.trg_Processing ON job.Processing
   AFTER INSERT, UPDATE
AS
   BEGIN
       SET NOCOUNT ON;
       INSERT  job.JobHistory
               ( JobTypeId
               , Context
               , ContextName
               , LastJobStatus
               , LastProcessed
               )
               SELECT
                   I.JobTypeId
                 , I.Context
                 , COALESCE(A.OverrideName, A.ShortName, A.Name)
                 , I.LastJobStatus
                 , I.LastProcessed
               FROM
                   [INSERTED] AS I
               INNER JOIN core.Asset AS A
                   ON I.ContextId = A.AssetId
               WHERE
                   I.Context = 1 --Asset
                   AND I.LastProcessed IS NOT NULL
       INSERT  job.JobHistory
               ( JobTypeId
               , Context
               , ContextName
               , LastJobStatus
               , LastProcessed
               )
               SELECT
                   I.JobTypeId
                 , I.Context
                 , COALESCE(A.OverrideName, A.ShortName, A.Name)
                 , I.LastJobStatus
                 , I.LastProcessed
               FROM
                   [INSERTED] AS I
               INNER JOIN core.Asset AS A
                   ON I.ContextId = A.AssetId
               WHERE
                   I.Context = 2097152 --Scheduled Reports
                   AND I.LastProcessed IS NOT NULL

   END
GO
