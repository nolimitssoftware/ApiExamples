CREATE TRIGGER core.trg_CustomAssetPropertySearch ON core.CustomAssetProperty
   AFTER INSERT, UPDATE, DELETE
AS
   BEGIN
       SET NOCOUNT ON;
       DELETE
           search.SearchValue
       FROM
           search.SearchValue AS SV
       INNER JOIN [Deleted] AS D
           ON SV.Id = D.Id;

       INSERT  search.SearchValue
               ( Id
               , AssetId
               , SearchValueType
               , Type
               , Name
               , SystemValue
               , SystemValueBaseType
               , SystemValueDateValue
               , SystemValueNumericValue
               , SystemValueBoolValue
               , SystemValueStringValue
               , UserValue
               , UserValueBaseType
               , UserValueDateValue
               , UserValueNumericValue
               , UserValueBoolValue
               , UserValueStringValue
               , DefaultValue
               , DefaultValueBaseType
               , DefaultValueDateValue
               , DefaultValueNumericValue
               , DefaultValueBoolValue
               , DefaultValueStringValue
               , LastUpdate
               , Value
               , ValueBaseType
               , Unit
               )
               SELECT
                   SV.[Id]
                 , SV.AssetId
                 , 3
                 , COALESCE(T1.Value, SV.[Type]) AS [Type]
                 , SV.[Name]
                 , SV.SystemValue
                 , SV.SystemValueBaseType
                 , SV.SystemValueDateValue
                 , SV.SystemValueNumericValue
                 , SV.SystemValueBoolean
                 , COALESCE(T2.Value, SV.SystemValueStringValue, NULL) AS [SystemValueStringValue]
                 , SV.UserValue
                 , SV.UserValueBaseType
                 , SV.UserValueDateValue
                 , SV.UserValueNumericValue
                 , SV.UserValueBoolean
                 , COALESCE(T3.Value, SV.UserValueStringValue, NULL) AS [UserValueStringValue]
                 , SV.[DefaultValue]
                 , SV.[DefaultValueBaseType]
                 , SV.[DefaultValueDateValue]
                 , SV.[DefaultValueNumericValue]
                 , SV.[DefaultValueBoolean]
                 , COALESCE(T4.Value, SV.DefaultValueStringValue, NULL) AS [DefaultValueStringValue]
                 , GETUTCDATE()
                 , CAST(COALESCE(SV.SystemValue, SV.UserValue,
                                 SV.DefaultValue) AS VARBINARY(MAX)) AS [Value]
                 , CASE WHEN SV.SystemValue IS NOT NULL
                        THEN SV.SystemValueBaseType
                        WHEN SV.UserValue IS NOT NULL
                        THEN SV.UserValueBaseType
                        WHEN SV.DefaultValue IS NOT NULL
                        THEN SV.DefaultValueBaseType
                   END AS [ValueBaseType]
                 , COALESCE(T5.Value, SV.Unit, NULL) AS [Unit]
               FROM
                   ( SELECT
                       VCAP.Id
                     , VCAP.AssetId
                     , A.[Type]
                     , VCAP.Name
                     , VCAP.SystemValue
                     , CONVERT(NVARCHAR(32), VCAP.SystemValueBaseType) AS [SystemValueBaseType]
                     , VCAP.SystemValueDateValue
                     , VCAP.SystemValueNumericValue
                     , VCAP.SystemValueBoolean
                     , VCAP.SystemValueStringValue
                     , VCAP.UserValue
                     , CONVERT(NVARCHAR(32), VCAP.UserValueBaseType) AS UserValueBaseType
                     , VCAP.UserValueDateValue
                     , VCAP.UserValueNumericValue
                     , VCAP.UserValueBoolean
                     , VCAP.UserValueStringValue
                     , VCAP.DefaultValue
                     , CONVERT(NVARCHAR(32), VCAP.DefaultValueBaseType) AS DefaultValueBaseType
                     , VCAP.DefaultValueDateValue
                     , VCAP.DefaultValueNumericValue
                     , VCAP.DefaultValueBoolean
                     , VCAP.DefaultValueStringValue
                     , VCAP.Unit
                     FROM
                       core.vw_CustomAssetProperty AS VCAP
                     INNER JOIN core.Asset AS A
                       ON VCAP.AssetId = A.AssetId
                     INNER JOIN [INSERTED] AS I
                       ON VCAP.Id = I.Id
                   ) AS SV
               LEFT JOIN settings.Translation AS T1 WITH ( NOLOCK )
                   ON SV.Type COLLATE SQL_Latin1_General_CP1_CS_AS = T1.Name
                      AND T1.Context = 'text'
                      AND T1.Locale = 'en'
               LEFT JOIN settings.Translation AS T2 WITH ( NOLOCK )
                   ON SV.SystemValueStringValue COLLATE SQL_Latin1_General_CP1_CS_AS = T2.Name
                      AND T2.Context = 'text'
                      AND T2.Locale = 'en'
               LEFT JOIN settings.Translation AS T3 WITH ( NOLOCK )
                   ON SV.UserValueStringValue COLLATE SQL_Latin1_General_CP1_CS_AS = T3.Name
                      AND T3.Context = 'text'
                      AND T3.Locale = 'en'
               LEFT JOIN settings.Translation AS T4 WITH ( NOLOCK )
                   ON SV.DefaultValueStringValue COLLATE SQL_Latin1_General_CP1_CS_AS = T4.Name
                      AND T4.Context = 'text'
                      AND T4.Locale = 'en'
               LEFT JOIN settings.Translation AS T5 WITH ( NOLOCK )
                   ON SV.Unit COLLATE SQL_Latin1_General_CP1_CS_AS = T5.Name
                      AND T5.Context = 'text'
                      AND T5.Locale = 'en'
   END
GO
