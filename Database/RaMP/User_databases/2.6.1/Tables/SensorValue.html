CREATE TABLE [monitoring].[SensorValue]
(
[Id] [uniqueidentifier] NOT NULL CONSTRAINT [DF_SensorValue_Id] DEFAULT (newsequentialid()),
[SensorId] [uniqueidentifier] NOT NULL,
[RollupType] [tinyint] NOT NULL,
[Value] [decimal] (30, 2) NULL,
[LastUpdate] [datetime2] NOT NULL CONSTRAINT [DF_SensorValue_LastUpdate] DEFAULT (getutcdate())
) ON [PRIMARY]
GO
CREATE TRIGGER monitoring.trg_SensorValueSearch_Delete ON monitoring.SensorValue
   AFTER DELETE
AS
   BEGIN
       SET NOCOUNT ON;
       DELETE
           search.SearchValue
       FROM
           search.SearchValue AS SV
       INNER JOIN [Deleted] AS D
           ON SV.Id = D.Id;
   END
GO
CREATE TRIGGER monitoring.trg_SensorValueSearch_Insert ON monitoring.SensorValue
   AFTER INSERT
AS
   BEGIN
       SET NOCOUNT ON;
      
       INSERT  search.SearchValue
               ( Id
               , AssetId
               , SearchValueType
               , Type
               , Name
               , SystemValue
               , SystemValueBaseType
               , SystemValueNumericValue
               , LastUpdate
               , Value
               , ValueBaseType
               , Unit
               )
               SELECT
                   SV.[Id]
                 , SV.AssetId
                 , 5
                 , COALESCE(T1.Value, SV.[Type]) AS [Type]
                 , SV.[Name]
                 , SV.SystemValue
                 , SV.SystemValueBaseType
                 , SV.SystemValueNumericValue
                 , SV.LastUpdate
                 , CAST(SV.SystemValue AS VARBINARY(MAX)) AS [Value]
                 , SV.SystemValueBaseType AS [ValueBaseType]
                 , COALESCE(T5.Value, SV.Unit, NULL) AS [Unit]
               FROM
                   ( SELECT
                       SV.Id
                     , A.AssetId
                     , A.[Type]
                     , S.Type + '_' + CASE SV.RollupType
                                        WHEN 1 THEN 'current'
                                        WHEN 2 THEN 'hourlyAvg'
                                        WHEN 4 THEN 'hourlyMax'
                                        WHEN 8 THEN 'dailyAvg'
                                        WHEN 16 THEN 'dailyMax'
                                        ELSE 'none'
                                      END AS [Name]
                     , SV.Value AS [SystemValue]
                     , 'decimal' AS [SystemValueBaseType]
                     , SV.Value AS [SystemValueNumericValue]
                     , SV.LastUpdate
                     , S.Unit
                     FROM
                       monitoring.SensorValue AS SV
                     INNER JOIN core.Sensor AS S
                       ON SV.SensorId = S.SensorId
                     INNER JOIN core.Asset AS A
                       ON S.DeviceId = A.AssetId
                     INNER JOIN [INSERTED] AS I
                       ON SV.Id = I.Id
                     UNION
                     SELECT
                       SV.Id
                     , LA.AssetId
                     , LA.[Type]
                     , NAS.Type + '_' + CASE SV.RollupType
                                          WHEN 1 THEN 'current'
                                          WHEN 2 THEN 'hourlyAvg'
                                          WHEN 4 THEN 'hourlyMax'
                                          WHEN 8 THEN 'dailyAvg'
                                          WHEN 16 THEN 'dailyMax'
                                          ELSE 'none'
                                        END
                     , SV.Value
                     , 'decimal'
                     , SV.Value
                     , SV.LastUpdate
                     , NAS.Unit
                     FROM
                       monitoring.SensorValue AS SV
                     INNER JOIN core.NonAssetSensor AS NAS
                       ON SV.SensorId = NAS.SensorId
                     INNER JOIN core.LogicalAsset AS LA
                       ON NAS.DeviceId = LA.AssetId
                     INNER JOIN [INSERTED] AS I
                       ON SV.Id = I.Id
                   ) AS SV
               LEFT JOIN settings.Translation AS T1 WITH ( NOLOCK )
                   ON SV.Type COLLATE SQL_Latin1_General_CP1_CS_AS = T1.Name
                      AND T1.Context = 'text'
                      AND T1.Locale = 'en'
               LEFT JOIN settings.Translation AS T5 WITH ( NOLOCK )
                   ON SV.Unit COLLATE SQL_Latin1_General_CP1_CS_AS = T5.Name
                      AND T5.Context = 'text'
                      AND T5.Locale = 'en'
   END
GO
CREATE TRIGGER [monitoring].[trg_SensorValueSearch_Update] ON [monitoring].[SensorValue]
   AFTER UPDATE
AS
   BEGIN
       SET NOCOUNT ON;
       
       UPDATE
           search.SearchValue
       SET
           SystemValue = SV2.Value
         , Value = CAST(SV2.Value AS VARBINARY(MAX))
         , LastUpdate = SV2.LastUpdate
       FROM
           [Inserted] AS I
       inner join search.SearchValue as SV1
           ON I.Id = SV1.Id
       INNER JOIN monitoring.SensorValue AS SV2
           ON I.Id = SV2.Id
   END
GO
