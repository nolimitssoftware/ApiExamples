CREATE TABLE [core].[Asset]
(
[AssetId] [uniqueidentifier] NOT NULL CONSTRAINT [DF_Asset_AssetId] DEFAULT (newsequentialid()),

    CREATE TRIGGER core.trg_AssetSearch ON core.Asset
   AFTER INSERT, UPDATE, DELETE
AS
   BEGIN
       SET NOCOUNT ON;
       DELETE
           search.SearchValue
       FROM
           search.SearchValue AS SV
       INNER JOIN [Deleted] AS D
           ON SV.AssetId = D.AssetId
       WHERE
           SV.SearchValueType = 1;

       INSERT  search.SearchValue
               ( Id
               , AssetId
               , SearchValueType
               , Type
               , Name
               , SystemValue
               , SystemValueBaseType
               , SystemValueStringValue
               , LastUpdate
               , Value
               , ValueBaseType
               )
               SELECT
                   NEWID() AS [Id]
                 , SV.AssetId
                 , 1 AS [SystemValueType]
                 , COALESCE(T1.Value, SV.[Type]) AS [Type]
                 , SV.[Name]
                 , SV.SystemValue
                 , SV.SystemValueBaseType
                 , COALESCE(T2.Value, SV.SystemValueStringValue, NULL) AS [SystemValueStringValue]
                 , GETUTCDATE() AS [LastUpdate]
                 , CAST(SV.SystemValue AS VARBINARY(MAX)) AS [Value]
                 , SV.SystemValueBaseType AS [ValueBaseType]
               FROM
                   ( SELECT
                       A.AssetId
                     , A.[Type]
                     , 'name' AS [Name]
                     , A.Name AS [SystemValue]
                     , 'nvarchar' AS [SystemValueBaseType]
                     , A.Name AS [SystemValueStringValue]
                     FROM
                       core.Asset AS A
                     INNER JOIN [INSERTED] AS I
                       ON A.AssetId = I.AssetId
                     UNION
                     SELECT
                       A.AssetId
                     , A.[Type]
                     , 'name'
                     , A.ShortName
                     , 'nvarchar'
                     , A.ShortName
                     FROM
                       core.Asset AS A
                     INNER JOIN [INSERTED] AS I
                       ON A.AssetId = I.AssetId
                     UNION
                     SELECT
                       A.AssetId
                     , A.[Type]
                     , 'name'
                     , A.OverrideName
                     , 'nvarchar'
                     , A.OverrideName
                     FROM
                       core.Asset AS A
                     INNER JOIN [INSERTED] AS I
                       ON A.AssetId = I.AssetId
                     UNION
                     SELECT
                       A.AssetId
                     , A.[Type]
                     , 'deviceName'
                     , A.Name
                     , 'nvarchar'
                     , A.Name
                     FROM
                       core.Asset AS A
                     INNER JOIN [INSERTED] AS I
                       ON A.AssetId = I.AssetId
                     UNION
                     SELECT
                       A.AssetId
                     , A.[Type]
                     , 'displayName'
                     , COALESCE(A.OverrideName, A.ShortName, A.Name)
                     , 'nvarchar'
                     , COALESCE(A.OverrideName, A.ShortName, A.Name)
                     FROM
                       core.Asset AS A
                     INNER JOIN [INSERTED] AS I
                       ON A.AssetId = I.AssetId
                     UNION
                     SELECT
                       A.AssetId
                     , A.[Type]
                     , 'status'
                     , ST.Name
                     , 'nvarchar'
                     , ST.Name
                     FROM
                       core.Asset AS A
                     INNER JOIN core.StatusType AS ST
                       ON A.Status = ST.Id
                     INNER JOIN [INSERTED] AS I
                       ON A.AssetId = I.AssetId
                     UNION
                     SELECT
                       A.AssetId
                     , A.[Type]
                     , 'manufacturer'
                     , M2.Name
                     , 'nvarchar'
                     , M2.Name
                     FROM
                       core.Asset AS A
                     INNER JOIN core.Model AS M
                       ON A.ModelId = M.Id
                     INNER JOIN core.Manufacturer AS M2
                       ON M.ManufacturerId = M2.Id
                     INNER JOIN [INSERTED] AS I
                       ON A.AssetId = I.AssetId
                     UNION
                     SELECT
                       A.AssetId
                     , A.[Type]
                     , 'model'
                     , M.Name
                     , 'nvarchar'
                     , M.Name
                     FROM
                       core.Asset AS A
                     INNER JOIN core.Model AS M
                       ON A.ModelId = M.Id
                     INNER JOIN [INSERTED] AS I
                       ON A.AssetId = I.AssetId
                     UNION
                     SELECT
                       A.AssetId
                     , A.[Type]
                     , 'type'
                     , A.[Type]
                     , 'varchar'
                     , A.[Type]
                     FROM
                       core.Asset AS A
                     INNER JOIN [INSERTED] AS I
                       ON A.AssetId = I.AssetId
                     UNION
                     SELECT
                       A.AssetId
                     , A.[Type]
                     , 'path'
                     , A.AssetNode.ToString()
                     , 'varchar'
                     , A.AssetNode.ToString()
                     FROM
                       core.Asset AS A
                     INNER JOIN [INSERTED] AS I
                       ON A.AssetId = I.AssetId
                     UNION
                     SELECT
                       A.AssetId
                     , A.[Type]
                     , 'location'
                     , core.GetSimpleAssetPath(A.AssetId)
                     , 'nvarchar'
                     , core.GetSimpleAssetPath(A.AssetId)
                     FROM
                       core.Asset AS A
                     INNER JOIN [INSERTED] AS I
                       ON A.AssetId = I.AssetId
                   ) AS SV
               LEFT JOIN settings.Translation AS T1 WITH ( NOLOCK )
                   ON SV.Type COLLATE SQL_Latin1_General_CP1_CS_AS = T1.Name
                      AND T1.Context = 'text'
                      AND T1.Locale = 'en'
               LEFT JOIN settings.Translation AS T2 WITH ( NOLOCK )
                   ON SV.SystemValueStringValue COLLATE SQL_Latin1_General_CP1_CS_AS = T2.Name
                      AND T2.Context = 'text'
                      AND T2.Locale = 'en';                
               
   END
