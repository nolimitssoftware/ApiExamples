CREATE TRIGGER core.trg_SerialNumberSearch ON core.SerialNumber
   AFTER INSERT, UPDATE, DELETE
AS
   BEGIN
       SET NOCOUNT ON;
       DELETE
           search.SearchValue
       FROM
           search.SearchValue AS SV
       INNER JOIN [Deleted] AS D
           ON SV.Id = D.Id;

       INSERT  search.SearchValue
               ( Id
               , AssetId
               , SearchValueType
               , Type
               , Name
               , SystemValue
               , SystemValueBaseType
               , SystemValueStringValue
               , LastUpdate
               , Value
               , ValueBaseType
               )
               SELECT
                   SV.[Id]
                 , SV.AssetId
                 , 6
                 , COALESCE(T1.Value, SV.[Type]) AS [Type]
                 , SV.[Name]
                 , SV.SystemValue
                 , SV.SystemValueBaseType
                 , SV.SystemValueStringValue
                 , GETUTCDATE()
                 , CAST(SV.SystemValue AS VARBINARY(MAX)) AS [Value]
                 , SV.SystemValueBaseType AS [ValueBaseType]
               FROM
                   ( SELECT
                       SN.Id
                     , A.AssetId
                     , A.[Type]
                     , 'serialNumber' AS [Name]
                     , SN.SerialNumber AS [SystemValue]
                     , 'varchar' AS [SystemValueBaseType]
                     , SN.SerialNumber AS [SystemValueStringValue]
                     FROM
                       core.SerialNumber AS SN
                     INNER JOIN core.Asset AS A
                       ON SN.AssetId = A.AssetId
                     INNER JOIN [INSERTED] AS I
                       ON SN.Id = I.Id
                   ) AS SV
               LEFT JOIN settings.Translation AS T1 WITH ( NOLOCK )
                   ON SV.Type COLLATE SQL_Latin1_General_CP1_CS_AS = T1.Name
                      AND T1.Context = 'text'
                      AND T1.Locale = 'en';
                              
   END
GO
