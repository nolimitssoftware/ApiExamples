CREATE TRIGGER core.trg_OperatingSystemSearch ON core.OperatingSystem
   AFTER INSERT, UPDATE, DELETE
AS
   BEGIN
       SET NOCOUNT ON;
       DELETE
           search.SearchValue
       FROM
           search.SearchValue AS SV
       INNER JOIN [Deleted] AS D
           ON SV.Id = D.Id;

       INSERT  search.SearchValue
               ( Id
               , AssetId
               , SearchValueType
               , Type
               , Name
               , SystemValue
               , SystemValueBaseType
               , SystemValueStringValue
               , LastUpdate
               , Value
               , ValueBaseType
               )
               SELECT
                   ISNULL(SV.[Id], NEWID()) AS [Id]
                 , SV.AssetId
                 , 8
                 , COALESCE(T1.Value, SV.[Type]) AS [Type]
                 , SV.[Name]
                 , SV.SystemValue
                 , SV.SystemValueBaseType
                 , SV.SystemValueStringValue
                 , GETUTCDATE()
                 , CAST(SV.SystemValue AS VARBINARY(MAX)) AS [Value]
                 , SV.SystemValueBaseType AS [ValueBaseType]
               FROM
                   ( SELECT
                       OS.Id
                     , A.AssetId
                     , A.[Type]
                     , 'operatingSystem' AS [Name]
                     , SOS.Name AS [SystemValue]
                     , 'varchar' AS [SystemValueBaseType]
                     , SOS.Name AS [SystemValueStringValue]
                     FROM
                       core.Asset AS A
                     INNER JOIN core.OperatingSystem AS OS
                       ON A.AssetId = OS.AssetId
                     INNER JOIN core.ServerOperatingSystem AS SOS
                       ON OS.ServerOperatingSystemId = SOS.Id
                     INNER JOIN [INSERTED] AS I
                       ON OS.Id = I.Id
                   ) AS SV
               LEFT JOIN settings.Translation AS T1 WITH ( NOLOCK )
                   ON SV.Type COLLATE SQL_Latin1_General_CP1_CS_AS = T1.Name
                      AND T1.Context = 'text'
                      AND T1.Locale = 'en';
               
   END
GO
