   CREATE TRIGGER core.trg_AssetStatisticSearch ON core.AssetStatistic
   AFTER INSERT, UPDATE, DELETE
AS
   BEGIN
       SET NOCOUNT ON;
       DELETE
           search.SearchValue
       FROM
           search.SearchValue AS SV
       INNER JOIN [Deleted] AS D
           ON SV.Id = D.Id;

       INSERT  search.SearchValue
               ( Id
               , AssetId
               , SearchValueType
               , Type
               , Name
               , SystemValue
               , SystemValueBaseType
               , SystemValueNumericValue
               , LastUpdate
               , Value
               , ValueBaseType
               , Unit
               )
               SELECT
                   ISNULL(SV.[Id], NEWID()) AS [Id]
                 , SV.AssetId
                 , 9
                 , COALESCE(T1.Value, SV.[Type]) AS [Type]
                 , SV.[Name]
                 , SV.SystemValue
                 , SV.SystemValueBaseType
                 , SV.SystemValueNumericValue
                 , GETUTCDATE()
                 , CAST(SV.SystemValue AS VARBINARY(MAX)) AS [Value]
                 , SV.SystemValueBaseType AS [ValueBaseType]
                 , COALESCE(T5.Value, SV.Unit, NULL) AS [Unit]
               FROM
                   ( SELECT
                       [AS].Id
                     , A.AssetId
                     , A.[Type]
                     , [AS].PropertyName AS [Name]
                     , [AS].PropertyValue AS [SystemValue]
                     , 'decimal' AS [SystemValueBaseType]
                     , [AS].PropertyValue AS [SystemValueNumericValue]
                     , [AS].Unit
                     FROM
                       core.Asset AS A
                     INNER JOIN core.AssetStatistic AS [AS]
                       ON A.AssetId = [AS].AssetId
                     INNER JOIN [INSERTED] AS I
                       ON [AS].Id = I.Id
                   ) AS SV
               LEFT JOIN settings.Translation AS T1 WITH ( NOLOCK )
                   ON SV.Type COLLATE SQL_Latin1_General_CP1_CS_AS = T1.Name
                      AND T1.Context = 'text'
                      AND T1.Locale = 'en'
               LEFT JOIN settings.Translation AS T5 WITH ( NOLOCK )
                   ON SV.Unit COLLATE SQL_Latin1_General_CP1_CS_AS = T5.Name
                      AND T5.Context = 'text'
                      AND T5.Locale = 'en';
               
   END
GO
