CREATE TRIGGER core.trg_LogicalAssetSearch ON core.LogicalAssetMap
   AFTER INSERT, UPDATE, DELETE
AS
   BEGIN
       SET NOCOUNT ON;
       DELETE
           search.SearchValue
       FROM
           search.SearchValue AS SV
       INNER JOIN [Deleted] AS D
           ON SV.Id = D.Id;

       INSERT  search.SearchValue
               ( Id
               , AssetId
               , SearchValueType
               , Type
               , Name
               , SystemValue
               , SystemValueBaseType
               , SystemValueStringValue
               , LastUpdate
               , Value
               , ValueBaseType
               )
               SELECT
                   ISNULL(SV.[Id], NEWID()) AS [Id]
                 , SV.AssetId
                 , 10
                 , COALESCE(T1.Value, SV.[Type]) AS [Type]
                 , SV.[Name]
                 , SV.SystemValue
                 , SV.SystemValueBaseType
                 , SV.SystemValueStringValue
                 , GETUTCDATE()
                 , CAST(SV.SystemValue AS VARBINARY(MAX)) AS [Value]
                 , SV.SystemValueBaseType AS [ValueBaseType]
               FROM
                   ( SELECT
                       LAM.Id
                     , A.AssetId
                     , A.[Type]
                     , 'logicalGroup' AS [Name]
                     , LA.ShortName AS [SystemValue]
                     , 'nvarchar' AS [SystemValueBaseType]
                     , LA.ShortName AS [SystemValueStringValue]
                     FROM
                       core.LogicalAssetMap AS LAM
                     INNER JOIN core.LogicalAsset AS LA
                       ON LAM.GroupAssetId = LA.AssetId
                     INNER JOIN core.Asset AS A
                       ON A.AssetId = LAM.AssetId
                     INNER JOIN [INSERTED] AS I
                       ON LAM.Id = I.Id
                   ) AS SV
               LEFT JOIN settings.Translation AS T1 WITH ( NOLOCK )
                   ON SV.Type COLLATE SQL_Latin1_General_CP1_CS_AS = T1.Name
                      AND T1.Context = 'text'
                      AND T1.Locale = 'en';
               
   END
GO
