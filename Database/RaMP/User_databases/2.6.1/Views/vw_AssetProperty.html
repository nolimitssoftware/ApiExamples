SQL Script
CREATE VIEW core.vw_AssetProperty
WITH
    SCHEMABINDING
AS
   SELECT
       AP.Id
     , AP.AssetId
     , AP.Name
     , AP.SystemValueCheckSum
     , AP.SystemValueSource
     , AP.UserValueCheckSum
     , AP.UserValueSource
     , AP.Display
     , AP.Editable
     , CSV.Value AS 'SystemValue'
     , CSV.BaseType AS 'SystemValueBaseType'
     , CSV.DateValue AS 'SystemValueDateValue'
     , CSV.NumericValue AS 'SystemValueNumericValue'
     , CSV.BooleanValue AS 'SystemValueBoolean'
     , CSV.StringValue AS 'SystemValueStringValue'
     , USV.Value AS 'UserValue'
     , USV.BaseType AS 'UserValueBaseType'
     , USV.DateValue AS 'UserValueDateValue'
     , USV.NumericValue AS 'UserValueNumericValue'
     , USV.BooleanValue AS 'UserValueBoolean'
     , USV.StringValue AS 'UserValueStringValue'
   FROM
       core.AssetProperty AS AP
   OUTER APPLY util.ConvertSqlVariant(AP.SystemValue) AS CSV
   OUTER APPLY util.ConvertSqlVariant(AP.UserValue) AS USV
GO
