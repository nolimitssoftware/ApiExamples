SQL Script
CREATE VIEW core.vw_CustomAssetProperty
WITH
    SCHEMABINDING
AS
   SELECT
       CAP.Id
     , CP.Name
     , CP.Type
     , CP.[Values]
     , CP.Unit
     , CP.AssetTypes
     , CP.IsActive
     , CP.IsSearchField
     , A.AssetId
     , CAP.CustomPropertyId
     , CAP.SystemValueCheckSum
     , CAP.SystemValueSource
     , CAP.UserValueCheckSum
     , CAP.UserValueSource
     , CAP.Display
     , CAP.Editable
     , CSV.Value AS 'SystemValue'
     , CSV.BaseType AS 'SystemValueBaseType'
     , CSV.DateValue AS 'SystemValueDateValue'
     , CSV.NumericValue AS 'SystemValueNumericValue'
     , CSV.BooleanValue AS 'SystemValueBoolean'
     , CSV.StringValue AS 'SystemValueStringValue'
     , USV.Value AS 'UserValue'
     , USV.BaseType AS 'UserValueBaseType'
     , USV.DateValue AS 'UserValueDateValue'
     , USV.NumericValue AS 'UserValueNumericValue'
     , USV.BooleanValue AS 'UserValueBoolean'
     , USV.StringValue AS 'UserValueStringValue'
     , CPV.Value AS 'DefaultValue'
     , CPV.BaseType AS 'DefaultValueBaseType'
     , CPV.DateValue AS 'DefaultValueDateValue'
     , CPV.NumericValue AS 'DefaultValueNumericValue'
     , CPV.BooleanValue AS 'DefaultValueBoolean'
     , CPV.StringValue AS 'DefaultValueStringValue'
   FROM
       core.Asset AS A
   INNER JOIN core.CustomAssetProperty AS CAP
       ON A.AssetId = CAP.AssetId
   INNER JOIN settings.CustomProperty AS CP
       ON CAP.CustomPropertyId = CP.Id
   CROSS APPLY util.ConvertSqlVariant(CP.DefaultValue) AS CPV
   CROSS APPLY util.ConvertSqlVariant(CAP.SystemValue) AS CSV
   CROSS APPLY util.ConvertSqlVariant(CAP.UserValue) AS USV
   UNION
   SELECT
       NEWID()
     , CP.Name
     , CP.Type
     , CP.[Values]
     , CP.Unit
     , CP.AssetTypes
     , CP.IsActive
     , CP.IsSearchField
     , A.AssetId
     , NULL
     , NULL
     , NULL
     , NULL
     , NULL
     , CAST (1 AS BIT)
     , CAST (1 AS BIT)
     , CSV.Value AS 'SystemValue'
     , CSV.BaseType AS 'SystemValueBaseType'
     , CSV.DateValue AS 'SystemValueDateValue'
     , CSV.NumericValue AS 'SystemValueNumericValue'
     , CSV.BooleanValue AS 'SystemValueBoolean'
     , CSV.StringValue AS 'SystemValueStringValue'
     , USV.Value AS 'UserValue'
     , USV.BaseType AS 'UserValueBaseType'
     , USV.DateValue AS 'UserValueDateValue'
     , USV.NumericValue AS 'UserValueNumericValue'
     , USV.BooleanValue AS 'UserValueBoolean'
     , USV.StringValue AS 'UserValueStringValue'
     , CPV.Value AS 'DefaultValue'
     , CPV.BaseType AS 'DefaultValueBaseType'
     , CPV.DateValue AS 'DefaultValueDateValue'
     , CPV.NumericValue AS 'DefaultValueNumericValue'
     , CPV.BooleanValue AS 'DefaultValueBoolean'
     , CPV.StringValue AS 'DefaultValueStringValue'
   FROM
       core.Asset AS A
   INNER JOIN settings.CustomProperty AS CP
       ON ( core.GetAssetType(A.Type) & CP.AssetTypes ) = core.GetAssetType(A.Type)
          OR CP.AssetTypes IS NULL
   CROSS APPLY util.ConvertSqlVariant(CP.DefaultValue) AS CPV
   CROSS APPLY util.ConvertSqlVariant(NULL) AS CSV
   CROSS APPLY util.ConvertSqlVariant(NULL) AS USV
   WHERE
       A.AssetId NOT IN ( SELECT
                           CAP.AssetId
                          FROM
                           core.CustomAssetProperty AS CAP INNER JOIN settings.CustomProperty ON CAP.CustomPropertyId = CP.Id );
GO
