SQL Script
CREATE VIEW settings.vw_CustomProperty
WITH
    SCHEMABINDING
AS
   SELECT
       CP.Id
     , CP.Name
     , CP.Type
     , CP.[Values]
     , CP.Unit
     , CP.AssetTypes
     , CP.IsActive
     , CP.IsSearchField
     , CAST(CP.DefaultValue AS VARBINARY(MAX)) AS [DefaultValueRaw]
     , CSV.Value AS 'DefaultValue'
     , CSV.BaseType AS 'DefaultValueBaseType'
     , CSV.DateValue AS 'DefaultValueDateValue'
     , CSV.NumericValue AS 'DefaultValueNumericValue'
     , CSV.BooleanValue AS 'DefaultValueBoolean'
     , CSV.StringValue AS 'DefaultValueStringValue'
   FROM
       settings.CustomProperty AS CP
   CROSS APPLY util.ConvertSqlVariant(CP.DefaultValue) AS CSV
GO
