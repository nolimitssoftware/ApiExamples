CREATE VIEW [core].[vw_ParentAsset]
WITH
    SCHEMABINDING
AS
   SELECT
       PA.[Child.AssetId] AS 'ChildAssetId'
     , GPA.[Parent.AssetId] AS 'AssetId'
     , GPA.[Parent.Name] AS 'Name'
     , GPA.[Parent.ShortName] AS 'ShortName'
     , GPA.[Parent.OverrideName] AS 'OverrideName'
     , GPA.[Parent.Type] AS 'Type'
     , GPA.[Parent.Status] AS 'Status'
     , GPA.[Parent.AssetType] AS 'AssetType'
     , GPA.[Parent.ParentId] AS 'ParentId'
     , GPA.[Parent.ExtendedProperties] AS 'ExtendedProperties'
   FROM
       (
         SELECT
           A.AssetId AS 'Child.AssetId'
         , A.[Name] AS 'Child.Name'
         , A.ShortName AS 'Child.ShortName'
         , A.OverrideName AS 'Child.OverrideName'
         , A.[Type] AS 'Child.Type'
         , A.[Status] AS 'Child.Status'
         , A.AssetType AS 'Child.AssetType'
         , (
             SELECT
               A2.AssetId
             FROM
               core.Asset AS A2
             WHERE
               A2.AssetNode = A.AssetNode.GetAncestor(1)
           ) AS 'Parent.ParentId'
         , (
             SELECT
               property.Name AS 'name'
             , property.SystemValue AS 'systemValue'
             , SQL_VARIANT_PROPERTY(property.SystemValue, 'basetype') AS 'systemValueType'
             , property.SystemValueSource AS 'systemValueSource'
             , property.UserValue AS 'userValue'
             , SQL_VARIANT_PROPERTY(property.UserValue, 'basetype') AS 'userValueType'
             , property.UserValueSource AS 'userValueSource'
             , property.Display AS 'display'
             , property.Editable AS 'editable'
             FROM
               core.AssetProperty AS property
             WHERE
               property.AssetId = A.AssetId
           FOR
             XML PATH('property')
               , ROOT('properties')
           ) AS 'Child.ExtendedProperties'
         FROM
           core.Asset AS A
         --WHERE
         --  A.AssetId = @Id
         
       ) AS PA
   CROSS APPLY core.GetParentAsset(PA.[Parent.ParentId]) AS GPA
GO
