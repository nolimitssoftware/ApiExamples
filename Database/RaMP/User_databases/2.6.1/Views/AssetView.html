SQL Script
CREATE VIEW core.AssetView
WITH
    SCHEMABINDING
AS
   SELECT
       A.AssetId
     , A.Name
     , A.ShortName
     , A.OverrideName
     , COALESCE(A.OverrideName, A.ShortName, A.Name) AS [DisplayName]
     , ISNULL(core.GetSimpleAssetPath(A.AssetId),'') AS [Location]
     , a.AssetNode.ToString() as AssetNode
     , M2.[Name] AS 'Manufacturer'
     , M.[Name] AS 'Model'
     , COALESCE(T1.Value, A.Type) AS [Type]
     , A.Type AS [TypeValue]
     , COALESCE(T2.Value, ST.Name) AS [Status]
     , A.ModelId
     , A.AssetType
     , A.AssetLevel
     , A.Notes
     , A.ExtendedProperties
     , A.TypeAtDiscovery
     , A.LifecycleStatus
   FROM
       core.Asset AS A
   LEFT JOIN core.Model AS M
       ON A.ModelId = M.Id
   LEFT JOIN core.Manufacturer AS M2
       ON M.ManufacturerId = M2.Id
   LEFT JOIN core.StatusType AS ST
       ON A.Status = ST.Id
   LEFT JOIN settings.Translation AS T1
   ON A.Type COLLATE SQL_Latin1_General_CP1_CS_AS = T1.Name
      AND T1.Context = 'text'
      AND T1.Locale = 'en'
   LEFT JOIN settings.Translation AS T2
   ON ST.Name COLLATE SQL_Latin1_General_CP1_CS_AS = T2.Name
      AND T2.Context = 'text'
      AND T2.Locale = 'en'
GO
