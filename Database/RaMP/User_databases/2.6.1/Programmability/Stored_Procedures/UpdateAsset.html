SQL Script
CREATE PROC core.UpdateAsset
   @AssetId UNIQUEIDENTIFIER
 , @Type NVARCHAR(32)
 , @TypeAtDiscovery NVARCHAR(32) = NULL
 , @Name NVARCHAR(256)
 , @ShortName NVARCHAR(256)
 , @OverrideName NVARCHAR(256)
 , @ModelId UNIQUEIDENTIFIER
 , @Notes VARCHAR(MAX) = NULL
 , @AssetXml XML
AS
   BEGIN
         --NOTE: DO NOT update status here.  It will be updated by its active messages.

       UPDATE
           core.Asset
       SET
           [Type] = @Type
         , TypeAtDiscovery = ISNULL(@TypeAtDiscovery, A.TypeAtDiscovery)
         , [Name] = @Name
         , [ShortName] = @ShortName
         , [OverrideName] = @OverrideName
         , ModelId = @ModelId
         , Notes = @Notes
       FROM
           core.Asset AS A
       WHERE
           A.AssetId = @AssetId;
   END
GO
