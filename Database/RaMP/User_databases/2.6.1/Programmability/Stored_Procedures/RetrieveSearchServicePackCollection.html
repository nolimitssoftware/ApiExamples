CREATE PROCEDURE settings.RetrieveSearchServicePackCollection
   @Search VARCHAR(124) = NULL
 , @SortField VARCHAR(512) = NULL
 , @Start INT
 , @Limit INT
 , @IsAscending BIT = 1
 , @IsDebug BIT = 0
 , @TotalCount INT OUT
AS
   BEGIN
   
       SET NOCOUNT ON ;
       DECLARE @OperatingSystemId UNIQUEIDENTIFIER ;
       
       SELECT
           @OperatingSystemId = SOS.Id
       FROM
           core.ServerOperatingSystem AS SOS
       WHERE
           SOS.Name = @Search ;
       
       IF ( @OperatingSystemId IS NOT NULL )
           BEGIN
       
               DECLARE @End INT ;
       
               SELECT
                   @TotalCount = COUNT(1)
               FROM
                   core.ServicePack AS SP
               WHERE
                   SP.ServerOperatingSystemId = @OperatingSystemId
   
               IF ( @Limit IS NULL )
                   BEGIN
                       SET @Limit = @TotalCount ;
                   END
       
               SET @End = @Start + @Limit - 1 ;
   
               DECLARE
                   @sql NVARCHAR(MAX)
                 , @params NVARCHAR(4000)
                 , @sort VARCHAR(1024) ;     
               IF (
                    @SortField IS NULL
                    OR @SortField = 'Name'
                  )
                   BEGIN
                       SET @SortField = 'ServicePack.Name' ;
                   END
   
               SET @sort = ', ROW_NUMBER() OVER ( ORDER BY [' + @SortField
                   + ']' ;
               IF ( @IsAscending = 1 )
                   BEGIN
                       SET @sort += ' ASC ' ;
                   END
               ELSE
                   BEGIN
                       SET @sort += ' DESC ' ;
                   END
       
               SET @sort += ' ) AS RowNum ' ;
               SET @sql = '
   SELECT
       *
   FROM
       (
       SELECT temp.* '
               SET @sql += @sort
       
               SET @sql += '
       FROM(
         SELECT
           M.Id AS ''ManufacturerId''
         , M.Name AS ''Manufacturer''
         , SUBSTRING(temp.Names, 0, LEN(temp.Names)) AS ''ManufacturerAlternateNames''
         , SP.ServerOperatingSystemId AS ''ServerOperatingSystem.Id''
         , SP.Id AS ''ServicePack.Id''
         , SOS.Name AS ''ServerOperatingSystem.Name''
         , CASE WHEN SP.Name IS NULL THEN ''emptyServicePack'' WHEN LEN(SP.Name) = 0 THEN ''emptyServicePack'' ELSE SP.Name END AS ''ServicePack.Name''
         , SP.VersionString
         , SP.SearchString
         , SP.Major
         , SP.Minor
         , SP.Revision
         , SP.Build
         , SP.Other
         , SP.Notes
         , SP.ExtendedProperties
         , SP.IsSystemDefined
         , SP.SentToNLS
         , SP.InternalId
         , 0 AS ''LicenseCount''
         , CASE SOS.IsApproved
             WHEN CAST(1 AS BIT) THEN SP.IsApproved
             ELSE CAST(0 AS BIT)
           END AS ''IsApproved''
         , SP.IsApproved AS [HasUnapprovedServicePack]
         , core.CountOSInstallations(SOS.Id, SP.Id, 1) AS ''InstallationCount''
            , 0 as ServicePackCount
       FROM
           core.Manufacturer AS M
       INNER JOIN core.ServerOperatingSystem AS SOS
           ON M.Id = SOS.ManufacturerId
       INNER JOIN core.ServicePack AS SP
           ON SOS.Id = SP.ServerOperatingSystemId
       CROSS APPLY (
                     SELECT
                       CONVERT(VARCHAR(256), ISNULL(MAN.[Name], '''')) + '','' AS [text()]
                     FROM
                       core.ManufacturerAlternateName AS MAN
                     WHERE
                       MAN.ManufacturerId = M.Id
                   FOR
                     XML PATH('''')
                   ) AS temp ( [Names] )                      
       WHERE 1=1
       AND SP.ServerOperatingSystemId = @OperatingSystemId '                           
                     
               SET @sql += '
       ) AS temp ) AS OS
   WHERE
       OS.RowNum BETWEEN @Start AND @End
   ORDER BY
       OS.RowNum ;
   '
       
               SET @params = '
 @OperatingSystemId UNIQUEIDENTIFIER
, @Search VARCHAR(124)
, @Start INT
, @End INT    
'        

               IF ( @IsDebug = 1 )
                   BEGIN
                       PRINT @params ;
                       PRINT @sql ;
                   END

               EXEC sp_executesql
                   @sql
                 , @params
                 , @OperatingSystemId
                 , @Search
                 , @Start
                 , @End
       
           END
   END
GO
