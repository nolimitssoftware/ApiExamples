PropertyValue
ANSI Nulls OnTrue

SQL Script
CREATE PROC notification.CleanUpOrphanedMessageNotifications
AS
   BEGIN
   
       DECLARE @Parents TABLE
           (
             AssetId UNIQUEIDENTIFIER
           , NotificationGroupId UNIQUEIDENTIFIER
           , Critical BIT
           , Warning BIT
           , Unreachable BIT
           , Administration BIT
           , Maintenance BIT
           , [Row] INT IDENTITY(1, 1)
           );

       WITH    tree ( [Path], [PathName], AssetId, [Level], [Type], NotificationGroupId, Critical, Warning, Unreachable, Administration, Maintenance )
                 AS ( SELECT
                       A.AssetNode AS [Path]
                     , A.AssetNode.ToString() AS [PathName]
                     , A.AssetId
                     , A.AssetLevel AS [Level]
                     , A.Type
                     , ANS.NotificationGroupId
                     , ANS.Critical
                     , ANS.Warning
                     , ANS.Unreachable
                     , ANS.Administration
                     , ANS.Maintenance
                      FROM
                       core.Asset AS A
                      INNER JOIN notification.AssetNotificationSetting
                       AS ANS
                       ON A.AssetId = ANS.AssetId
                      WHERE
                       A.Type = 'physicalGroup'
                      UNION ALL
                      SELECT
                       A.AssetNode AS [Path]
                     , A.AssetNode.ToString() AS [PathName]
                     , A.AssetId
                     , CAST(( t.[Level] - 1 ) AS SMALLINT) AS [Level]
                     , A.[Type]
                     , t.NotificationGroupId
                     , t.Critical
                     , t.Warning
                     , t.Unreachable
                     , t.Administration
                     , t.Maintenance
                      FROM
                       core.Asset AS A
                      INNER JOIN tree t
                       ON A.AssetNode.IsDescendantOf(t.Path) = 1
                      WHERE
                       A.[AssetLevel] = t.[Path].GetLevel()
                       + 1
                    )
           INSERT  @Parents
                   ( AssetId
           --, [Type]
                   , NotificationGroupId
                   , Critical
                   , Warning
                   , Unreachable
                   , Administration
                   , Maintenance
                   
                   )
                   SELECT DISTINCT
                       t.AssetId
             --, t.[Type]
                     , t.NotificationGroupId
                     , t.Critical
                     , t.Warning
                     , t.Unreachable
                     , t.Administration
                     , t.Maintenance
                   FROM
                       tree AS t;
               
       INSERT  @Parents
               ( AssetId
       --, [Type]
               , NotificationGroupId
               , Critical
               , Warning
               , Unreachable
               , Administration   
               , Maintenance             
               )
               SELECT
                   missing.AssetId
         --, missing.[Type]
                 , missing.NotificationGroupId
                 , missing.Critical
                 , missing.Warning
                 , missing.Unreachable
                 , missing.Administration
                 , missing.Maintenance
               FROM
                   ( SELECT
                       A.AssetId
                     , A.[Type]
                     , ANS.NotificationGroupId
                     , ANS.Critical
                     , ANS.Warning
                     , ANS.Unreachable
                     , ANS.Administration
                     , ANS.Maintenance
                     FROM
                       notification.AssetNotificationSetting
                       AS ANS
                     INNER JOIN core.Asset AS A
                       ON ANS.AssetId = A.AssetId
                   ) AS missing
               LEFT JOIN @Parents AS P
                   ON P.AssetId = missing.AssetId
                      AND P.NotificationGroupId = missing.NotificationGroupId
               WHERE
                   P.AssetId IS NULL

       DELETE
           notification.MessageNotification
       FROM
           notification.MessageNotification AS MN
       INNER JOIN log.Message AS M
           ON MN.MessageId = M.Id
       WHERE
           MN.MessageId NOT IN (
           SELECT
               MN.MessageId
           FROM
               notification.MessageNotification AS MN
           INNER JOIN log.Message AS M
               ON MN.MessageId = M.Id
           LEFT JOIN @Parents AS P
               ON M.AssetId = P.AssetId
           WHERE
               P.Unreachable = 1
               AND M.Severity = 16 )
           AND M.Severity = 16

       DELETE
           notification.MessageNotification
       FROM
           notification.MessageNotification AS MN
       INNER JOIN log.Message AS M
           ON MN.MessageId = M.Id
       WHERE
           MN.MessageId NOT IN (
           SELECT
               MN.MessageId
           FROM
               notification.MessageNotification AS MN
           INNER JOIN log.Message AS M
               ON MN.MessageId = M.Id
           LEFT JOIN @Parents AS P
               ON M.AssetId = P.AssetId
           WHERE
               P.Critical = 1
               AND M.Severity = 32 )
           AND M.Severity = 32

       DELETE
           notification.MessageNotification
       FROM
           notification.MessageNotification AS MN
       INNER JOIN log.Message AS M
           ON MN.MessageId = M.Id
       WHERE
           MN.MessageId NOT IN (
           SELECT
               MN.MessageId
           FROM
               notification.MessageNotification AS MN
           INNER JOIN log.Message AS M
               ON MN.MessageId = M.Id
           LEFT JOIN @Parents AS P
               ON M.AssetId = P.AssetId
           WHERE
               P.Warning = 1
               AND M.Severity = 64 )
           AND M.Severity = 64

       DELETE
           notification.MessageNotification
       FROM
           notification.MessageNotification AS MN
       INNER JOIN log.Message AS M
           ON MN.MessageId = M.Id
       WHERE
           MN.MessageId NOT IN (
           SELECT
               MN.MessageId
           FROM
               notification.MessageNotification AS MN
           INNER JOIN log.Message AS M
               ON MN.MessageId = M.Id
           LEFT JOIN @Parents AS P
               ON M.AssetId = P.AssetId
           WHERE
               P.Administration = 1
               AND M.Severity = 2048 )
           AND M.Severity = 2048

       DELETE
           notification.MessageNotification
       FROM
           notification.MessageNotification AS MN
       INNER JOIN log.Message AS M
           ON MN.MessageId = M.Id
       WHERE
           MN.MessageId NOT IN (
           SELECT
               MN.MessageId
           FROM
               notification.MessageNotification AS MN
           INNER JOIN log.Message AS M
               ON MN.MessageId = M.Id
           LEFT JOIN @Parents AS P
               ON M.AssetId = P.AssetId
           WHERE
               P.Maintenance = 1
               AND M.Severity = 4 )
           AND M.Severity = 4

   END
GO
Uses

  • [log].[Message]
