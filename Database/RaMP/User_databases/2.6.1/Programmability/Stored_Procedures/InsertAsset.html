SQL Script
CREATE PROCEDURE [core].[InsertAsset]
    @AssetId UNIQUEIDENTIFIER = NULL
 , @AssetType NVARCHAR(64)
 , @AssetXml XML = NULL
 , @Type NVARCHAR(32) = 'device'
 , @TypeAtDiscovery NVARCHAR(32) = NULL
 , @Status INT = 65536
 , @LifecycleStatus INT = 0
 , @Name NVARCHAR(256) = 'Unknown'
 , @ShortName NVARCHAR(256) = 'Unknown'
 , @OverrideName NVARCHAR(256) = NULL
 , @ModelId UNIQUEIDENTIFIER = NULL
 , @ParentAssetId UNIQUEIDENTIFIER = NULL
 , @RampXml XML = NULL  
AS
    BEGIN
           
       IF ( @ParentAssetId IS NULL )
           BEGIN
                   SELECT
                       @ParentAssetId = L.AssetId
                   FROM
                       core.Asset AS L
                   WHERE
                       L.AssetNode = HIERARCHYID::GetRoot() ;
           END
               
       DECLARE @ParentNode HIERARCHYID ;
               
       SELECT
           @ParentNode = L.AssetNode
       FROM
           core.Asset AS L
       WHERE
           L.AssetId = @ParentAssetId ;
               
       DECLARE @LastAsset HIERARCHYID ;
               
       SELECT
           @LastAsset = MAX(L.AssetNode)
       FROM
           core.Asset AS L
       WHERE
           L.AssetNode.GetAncestor(1) = @ParentNode ;

       DECLARE @Output TABLE(AssetId UNIQUEIDENTIFIER);

       IF (@AssetId IS NULL)
           BEGIN
                   INSERT  core.Asset
                   (
                     AssetType
                   , AssetXml
                   , AssetNode
                   , [Type]
                   , [Name]
                   , [ShortName]
                   , OverrideName
                   , [Status]
                   , ModelId
                   , RampXml                                
                   , TypeAtDiscovery
                   , LifecycleStatus
                   )
           OUTPUT  INSERTED.AssetId
           INTO @Output
           VALUES
                   (
                     @AssetType
                   , @AssetXml
                   , @ParentNode.GetDescendant(@LastAsset, NULL)
                   , @Type
                   , @Name
                   , @ShortName
                   , @OverrideName
                   , @Status
                   , @ModelId
                   , @RampXml            
                   , ISNULL(@TypeAtDiscovery, @Type)
                   , @LifecycleStatus
                   ) ;
           END
       ELSE
           BEGIN
                   INSERT  core.Asset
                   (
                     AssetId
                   , AssetType
                   , AssetXml
                   , AssetNode
                   , [Type]
                   , [Name]
                   , [ShortName]
                   , OverrideName
                   , [Status]
                   , ModelId
                   , RampXml                                
                   , TypeAtDiscovery
                   , LifecycleStatus
                   )
           OUTPUT  INSERTED.AssetId
           INTO @Output
           VALUES
                   (
                     @AssetId
                   , @AssetType
                   , @AssetXml
                   , @ParentNode.GetDescendant(@LastAsset, NULL)
                   , @Type
                   , @Name
                   , @ShortName
                   , @OverrideName
                   , @Status
                   , @ModelId
                   , @RampXml            
                   , ISNULL(@TypeAtDiscovery, @Type)
                   , @LifecycleStatus
                   ) ;
           END

       SELECT * FROM @Output AS O;

    END
GO
