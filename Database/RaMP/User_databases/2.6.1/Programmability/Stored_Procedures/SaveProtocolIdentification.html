CREATE PROCEDURE [settings].[SaveProtocolIdentification]
   @ProtocolId UNIQUEIDENTIFIER
 , @ProtocolDesignId UNIQUEIDENTIFIER
 , @Identifier TINYINT
 , @Comparison TINYINT
 , @Value VARCHAR(256) = NULL
 , @UserDefined BIT
 , @Order INT = 1
 , @Oid VARCHAR(256) = NULL
 , @Class VARCHAR(128) = NULL
 , @Property VARCHAR(128) = NULL
 , @Command VARCHAR(512) = NULL
AS
   BEGIN
       DECLARE @ProtocolIdentificationId UNIQUEIDENTIFIER = (
                                                              SELECT
                                                               PI.ProtocolIdentificationId
                                                              FROM
                                                               settings.ProtocolIdentification AS PI
                                                              WHERE
                                                               PI.ProtocolId = @ProtocolId
                                                               AND PI.ProtocolDesignId = PI.ProtocolDesignId
                                                               AND PI.UserDefined = @UserDefined
                                                               AND PI.Identifier = @Identifier
                                                               AND PI.Comparison = @Comparison
                                                               AND (
                                                                     @Value IS NULL
                                                                     OR (
                                                                          @Value IS NOT NULL
                                                                          AND PI.Value = @Value
                                                                        )
                                                                   )
                                                               AND (
                                                                     @Oid IS NULL
                                                                     OR (
                                                                          @Oid IS NOT NULL
                                                                          AND @Oid = @Oid
                                                                        )
                                                                   )
                                                            )

       UPDATE
           settings.ProtocolIdentification
       SET
           [Order] = PI.[Order] + 1
       FROM
           settings.ProtocolIdentification AS PI
       WHERE
           PI.ProtocolId = @ProtocolId
           AND PI.[Order] >= @Order;
           
       IF ( @ProtocolIdentificationId IS NOT NULL )
           BEGIN
                           
               UPDATE
                   settings.ProtocolIdentification
               SET
                   Identifier = @Identifier
                 , Comparison = @Comparison
                 , Value = @Value
                 , Oid = @Oid
                 , Class = @Class
                 , Property = @Property
                 , Command = @Command
               WHERE
                   settings.ProtocolIdentification.ProtocolIdentificationId = @ProtocolIdentificationId;
                   
           END                                                             
       ELSE
           BEGIN
               INSERT  settings.ProtocolIdentification
                       (
                         ProtocolId
                       , ProtocolDesignId
                       , Identifier
                       , Comparison
                       , Value
                       , [Order]
                       , UserDefined
                       , Oid
                       , Class
                       , Property
                       , Command
                       )
               VALUES
                       (
                         @ProtocolId
                       , @ProtocolDesignId
                       , @Identifier
                       , @Comparison
                       , @Value
                       , @Order
                       , @UserDefined
                       , @Oid
                       , @Class
                       , @Property
                       , @Command
                       )
           END
   END
GO
