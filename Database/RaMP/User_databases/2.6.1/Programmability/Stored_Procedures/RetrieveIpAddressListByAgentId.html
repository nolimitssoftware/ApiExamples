CREATE PROCEDURE core.RetrieveIpAddressListByAgentId
   @DiscoveryAgentId UNIQUEIDENTIFIER
 , @MinTimeBetweenUpDownMonitoring INT
 , @NumberOfMonitoredDeviceInList INT
AS
   BEGIN
       SELECT TOP ( @NumberOfMonitoredDeviceInList )
           H.Id
         , H.[Address]
         , H.AssetId
       FROM
           core.Host AS H
           JOIN core.Asset AS A ON H.AssetId = A.AssetId
       LEFT JOIN monitoring.Host AS H2
           ON H.Id = H2.HostId
       WHERE
           H.AgentId = @DiscoveryAgentId
           AND GETUTCDATE() > DATEADD(Second,
                                      @MinTimeBetweenUpDownMonitoring,
                                      H2.LastCheck)
           AND core.IsInExcludedAssetNode(A.AssetNode) = 0
       ORDER BY
           H2.LastCheck
           
   END
GO
