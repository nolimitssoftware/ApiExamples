CREATE PROCEDURE [settings].[InsertDiscovery]
   @Name NVARCHAR(32) ,
   @DiscoveryAgentId UNIQUEIDENTIFIER = '510E20A0-03D5-4D3C-9936-3EE817C65189' ,
   @DefaultPhysicalGroupId UNIQUEIDENTIFIER = 'BC4C8650-810D-44D4-BA11-AFEFE75EF939' -- Unknown Group
AS
   BEGIN
       
       DECLARE @DefaultPhysicalGroupName VARCHAR(256)
       SELECT @DefaultPhysicalGroupName = [Name] FROM core.asset WHERE assetId = @DefaultPhysicalGroupId

       INSERT INTO [settings].[Discovery]
             ([Name]
             ,[DiscoveryAgentId]
             ,[DefaultPhysicalGroupId])
       OUTPUT
           INSERTED.*, @DefaultPhysicalGroupName AS 'DefaultPhysicalGroupName'
       VALUES
             (@Name
             ,@DiscoveryAgentId
             ,@DefaultPhysicalGroupId)
   END
GO
