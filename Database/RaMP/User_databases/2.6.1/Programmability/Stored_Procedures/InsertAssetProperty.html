CREATE PROCEDURE core.InsertAssetProperty
    @AssetId UNIQUEIDENTIFIER
 , @Name VARCHAR(256)
 , @SystemValue SQL_VARIANT
 , @UserValue SQL_VARIANT
 , @SystemValueSource VARCHAR(32) = NULL
 , @UserValueSource VARCHAR(32) = NULL
 , @Display BIT = 1
 , @Editable BIT = 0
AS
    BEGIN
         INSERT  core.AssetProperty
                    (
                       AssetId
                    , Name
                    , SystemValue
                    , UserValue
                    , Display
                    , Editable
                   , SystemValueSource
                    , UserValueSource
                )          
         VALUES
                    (
                       @AssetId
                    , @Name
                    , @SystemValue
                    , @UserValue
                    , @Display
                    , @Editable
                    , @SystemValueSource
                    , @UserValueSource
                )
    END
GO
