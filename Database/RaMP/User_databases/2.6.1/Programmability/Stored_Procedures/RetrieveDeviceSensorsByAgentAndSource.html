CREATE PROCEDURE core.RetrieveDeviceSensorsByAgentAndSource
   @DiscoveryAgentId UNIQUEIDENTIFIER
 , @Source VARCHAR(32)
 , @MinTimeBetweenSensorMonitoring INT
 , @NumberOfMonitoredSensorsInList INT
AS
   BEGIN
       SELECT TOP ( @NumberOfMonitoredSensorsInList )
           S.SensorId
         , S.DeviceId
         , S.DiscoveryAgentId
         , S.Source
         , S.Type
         , S.SystemName
         , S.UserName
         , S.Unit
         , S.Element
         , S.OriginType
         , S.LocationAssetId
         , S.RackLocation
         , S.Position
         , CONVERT(NVARCHAR(MAX), S.SensorXml) AS [SensorXml]
         , CONVERT(NVARCHAR(MAX), S.ExtendedProperties) AS [ExtendedProperties]
         , D.LastUpdate
       FROM
           core.Sensor AS S
       INNER JOIN core.Host AS H
           ON S.DeviceId = H.AssetId
       INNER JOIN core.Asset AS A
           ON H.AssetId = A.AssetId
       INNER JOIN monitoring.Host AS H2
           ON H.Id = H2.HostId
       CROSS APPLY ( SELECT TOP 1
                       SV.LastUpdate
                     , SV.SensorId
                     FROM
                       monitoring.SensorValue AS SV
                     WHERE
                       S.SensorId = SV.SensorId
                       AND S.Unit <> 'value' --rrd value
                     ORDER BY
                       SV.LastUpdate
                   ) AS D
       WHERE
           H2.Available = 1
           AND S.DiscoveryAgentId = @DiscoveryAgentId
           AND S.Source = @Source
           AND GETUTCDATE() > DATEADD(Second, @MinTimeBetweenSensorMonitoring,
                                      D.LastUpdate)
           AND core.IsInExcludedAssetNode(A.AssetNode) = 0
       UNION
       SELECT
           S.SensorId
         , S.DeviceId
         , S.DiscoveryAgentId
         , S.Source
         , S.Type
         , S.SystemName
         , S.UserName
         , S.Unit
         , S.Element
         , S.OriginType
         , S.LocationAssetId
         , S.RackLocation
         , S.Position
         , CONVERT(NVARCHAR(MAX), S.SensorXml) AS [SensorXml]
         , CONVERT(NVARCHAR(MAX), S.ExtendedProperties) AS [ExtendedProperties]
         , D.LastUpdated
       FROM
           core.Sensor AS S
       INNER JOIN core.Host AS H
           ON S.DeviceId = H.AssetId
       INNER JOIN core.Asset AS A
           ON H.AssetId = A.AssetId
       INNER JOIN monitoring.Host AS H2
           ON H.Id = H2.HostId
       CROSS APPLY ( SELECT TOP 1
                       AP.LastUpdated
                     , AP.Name
                     FROM
                       core.AssetProperty AS AP
                     WHERE
                       AP.AssetId = S.DeviceId
                       AND AP.[Name] = S.Type
                       AND S.Unit = 'value'
                     ORDER BY
                       AP.LastUpdated
                   ) AS D
       WHERE
           H2.Available = 1
           AND S.DiscoveryAgentId = @DiscoveryAgentId
           AND S.Source = @Source
           AND GETUTCDATE() > DATEADD(Second, @MinTimeBetweenSensorMonitoring,
                                      D.LastUpdated)
           AND core.IsInExcludedAssetNode(A.AssetNode) = 0
       ORDER BY
           D.LastUpdate
   END
GO
