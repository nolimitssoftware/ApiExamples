CREATE PROC core.FindAssetByPropertyAndSystemValue
   @PropertyName VARCHAR(256)
 , @SystemValue SQL_VARIANT = NULL
AS
   BEGIN
       SELECT
             AP.AssetId              
       FROM
           core.AssetProperty AS AP
       WHERE
           AP.Name = @PropertyName
           AND AP.SystemValueCheckSum = CHECKSUM(@SystemValue)
           AND AP.SystemValue = @SystemValue;
   END
GO
