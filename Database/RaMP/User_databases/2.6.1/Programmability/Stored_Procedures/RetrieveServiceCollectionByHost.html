SQL Script
CREATE PROC core.RetrieveServiceCollectionByHost
   @IpAddress VARCHAR(56)
 , @AssetId UNIQUEIDENTIFIER
AS
   BEGIN
       SELECT
           S.Id
         , S.IpAddress
         , S.AssetId
         , S.Name
         , S.Port
         , S.[Source]
         , S.ExtendedProperties
         --, core.RetrieveHostNameListByHost(S.HostId) AS 'Hostnames'
         , NULL AS 'Hostnames'
       FROM
           core.Service AS S
       INNER JOIN core.Asset AS A
           ON A.AssetId = S.AssetId
       WHERE
           S.AssetId = @AssetId
           AND S.IpAddress = @IpAddress;
   END
GO
