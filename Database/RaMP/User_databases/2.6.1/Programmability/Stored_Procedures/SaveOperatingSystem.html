CREATE PROC core.SaveOperatingSystem
   @Id UNIQUEIDENTIFIER = NULL
 , @AssetId UNIQUEIDENTIFIER
 , @Name VARCHAR(256) = NULL
 , @Manufacturer VARCHAR(256) = NULL
 , @ServerOperatingSystemId UNIQUEIDENTIFIER = NULL
 , @Architecture VARCHAR(16) = NULL
 , @ServicePackId UNIQUEIDENTIFIER = NULL
 , @Source VARCHAR(32)
 , @ExtendedProperties XML = NULL
 , @AgentId UNIQUEIDENTIFIER = NULL
 , @PropertyXml XML = NULL
AS
   BEGIN          

       DECLARE @Output TABLE(Id UNIQUEIDENTIFIER);

       IF ( @Id IS NULL )
           BEGIN
           
               IF ( @ServerOperatingSystemId IS NULL )
                   BEGIN
                       SELECT
                           @ServerOperatingSystemId = M.Id
                       FROM
                           core.ServerOperatingSystem AS SOS
                       INNER JOIN core.Manufacturer AS M
                           ON SOS.ManufacturerId = M.Id
                       LEFT JOIN core.ManufacturerAlternateName AS MAN
                           ON M.Id = MAN.ManufacturerId
                       WHERE
                           (
                             M.Name = @Manufacturer
                             OR MAN.Name = @Manufacturer
                           )
                           AND SOS.Name = @Name
                   END
               
               DELETE
                   core.OperatingSystem
               WHERE
                   AssetId = @AssetId ;
               
               INSERT  core.OperatingSystem
                       (
                         AssetId
                       , ServerOperatingSystemId
                       , Architecture
                       , ServicePackId
                       , [Source]
                       , ExtendedProperties
                       , AgentId
                       , PropertyXml                      
                       )
               OUTPUT  INSERTED.Id
               INTO @Output
               VALUES
                       (
                         @AssetId
                       , @ServerOperatingSystemId
                       , @Architecture
                       , @ServicePackId
                       , @Source
                       , @ExtendedProperties
                       , @AgentId
                       , @PropertyXml                      
                       )
           END
       ELSE
           BEGIN
       
               UPDATE
                   core.OperatingSystem
               SET
                   ServerOperatingSystemId = @ServerOperatingSystemId
                 , Architecture = @Architecture
                 , ServicePackId = @ServicePackId
                 , ExtendedProperties = @ExtendedProperties
                 , AgentId = @AgentId
                 , PropertyXml = @PropertyXml
               OUTPUT
                   INSERTED.Id
               INTO @Output
               WHERE
                   core.OperatingSystem.Id = @Id ;
       
           END
           
       SELECT * FROM @Output;            
   END
GO
