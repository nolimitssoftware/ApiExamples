CREATE PROCEDURE [settings].[RetrieveCustomPropertyByName] @Name NVARCHAR(256)
AS
   BEGIN
       SELECT
           CP.Id
         , CP.Name
         , CP.Type
         , CP.[Values]
         , CP.Unit
         , CP.AssetTypes
         , CP.IsActive
         , CP.IsSearchField
         , CP.DefaultValue
         , SQL_VARIANT_PROPERTY(CP.DefaultValue, 'baseType') AS 'DefaultValueType'
       FROM
           settings.CustomProperty AS CP
       WHERE
           CP.[Name] = @Name ;
   END
GO
