CREATE PROCEDURE [doc].[RetrieveIndirectDocumentLinks]
    @ContextId UNIQUEIDENTIFIER
AS
   BEGIN
       -- Indirect links are the following:
       --    * For locations (racks, physicalGroups) and assets: all ancestor documents.
       --    * If it has a model then the model's direct documents.
       --    * One level up the hierarchy chain model documents (if applicable)
       --    * For assets in logical group(s): The logical group(s) direct documents.
   
       DECLARE
           @Type NVARCHAR(32),
           @AssetNode HIERARCHYID,
           @AssetLevel SMALLINT,
           @AssetModelId UNIQUEIDENTIFIER ;
       
       SELECT
           @Type = A.[Type],
           @AssetNode = A.AssetNode,
           @AssetLevel = A.AssetLevel,
           @AssetModelId = A.ModelId
       FROM
           [core].[Asset] AS A
       WHERE
           A.AssetId = @ContextId
           
       IF (@Type IS NOT NULL)
           BEGIN
               --Build a temp table hold the document ids
               DECLARE @IndirectDocument TABLE
                   (
                   DocumentId UNIQUEIDENTIFIER,
                   ContextId UNIQUEIDENTIFIER,
                   Context INT,
                   AssetType NVARCHAR (32),
                   ContextName NVARCHAR (256),
                   ContextShortName NVARCHAR (256),
                   ContextOverrideName NVARCHAR (256)
                   )
                   
               --Get ancestor docs
               INSERT  INTO @IndirectDocument
                   (
                       DocumentId
                   , ContextId
                   , Context
                   , AssetType
                   , ContextName
                   , ContextShortName
                   , ContextOverrideName
                   )
                   SELECT
                       DDL.DocumentId,
                       A.AssetId,
                       DDL.Context,
                       A.[Type],
                       A.Name,
                       A.ShortName,
                       A.OverrideName
                   FROM
                       doc.DocumentDirectLink AS DDL
                   INNER JOIN
                       core.asset AS A ON DDL.ContextId = A.AssetId
                   WHERE
                       @AssetNode.IsDescendantOf(A.[AssetNode]) = 1 AND A.AssetId <> @ContextId ;
                           
               --Get logical group docs
               INSERT  INTO @IndirectDocument
                   (
                       DocumentId
                   , ContextId
                   , Context
                   , AssetType
                   , ContextName
                   , ContextShortName
                   , ContextOverrideName
                   )
                   SELECT
                       DDL.DocumentId,
                       LA.AssetId,
                       DDL.Context,
                       LA.[Type],
                       LA.Name,
                       LA.ShortName,
                       LA.OverrideName
                   FROM
                       doc.DocumentDirectLink AS DDL
                   INNER JOIN
                       core.LogicalAsset AS LA ON DDL.ContextId = LA.AssetId AND Context = 2 -- LogicalGroup Context
                   INNER JOIN
                       core.LogicalAssetMap AS LAM ON LAM.GroupAssetId = LA.AssetId
                   WHERE
                       LAM.AssetId = @ContextId ;
                       
               --Get model docs    
               INSERT  INTO @IndirectDocument
                   (
                       DocumentId
                   , ContextId
                   , Context
                   , AssetType
                   , ContextName
                   , ContextShortName
                   , ContextOverrideName
                   )
                   SELECT
                       DDL.DocumentId,
                       M.Id,
                       DDL.Context,
                       'model',
                       COALESCE(M.UserName, M.Name),
                       NULL,
                       NULL
                   FROM
                       doc.DocumentDirectLink AS DDL
                   INNER JOIN
                       core.Model AS M ON DDL.ContextId = M.Id AND Context = 8 -- Model Context
                   WHERE
                       M.Id = @AssetModelId ;
               
               --Get parent's model docs
               DECLARE @ParentModelId UNIQUEIDENTIFIER ;
               SELECT @ParentModelId = ModelId FROM core.Asset WHERE AssetNode = @AssetNode.GetAncestor(1) ;
               INSERT  INTO @IndirectDocument
                   (
                       DocumentId
                   , ContextId
                   , Context
                   , AssetType
                   , ContextName
                   , ContextShortName
                   , ContextOverrideName
                   )
                   SELECT
                       DDL.DocumentId,
                       M.Id,
                       DDL.Context,
                       'model',
                       COALESCE(M.UserName, M.Name),
                       NULL,
                       NULL
                   FROM
                       doc.DocumentDirectLink AS DDL
                   INNER JOIN
                       core.Model AS M ON DDL.ContextId = M.Id AND Context = 8 -- Model Context
                   WHERE
                       M.Id = @ParentModelId ;
               
               --Select document info        
               SELECT
                   D.[Id]
                   ,D.[Name]
                   ,D.[Extension]
                   ,D.[MimeType]
                   ,D.[FileType]
                   ,D.[AssociationType]
                   ,D.[Size]
                   ,D.[Address]
                   ,ID.ContextId
                   ,ID.Context
                   ,ID.AssetType
                   ,ID.ContextName
                   ,ID.ContextShortName
                   ,ID.ContextOverrideName
               FROM
                   [doc].[Document] AS D
               INNER JOIN
                   @IndirectDocument AS ID ON D.Id = ID.DocumentId ;
           
           END
   END
GO
