CREATE PROC [core].[RetrieveLogicalGroupAsset]
   @AssetId UNIQUEIDENTIFIER
AS
   BEGIN        
       SELECT
           A.AssetId
         , A.Name
         , A.ShortName
         , A.OverrideName
         , A.[Type]
         , A.[Status]
         , A.[Status] AS 'DerivedStatus'
            , A.AssetType
         , A.AssetNode.ToString() AS 'AssetPath'
         , A.AssetLevel
         , A.Notes
         , NULL as 'AssetXml'
         , (
             SELECT
               COUNT(*)
             FROM
               core.LogicalAsset A2
             WHERE
               A2.AssetNode.IsDescendantOf(A.AssetNode) = 1
               AND A2.AssetLevel = A.AssetLevel + 1
           ) AS 'ChildCount'
         , (
             SELECT
               A2.AssetId
             FROM
               core.LogicalAsset AS A2
             WHERE
               A2.AssetNode = A.AssetNode.GetAncestor(1)
           ) AS 'ParentId'
         , (
             SELECT
               property.Name AS 'name'
             , property.SystemValue AS 'systemValue'
             , SQL_VARIANT_PROPERTY(property.SystemValue, 'basetype') AS 'systemValueType'
             , property.UserValue AS 'userValue'
             , SQL_VARIANT_PROPERTY(property.UserValue, 'basetype') AS 'userValueType'
             , property.Display AS 'display'
             , property.Editable AS 'editable'
             FROM
               core.LogicalAssetProperty AS property
             WHERE
               property.AssetId = @AssetId
           FOR
             XML PATH('property')
               , ROOT('properties')
           ) AS ExtendedProperties
       FROM
           core.LogicalAsset AS A
       WHERE
           A.AssetId = @AssetId ;
   END
GO
