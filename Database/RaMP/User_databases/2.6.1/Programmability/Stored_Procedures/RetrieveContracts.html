CREATE PROCEDURE [core].[RetrieveContracts]
   @Enabled BIT = NULL
 , @EndDate DATE = NULL
 , @ContractId UNIQUEIDENTIFIER = NULL
AS
   BEGIN
       SELECT
           C.ContractId
         , C.AssetId
         , C.StartDate
         , C.EndDate
         , C.Enabled
         , C.TrackingNumber
         , C.HostedCompany
         , C.ContactName
         , C.ContactPhone
         , C.ContactEmail
         , CT.Name AS ContractTypeName
         , CT.ExpiringMessageType
         , CT.ExpiredMessageType
       FROM core.Contract AS C
       JOIN core.ContractType AS CT ON C.ContractTypeId = CT.ContractTypeId
       JOIN core.Asset AS A ON C.AssetId = A.AssetId
       WHERE
           ( @ContractId IS NULL
             OR C.ContractId = @ContractId
           )
           AND ( @Enabled IS NULL
                 OR C.Enabled = @Enabled
               )
           AND ( @EndDate IS NULL
                 OR C.EndDate < @EndDate
               )
           AND core.IsInExcludedAssetNode(A.AssetNode) = 0
   END
GO
