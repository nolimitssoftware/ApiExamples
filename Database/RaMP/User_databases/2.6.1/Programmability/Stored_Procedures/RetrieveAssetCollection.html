CREATE PROC [core].[RetrieveAssetCollection]
    @ParentId UNIQUEIDENTIFIER = NULL ,   
    @ParentNode HIERARCHYID = NULL,
    @AssetType VARCHAR(256) = NULL ,
    @ExcludedAssetType VARCHAR(256) = NULL ,
    @AssetName VARCHAR(128) = NULL ,
    @DisplayName varchar(128) = NULL,
    @PropertyFilter VARCHAR(1024) = NULL,
    @AssetStatus INT = NULL,
    @AssetIndividualStatus INT = NULL,
    @Group UNIQUEIDENTIFIER = NULL,
    @IsDebug BIT = 0 ,
    @IgnoreHierarchy BIT = NULL,
    @RackLocation INT = NULL,
    @IncludeUndefinedDevices BIT = NULL,
    @ExcludeIcuDevices BIT = NULL,
    @SortField VARCHAR(32) = NULL,
    @Start INT = 1,
    @Limit INT = NULL,
    @IsAscending BIT = 1,
    @TotalCount INT OUTPUT
AS
    BEGIN
    SET NOCOUNT ON;
         SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED ;
         DECLARE @ChildLevel INT ;
         DECLARE @OutParentId UNIQUEIDENTIFIER = @ParentId ;

       DECLARE @End INT = @Start + @Limit - 1;
       DECLARE @SortDir CHAR(6);
       DECLARE @OrderBy VARCHAR(128);        
       
       IF(@IsAscending = 1)
               SET @SortDir = ' ASC  ';
           ELSE
               SET @SortDir = ' DESC ';
               
       IF(@SortField IS NULL OR @SortField = 'DisplayName')
       BEGIN
           SET @OrderBy = ' COALESCE(A.OverrideName,A.ShortName,A.Name) ' + @SortDir + ' ';
       END
       ELSE
       BEGIN            
           SET @OrderBy = 'A.' + @SortField + @SortDir + ', COALESCE(A.OverrideName,A.ShortName,A.Name) ASC ' ;
       END            
       
         IF ( @ParentId IS NULL AND @ParentNode IS NOT NULL)
               BEGIN
                    SELECT
                         --@ParentId = A.AssetId ,                    
                         @ChildLevel = A.AssetLevel + 1
                    FROM
                         core.Asset AS A
                    WHERE
                         A.AssetNode = HIERARCHYID::GetRoot()
               END
         ELSE IF(@ParentId IS NOT NULL)
               BEGIN
                    SELECT
                         @ParentNode = A.AssetNode ,
                         @ChildLevel = A.AssetLevel + 1
                    FROM
                         core.Asset AS A
                    WHERE
                         A.AssetId = @ParentId ;    
               END            
   
         
         DECLARE @sql NVARCHAR(MAX) ;
         DECLARE @params NVARCHAR(4000) ;
   
         SET @sql = '
         SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED ;SET NOCOUNT ON; DECLARE @FilteredAssets TABLE(AssetId UNIQUEIDENTIFIER) ;DECLARE @Assets TABLE(AssetId UNIQUEIDENTIFIER) ; '
IF(@PropertyFilter IS NOT NULL)
   SET @sql += ' DECLARE @Properties TABLE(name VARCHAR(128), value NVARCHAR(1024)); INSERT @Properties
         SELECT SUBSTRING(SD_PF.str,0, CHARINDEX('':'', SD_PF.str)), SUBSTRING(SD_PF.str,CHARINDEX('':'', SD_PF.str)+1,len(SD_PF.str) - CHARINDEX('':'', SD_PF.str))
         FROM util.split_delimiter(@PropertyFilter, '','') AS SD_PF; '
         
SET @sql +=
'
INSERT @Assets(
   AssetId
)
SELECT DISTINCT
    A.AssetId
FROM core.Asset AS A  
LEFT JOIN core.AssetProperty AS AP
ON A.AssetId = AP.AssetId '
IF(@PropertyFilter IS NOT NULL)
   BEGIN
   SET @sql += '
INNER JOIN @Properties AS P
ON AP.Name = P.name
and (Convert(nvarchar(1024), AP.SystemValue) like P.value OR convert(nvarchar(1024), AP.UserValue) like P.value) '
   END

IF(@Group IS NOT NULL)
BEGIN
   SET @sql += ' INNER JOIN core.LogicalAssetMap AS LAM ON A.AssetId = LAM.AssetId '        
END

IF(@RackLocation IS NOT NULL)
BEGIN
SET @sql += '
LEFT JOIN core.Location AS L
   ON A.AssetId = L.Id '

IF(@RackLocation = 0)
BEGIN
SET @sql += '
INNER JOIN core.Model AS M
   ON A.ModelId = M.Id '
END
END
    
SET @sql +=
' LEFT JOIN core.Host as H
   ON A.AssetId = H.AssetId
LEFT JOIN core.Hostname as HN
   ON A.AssetId = HN.AssetId '

IF(CHARINDEX(',', @AssetType) > 0)
   SET @sql +=  '
   INNER JOIN util.split_delimiter(@AssetType, '','') AS SD_T
       ON SD_T.str = A.Type '
           
SET @sql += ' WHERE 1 = 1 ';
SET @sql += ' AND core.IsInExcludedAssetNode(A.AssetNode) = 0 ';
           
       IF(@ParentId IS NOT NULL)
           SET @sql += ' AND
               A.AssetId <> @ParentId ';
       IF(@ParentNode IS NOT NULL AND @ParentId IS NOT NULL)
           SET @sql += ' AND A.AssetNode.IsDescendantOf(@ParentNode) = 1 ';        
           IF(@IgnoreHierarchy = 1 )
           BEGIN
               PRINT 'ignoring hierarchy'
           END
           ELSE    
           BEGIN
               IF(@ParentNode IS NOT NULL AND @ParentId IS NOT NULL)            
                   SET @sql += ' AND A.AssetLevel = @ChildLevel ' ;
       
               IF(@ParentNode IS NOT NULL AND @ParentId IS NULL)            
                   SET @sql += ' AND A.AssetNode = @ParentNode ' ;
           END
           
       IF ( @AssetType IS NOT NULL AND CHARINDEX(',', @AssetType ) = 0)
           SET @sql += ' AND A.Type = @AssetType ' ;
       
       IF ( @ExcludedAssetType IS NOT NULL AND CHARINDEX(',', @ExcludedAssetType ) = 0)
           SET @sql += ' AND A.Type <> @ExcludedAssetType ' ;
                   
       IF(CHARINDEX(',', @ExcludedAssetType) > 0)
           SET @sql +=  ' AND A.Type NOT IN (SELECT SD_ET.str FROM util.split_delimiter(@ExcludedAssetType, '','') AS SD_ET) ';

       IF ( @AssetType IS NULL AND @IncludeUndefinedDevices IS NULL )            
           SET @sql += ' AND A.Type <> ''unknown'' ';
       
       IF(@Group IS NOT NULL)
           SET @sql += ' AND LAM.GroupAssetId = @Group ' ;
           
       IF(@AssetStatus IS NOT NULL) --Derived Asset Status
           SET @sql += ' AND ( ISNULL(core.GetStatus(A.AssetNode), A.Status) & @AssetStatus ) > 0 '
       
       IF(@AssetIndividualStatus IS NOT NULL)
           SET @sql += ' AND ( (A.Status & @AssetIndividualStatus ) = A.Status ) '
       
       IF ( @AssetName IS NOT NULL )             
               SET @sql += ' AND ((A.Name LIKE @AssetName) OR (A.ShortName LIKE @AssetName) OR (A.OverrideName LIKE @AssetName) OR ( H.Address LIKE @AssetName) OR (HN.Name LIKE @AssetName)) '
           
       IF(@DisplayName IS NOT NULL)
               SET @sql += ' AND ( COALESCE(A.OverrideName,A.ShortName) LIKE @DisplayName ) '
                       
       IF(@ExcludeIcuDevices = 1)
               SET @sql += ' AND ( AP.Name = ''IsFromICU'' AND AP.SystemValue = 0 )';

       IF(@RackLocation IS NOT NULL)
           BEGIN
               IF(@RackLocation = 0)
                   BEGIN
                       SET @sql += ' AND ((L.RackLocation = 0 OR L.RackLocation IS NULL) OR ((M.RackUnits = 0 AND AP.Name = ''rackUnits'' AND (AP.UserValue IS NULL OR AP.UserValue = 0)) OR (AP.Name = ''rackUnits'' AND AP.UserValue = 0))) ';
                   END
                   ELSE                        
                   BEGIN
                       SET @sql += ' AND (L.RackLocation = @RackLocation) ';
                   END                                                                    
           END            
       
         SET @sql += ';' ;
   
        
         SET @sql +=' SElECT @TotalCount = COUNT(*) FROM @Assets; IF(@TotalCount IS NULL) BEGIN SET @TotalCount = 0; END; IF(@End IS NULL) BEGIN SET @End = @TotalCount; END;
       
SELECT  
   A.*
    , A.AssetNode.ToString() AS ''AssetPath''    
         FROM(
           SELECT DISTINCT
               A.AssetId                      
             , A.ModelId              
             , (SELECT COUNT(*) FROM core.Asset A2 ' ;
       IF(CHARINDEX(',', @AssetType) > 0)
           SET @sql +=  '
               INNER JOIN util.split_delimiter(@AssetType, '','') AS SD_T
                       ON SD_T.str = A2.Type ';
                 
       SET @sql += ' WHERE 1 = 1 '
       IF ( @AssetType IS NOT NULL AND CHARINDEX(',', @AssetType ) = 0)
           SET @sql += ' AND A2.Type = @AssetType ' ;
       
       SET @sql += ' AND A2.AssetNode.IsDescendantOf(A.AssetNode) = 1 AND A2.AssetLevel = A.AssetLevel +1) AS ''ChildCount'' '
       
                      
             SET @sql +=', '''' AS ''AssetXml''                             
             , COALESCE(@OutParentId,     (
                 SELECT TOP 1
                    A2.AssetId
                 FROM
                    core.Asset AS A2
                 WHERE                     
                    A2.AssetNode = A.AssetNode.GetAncestor(1)
               ) ) AS ''ParentId''
             ,    ROW_NUMBER() OVER(ORDER BY ' + @OrderBy + ' '
                       
       SET @sql += ') AS ''row_num''';        
         SET @sql += '
         FROM
               @Assets AS A2
         INNER JOIN core.Asset AS A   
           ON A2.AssetId = A.AssetId
       ) AS A2
       CROSS APPLY core.GetAsset(A2.AssetId) AS A        
       WHERE A2.row_num BETWEEN @Start AND @End
       ORDER BY A2.Row_Num ASC;';
                   
         
         
         SET @params = '
         @ParentId UNIQUEIDENTIFIER,
         @ParentNode HIERARCHYID,
         @ChildLevel INT,
         @OutParentId UNIQUEIDENTIFIER,
         @AssetType VARCHAR(256),
         @ExcludedAssetType VARCHAR(256),
         @PropertyFilter VARCHAR(1024),
         @AssetStatus INT,
         @AssetIndividualStatus INT,
         @Group UNIQUEIDENTIFIER,
         @RackLocation INT,
         @AssetName VARCHAR(128),         
         @DisplayName VARCHAR(128),         
         @Start INT,
         @End INT,
         @TotalCount INT OUTPUT';
         
         IF ( @IsDebug = 1 )
               BEGIN
                    print @params;
                    PRINT @Sql ;
               END
         EXEC sp_executesql @sql, @params, @ParentId, @ParentNode, @ChildLevel, @OutParentId,
               @AssetType, @ExcludedAssetType, @PropertyFilter, @AssetStatus, @AssetIndividualStatus, @Group, @RackLocation,@AssetName, @DisplayName, @Start, @End, @TotalCount = @TotalCount OUTPUT;
    END
GO
