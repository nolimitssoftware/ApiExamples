CREATE PROCEDURE [log].[RetrieveLogCollection]
   @SortExpression NVARCHAR(512) = 'EL.Timestamp DESC'
 , @Start INT
 , @Limit INT
 , @Level NVARCHAR(10) = NULL
 , @Logger NVARCHAR(128) = NULL
 , @Message NVARCHAR(512) = NULL
 , @MinTimestamp DATETIME2 = '1/1/1753'
 , @MaxTimestamp DATETIME2 = '12/31/9999'
 , @Debug BIT = 0
 , @TotalCount INT OUTPUT
AS
   BEGIN
               
       SET NOCOUNT ON ;
       DECLARE @End INT = @Start + @Limit - 1 ;
       SELECT
           @TotalCount = COUNT(1)
       FROM
           log.EventLog AS EL
       WHERE
           (
             @Level IS NULL
             OR (
                  @Level IS NOT NULL
                  AND EL.Level = @Level
                )
           )
           AND (
                 @Message IS NULL
                 OR (
                      @Message IS NOT NULL
                      AND EL.Message LIKE '%' + @Message + '%'
                    )
               )
           AND (
                 @Logger IS NULL
                 OR (
                      @Logger IS NOT NULL
                      AND EL.Logger LIKE '%' + @Logger + '%'
                    )
               )
           AND ( EL.Timestamp BETWEEN @MinTimestamp
                              AND     @MaxTimestamp ) ;
       
    
       
       DECLARE @sql NVARCHAR(4000) ;
       DECLARE @paramlist NVARCHAR(4000) = '
   @Start INT,
   @End INT,
   @Level NVARCHAR(10),    
   @Message NVARCHAR(512),
   @Logger NVARCHAR(128),
   @MinTimestamp DATETIME2,
   @MaxTimestamp DATETIME2'
   
       IF ( LEN(@SortExpression) = 0 )
           BEGIN
               SET @SortExpression = 'EL.Timestamp DESC'
           END
    
       IF ( PATINDEX('EL.Timestamp%', @SortExpression) = 0 )
           BEGIN
               SET @SortExpression = 'EL.' + @SortExpression
                   + ', EL.Timestamp DESC' ;
           END    
       SET @sql = '
       SELECT
       *
     FROM
     (
       SELECT DISTINCT
           EL.EventLogId ,
           EL.Level ,
           EL.Message,
           EL.Logger ,
           EL.CallSite,
           EL.Timestamp ,         
           '''' as FlatStackTrace,
           '''' as RawStackTrace,            
           '''' as Exception,
           '''' as Context,
           ROW_NUMBER() OVER (ORDER BY ' + @SortExpression
           + ') as RowNum  
       FROM
           log.EventLog AS EL
       WHERE
           (
             @Level IS NULL OR
             (
                @Level IS NOT NULL AND
                EL.Level = @Level
             )
           ) AND
           (
             @Message IS NULL OR
             (
                @Message IS NOT NULL AND
                EL.Message LIKE ''%'' + @Message + ''%''
             )
           )AND ( @Logger IS NULL
               OR ( @Logger IS NOT NULL
                    AND EL.Logger LIKE ''%'' + @Logger + ''%''
               )
           ) AND
           ( EL.Timestamp BETWEEN @MinTimestamp
                             AND     @MaxTimestamp )
       ) AS entries
       WHERE
       entries.RowNum BETWEEN @Start AND @End
       ORDER BY
       entries.RowNum ASC'
   
       IF ( @Debug = 1 )
           BEGIN
               PRINT @paramlist ;
               PRINT @sql ;
           END
   
       EXEC dbo.sp_executesql
           @sql
         , @paramList
         , @Start
         , @End
         , @Level
         , @Message
         , @Logger
         , @MinTimestamp
         , @MaxTimestamp
   
   
   END
GO
