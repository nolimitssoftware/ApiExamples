CREATE PROC [core].[BuildDiscoveredSensorWithInternalId]
    @DeviceId UNIQUEIDENTIFIER
 , @InternalDeviceId BIGINT
 , @DiscoveryAgentId UNIQUEIDENTIFIER
 , @Source VARCHAR(32)
 , @Type VARCHAR(128)
 , @SystemName VARCHAR(128)
 , @UserName VARCHAR(128) = NULL
 , @Unit VARCHAR(16)
 , @Element VARCHAR(32) = NULL
 , @OriginType VARCHAR(32)
 , @LocationAssetId UNIQUEIDENTIFIER = NULL
 , @RackLocation TINYINT = NULL
 , @Position INT = NULL
 , @SensorXml XML = NULL
 , @ExtendedProperties XML = NULL
AS
    BEGIN
    
         DECLARE @SensorId UNIQUEIDENTIFIER
         SELECT TOP 1
               @SensorId = S.SensorId
         FROM
               [core].[Sensor] AS S
               INNER JOIN  [core].SensorDeviceInformation AS SDI on S.SensorId = SDI.SensorId
         WHERE
               SDI.InternalDeviceId = @InternalDeviceId;

         IF ( @SensorId IS NOT NULL )
               BEGIN
                    UPDATE
                         [core].[Sensor]
                    SET
                       [DiscoveryAgentId] = @DiscoveryAgentId
                       , [Source] = @Source
                       , [Type] = @Type
                       , [SystemName] = @SystemName
                       , [Element] = @Element
                       , [Unit] = @Unit
                       , [OriginType] = @OriginType
                       , [LocationAssetId] = COALESCE(@LocationAssetId, [LocationAssetId])
                       , [RackLocation] = @RackLocation
                       , [Position] = @Position
                       , [ExtendedProperties] = @ExtendedProperties
                    WHERE
                       SensorId = @SensorId
               END
         ELSE
               BEGIN
                   --Determine default position.
                   --DECLARE @Position INT = NULL
                   --SELECT TOP 1 @Position = 1 FROM core.Asset WHERE @Type IN ('humidity', 'ambientTemperature') AND AssetId = @DeviceId AND [Type] IN ('server', 'bladeServer')

                   --First time this sensor has been found during a discovery
       
                    DECLARE @Temp TABLE ( Id UNIQUEIDENTIFIER ) ;
                                       
                    INSERT  INTO [core].[Sensor]
                               (
                                 [DeviceId]
                               , [DiscoveryAgentId]
                               , [Source]
                               , [Type]
                               , [SystemName]
                               , [UserName]
                               , [Unit]
                               , [Element]
                               , [OriginType]
                               , [LocationAssetId]
                               , [RackLocation]
                               , [Position]
                               , [ExtendedProperties]
                               )
                    OUTPUT  INSERTED.SensorId
                               INTO @Temp
                    VALUES
                               (
                                 @DeviceId
                               , @DiscoveryAgentId
                               , @Source
                               , @Type
                               , @SystemName
                               , @UserName
                               , @Unit
                               , @Element
                               , @OriginType
                               , COALESCE(@LocationAssetId, @DeviceId)
                               , @RackLocation
                               , @Position
                               , @ExtendedProperties
                               )
       
                    DECLARE @NewId UNIQUEIDENTIFIER = (
                                                                    SELECT TOP 1
                                                                         T.Id
                                                                    FROM
                                                                         @Temp AS T
                                                                 ) ;

                   SET @SensorId = @NewId ;    

                   --Set Extra Sensor Device Info (like internal device id)
                    INSERT  INTO [core].[SensorDeviceInformation]
                               (
                                 [SensorId]
                               , [InternalDeviceId]
                               )
                    VALUES
                               (
                                 @SensorId
                               , @InternalDeviceId
                               )

               END
           
           IF ( @SensorXml IS NOT NULL )
               BEGIN
                   -- This needs to run for both updates and inserts.  It stuffs the sensor ID into the XML and in the case of the update will update the sensor XML with any design definition changes that we may made.
                   UPDATE
                           core.Sensor
                   SET
                           SensorXml = [core].[InsertSensorIdIntoSensorXml](@NewId,
                                                                       @SensorXml)
                   WHERE
                           SensorId = @NewId ;
               END

           --Return sensor ID
           SELECT
               @SensorId
    END
GO
