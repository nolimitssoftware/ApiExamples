CREATE PROCEDURE [settings].[SaveModel]
    @Id UNIQUEIDENTIFIER = NULL
 , @ManufacturerId UNIQUEIDENTIFIER
 , @Name NVARCHAR(128) = NULL
 , @SearchName NVARCHAR(256) = NULL
 , @UserName NVARCHAR(128) = NULL
 , @UserPartNumber NVARCHAR(256) = NULL
 , @Type VARCHAR(32)
 , @Power FLOAT = NULL
 , @Weight FLOAT = NULL
 , @RackUnits TINYINT = NULL
 , @Width FLOAT = NULL
 , @Height FLOAT = NULL
 , @Depth FLOAT = NULL
 , @Notes VARCHAR(MAX) = NULL
 , @ImagePath VARCHAR(128) = NULL
 , @Current float = NULL
 , @SupportsRaMP BIT = 0
 , @IsSystemDefined BIT  
 , @ExtendedProperties XML = NULL
AS
    BEGIN
         IF ( @Name IS NULL AND @UserName IS NULL )
               BEGIN
                    SET @Name = 'Unknown' ;
                    SET @SearchName = 'Unknown' ;
               END
               
         IF ( @SearchName IS NULL )
               BEGIN
                    SET @SearchName = 'Unknown' ;
               END
           
         IF ( @Id IS NULL )
               BEGIN
                    SELECT TOP 1
                         @Id = M.Id
                    FROM
                         core.Model AS M
                    WHERE
                         M.ManufacturerId = @ManufacturerId
                         AND
                           (@Type IS NULL OR M.[Type] = @Type)
                         AND
                           (
                               (@Name IS NULL AND M.UserName = @UserName)
                               OR
                               (@Name IS NOT NULL AND M.Name = @Name));
               END

           --This is being included in order to be cautious
         If (@Current is null)
           Set @Current = ROUND(@Power / 208, 4)


         DECLARE @Manufacturer VARCHAR(256)
         SELECT
               @Manufacturer = Name
         FROM
               core.Manufacturer
         WHERE
               Id = @ManufacturerId ;
               
         IF ( @Id IS NOT NULL )
               BEGIN
                   -- Do not update searchName at update.  We don't want the user to "pollute" it.

                    UPDATE
                         core.Model
                    SET
                         [Type] = @Type
                       , ManufacturerId = @ManufacturerId    
                       , [Name] = COALESCE(@Name, [Name])
                       , [UserName] = COALESCE(@UserName, [UserName])
                       , [UserPower] = COALESCE(@Power, [UserPower])    
                       , [UserPartNumber] = COALESCE(@UserPartNumber, [UserPartNumber])                        
                       , Height = @Height
                       , Depth = @Depth
                       , Width = @Width
                       , [Weight] = @Weight
                       , RackUnits = @RackUnits
                       , Notes = @Notes
                       , ImagePath = @ImagePath
                       , ExtendedProperties = @ExtendedProperties
                       , SupportsRaMP = @SupportsRaMP
                       , IsSystemDefined = 0    -- It was decided to help with upgrades that when a user updates a model to flip this flag.
                       , [Current] = @Current
                    OUTPUT
                         INSERTED.*
                       , @Manufacturer AS 'Manufacturer'
                    WHERE
                         Id = @Id ;
               
               END
                    
         ELSE
               BEGIN
           
                    INSERT  core.Model
                               (
                                 ManufacturerId
                               , Name
                               , SearchName
                               , [Type]
                               , [UserPower]
                               , [Current]
                               , [UserPartNumber]
                               , Height
                               , Depth
                               , Width
                               , [Weight]
                               , RackUnits
                               , Notes
                               , ImagePath
                               , ExtendedProperties
                               , IsSystemDefined
                               , IsBasic
                               , SupportsRaMP
                               
                               )
                    OUTPUT  INSERTED.*
                            , @Manufacturer AS 'Manufacturer'
                    VALUES
                               (
                                 @ManufacturerId
                               , COALESCE(@Name, @UserName)
                               , @SearchName
                               , @Type
                               , @Power
                               , @Current
                               , @UserPartNumber
                               , @Height
                               , @Depth
                               , @Width
                               , @Weight
                               , @RackUnits
                               , @Notes
                               , @ImagePath
                               , @ExtendedProperties
                               , @IsSystemDefined
                               , 0
                               , @SupportsRaMP                                                            
                               
                               )
               END
       
    END
GO
