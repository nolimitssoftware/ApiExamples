CREATE PROC core.RetrieveSerialNumberCollection
   @AssetId UNIQUEIDENTIFIER
AS
   BEGIN
       SELECT
           SN.Id
         , SN.AssetId
         , SN.SerialNumber
         , SN.[Source]
       FROM
           core.SerialNumber AS SN
       INNER JOIN core.Asset AS A ON A.AssetId = SN.AssetId
       WHERE
           SN.AssetId = @AssetId
       ORDER BY
           SN.SerialNumber ASC ;
   END
GO
