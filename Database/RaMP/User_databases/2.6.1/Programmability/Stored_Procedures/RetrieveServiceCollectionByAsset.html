CREATE PROC core.RetrieveServiceCollectionByAsset
   @AssetId UNIQUEIDENTIFIER
AS
   BEGIN
       SELECT
           S.Id
         , S.IpAddress
         , S.AssetId
         , S.Name
         , S.Port
         , S.[Source]
         , S.ExtendedProperties          
         , '' AS 'Hostnames'
         --, core.RetrieveHostNameListByHost(S.HostId) AS 'Hostnames'
       FROM
           core.Service AS S        
       WHERE
           S.AssetId = @AssetId ;
   END
GO
