CREATE PROCEDURE [doc].[RetrieveDocumentsInfoByAssociationType]
   @AssociationType INT,
   @StartRow bigint,
   @PageSize int
AS
BEGIN
   With docs as (
       SELECT
           ROW_NUMBER() OVER(ORDER BY d.Name) as rownum,
           d.Id,
           d.Name
       FROM
           doc.Document AS d
       WHERE
           d.AssociationType & @AssociationType > 0
   )
   Select
           D.[Id],
           D.[Name],
           D.[Extension],
           D.[MimeType],
           D.[FileType],
           D.[AssociationType],
           D.[Size],
           D.[Address]
   From docs
   join doc.Document d on docs.Id = d.Id
   Where @PageSize is null or docs.rownum between @StartRow + 1 and (@StartRow + @PageSize);
END
GO
