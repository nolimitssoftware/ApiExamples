CREATE PROCEDURE [core].RetrieveNetworkDirectDownstream
   @AssetId UNIQUEIDENTIFIER
 , @ShowAll BIT
AS
   BEGIN

       DECLARE @Temp TABLE
           (
             RemoteAssetId UNIQUEIDENTIFIER
           , MAC CHAR(17)
           , Port INT
           , PortName NVARCHAR(256)
           , NetworkInterfaceId UNIQUEIDENTIFIER
           , NetworkInterfaceSource VARCHAR(64)
           , Technology VARCHAR(64)
           , Speed INT
           , ConnectionFlag VARCHAR(8)
           , Vlan INT
           , IndirectCount INT
           , FlaggedPortCount INT
           )            
       
       --First get information about what is connected to the device
;
       WITH    c AS ( SELECT
                       COUNT(Id) AS [FlaggedPortCount]
                     , SUM(CASE WHEN RM.ConnectionFlag = 'indirect' THEN 1
                                ELSE 0
                           END) AS [IndirectCount]
                     , RM.Port
                      FROM
                       core.RemoteMac AS RM
                      WHERE
                       RM.AssetId = @AssetId
                      GROUP BY
                       RM.Port
                    )
           INSERT  @Temp
                   ( RemoteAssetId
                   , MAC
                   , Port
                   , PortName
                   , NetworkInterfaceId
                   , NetworkInterfaceSource
                   , Technology
                   , Speed
                   , ConnectionFlag
                   , Vlan
                   , IndirectCount
                   , FlaggedPortCount
                   )
                   SELECT
                       RM.RemoteAssetId
                     , RM.Mac
                     , NI.Port
                     , NI.Name AS 'PortName'
                     , NI.Id
                     , NI.Source
                     , NI.Technology
                     , NI.Speed
                     , RM.ConnectionFlag
                     , RM.Vlan
                     , ISNULL(c.IndirectCount, 0) AS [IndirectCount]
                     , ISNULL(c.FlaggedPortCount, 0) AS [FlaggedPortCount]
                   FROM
                       core.RemoteMac AS RM
                   INNER JOIN c
                       ON c.Port = RM.Port
                   RIGHT JOIN core.NetworkInterface AS NI
                       ON NI.AssetId = RM.AssetId
                          AND ( NI.Port = RM.Port
                                OR NI.IpAddressIndex = RM.Port
                              )
                   WHERE
                       NI.AssetId = @AssetId;

       -- Now we will need to filter down to what is directly connected.
       DECLARE @Temp2 TABLE
           (
             RemoteAssetId UNIQUEIDENTIFIER
           , MAC CHAR(17)
           , Port INT
           , PortName NVARCHAR(256)
           , NetworkInterfaceId UNIQUEIDENTIFIER
           , NetworkInterfaceSource VARCHAR(64)
           , Technology VARCHAR(64)
           , Speed INT
           , ConnectionFlag VARCHAR(8)
           , Vlan INT
           , PortTaken BIT
           , ManualSet BIT DEFAULT 0
           )

       -- Get empty ports (ie. nothing is connected to them.)
       INSERT  @Temp2
               ( RemoteAssetId
               , MAC
               , Port
               , PortName
               , NetworkInterfaceId
               , NetworkInterfaceSource
               , Technology
               , Speed
               , Vlan
               , ConnectionFlag
               , PortTaken
               
               )
               SELECT
                   RemoteAssetId
                 , Mac
                 , Port
                 , PortName
                 , NetworkInterfaceId
                 , NetworkInterfaceSource
                 , Technology
                 , Speed
                 , Vlan
                 , ConnectionFlag
                 , 0
               FROM
                   @Temp
               WHERE
                   IndirectCount = 0
                   AND FlaggedPortCount = 0

       -- Get directly taken ports
       --    Taking into account to filter out anything that is connected to them.
       INSERT  @Temp2
               ( RemoteAssetId
               , MAC
               , Port
               , PortName
               , NetworkInterfaceId
               , NetworkInterfaceSource
               , Technology
               , Speed
               , Vlan
               , ConnectionFlag
               , PortTaken
               
               )
               SELECT
                   RemoteAssetId
                 , Mac
                 , Port
                 , PortName
                 , NetworkInterfaceId
                 , NetworkInterfaceSource
                 , Technology
                 , Speed
                 , Vlan
                 , ConnectionFlag
                 , 1
               FROM
                   @Temp
               WHERE
                   FlaggedPortCount <> 0
                   AND IndirectCount <> FlaggedPortCount
                   AND ConnectionFlag = 'direct'

       --Add the ports that we known are taken but all we know are the indirect connections to the port.
       --    So...something has to be there at the port.
       --    How this might happen: 1) directly connected to a network devices that we don't get much info from. 2) Directly connected to a server with 2 hosts that we could not merge together.    
       INSERT  @Temp2
               ( Port
               , PortName
               , NetworkInterfaceId
               , NetworkInterfaceSource
               , Technology
               , Speed
               , Vlan
               , ConnectionFlag
               , PortTaken
               
               )
               SELECT
                   Port
                 , PortName
                 , NetworkInterfaceId
                 , NetworkInterfaceSource
                 , Technology
                 , Speed
                 , Vlan
                 , 'indirect'
                 , 1
               FROM
                   @Temp
               WHERE
                   FlaggedPortCount <> 0
                   AND IndirectCount = FlaggedPortCount
               GROUP BY
                   Port
                 , PortName
                 , NetworkInterfaceId
                 , NetworkInterfaceSource
                 , Technology
                 , Speed
                 , Vlan
       
       ----Add any manually set connections
       UPDATE
           @Temp2
       SET
           RemoteAssetId = MDNC.DownstreamAssetId
         , ManualSet = 1
         , PortTaken = 1
       FROM
           @Temp2 AS T2
       RIGHT JOIN core.ManualDownstreamNetworkConnection AS MDNC
           ON T2.NetworkInterfaceId = MDNC.NetworkInterfaceId
       WHERE
           RemoteAssetId IS NULL

       SELECT DISTINCT
           T2.RemoteAssetId
         , T2.MAC
         , T2.Port
         , T2.PortName
         , T2.NetworkInterfaceId
         , T2.NetworkInterfaceSource
         , T2.Technology
         , T2.Speed
         , T2.Vlan
         , T2.ConnectionFlag
         , T2.PortTaken
         , T2.ManualSet
         , A.[Name]
         , A.ShortName
         , A.OverrideName
         , A.[Type] AS 'ConnectedType'
         , A.[Status]
         , A.LifecycleStatus
       FROM
           @Temp2 AS T2
       LEFT JOIN core.Asset AS A
           ON T2.RemoteAssetId = A.AssetId
       WHERE
           ( MAC IS NOT NULL )
           OR ( T2.ConnectionFlag IS NOT NULL )
           OR ( @ShowAll = 1
                AND MAC IS NULL
                AND T2.RemoteAssetId IS NULL
              )
           OR ( T2.ManualSet = 1 )
       ORDER BY
           Port
         , T2.Vlan
       
   END
GO
