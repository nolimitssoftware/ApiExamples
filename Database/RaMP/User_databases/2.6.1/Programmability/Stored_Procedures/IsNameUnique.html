CREATE PROC [core].[IsNameUnique]
    @AssetId UNIQUEIDENTIFIER = NULL
 , @NameToCheck VARCHAR(256)
 , @ParentAssetIdToCheckWith UNIQUEIDENTIFIER = NULL
AS
    BEGIN
       -- This will be the parent physical group node of the given location
       DECLARE @GroupNode HIERARCHYID
       
       IF (@ParentAssetIdToCheckWith IS NULL)
           BEGIN
               --@ParentAssetIdToCheckWith was not given so use the asset's currently set parent.
               DECLARE @AssetNode HIERARCHYID
               SELECT TOP 1 @AssetNode=AssetNode FROM core.Asset WHERE AssetId = @AssetId
               SELECT TOP 1 @ParentAssetIdToCheckWith = A.AssetId FROM core.Asset AS A WHERE A.AssetNode = @AssetNode.GetAncestor(1)            
           END
       
       IF (@ParentAssetIdToCheckWith IS NOT NULL)
           BEGIN
               --Uniqueness is enforced at the location level (physcial group).
               --    Example: So it would NOT be valid to have a device of the same name under the "Saint Louis" group even if both devices are in different racks under "Saint Louis"
           
               DECLARE @ParentAssetIdToCheckWithNode HIERARCHYID
               SELECT TOP 1 @ParentAssetIdToCheckWithNode=AssetNode FROM core.Asset WHERE AssetId = @ParentAssetIdToCheckWith
               
               SELECT
                   TOP 1
                   @GroupNode = AssetNode
               FROM [core].[Asset]
               WHERE
                   @ParentAssetIdToCheckWithNode.IsDescendantOf(AssetNode) = 1
                   AND
                   (@AssetId IS NULL OR AssetId <> @AssetId)
                   AND
                   AssetType = 'container'
               ORDER BY AssetLevel DESC
           END
       ELSE
           BEGIN
               --Set a default
               SET @GroupNode = HIERARCHYID::GetRoot()
           END
       
       IF EXISTS ( SELECT
                       1
                   FROM
                       core.Asset AS A
                   WHERE
                           AssetNode.IsDescendantOf(@GroupNode) = 1
                       AND -- This allows something to have the same name in a parent physical group and in a child physical group.
                           @GroupNode = (SELECT TOP 1 A2.AssetNode FROM core.Asset AS A2 WHERE A.AssetNode.IsDescendantOf(A2.AssetNode) = 1 AND AssetType = 'container' ORDER BY AssetLevel DESC)
                       AND
                           ((A.OverrideName IS NULL AND A.ShortName = @NameToCheck) OR (A.OverrideName = @NameToCheck))
                       AND
                           (@AssetId IS NULL OR AssetId <> @AssetId) )
       BEGIN                 
               SELECT
                   0 ;
       END
       ELSE
       BEGIN
               SELECT
                   1 ;
       END

    END
GO
