CREATE PROC core.RetrieveParentAsset
   @AssetId UNIQUEIDENTIFIER
AS
   BEGIN
       DECLARE @ParentNode HIERARCHYID ;
       SELECT
           @ParentNode = A.AssetNode.GetAncestor(1)
       FROM
           core.Asset AS A
       WHERE
           A.AssetId = @AssetId ;
       SELECT
           PA.[Parent.AssetId] AS 'AssetId'
         , PA.[Parent.Name] AS 'Name'
         , PA.[Parent.ShortName] AS 'ShortName'
         , PA.[Parent.OverrideName] AS 'OverrideName'
         , PA.[Parent.Type] AS 'Type'
         , PA.[Parent.Status] AS 'Status'
         , PA.[Parent.AssetType] AS 'AssetType'
         , PA.[Parent.ParentId] AS 'ParentId'
         , PA.[Parent.ExtendedProperties] AS 'ExtendedProperties'
       FROM
           core.Asset AS A
       CROSS APPLY core.GetParentAsset(A.AssetId) AS PA
       WHERE
           A.AssetNode = @ParentNode ;
   END
GO
