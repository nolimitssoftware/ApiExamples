CREATE PROC [core].[HandleRackModelRackUnitSizeShrinkage]
   @ModelId UNIQUEIDENTIFIER
 , @NewRackSize INT
AS
   BEGIN            
       -- Places devices in a "unplaced" state that were in a slot that has since disappeared with the user shrinking a rack size.
       --    This join also makes sure not to touch racks where we read the size from the rack or it has been set by the user.
       
       UPDATE
           core.Location
       SET
           RackLocation = NULL
       FROM
           core.Asset AS A
           LEFT JOIN core.AssetProperty AS AP ON A.AssetId = AP.AssetId AND AP.Name = 'rackUnits'        
           INNER JOIN core.Location AS L ON L.AssetId = A.AssetId        
       WHERE
           A.[Type] = 'rack'
           AND
           A.ModelId = @ModelId
           AND
           AP.SystemValue IS NULL
           AND
           AP.UserValue IS NULL
           AND
           L.RackLocation > @NewRackSize
   END
GO
