CREATE PROC [core].[UpdateAssetStatusFromActiveMessages]
    @AssetId UNIQUEIDENTIFIER,
    @CurrentMessageSeverity INT = NULL
AS
    BEGIN
       
       DECLARE @AssetStatus INT
       DECLARE @OldAssetStatus INT
       
       --Get the "lowest" severity for the asset's active messages (note informational (32768) does NOT count)
       SELECT
           @AssetStatus = MIN(M.Severity)
       FROM
           [log].Message AS M
       WHERE
           M.AssetId = @AssetId
           AND
           M.EndTimestamp IS NULL
           AND
           M.Severity <> 32768;
           
       IF(@CurrentMessageSeverity IS NOT NULL AND @CurrentMessageSeverity <> 32768)
       BEGIN
           IF(@CurrentMessageSeverity < @AssetStatus)    
               BEGIN    
                   SET @AssetStatus = @CurrentMessageSeverity;
               END                                            
       END
       -- if NULL then it is normal
       IF (@AssetStatus IS NULL)
           BEGIN                
               SET @AssetStatus = 65536
           END
         
         --Get old status
         SELECT
           @OldAssetStatus = [Status]
       FROM
           [core].[asset]
       WHERE
           AssetId = @AssetId
           
       --If different then set and work your way up the tree
       IF (@AssetStatus <> @OldAssetStatus)
           BEGIN
           
           --Update asset table
           UPDATE
               [core].[asset]
           SET
               [Status] = @AssetStatus
           WHERE
               AssetId = @AssetId            
       
       -- Currently we will not be bubbling up at this level.
       -- Update ancestors' status
       --DECLARE @CurrentAssetNode HIERARCHYID
       --SET @CurrentAssetNode = (SELECT AssetNode.GetAncestor(1) FROM [core].[asset] WHERE AssetId = @AssetId)
       --WHILE (@CurrentAssetNode IS NOT NULL)
       --    BEGIN
       --        --TODO may want to get old status and compare with new status and if they are the same do not continue to walk the tree up.
               
       --        -- Figure out the new status (note that unreachable (1) will not be taken into account)
       --        SELECT @AssetStatus = MIN([Status]) FROM [core].[asset] WHERE AssetNode.GetAncestor(1) = @CurrentAssetNode AND [Status] <> 1
               
       --        UPDATE
       --            [core].[asset]
       --        SET
       --            [Status] = @AssetStatus
       --        WHERE
       --            AssetNode = @CurrentAssetNode                
               
       --        SELECT @CurrentAssetNode = AssetNode.GetAncestor(1) FROM [core].[asset] WHERE AssetNode = @CurrentAssetNode
       --    END            
       END        
    END
GO
