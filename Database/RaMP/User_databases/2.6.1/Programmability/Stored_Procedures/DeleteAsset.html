CREATE PROC [core].[DeleteAsset]
   @AssetId UNIQUEIDENTIFIER
AS
   BEGIN
       --We do not want to delete the root asset node or the unknown group asset node.
       DECLARE @UndeletableAsset BIT = 0
       SELECT
           @UndeletableAsset = 1
       FROM
           core.Asset
       WHERE
           core.Asset.AssetId = @AssetId
           AND ( core.Asset.AssetNode = HIERARCHYID::GetRoot()
                 OR core.Asset.AssetNode = 0x58 -- Unknown Group Asset Node
                 
               )
       IF ( @UndeletableAsset = 1 )
           BEGIN
               RAISERROR (
               N'Cannot delete the root asset or unknown group asset.'
               ,0
               ,1);        
           END
       ELSE
           BEGIN
               --Clean up tables
               DELETE
                   driver.[AssetCommunicationSetting]
               WHERE
                   driver.[AssetCommunicationSetting].AssetId = @AssetId;

               DELETE
                   settings.ReportSettings
               WHERE
                   settings.ReportSettings.Id = @AssetId;

               DELETE
                   settings.ReportSettings
               WHERE
                   settings.ReportSettings.ContextId = @AssetId;

               DELETE
                   core.RampManager
               WHERE
                   core.RampManager.RackId = @AssetId;

               DELETE
                   core.UpstreamPower
               WHERE
                   core.UpstreamPower.UpstreamAssetId = @AssetId;
               
               DELETE
                   core.VirtualMachine
               WHERE
                   core.VirtualMachine.DeviceId = @AssetId;

               DELETE
                   [core].[PduBreaker]
               WHERE
                   [core].[PduBreaker].Assetid = @AssetId;

               DELETE
                   core.Location
               WHERE
                   core.Location.AssetId = @AssetId;    

                   --Clean up sensor location table
               UPDATE
                   [core].[Sensor]
               SET
                   [LocationAssetId] = [DeviceId]
               WHERE
                   [LocationAssetId] = @AssetId ;
               
               --Clean up blade info table
               --If placed, unassociate asset
               UPDATE
                   [core].Blade
               SET
                   BladeAssetId = NULL
               WHERE
                   BayLocation IS NOT NULL
                   AND BladeAssetId = @AssetId

               --If it is unplaced, delete it
               DELETE
                   [core].Blade
               WHERE
                   BayLocation IS NULL
                   AND BladeAssetId = @AssetId;
                       
               --Clean up Sensor values                
               DELETE
                   monitoring.SensorValue
               FROM
                   monitoring.SensorValue AS SV
               INNER JOIN core.Sensor AS S
                   ON SV.SensorId = S.SensorId
               WHERE
                   S.DeviceId = @AssetId;            
                   
               DELETE
                   search.SearchValue
               WHERE
                   search.SearchValue.AssetId = @AssetId;

               DELETE
                   core.IpAddress
               FROM
                   core.IpAddress AS IA
               INNER JOIN core.NetworkInterface AS NI
                   ON IA.NetworkInterfaceId = NI.Id
               WHERE
                   NI.AssetId = @AssetId;
                   
               DELETE
                   core.IpAddress
               WHERE
                   core.IpAddress.AssetId = @AssetId;
               
               DELETE
                   core.NetworkInterface
               WHERE
                   core.NetworkInterface.AssetId = @AssetId;
                                       
               DELETE
                   core.[ManualDownstreamNetworkConnection]
               WHERE
                   core.[ManualDownstreamNetworkConnection].DownstreamAssetId = @AssetId;
                                       
               DELETE
                   core.Asset
               WHERE
                   core.Asset.AssetId = @AssetId;
                   
               --Clean up visual layout items            
               DELETE
                   visual.LayoutItem
               WHERE
                   visual.LayoutItem.Id = @AssetId;

               --Clean up analyzer job processing            
               DELETE
                   [job].[Processing]
               WHERE
                   [job].[Processing].[ContextId] = @AssetId;        

           END
   END
GO
