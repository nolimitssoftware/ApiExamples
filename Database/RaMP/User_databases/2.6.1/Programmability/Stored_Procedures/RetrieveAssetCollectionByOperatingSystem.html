CREATE PROC [core].[RetrieveAssetCollectionByOperatingSystem]
   @OperatingSystemName NVARCHAR(256) ,
   @ServicePackName NVARCHAR(256) = NULL ,
   @Start INT = 1 ,
   @Limit INT = NULL ,
   @IsAscending BIT = 1 ,
   @IsDebug BIT = 1 ,
   @TotalCount INT OUTPUT
AS
   BEGIN
       SET NOCOUNT ON;
       SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;
       DECLARE @End INT;
       DECLARE @OperatingSystemId UNIQUEIDENTIFIER ,
               @ServicePackId UNIQUEIDENTIFIER;
       
       SELECT  @OperatingSystemId = SOS.Id
       FROM    core.ServerOperatingSystem AS SOS
       WHERE   SOS.Name = @OperatingSystemName;        
       
       IF ( @ServicePackName IS NOT NULL )
           BEGIN
               SELECT  @ServicePackId = SP.Id
               FROM    core.ServicePack AS SP
               WHERE   SP.ServerOperatingSystemId = @OperatingSystemId
                       AND SP.Name = @ServicePackName;
           END

       IF ( @ServicePackId IS NOT NULL )
           BEGIN
               SELECT  @TotalCount = COUNT(1)
               FROM    core.OperatingSystem AS OS
               WHERE   OS.ServerOperatingSystemId = @OperatingSystemId
                       AND OS.ServicePackId = OS.ServicePackId;
           END
       ELSE
           BEGIN
               IF ( @ServicePackName IS NOT NULL )
                   BEGIN
                       SELECT  @ServicePackId = SP.Id
                       FROM    core.ServicePack AS SP
                       WHERE   SP.ServerOperatingSystemId = @OperatingSystemId
                               AND SP.Name = N''
                   END
                           
               SELECT  @TotalCount = COUNT(1)
               FROM    core.OperatingSystem AS OS
               WHERE   OS.ServerOperatingSystemId = @OperatingSystemId;
           END    

       IF ( @Limit IS NULL )
           BEGIN
               SET @Limit = @TotalCount;
           END
       
       SET @End = @Start + @Limit - 1;
   
       DECLARE @sql NVARCHAR(MAX) ,
           @params NVARCHAR(4000) ,
           @sort VARCHAR(1024);
    
       
       SET @sort = ', ROW_NUMBER() OVER ( ORDER BY COALESCE(A.OverrideName, A.ShortName, A.Name) ';
       IF ( @IsAscending = 1 )
           BEGIN
               SET @sort += ' ASC ';
           END
       ELSE
           BEGIN
               SET @sort += ' DESC ';
           END
           
       SET @sort += ' ) AS RowNum ';
           
       SET @sql = '
               SELECT
                   GA.*
                 , GA.AssetNode.ToString() AS ''AssetPath''
               FROM
                   (
                     SELECT
                       A.AssetId
                     ';
       SET @sql += @sort;
       SET @sql += '
                    FROM
                       core.Asset AS A
                     INNER JOIN core.OperatingSystem AS OS
                       ON A.AssetId = OS.AssetId
                     WHERE
                       OS.ServerOperatingSystemId = @OperatingSystemId
                     AND core.IsInExcludedAssetNode(A.AssetNode) = 0 '
       IF ( @ServicePackId IS NOT NULL )
           BEGIN
               SET @sql += ' AND OS.ServicePackId = @ServicePackId '
           END                        
       
       SET @sql += '
) AS A
               CROSS APPLY core.GetAsset(A.AssetId) AS GA
               WHERE
                   A.RowNum BETWEEN @Start AND @End
               ORDER BY
                   A.RowNum ;'
                   
       SET @params = '
       @OperatingSystemId UNIQUEIDENTIFIER,
       @ServicePackId UNIQUEIDENTIFIER,
       @Start INT,
       @End INT'
   
       IF ( @IsDebug = 1 )
           BEGIN
               PRINT @params;
               PRINT @sql;
           END
   
       EXEC sp_executesql @sql, @params, @OperatingSystemId, @ServicePackId,
           @Start, @End
   END
GO
