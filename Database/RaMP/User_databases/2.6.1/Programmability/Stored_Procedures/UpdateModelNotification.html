CREATE PROCEDURE [core].[UpdateModelNotification]
AS
   BEGIN
       UPDATE
           core.Model
       SET
           SentToNLS = 1
       FROM
           core.Model AS M
       WHERE
           M.IsSystemDefined = 0
           AND M.SentToNLS = 0
   END
GO
