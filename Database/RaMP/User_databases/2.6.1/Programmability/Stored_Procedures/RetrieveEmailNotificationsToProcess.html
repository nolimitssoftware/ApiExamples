CREATE PROC [notification].[RetrieveEmailNotificationsToProcess]
   @BatchSize INT = 0
 , @Severity INT = NULL
AS
   BEGIN
       SET ROWCOUNT @BatchSize;
       DECLARE @NotificationType TINYINT = 1;
       SELECT
           NTP.*
         , GA.*
         , core.GetIpAddressListByAsset(NTP.AssetId) AS [IPAddresses]
         , core.GetSerialNumberListByAsset(NTP.AssetId) AS [SerialNumbers]
         , core.GetSimpleAssetPath(NTP.AssetId) AS [AssetPath]
       FROM
           (
             SELECT DISTINCT --new messages and resend (nag) messages
               N.Id
             , MN.MessageId
             , N.NotificationGroupId
             , N.NotificationType AS [NotificationGroupType]
             , N.NotificationTypeId
             , MN.LastNotified
             , MN.LastNotifiedStatus
             , MN.LastNotifiedMessage
             , COALESCE(AM.Email, NGR.Recipient) AS [Email]
             , AU.UserName
             , COALESCE(A.OverrideName, A.ShortName) AS [DisplayName]
             , A.[Status] AS [CurrentStatus]
             , M.AssetId
             , M.StartTimestamp
             , M.EndTimestamp
             , M.ClearedByUserName
             , CONVERT(VARCHAR(MAX), M.ExtendedProperties) AS [ExtendedProperties]
             , M.Severity
             , M.Text AS [Description]
             , MM.NotificationRepeatInterval
             , NG.NotificationFormat
             , NG.BatchMsg
             , NG.ClearMessages
             , NG.EmailSubjectLine
             , N.NewMessagesSetting AS [NewMessages]
             , NG.SetMessages
             , CAST(0 AS TINYINT) AS [NotificationType]
             FROM
               notification.MessageNotification AS MN
             INNER JOIN [notification].[Notification] AS N
               ON MN.NotificationId = N.Id
             INNER JOIN log.Message AS M
               ON MN.MessageId = M.Id
             INNER JOIN log.MessageMaster AS MM
               ON MM.Id = M.MessageMasterId
             INNER JOIN core.Asset AS A
               ON M.AssetId = A.AssetId
             INNER JOIN settings.NotificationGroup AS NG
               ON N.NotificationGroupId = NG.Id
             INNER JOIN settings.NotificationGroupRecipient AS NGR
               ON N.NotificationTypeId = NGR.Id
             LEFT JOIN dbo.aspnet_Membership AS AM
               ON NGR.UserId = AM.UserId
             LEFT JOIN dbo.aspnet_Users AS AU
               ON AM.UserId = AU.UserId
             WHERE
               N.NotificationType = @NotificationType
               AND A.[Type] <> 'unknown'
               AND ( M.Severity & @Severity ) = M.Severity
               AND ( MN.LastNotified IS NULL
                     OR MN.LastNotifiedStatus = 4
                     OR (MN.LastNotifiedStatus <> 8 AND MM.NotificationRepeatInterval > 0 AND MN.LastNotified <= DATEADD(mi, -MM.NotificationRepeatInterval, GETUTCDATE()))--nag messages
                   )
             UNION
             SELECT DISTINCT --chnage messages
               N.Id
             , C.Id AS [ChangeId]
             , N.NotificationGroupId
             , N.NotificationType AS [NotificationGroupType]
             , N.NotificationTypeId
             , CN.LastNotified
             , CN.LastNotifiedStatus
             , CN.LastNotifiedMessage
             , COALESCE(AM.Email, NGR.Recipient) AS [Email]
             , AU.UserName
             , COALESCE(A.OverrideName, A.ShortName) AS [DisplayName]
             , A.[Status] AS [CurrentStatus]
             , C.ContextId AS [AssetId]
             , C.StartTimestamp
             , NULL AS [EndTimestamp]
             , C.ChangedByUserName AS [ClearedByUserName]
             , CONVERT(VARCHAR(MAX), C.ExtendedProperties) AS [ExtendedProperties]
             , 0 AS [Severity]
             , CM.[Description]
             , CAST(0 AS SMALLINT) AS [NotificationRepeatInterval]
             , NG.NotificationFormat
             , NG.BatchMsg
             , NG.ClearMessages
             , NG.EmailSubjectLine
             , N.NewMessagesSetting AS [NewMessages]
             , NG.SetMessages
             , CAST(1 AS TINYINT) [NotificationType]
             FROM
               notification.ChangeNotification AS CN
             INNER JOIN [notification].[Notification] AS N
               ON CN.NotificationId = N.Id
             INNER JOIN log.Change AS C
               ON C.ContextId = CN.AssetId
             INNER JOIN log.ChangeMaster AS CM
               ON C.ChangeMasterId = CM.Id
             INNER JOIN core.Asset AS A
               ON C.ContextId = A.AssetId
             INNER JOIN settings.NotificationGroup AS NG
               ON N.NotificationGroupId = NG.Id
             INNER JOIN settings.NotificationGroupRecipient AS NGR
               ON N.NotificationTypeId = NGR.Id
             LEFT JOIN dbo.aspnet_Membership AS AM
               ON NGR.UserId = AM.UserId
             LEFT JOIN dbo.aspnet_Users AS AU
               ON AM.UserId = AU.UserId
             WHERE
               N.NotificationType = @NotificationType
               AND CM.ChangeRecordType = 1
               AND A.[Type] <> 'unknown'
               AND @Severity = 0
               AND ( ( C.StartTimestamp > CN.LastNotified
                       OR CN.LastNotified IS NULL
                     )
                     OR CN.LastNotifiedStatus = 4
                   )
             UNION
             SELECT DISTINCT --cleared messages
               N.Id
             , MN.MessageId
             , N.NotificationGroupId
             , N.NotificationType AS [NotificationGroupType]
             , N.NotificationTypeId
             , MN.LastNotified
             , MN.LastNotifiedStatus
             , MN.LastNotifiedMessage
             , COALESCE(AM.Email, NGR.Recipient) AS [Email]
             , AU.UserName
             , COALESCE(A.OverrideName, A.ShortName) AS [DisplayName]
             , A.[Status] AS [CurrentStatus]
             , M.AssetId
             , M.StartTimestamp
             , M.EndTimestamp
             , M.ClearedByUserName
             , CONVERT(VARCHAR(MAX), M.ExtendedProperties) AS [ExtendedProperties]
             , M.Severity
             , M.Text AS [Description]
             , CAST(0 AS SMALLINT) AS [NotificationRepeatInterval]
             , NG.NotificationFormat
             , NG.BatchMsg
             , NG.ClearMessages
             , NG.EmailSubjectLine
             , N.NewMessagesSetting AS [NewMessages]
             , NG.SetMessages
             , CAST(0 AS TINYINT) AS [NotificationType]
             FROM
               notification.MessageNotification AS MN
             INNER JOIN [notification].[Notification] AS N
               ON MN.NotificationId = N.Id
             INNER JOIN log.Message AS M
               ON MN.MessageId = M.Id
             INNER JOIN core.Asset AS A
               ON M.AssetId = A.AssetId
             INNER JOIN settings.NotificationGroup AS NG
               ON N.NotificationGroupId = NG.Id
             INNER JOIN settings.NotificationGroupRecipient AS NGR
               ON N.NotificationTypeId = NGR.Id
             LEFT JOIN dbo.aspnet_Membership AS AM
               ON NGR.UserId = AM.UserId
             LEFT JOIN dbo.aspnet_Users AS AU
               ON AM.UserId = AU.UserId
             WHERE
               N.NotificationType = @NotificationType
               AND A.[Type] <> 'unknown'
               AND ( M.Severity & @Severity ) = M.Severity
               AND MN.LastNotifiedStatus <> 8
               AND M.EndTimestamp IS NOT NULL
           ) AS NTP
       CROSS APPLY core.GetAsset(NTP.AssetId) AS GA
       ORDER BY
           NTP.ClearedByUserName
         , NTP.Severity
       SET ROWCOUNT 0;
   END
GO
