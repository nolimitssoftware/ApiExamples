CREATE PROC notification.UpdateMessageNotification
   @Id UNIQUEIDENTIFIER
 , @NotificationId UNIQUEIDENTIFIER
 , @LastNotifiedStatus TINYINT
 , @LastNotifiedMessage VARCHAR(1024) = NULL
AS
   BEGIN
       UPDATE
           notification.[MessageNotification]
       SET
           LastNotified = GETUTCDATE()
         , LastNotifiedStatus = @LastNotifiedStatus
         , LastNotifiedMessage = @LastNotifiedMessage
       WHERE
           NotificationId = @NotificationId
           AND MessageId = @Id;
   END
GO
