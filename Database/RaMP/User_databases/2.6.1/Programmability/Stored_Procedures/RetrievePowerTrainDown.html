CREATE PROCEDURE [core].[RetrievePowerTrainDown]
   @AssetId UNIQUEIDENTIFIER
AS
   BEGIN
   
       DECLARE @DownPowerPath TABLE
           (
             Id UNIQUEIDENTIFIER
           , AssetId UNIQUEIDENTIFIER
           , UpstreamAssetId UNIQUEIDENTIFIER
           , NumberOfPowerParentsForAssetId INT
           , HasParentOutsideOfThisDownPath BIT DEFAULT 0
           , IsFailableForGivenAsset BIT DEFAULT 1
           ) ;

       -- Get the power train of what all is down (directly and indirectly plugged into) the given asset
       WITH DownstreamPower_CTE (Id, AssetId, UpstreamAssetId)
       AS
           (
               SELECT Id, AssetId, UpstreamAssetId
               FROM core.UpstreamPower AS UP
               WHERE UpstreamAssetId = @AssetId
               UNION ALL
               SELECT UP2.Id, UP2.AssetId, UP2.UpstreamAssetId
               FROM core.UpstreamPower AS UP2
               INNER JOIN DownstreamPower_CTE AS DPCTE ON DPCTE.AssetId = UP2.UpstreamAssetId
           )
           INSERT INTO @DownPowerPath
               (
                   Id
                   ,AssetId
                   ,UpstreamAssetId
                   ,NumberOfPowerParentsForAssetId
               )
               SELECT DISTINCT
                   DPCTE.Id, DPCTE.AssetId ,DPCTE.UpstreamAssetId
                   , (SELECT COUNT(1) FROM core.UpstreamPower AS UP WHERE UP.AssetId = DPCTE.AssetId) AS 'NumberOfPowerParentsForAssetId'
               FROM DownstreamPower_CTE AS DPCTE
       
       DECLARE @AssetsWithPowerParentsOutsideGivenPath TABLE
           (
               AssetId UNIQUEIDENTIFIER
           )        
       
       --Figure out what assets in the path have power parents outside this path.  (IE. what assets in the path are plugged into things outside this path)
       --    Those devices are not devices that will fail when the given device fails.
       INSERT INTO @AssetsWithPowerParentsOutsideGivenPath
           (AssetId)
           SELECT
               AssetId
           FROM @DownPowerPath AS DPP
           WHERE
               DPP.NumberOfPowerParentsForAssetId > 1
               AND (SELECT COUNT (1) FROM @DownPowerPath WHERE AssetId = DPP.AssetId) <> DPP.NumberOfPowerParentsForAssetId ;

       --From each of those devices walk down from them in the path to find any other devices that they make power redundant.
       WITH DownstreamPower2_CTE (Id, AssetId, UpstreamAssetId)
       AS
           (
               SELECT Id, AssetId, UpstreamAssetId
               FROM core.UpstreamPower AS UP
               WHERE AssetId IN (SELECT AssetId FROM @AssetsWithPowerParentsOutsideGivenPath)
               UNION ALL
               SELECT UP2.Id, UP2.AssetId, UP2.UpstreamAssetId
               FROM core.UpstreamPower AS UP2
               INNER JOIN DownstreamPower2_CTE AS DPCTE ON DPCTE.AssetId = UP2.UpstreamAssetId
           )
               UPDATE @DownPowerPath
                   SET IsFailableForGivenAsset = 0
               FROM
                   @DownPowerPath AS DPP
                   LEFT JOIN DownstreamPower2_CTE AS DPCTE ON DPP.AssetId = DPCTE.AssetId
               WHERE DPCTE.AssetId IS NOT NULL
                       
       SELECT
           DPP.Id
           ,DPP.AssetId
           ,DPP.UpstreamAssetId
           ,DPP.IsFailableForGivenAsset
           ,A1.Name AS 'DownstreamName'
           ,A1.ShortName AS 'DownstreamShortName'
           ,A1.OverrideName AS 'DownstreamOverrideName'
           ,A1.[Type] AS 'DownstreamType'
           ,A1.LifecycleStatus AS 'DownstreamLifecycleStatus'
           ,M.[Power] AS 'DownstreamSystemModelPower'
           ,M.UserPower AS 'DownstreamUserModelPower'
           ,(SELECT TOP 1 S.SensorId FROM core.Sensor AS S WHERE S.DeviceId = DPP.AssetId AND S.[Type] IN ('power', 'totalPower', 'outputTotalPower') ORDER BY CASE S.[Type] WHEN  'outputTotalPower' THEN 1 WHEN 'totalPower' THEN 2 WHEN 'power' THEN 3 ELSE 4 END) AS 'TotalPowerSensorId'
           ,(SELECT TOP 1 S.OriginType FROM core.Sensor AS S WHERE S.DeviceId = DPP.AssetId AND S.[Type] IN ('power', 'totalPower', 'outputTotalPower') ORDER BY CASE S.[Type] WHEN  'outputTotalPower' THEN 1 WHEN 'totalPower' THEN 2 WHEN 'power' THEN 3 ELSE 4 END) AS 'TotalPowerOriginType'
           ,(SELECT TOP 1 S.SensorId FROM core.Sensor AS S WHERE S.DeviceId = DPP.AssetId AND S.[Type] IN ('totalLoad', 'outputTotalLoad') ORDER BY CASE S.[Type] WHEN  'outputTotalLoad' THEN 1 WHEN 'totalLoad' THEN 2 ELSE 3 END) AS 'TotalLoadSensorId'
           ,(SELECT TOP 1 S.OriginType FROM core.Sensor AS S WHERE S.DeviceId = DPP.AssetId AND S.[Type] IN ('totalLoad', 'outputTotalLoad') ORDER BY CASE S.[Type] WHEN  'outputTotalLoad' THEN 1 WHEN 'totalLoad' THEN 2 ELSE 3 END) AS 'TotalLoadOriginType'
       FROM
           @DownPowerPath AS DPP
           INNER JOIN core.Asset AS A1 ON DPP.AssetId = A1.AssetId
           INNER JOIN CORE.Model as M on M.Id = A1.ModelId
           
   END
GO
