CREATE PROCEDURE core.InsertLogicalAssetProperty
    @AssetId UNIQUEIDENTIFIER
 , @Name VARCHAR(256)
 , @SystemValue SQL_VARIANT
 , @UserValue SQL_VARIANT
 , @Display BIT = 1
 , @Editable BIT = 0
AS
    BEGIN
       DECLARE @Output TABLE(Id UNIQUEIDENTIFIER);
         INSERT  core.LogicalAssetProperty
                    (
                       AssetId
                    , Name
                    , SystemValue
                    , UserValue
                    , Display
                    , Editable
                )
         OUTPUT  INSERTED.Id
         INTO @Output
         VALUES
                    (
                       @AssetId
                    , @Name
                    , @SystemValue
                    , @UserValue
                    , @Display
                    , @Editable
                )

           SELECT * FROM @Output;
    END
GO
