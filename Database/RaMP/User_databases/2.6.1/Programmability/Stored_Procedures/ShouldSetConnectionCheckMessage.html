CREATE PROCEDURE [core].ShouldSetConnectionCheckMessage
   @AssetId UNIQUEIDENTIFIER
   ,@CheckPowerConnections BIT = 1
   ,@CheckManualNetworkConnections BIT = 1
AS
   BEGIN
       DECLARE @SetMessage BIT = 0 ;

       IF (@CheckPowerConnections = 1)
           BEGIN
               --This will check both sides of the power connection
               SELECT
                   @SetMessage = 1
               FROM
                   core.UpstreamPower AS UP
               WHERE
                   UP.AssetId = @AssetId
                   OR
                   UP.UpstreamAssetId = @AssetId ;
           END

       IF (@CheckManualNetworkConnections = 1 AND @SetMessage = 0)
           BEGIN    
               --This will check both sides of the manual network connection
               SELECT
                   @SetMessage = 1
               FROM
                   core.ManualDownstreamNetworkConnection AS MDNC
                   INNER JOIN core.NetworkInterface AS N ON MDNC.NetworkInterfaceId = N.Id
               WHERE
                   MDNC.DownstreamAssetId = @AssetId
                   OR
                   N.AssetId = @AssetId        
           END

       DECLARE @CheckConnectionsMessageMasterId UNIQUEIDENTIFIER = 'b2a48769-6d8f-45a2-ae1b-ce83bdd0f5b7';
       IF (@SetMessage = 0)
           BEGIN
               SELECT CAST(0 AS BIT);
           END
       ELSE IF (EXISTS (SELECT 1 FROM log.Message AS M WHERE M.MessageMasterId = @CheckConnectionsMessageMasterId AND M.AssetId = @AssetId AND M.EndTimestamp IS NULL))
           BEGIN
               --Message is already active
               SELECT CAST(0 AS BIT);
           END
       ELSE
           BEGIN
               SELECT CAST(1 AS BIT);
           END
   END
GO
