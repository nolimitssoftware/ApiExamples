CREATE PROC [core].[SaveRackSpaceCounts]
   @rackId UNIQUEIDENTIFIER
 , @unit INT
 , @number INT
AS
   BEGIN
       IF EXISTS ( SELECT
                       1
                   FROM
                       [core].[RackSpace]
                   WHERE
                       assetId = @rackId
                       AND Unit = @unit )
           BEGIN
               UPDATE
                   [core].[RackSpace]
               SET
                   AssetId = @rackId
                 , Number = @number
               WHERE
                   AssetId = @rackId
                   AND Unit = @unit
           END
       ELSE
           BEGIN
               INSERT  INTO [core].[RackSpace]
                       ( AssetId, Unit, Number )
               VALUES
                       ( @rackId, @unit, @number )
           END
   END
GO
