CREATE PROCEDURE [core].[RetrieveSensorCollection]
    @AssetId UNIQUEIDENTIFIER
 , @ShowHidden BIT = 0
AS -- Summary: We will need to get three types of sensors
   --    * Sensors that belong to the given asset.
   --    * Sensors that have an explicit location of the given asset (The given asset should be a rack or physically group)
   --    * Temperature or Humidity Sensors that do not have an explicit location of the given asset BUT whose asset is inside the given asset.  (Again really only for when the given asset is a rack or physical group)
   
   DECLARE    @Sensors TABLE
       (
         SensorId UNIQUEIDENTIFIER
       , RackLocation TINYINT NULL
       );

   -- Get the sensors that belong to the asset (this will be all that is really needed for non rack/physical groups).
   INSERT    @Sensors
           (
             SensorId
           , RackLocation
           )
           SELECT
               S.SensorId
             , S.RackLocation
           FROM
               core.Sensor AS S
           WHERE
               DeviceId = @AssetId

   --Get some things we may need
   DECLARE    @AssetNode HIERARCHYID
   DECLARE    @LocationAssetName VARCHAR(256)
   DECLARE    @LocationAssetShortName VARCHAR(256)
   DECLARE    @LocationAssetOverrideName VARCHAR(256)
   DECLARE    @LocationAssetType VARCHAR(64)
   SELECT
       @AssetNode = A.AssetNode
     , @LocationAssetName = A.Name
     , @LocationAssetShortName = A.ShortName
     , @LocationAssetOverrideName = A.OverrideName
     , @LocationAssetType = A.[Type]
   FROM
       core.Asset AS A
   WHERE
       A.AssetId = @AssetId
   
   --We only need to grab the other sensors if the given asset is a rack.
   IF (
        @LocationAssetType = 'rack'
        OR @LocationAssetType = 'physicalGroup'
      )
       BEGIN
       --Insert anything that has a location of the given asset (but is not the asset's sensor)
           INSERT    @Sensors
                   (
                     SensorId
                   , RackLocation
                   )
                   SELECT
                       S.SensorId
                     , S.RackLocation
                   FROM
                       core.Sensor AS S
                   WHERE
                       LocationAssetId = @AssetId
                       AND DeviceId <> @AssetId
       
       --Insert any TEMPERATURE or HUMIDITY that does not have different location from itself but its parent is the same as the given asset.  
       --    IE. Give all the TEMPERATURE or HUMIDITY sensors who belong to an asset whose location is the given asset.  This will give us all the sensors that have not been explicitly set to the asset.
       --    NOTE: If we ever come up with a new temperature or humidity like sensor that we want to support they will have to be added to this where clause.
           INSERT    @Sensors
                   (
                     SensorId
                   , RackLocation
                   )
                   SELECT
                       S.SensorId
                     , L.RackLocation
                   FROM
                       core.Sensor AS S
                   INNER JOIN core.Asset AS A
                       ON S.DeviceId = A.AssetId
                   LEFT JOIN core.Location AS L
                       ON L.Id = A.AssetId
                   WHERE
                       S.LocationAssetId = A.AssetId
                       AND A.AssetNode.GetAncestor(1) = @AssetNode
                       AND (
                             S.[Type] = 'ambientTemperature'
                             OR S.[Type] = 'humidity'
                           )

           SELECT
               S2.[SensorId]
             , S2.[SensorId] AS 'Id'
             , S2.[DeviceId] AS 'AssetId'
             , A2.ShortName AS 'ParentShortName'
             , A2.OverrideName AS 'ParentOverrideName'
             , S2.[DiscoveryAgentId]
             , S2.[Source]
             , S2.[Type]
             , S2.[SystemName] AS 'Name'
             , S2.[SystemName]
             , S2.[UserName]
             , S2.[Unit]
             , S2.[Element]
             , S2.[OriginType]
             , S2.[SensorXml]
             , S2.[ExtendedProperties]
             , @AssetId AS 'LocationAssetId'
             , @LocationAssetName AS 'LocationAssetName'
             , @LocationAssetShortName AS 'LocationAssetShortName'
             , @LocationAssetOverrideName AS 'LocationAssetOverrideName'
             , @LocationAssetType AS 'LocationAssetType'
             , S1.RackLocation
             , COALESCE(S2.[Position], SC.DefaultPosition) AS [Position]
             , SV.Value AS 'CurrentValue'
             , SV.LastUpdate AS 'CurrentValueLastUpdate'
             , SC.LocationEditable
             , SC.DisplayInAssetSensorList
             , COALESCE (SSV.Value, CAST(SV.Value AS VARCHAR(32))) AS 'RawValue'
           FROM
               @Sensors AS S1
           INNER JOIN core.Sensor AS S2
               ON S1.SensorId = S2.SensorId
           INNER JOIN [core].Asset AS A2
               ON S2.DeviceId = A2.AssetId
           LEFT JOIN settings.SensorConfiguration AS SC
               ON S2.[Type] = SC.SensorType
           LEFT JOIN monitoring.SensorValue AS SV
               ON S2.SensorId = SV.SensorId AND SV.RollupType = 1
           LEFT JOIN monitoring.SensorStringValue AS SSV
               ON S2.SensorId = SSV.SensorId AND S2.Unit = 'value'
           WHERE
               @ShowHidden = 1
               OR SC.IsHidden = 0

       END
   ELSE
       BEGIN
   
           SELECT
               S2.[SensorId]
             , S2.[SensorId] AS 'Id'
             , S2.[DeviceId] AS 'AssetId'
             , A2.ShortName AS 'ParentShortName'
             , A2.OverrideName AS 'ParentOverrideName'
             , S2.[DiscoveryAgentId]
             , S2.[Source]
             , S2.[Type]
             , S2.[SystemName] AS 'Name'
             , S2.[SystemName]
             , S2.[UserName]
             , S2.[Unit]
             , S2.[Element]
             , S2.[OriginType]
             , S2.[SensorXml]
             , S2.[ExtendedProperties]
             , S2.LocationAssetId AS 'LocationAssetId'
             , A.Name AS 'LocationAssetName'
             , A.ShortName AS 'LocationAssetShortName'
             , A.OverrideName AS 'LocationAssetOverrideName'
             , A.[Type] AS 'LocationAssetType'
             , S1.RackLocation              
             , COALESCE(S2.[Position], SC.DefaultPosition) AS [Position]      
             , SV.Value AS 'CurrentValue'
             , SV.LastUpdate AS 'CurrentValueLastUpdate'      
             , SC.LocationEditable
             , SC.DisplayInAssetSensorList
             , COALESCE (SSV.Value, CAST(SV.Value AS VARCHAR(32))) AS 'RawValue'  
           FROM
               @Sensors AS S1
           INNER JOIN core.Sensor AS S2
               ON S1.SensorId = S2.SensorId
           INNER JOIN [core].Asset AS A
               ON S2.LocationAssetId = A.AssetId
           INNER JOIN [core].Asset AS A2
               ON S2.DeviceId = A2.AssetId
           LEFT JOIN settings.SensorConfiguration AS SC
               ON S2.[Type] = SC.SensorType
           LEFT JOIN monitoring.SensorValue AS SV
               ON S2.SensorId = SV.SensorId AND SV.RollupType = 1
           LEFT JOIN monitoring.SensorStringValue AS SSV
               ON S2.SensorId = SSV.SensorId AND S2.Unit = 'value'
           WHERE
               @ShowHidden = 1
               OR SC.IsHidden = 0
       END
GO
