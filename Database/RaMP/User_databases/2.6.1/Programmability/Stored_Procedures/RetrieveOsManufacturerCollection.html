[settings].[RetrieveOsManufacturerCollection]
Properties
PropertyValue

SQL Script
CREATE PROC settings.RetrieveOsManufacturerCollection
AS
   BEGIN
       SELECT
           M.Id
         , M.Name
         , M.AlternateNames
         , M.IsSystemDefined
         , M.InternalId
       FROM
           core.Manufacturer AS M
       INNER JOIN core.ServerOperatingSystem AS SOS
           ON M.Id = SOS.ManufacturerId
   
   END
GO
Uses
  • [core].[Manufacturer]
  • [core].[ServerOperatingSystem]
  • [settings]
