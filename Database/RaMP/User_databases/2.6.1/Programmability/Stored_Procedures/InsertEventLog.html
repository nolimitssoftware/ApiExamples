CREATE PROC [log].[InsertEventLog]
    @Level NVARCHAR(10)
 , @Logger NVARCHAR(128)
 , @Message VARCHAR(512) = NULL
 , @CallSite VARCHAR(256) = NULL
 , @FlatStackTrace VARCHAR(MAX) = NULL
 , @RawStackTrace VARCHAR(MAX) = NULL
 , @Exception VARCHAR(MAX) = NULL
 , @Context VARCHAR(MAX) = NULL
 , @Timestamp DATETIME = NULL
AS
    BEGIN
         INSERT  log.EventLog
                    (
                       Level
                    , Message
                    , CallSite
                    , FlatStackTrace
                    , RawStackTrace
                    , Exception
                    , Context
                    , [Timestamp]
                    , Logger
                    
                    )
         VALUES
                    (
                       @Level
                    , @Message
                    , @CallSite
                    , @FlatStackTrace
                    , @RawStackTrace
                    , @Exception
                    , @Context
                    , COALESCE(@Timestamp, GETUTCDATE())
                    , @Logger                    
                    
                    )
    END
GO
