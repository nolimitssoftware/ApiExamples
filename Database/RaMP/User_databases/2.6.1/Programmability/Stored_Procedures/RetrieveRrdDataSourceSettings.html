CREATE PROCEDURE [settings].[RetrieveRrdDataSourceSettings]
    @UseType VARCHAR(32) = 'device'
AS
   SELECT
        RDST.Name AS 'DataSourceType'
       ,Heartbeat
   FROM
       [settings].RrdDataSourceSetting RDSS
       INNER JOIN
       [settings].RrdDataSourceType RDST
       ON RDSS.RrdDataSourceTypeId = RDST.RrdDataSourceTypeId
   WHERE RDSS.UseType = @UseType
GO
