CREATE PROC core.RetrieveApplicationCollection
   @AssetId UNIQUEIDENTIFIER
 , @Start INT = 1
 , @Limit INT = NULL
 , @TotalCount INT OUTPUT
AS
   BEGIN

       IF ( @Limit IS NULL )
           BEGIN
               SET @Limit = 2500 ;
           END
       DECLARE @End INT = @Start + @Limit - 1 ;
  
       SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED ;
       SET NOCOUNT ON ;
  
       SELECT
           @TotalCount = COUNT(A.Id)
       FROM
           core.[Application] AS A
       WHERE
           A.AssetId = @AssetId ;      
  
       SELECT
           Id
         , AssetId
         , Name
         , Installed
         , [Source]
         , ExtendedProperties
         , Approved
       FROM
           (
             SELECT
               A.Id
             , A.AssetId
             , A.Name
             , A.Installed
             , A.[Source]
             , A.ExtendedProperties
             , COALESCE(S.Approved, (CAST(1 AS BIT))) AS 'Approved'
             , ROW_NUMBER() OVER ( ORDER BY S.Approved, A.Name ) AS 'row_num'
             FROM
               core.[Application] AS A
             LEFT JOIN settings.software AS S
               ON A.Name = S.Name
             WHERE
               A.AssetId = @AssetId
           ) AS temp
       WHERE
           temp.row_num BETWEEN @Start AND @End
       ORDER BY
           temp.row_num
   
   END
GO
