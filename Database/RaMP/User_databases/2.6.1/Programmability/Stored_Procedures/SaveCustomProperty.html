CREATE PROC settings.SaveCustomProperty
   @Id UNIQUEIDENTIFIER = NULL
 , @Name NVARCHAR(256)
 , @Type TINYINT
 , @Values NVARCHAR(MAX) = NULL
 , @Unit NVARCHAR(32) = NULL
 , @AssetTypes INT = NULL
 , @IsActive BIT
 , @IsSearchField BIT = 0
 , @DefaultValue SQL_VARIANT = NULL
AS
   BEGIN
       
       IF ( @Id IS NULL )
           BEGIN
               INSERT  settings.CustomProperty
                       (
                         Name
                       , Type
                       , [Values]
                       , Unit
                       , AssetTypes
                       , IsActive
                       , IsSearchField
                       , DefaultValue                           
                       )
               OUTPUT  INSERTED.*
                     , SQL_VARIANT_PROPERTY(INSERTED.DefaultValue, 'baseType') AS 'DefaultValueType'
               VALUES
                       (
                         @Name
                       , @Type
                       , @Values
                       , @Unit
                       , @AssetTypes
                       , @IsActive
                       , @IsSearchField
                       , @DefaultValue                           
                       ) ;
           END
       ELSE
           BEGIN
               UPDATE
                   settings.CustomProperty
               SET
                   Name = @Name
                 , Type = @Type
                 , [Values] = @Values
                 , Unit = @Unit
                 , AssetTypes = @AssetTypes
                 , IsActive = @IsActive
                 , IsSearchField = @IsSearchField
                 , DefaultValue = @DefaultValue
               OUTPUT
                   INSERTED.*
                 , SQL_VARIANT_PROPERTY(INSERTED.DefaultValue, 'baseType') AS 'DefaultValueType'
               WHERE
                   settings.CustomProperty.Id = @Id ;
           END
       
   END
GO
