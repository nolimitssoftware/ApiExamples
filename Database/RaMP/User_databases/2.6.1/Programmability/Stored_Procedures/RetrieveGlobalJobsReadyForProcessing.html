CREATE PROCEDURE [job].[RetrieveGlobalJobsReadyForProcessing] @JobTypeId INT
AS
   BEGIN
   
       DECLARE @Root UNIQUEIDENTIFIER = ( SELECT TOP 1
                                           A.AssetId
                                          FROM
                                           core.Asset AS A
                                          WHERE
                                           A.AssetNode = HIERARCHYID::GetRoot()
                                        );
       IF NOT EXISTS ( SELECT
                           1
                       FROM
                           job.Processing AS P
                       WHERE
                           P.JobTypeId = @JobTypeId
                           AND P.Context = 1 --Asset Context
                           AND P.ContextId = @Root )
           BEGIN
               INSERT  job.Processing
                       ( Context
                       , ContextId
                       , LastProcessed
                       , LastJobStatus
                       , JobTypeId
                        
                       )
               VALUES
                       ( 1 --Asset Context
                       , @Root
                       , NULL
                       , NULL
                       , @JobTypeId
                        
                       );
           END
           
       SELECT
           P.Id
         , P.Context
         , P.ContextId
         , P.LastProcessed
         , P.LastJobStatus
         , P.JobTypeId
         , NULL AS [CronExpression]
       FROM
           job.Processing AS P
       LEFT JOIN job.JobConfiguration AS JC
           ON P.JobTypeId = JC.JobTypeId
       WHERE
           P.JobTypeId = @JobTypeId
           AND P.Context = 1 --Asset Context
           AND P.ContextId = @Root;
       
   END
GO
Uses

  • [job].[JobConfiguration]
