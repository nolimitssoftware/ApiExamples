CREATE PROCEDURE [settings].[SaveOsModel]
   @Id UNIQUEIDENTIFIER = NULL
 , @ManufacturerId UNIQUEIDENTIFIER = NULL
 , @Manufacturer VARCHAR(256) = NULL
 , @Name NVARCHAR(256) = NULL
 , @VersionString VARCHAR(256) = NULL
 , @SearchString VARCHAR(128) = NULL
 , @Major VARCHAR(16) = NULL
 , @Minor VARCHAR(16) = NULL
 , @Revision VARCHAR(16) = NULL
 , @Build VARCHAR(16) = NULL
 , @Other VARCHAR(16) = NULL
 , @Notes VARCHAR(MAX) = NULL
 , @ExtendedProperties XML = NULL
 , @IsApproved BIT = 1
 , @LicenseCount INT = 0
AS
   BEGIN
       IF ( @Name IS NULL )
           BEGIN
               SET @Name = 'Unknown' ;
           END
       IF ( @ManufacturerId IS NULL )
           BEGIN
               IF ( @Manufacturer IS NULL )
                   BEGIN
                       SET @Manufacturer = 'Unknown' ;
                   END
               SELECT TOP 1
                   @ManufacturerId = M.Id
               FROM
                   core.Manufacturer AS M
               LEFT JOIN core.ManufacturerAlternateName AS MAN
                   ON M.Id = MAN.ManufacturerId
               WHERE
                   (
                     M.Name = @Manufacturer
                     OR MAN.Name = @Manufacturer
                   ) ;
           END
           
       IF ( @Id IS NULL )
           BEGIN
               SELECT TOP 1
                   @Id = SOS.Id
               FROM
                   core.ServerOperatingSystem AS SOS
               WHERE
                   SOS.ManufacturerId = @ManufacturerId
                   AND SOS.Name = @Name ;
           END
               
       IF ( @Id IS NOT NULL )
           BEGIN
               
               UPDATE
                   core.ServerOperatingSystem
               SET
                   ManufacturerId = @ManufacturerId
                 , [Name] = @Name
                 , Major = @Major
                 , Minor = @Minor
                 , Revision = @Revision
                 , Build = @Build
                 , Other = @Other
                 , Notes = @Notes
                 , ExtendedProperties = @ExtendedProperties
                 , IsApproved = @IsApproved
                 , LicenseCount = @LicenseCount
               OUTPUT
                   INSERTED.Id
               WHERE
                   Id = @Id ;                
           END
                    
       ELSE
           BEGIN
               
               INSERT  core.ServerOperatingSystem
                       (
                         ManufacturerId
                       , Name
                       , VersionString
                       , SearchString
                       , Major
                       , Minor
                       , Revision
                       , Build
                       , Other
                       , Notes
                       , ExtendedProperties
                       , IsSystemDefined
                       , IsApproved
                       , LicenseCount
                       )
               OUTPUT  INSERTED.Id
               VALUES
                       (
                         @ManufacturerId
                       , @Name
                       , @VersionString
                       , @SearchString
                       , @Major
                       , @Minor
                       , @Revision
                       , @Build
                       , @Other
                       , @Notes
                       , @ExtendedProperties
                       , 0
                       , @IsApproved
                       , @LicenseCount
                       )
           END
       
   END
GO
