CREATE PROCEDURE [core].[SaveLocation]
   @Id UNIQUEIDENTIFIER
 , @AssetId UNIQUEIDENTIFIER
 , @Source VARCHAR(32)
 , @RackLocation TINYINT = NULL
 , @RaMPPort CHAR(4) = NULL
 , @ExtendedProperties XML
 , @AgentId UNIQUEIDENTIFIER = NULL
 , @PropertyXml XML = NULL
AS
   BEGIN
    
       IF EXISTS ( SELECT
                       1
                   FROM
                       core.Location
                   WHERE
                       Id = @Id )
           BEGIN
               UPDATE
                   core.Location
               SET
                   AssetId = @AssetId
                 , RackLocation = @RackLocation
                 , RaMPPort = @RaMPPort
                 , [Source] = @Source
                 , ExtendedProperties = @ExtendedProperties
                 , AgentId = @AgentId
                 , PropertyXml = @PropertyXml
               OUTPUT
                   INSERTED.Id
               WHERE
                   Id = @Id
           END
       ELSE
           BEGIN
           
               INSERT  core.Location
                       (
                         Id
                       , AssetId
                       , RackLocation
                       , RaMPPort
                       , Source
                       , ExtendedProperties
                       , AgentId
                       , PropertyXml                                          
                       )
               OUTPUT  INSERTED.Id
               VALUES
                       (
                         @Id
                       , @AssetId
                       , @RackLocation
                       , @RaMPPort
                       , @Source
                       , @ExtendedProperties
                       , @AgentId
                       , @PropertyXml                      
                       ) ;
           END

   END
GO
