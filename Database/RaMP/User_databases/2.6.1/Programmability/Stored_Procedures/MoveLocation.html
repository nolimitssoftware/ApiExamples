CREATE PROCEDURE [core].[MoveLocation]
   @Id UNIQUEIDENTIFIER
 , @OldAssetId UNIQUEIDENTIFIER
 , @NewAssetId UNIQUEIDENTIFIER
AS
   BEGIN
       DECLARE @ShouldUpdate BIT ;
       DECLARE @ExistingSource VARCHAR(32) = (
                                               SELECT
                                                   LOWER(L.Source)
                                               FROM
                                                   core.Location AS L
                                               WHERE
                                                   L.Id = @OldAssetId
                                             ) ;
       DECLARE @NewSource VARCHAR(32) = (
                                          SELECT
                                           LOWER(L.Source)
                                          FROM
                                           core.Location AS L
                                          WHERE
                                           L.Id = @NewAssetId
                                        ) ;
       SELECT
           @ExistingSource
         , @NewSource
       IF ( @ExistingSource IS NULL ) --no existing source        
           BEGIN
               PRINT 'no source' ;
               UPDATE
                   core.Location
               SET
                   Id = @NewAssetId
               WHERE
                   Id = @OldAssetId
                   AND AssetId = @Id ;
                   
               EXEC core.MoveAsset
                   @AssetId = @NewAssetId
                 , @NewParentId = @Id ;
                             
           END        
       ELSE
           IF ( @ExistingSource <> 'ramp' ) --source is not ramp
               BEGIN            
                   IF (
                        @NewSource IS NULL
                        OR (
                             @ExistingSource = 'api'
                             AND @NewSource <> 'api'
                           )
                      ) --new source is a serial communication or button
                       BEGIN
                           UPDATE
                               core.Location
                           SET
                               Id = @NewAssetId
                           WHERE
                               Id = @OldAssetId
                               AND AssetId = @Id ;    
                           
                           EXEC core.MoveAsset
                               @AssetId = @NewAssetId
                             , @NewParentId = @Id ;
                           
                       END                        
               END        
   END
GO
