CREATE PROCEDURE [settings].[CleanUpUserOwnedThings]
   @Id UNIQUEIDENTIFIER
AS
   BEGIN
   
       --When deleting a user reassign all tasks to the "admin" user if there or the first admin user if not.
       DECLARE @AdminId UNIQUEIDENTIFIER
       
       SELECT
           @AdminId = U.UserId
       FROM
           dbo.aspnet_Users AS U
           INNER JOIN dbo.aspnet_UsersInRoles AS UIR ON UIR.UserId = U.UserId
           INNER JOIN dbo.aspnet_Roles AS R ON UIR.RoleId = R.RoleId
       WHERE
           U.UserName = 'admin'
           AND U.UserId <> @Id
           AND R.LoweredRoleName = 'admin' ;
           
       IF (@AdminId IS NULL)
           BEGIN
           
               SELECT TOP 1
                   @AdminId = UserId
               FROM
                   dbo.aspnet_UsersInRoles AS UIR
                   INNER JOIN dbo.aspnet_Roles AS R ON UIR.RoleId = R.RoleId
               WHERE
                   UIR.UserId <> @Id
                   AND R.LoweredRoleName = 'admin' ;            
           
           END
           
       UPDATE
           work.Task
       SET
           OwnerUserId = @AdminId
       WHERE
           OwnerUserId = @Id ;
   
   END
GO
