CREATE PROCEDURE [job].[SetAssetDueForAnalysis]
   @AssetId UNIQUEIDENTIFIER
AS
   DECLARE @ParentId UNIQUEIDENTIFIER
   
   --Set dumb power device job so it will be processed next.
   UPDATE
       [job].[Processing]
   SET
       [LastProcessed] = NULL,
       LastJobStatus = 0
   WHERE
       JobTypeId = 212 ; -- DumbPowerDeviceJob

   IF NOT EXISTS(SELECT 1 FROM core.Asset WHERE assetid = @AssetId and [Type] = 'rack')
       BEGIN
           --It is not a rack so lets see if it is in a rack and get the rack id
           DECLARE @Node HIERARCHYID
           SELECT @Node = assetnode FROM core.Asset WHERE assetid = @AssetId ;
           
           SELECT
               @ParentId = assetid
           FROM
               core.Asset
           WHERE
               AssetNode = @node.GetAncestor(1) AND [Type] = 'rack' ;
       END
   ELSE
       BEGIN
           --It is a rack so let us update its record so it will process
           SET @ParentId = @AssetId ;
       END
   
   IF (@ParentId IS NOT NULL)
       BEGIN
           --Set rack job so it will be processed next.
           UPDATE
               [job].[Processing]
           SET
               [LastProcessed] = NULL,
               LastJobStatus = 0
           WHERE
               JobTypeId = 1 -- RackProcessingJob
               AND
               ContextId = @ParentId ;            
       END

   --Set physical group job so it will be processed next.
   UPDATE
       [job].[Processing]
   SET
       [LastProcessed] = NULL,
       LastJobStatus = 0
   WHERE
       JobTypeId = 201 ; -- PhysicalGroupJob

   -- Take care of the projected power jobs.
   EXEC [job].[SetProjectedPowerJobsDueForAnalysis] @AssetId
GO
Uses
  • [core].[Asset]
  • [job].[Processing]
  • [job].[SetProjectedPowerJobsDueForAnalysis]
  • [job]
