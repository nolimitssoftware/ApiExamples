CREATE PROCEDURE [log].RetrieveGlobalEventCollection
   @LocationId UNIQUEIDENTIFIER = NULL
 , @Severities VARCHAR(256) = 'critical,warning'
 , @SortExpression NVARCHAR(512) = 'M.StartTimestamp DESC, M.Severity DESC'
 , @Start INT
 , @Limit INT = NULL
 , @StartDate DATETIME = '1/1/1753'
 , @EndDate DATETIME = '12/31/9999'
 , @ShowActive BIT = 1
 , @Debug BIT = 0
 , @TotalCount INT OUTPUT
AS
   BEGIN    
       DECLARE @LocationNode HIERARCHYID ;
       DECLARE @StatusNumber INT ;
       
       SET @StatusNumber = core.GetStatusesAsNumber(@Severities);
       
       print @StatusNumber
       
       IF (@LocationId IS NULL)
           BEGIN
               SELECT TOP 1 @LocationId = AssetId, @LocationNode = AssetNode FROM core.Asset WHERE AssetNode = HIERARCHYID::GetRoot()
           END
       ELSE
           BEGIN
               SELECT TOP 1 @LocationNode = AssetNode FROM core.Asset WHERE AssetId = @LocationId
           END
       
       SET NOCOUNT ON ;
       DECLARE @End INT = @Start + @Limit - 1 ;
       SELECT
           @TotalCount = COUNT(1)
       FROM
           log.Message AS M
           INNER JOIN core.Asset AS A ON M.AssetId = A.AssetId
       WHERE
           A.AssetNode.IsDescendantOf(@LocationNode) = 1
           AND
           ( M.StartTimestamp BETWEEN @StartDate AND @EndDate )
           AND ((M.Severity & @StatusNumber) = M.Severity)
           AND ((@ShowActive = 1 AND M.EndTimestamp IS NULL) OR @ShowActive = 0)
           AND A.Type <> 'unknown' ;
       
       IF(@Limit IS NULL)
       BEGIN
           --For non paged results
           SET @End = @TotalCount;
       END
       
       DECLARE @sql NVARCHAR(4000) ;
       DECLARE @paramlist NVARCHAR(4000) = '
   @Start INT,
   @End INT,
   @StartDate DATETIME,
   @EndDate DATETIME,
   @LocationNode HIERARCHYID,
   @StatusNumber INT,
   @ShowActive BIT'
   
       IF ( LEN(@SortExpression) = 0 )
           BEGIN
               SET @SortExpression = 'M.StartTimestamp DESC, M.Severity DESC'
           END
    
       IF ( PATINDEX('M.StartTimestamp%', @SortExpression) = 0 )
           BEGIN
               SET @SortExpression = 'M.' + @SortExpression
                   + ', M.StartTimestamp DESC' ;
           END    
       SET @sql = '
       SELECT
       *
     FROM
     (
       SELECT
           M.Id ,
           M.MessageMasterId ,
           M.AssetId,
           A.Type AS ''AssetType'',
           A.Name AS ''AssetName'',
           A.ShortName AS ''AssetShortName'',
           A.OverrideName AS ''AssetOverrideName'',
           PA.AssetId AS ''ParentAssetId'',
           PA.Type AS ''ParentAssetType'',
           PA.Name AS ''ParentAssetName'',
           PA.ShortName AS ''ParentAssetShortName'',
           PA.OverrideName AS ''ParentAssetOverrideName'',
           M.StartTimestamp ,
           M.EndTimestamp,  
           M.ClearedByUserName,
           M.ExtendedProperties,
           M.Text,
           M.Severity,
           ROW_NUMBER() OVER (ORDER BY ' + @SortExpression + ') as RowNum  
       FROM
           log.Message AS M
           INNER JOIN core.Asset AS A ON M.AssetId = A.AssetId
           LEFT JOIN core.Asset as PA ON PA.AssetNode = A.AssetNode.GetAncestor(1)
       WHERE
           ( M.StartTimestamp BETWEEN @StartDate AND @EndDate )
           AND A.AssetNode.IsDescendantOf(@LocationNode) = 1
           AND ((M.Severity & @StatusNumber) = M.Severity)
           AND ((@ShowActive = 1 AND M.EndTimestamp IS NULL) OR @ShowActive = 0)
           AND A.Type <> ''unknown''
       ) AS entries
       WHERE
       entries.RowNum BETWEEN @Start AND @End
       ORDER BY
       entries.RowNum ASC'
   
       IF ( @Debug = 1 )
           BEGIN
               PRINT @paramlist ;
               PRINT @sql ;
           END
   
       EXEC dbo.sp_executesql
           @sql
         , @paramList
         , @Start
         , @End
         , @StartDate
         , @EndDate
         , @LocationNode
         , @StatusNumber
         , @ShowActive
   
   END
GO
