CREATE PROCEDURE [core].[ComputeChildReservedDevicePowerSummation]
   @AssetId UNIQUEIDENTIFIER -- The ID of a power asset with child devices plugged into it.
AS
   BEGIN
       DECLARE @ChildReservedDevices TABLE
               (
                AssetId UNIQUEIDENTIFIER
               , NumberOfDirectPowerParentsForAssetId INT
               , NumberOfDirectPowerParentsInPathForAssetId INT
               , ReservedPower DECIMAL
               ) ;

       WITH DownstreamPower_CTE (Id, AssetId, UpstreamAssetId)
       AS
           (
               SELECT Id, AssetId, UpstreamAssetId
               FROM core.UpstreamPower AS UP
               WHERE UpstreamAssetId = @AssetId
               UNION ALL
               SELECT UP2.Id, UP2.AssetId, UP2.UpstreamAssetId
               FROM core.UpstreamPower AS UP2
               INNER JOIN DownstreamPower_CTE AS DPCTE ON DPCTE.AssetId = UP2.UpstreamAssetId
           )
               INSERT INTO @ChildReservedDevices
                   (
                        AssetId
                       , NumberOfDirectPowerParentsForAssetId  
                       , NumberOfDirectPowerParentsInPathForAssetId
                       , ReservedPower
                   )                         
                       SELECT DISTINCT
                           DPCTE.AssetId
                           , (SELECT COUNT(1) FROM core.UpstreamPower AS UP WHERE UP.AssetId = DPCTE.AssetId) AS 'NumberOfDirectPowerParentsForAssetId'
                           , (SELECT COUNT(1) FROM DownstreamPower_CTE WHERE DownstreamPower_CTE.AssetId = DPCTE.AssetId) AS 'NumberOfDirectPowerParentsInPathForAssetId'
                           , COALESCE (M.UserPower, M.Power, 0) AS 'ReservedPower'
                       FROM DownstreamPower_CTE AS DPCTE
                       INNER JOIN core.Asset AS A ON DPCTE.AssetId = A.AssetId
                       INNER JOIN core.Model AS M ON A.ModelId = M.Id
                       WHERE
                           A.LifecycleStatus = 1
                           AND
                           A.Type NOT IN ('pdu', 'rpdu', 'ups', 'rackOrTowerUps', 'powerMeter')


       --select     *     from @ChildReservedDevices

       --We take the ModelPower and divide it by the number of power sources it has and then multiple that by the number power sources
       --    that are in the given path to figure out what portion of the reserved device's power the given power asset is using.
       SELECT
           @AssetId AS 'PowerAssetId',
           SUM ((ReservedPower / NumberOfDirectPowerParentsForAssetId) * NumberOfDirectPowerParentsInPathForAssetId) AS 'ChildrenReservedPowerSummation',
           (SELECT CAST (1 AS BIT) FROM [log].Message AS M WHERE M.AssetId = @AssetId AND M.EndTimestamp IS NULL AND M.MessageMasterId = '7e36f143-7adf-4d89-ae57-5a891520a9fc') AS 'IsUsablePowerExceededByProjectedPowerMessageActive'
       FROM
           @ChildReservedDevices
   END
GO
