CREATE PROCEDURE [visual].[RetrieveParentLayout]
   @AssetId UNIQUEIDENTIFIER
AS
   BEGIN
       DECLARE @AssetNode HIERARCHYID = (
                                          SELECT
                                           A.AssetNode
                                          FROM
                                           core.Asset AS A
                                          WHERE
                                           A.AssetId = @AssetId
                                        ) ;
       IF ( @AssetNode IS NOT NULL )
           BEGIN    
               DECLARE @ParentId UNIQUEIDENTIFIER = (
                                                      SELECT
                                                       A.AssetId
                                                      FROM
                                                       core.Asset AS A
                                                      WHERE
                                                       A.AssetNode = @AssetNode.GetAncestor(1)
                                                    ) ;
               IF ( @ParentId IS NOT NULL )
                   BEGIN
                       SELECT
                           A.AssetId
                         , A.[Name] AS 'Asset.Name'
                         , A.ShortName
                         , A.OverrideName
                         , A.Type
                         , A.[Status]
                                 , ISNULL(core.GetStatus(A.AssetNode), A.[Status]) AS [DerivedStatus]
                         , L.LayoutType
                         , L.BackgroundId
                         , CASE WHEN L.BackgroundId IS NULL THEN L.Width
                                ELSE B.Width
                           END AS 'Width'
                         , CASE WHEN L.BackgroundId IS NULL THEN L.Height
                                ELSE B.Height
                           END AS 'Height'
                         , L.ColumnFormat
                         , L.RowFormat
                         , B.Name
                         , B.ImageFormat
                         , B.UserDefined
                         , B.DateCreated
                         , B.DateModified
                       FROM
                           core.Asset AS A
                       LEFT JOIN visual.Layout AS L
                           ON A.AssetId = L.Id
                       LEFT JOIN visual.Background AS B
                           ON L.BackgroundId = B.BackgroundId
                       WHERE
                           A.AssetId = @ParentId ;
                   END
           END
   END
GO
