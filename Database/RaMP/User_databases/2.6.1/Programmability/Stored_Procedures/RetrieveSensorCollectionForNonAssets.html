CREATE PROCEDURE [core].[RetrieveSensorCollectionForNonAssets]
   @AssetId UNIQUEIDENTIFIER
 , @ShowHidden BIT = NULL
AS
   SELECT
       S.[SensorId]
     , S.[SensorId] AS 'Id'
     , S.[DeviceId] AS 'AssetId'
     , S.[DiscoveryAgentId]
     , S.[Source]
     , S.[Type]
     , S.[SystemName] AS 'Name'
     , S.[SystemName]
     , S.[UserName]
     , S.[Unit]
     , S.[Element]
     , S.[OriginType]
     , S.[SensorXml]
     , S.[ExtendedProperties]
     , SV.Value AS 'CurrentValue'
     , SV.LastUpdate AS 'CurrentValueLastUpdate'
   FROM
       [core].[NonAssetSensor] AS S
   LEFT JOIN settings.SensorConfiguration AS SC
       ON S.[Type] = SC.SensorType
   LEFT JOIN monitoring.SensorValue AS SV
       ON S.[SensorId] = SV.SensorId AND SV.RollupType = 1
   WHERE
       DeviceId = @AssetId
       AND (
             @ShowHidden = 1
             OR SC.IsHidden = 0
           )
GO
Uses
  • [core].[NonAssetSensor]
  • [monitoring].[SensorValue]
