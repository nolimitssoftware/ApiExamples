CREATE PROC core.RetrieveOperatingSystemCollection
   @AssetId UNIQUEIDENTIFIER
AS
   BEGIN
       SELECT
           OS.Id
         , OS.AssetId
         , OS.ServerOperatingSystemId
         , OS.ServicePackId
         , OS.Architecture
         , OS.Source
         , OS.ExtendedProperties
         , M.Id AS 'ManufacturerId'
         , M.Name AS 'Manufacturer'
         , SUBSTRING(temp.Names, 0, LEN(temp.Names)) AS 'ManufacturerAlternateNames'
         , SOS.Name
         , SOS.VersionString
         , SOS.SearchString
         , SOS.Major
         , SOS.Minor
         , SOS.Revision
         , SOS.Build
         , SOS.Other
         , SOS.Notes
         , SOS.ExtendedProperties
         , SOS.IsSystemDefined
         , SOS.SentToNLS
         , SOS.InternalId
       FROM
           core.OperatingSystem AS OS
       INNER JOIN core.ServerOperatingSystem AS SOS
           ON OS.ServerOperatingSystemId = SOS.Id
       INNER JOIN core.Manufacturer AS M
           ON SOS.ManufacturerId = M.Id
       CROSS APPLY (
                     SELECT
                       CONVERT(VARCHAR(256), ISNULL(MAN.[Name], '')) + ',' AS [text()]
                     FROM
                       core.ManufacturerAlternateName AS MAN
                     WHERE
                       MAN.ManufacturerId = M.Id
                   FOR
                     XML PATH('')
                   ) AS temp ( [Names] )
       WHERE
           OS.AssetId = @AssetId ;
   END
GO
Uses
