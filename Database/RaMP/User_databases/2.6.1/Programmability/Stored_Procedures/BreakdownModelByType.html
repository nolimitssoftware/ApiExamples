CREATE PROC core.BreakdownModelByType
   @Types VARCHAR(1024) = NULL
 , @ParentId UNIQUEIDENTIFIER = NULL
AS
   BEGIN
   
       DECLARE @ParentNode HIERARCHYID ;
       
       IF ( @ParentId IS NOT NULL )
           BEGIN
               SELECT
                   @ParentNode = A.AssetNode
               FROM
                   core.Asset AS A
               WHERE
                   A.AssetId = @ParentId ;
           END

       IF ( @ParentNode IS NULL )
           BEGIN
               SET @ParentNode = HIERARCHYID::GetRoot() ;
           END
       IF ( @Types IS NULL )
           BEGIN
               SELECT
                   CASE WHEN M2.Name = COALESCE(M.UserName, M.Name) THEN COALESCE(M.UserName, M.Name)
                        ELSE M2.Name + ' ' + COALESCE(M.UserName, M.Name)
                   END AS 'Model.Name'
                 , COUNT(M.Name) AS 'Count'
               FROM
                   core.Asset AS A
               INNER JOIN core.Model AS M
                   ON A.ModelId = M.Id
               INNER JOIN core.Manufacturer AS M2
                   ON M.ManufacturerId = M2.Id
               WHERE
                   A.AssetNode.IsDescendantOf(@ParentNode) = 1
                   AND core.IsInExcludedAssetNode(A.AssetNode) = 0
               GROUP BY
                   M2.Name
                 , M.Name
                       , M.UserName
               ORDER BY
                   'Model.Name'                
           END
       ELSE
           BEGIN
 
               SELECT
                   CASE WHEN M2.Name = COALESCE(M.UserName, M.Name) THEN COALESCE(M.UserName, M.Name)
                        ELSE M2.Name + ' ' + COALESCE(M.UserName, M.Name)
                   END AS 'Model.Name'
                 , COUNT(M.Name) AS 'Count'
               FROM
                   core.Asset AS A
               INNER JOIN util.split_delimiter(@Types, ',') AS SD
                   ON A.Type = SD.str
               INNER JOIN core.Model AS M
                   ON A.ModelId = M.Id
               INNER JOIN core.Manufacturer AS M2
                   ON M.ManufacturerId = M2.Id
               WHERE
                   A.AssetNode.IsDescendantOf(@ParentNode) = 1
                   AND core.IsInExcludedAssetNode(A.AssetNode) = 0
               GROUP BY
                   M2.Name
                 , M.NAME
                       , M.UserName
               ORDER BY
                   'Model.Name' ;
           END
   END
GO
