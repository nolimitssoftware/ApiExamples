CREATE PROCEDURE [core].[RetrieveRemoteMacCollectionByRemoteAsset]
   @RemoteAssetId UNIQUEIDENTIFIER
   , @DirectOnly BIT = 0
AS
   BEGIN
        SELECT
             A.[Name]
           , A.ShortName
           , A.OverrideName
           , RM.AssetId
           , RM.Mac
           , RM.Port
           , RM.ConnectionFlag
           , RM.Source
           , RM.Id
           , RM.ExtendedProperties
           , RM.RemoteAssetId
           , A.[Type] AS 'ConnectedType'
           , NI.Name AS 'PortName'
FROM
            core.RemoteMac AS RM
        INNER JOIN core.host AS H
             ON RM.AssetId = H.AssetId
        INNER JOIN core.asset AS A
             ON A.AssetId = H.AssetId
        INNER JOIN core.NetworkInterface AS NI
           ON NI.AssetId = RM.AssetId AND NI.Port = RM.Port
        WHERE
             RM.RemoteAssetId = @RemoteAssetId
             AND (@DirectOnly = 0 OR RM.ConnectionFlag = 'direct');
END
GO
