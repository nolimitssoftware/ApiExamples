CREATE PROCEDURE settings.RetrieveSoftwareSetting
   @SoftwareId UNIQUEIDENTIFIER
AS
   BEGIN
       SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED ;
       SET NOCOUNT ON ;
             
       SELECT
           SoftwareId
         , Name
         , Approved
         , NumberOfLicenses
         , NumberOfInstallations
         , CAST(CASE WHEN S.Approved = 0 THEN 1
                WHEN S.Approved = 1
                     AND (
                           S.NumberOfLicenses IS NULL
                           OR S.NumberOfLicenses = 0
                         ) THEN 0
                WHEN S.NumberOfLicenses < S.NumberOfInstallations THEN 1
                ELSE 0
           END AS BIT) AS [IsWarning]
       FROM
           settings.Software AS S
       WHERE
           S.SoftwareId = @SoftwareId ;
                       
   END
GO
