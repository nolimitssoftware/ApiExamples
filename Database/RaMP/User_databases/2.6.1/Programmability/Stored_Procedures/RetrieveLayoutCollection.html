CREATE PROCEDURE [visual].[RetrieveLayoutCollection]
   @ParentAssetId UNIQUEIDENTIFIER
 , @IncludeUnplaced BIT = 1
AS
   BEGIN
       DECLARE @ParentNode HIERARCHYID
      
       SELECT
           @ParentNode = A.AssetNode
       FROM
           core.Asset AS A
       WHERE
           A.AssetId = @ParentAssetId;
      
       IF ( @ParentNode IS NULL )
           BEGIN
               SET @ParentNode = HIERARCHYID::GetRoot()
           END
       IF ( @IncludeUnplaced = 1 )
           BEGIN
               SELECT
                   COALESCE(A.Name, LI.[Name]) AS 'Name'
                 , COALESCE(A.ShortName, LI.[Name]) AS 'ShortName'
                 , COALESCE(A.OverrideName, LI.[Name]) AS 'OverrideName'
                 , A.Type
                 , ISNULL(core.GetStatus(A.AssetNode), A.[Status]) AS [DerivedStatus]
                 , A.[Status]
                 , CASE WHEN LI.Id IS NULL THEN CAST (0 AS BIT)
                        ELSE CAST(1 AS BIT)
                   END AS 'IsPlaced'
                 , COALESCE(LI.Id, A.AssetId) AS 'Id'
                 , A.AssetId
                 , LI.LayoutId
                 , LI.LayoutItemType
                 , LI.PlacedId
                 , LI.X
                 , LI.Y
                 , LI.XOffset
                 , LI.YOffset
                 , LI.Radius
                 , LI.Width
                 , LI.Height
                 , LI.PhysicalWidth
                 , LI.PhysicalHeight
                 , LI.Color
                 , LI.Direction
                 , (
                     SELECT
                       COUNT(*)
                     FROM
                       core.Asset Children
                     WHERE
                       Children.AssetNode.IsDescendantOf(a.AssetNode) = 1
                       AND Children.Type != 'physicalGroup'
                   ) AS ChildCount
               FROM
                   core.Asset AS A
               LEFT JOIN core.Model AS M
                   ON M.Id = A.ModelId
               FULL JOIN visual.LayoutItem AS LI
                   ON A.AssetId = LI.Id
               WHERE
                   ( ( A.AssetLevel = @ParentNode.GetLevel() + 1
                       AND A.AssetNode.IsDescendantOf(@ParentNode) = 1
                     )
                     OR LI.LayoutId = @ParentAssetId
                   )
                   AND ( ( A.AssetType <> 'device' ) /*We only want to pull floor based devices */
                         OR ( ( A.[AssetType] = 'device'
                                OR A.AssetType IS NULL
                              )
                              AND ( M.RackUnits = 0
                                    OR M.RackUnits IS NULL
                                  )
                            )
                         OR ( ( A.[AssetType] = 'device'
                                OR A.AssetType IS NULL
                              )
                              AND M.RackUnits > 0
                              AND (
                                    SELECT TOP 1
                                       CONVERT(TINYINT, AP.UserValue) --must be converted to correct type
                                    FROM
                                       core.AssetProperty AS AP
                                    WHERE
                                       AP.AssetId = A.AssetId
                                       AND AP.Name = 'rackUnits'
                                  ) = 0
                            )
                       )
           END
       ELSE
           BEGIN
               SELECT
                   COALESCE(A.Name, LI.[Name]) AS 'Name'
                 , COALESCE(A.ShortName, LI.[Name]) AS 'ShortName'
                 , COALESCE(A.OverrideName, LI.[Name]) AS 'OverrideName'
                 , A.Type
                 , ISNULL(core.GetStatus(A.AssetNode), A.[Status]) AS [DerivedStatus]
                 , A.[Status]
                 , CASE WHEN LI.Id IS NULL THEN CAST (0 AS BIT)
                        ELSE CAST(1 AS BIT)
                   END AS 'IsPlaced'
                 , COALESCE(LI.Id, A.AssetId) AS 'Id'
                 , A.AssetId
                 , LI.LayoutId
                 , LI.LayoutItemType
                 , LI.PlacedId
                 , LI.X
                 , LI.Y
                 , LI.XOffset
                 , LI.YOffset
                 , LI.Radius
                 , LI.Width
                 , LI.Height
                 , LI.PhysicalWidth
                 , LI.PhysicalHeight
                 , LI.Color
                 , LI.Direction
                 , (
                     SELECT
                       COUNT(*)
                     FROM
                       core.Asset Children
                     WHERE
                       Children.AssetNode.IsDescendantOf(a.AssetNode) = 1
                       AND Children.Type != 'physicalGroup'
                   ) AS ChildCount
               FROM
                   core.Asset AS A
               FULL JOIN visual.LayoutItem AS LI
                   ON A.AssetId = LI.Id
               WHERE
                   ( @IncludeUnplaced = 1
                     OR LI.Id IS NOT NULL
                   )
                   AND ( ( A.AssetLevel = @ParentNode.GetLevel() + 1
                           AND A.AssetNode.IsDescendantOf(@ParentNode) = 1
                         )
                         OR LI.LayoutId = @ParentAssetId
                       )
               UNION
               SELECT
                   A.Name
                 , A.ShortName
                 , A.OverrideName
                 , A.Type
                 , ISNULL(core.GetStatus(A.AssetNode), A.[Status]) AS [DerivedStatus]
                 , A.[Status]
                 , CAST(0 AS BIT) AS 'IsPlaced'
                 , A.AssetId AS 'Id'
                 , A.AssetId
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , (
                     SELECT
                       COUNT(*)
                     FROM
                       core.Asset Children
                     WHERE
                       Children.AssetNode.IsDescendantOf(a.AssetNode) = 1
                       AND Children.Type != 'physicalGroup'
                   ) AS ChildCount
               FROM
                   core.Asset AS A
               INNER JOIN core.AssetProperty AS AP
                   ON A.AssetId = AP.AssetId
               WHERE
                   ( (
                 A.AssetLevel = @ParentNode.GetLevel() + 1
                     AND A.AssetNode.IsDescendantOf(@ParentNode) = 1
               )
                   )
                   AND ( AP.Name = 'row' )
               UNION
               SELECT
                   A.Name
                 , A.ShortName
                 , A.OverrideName
                 , A.Type
                 , ISNULL(core.GetStatus(A.AssetNode), A.[Status]) AS [DerivedStatus]
                 , A.[Status]
                 , CAST(0 AS BIT) AS 'IsPlaced'
                 , A.AssetId AS 'Id'
                 , A.AssetId
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , NULL
                 , (
                     SELECT
                       COUNT(*)
                     FROM
                       core.Asset Children
                     WHERE
                       Children.AssetNode.IsDescendantOf(a.AssetNode) = 1
                       AND Children.Type != 'physicalGroup'
                   ) AS ChildCount
               FROM
                   core.Asset AS A
               INNER JOIN core.CustomAssetProperty AS CAP
                   ON A.AssetId = CAP.AssetId
               INNER JOIN settings.CustomProperty AS CP
                   ON CAP.CustomPropertyId = CP.Id
               WHERE
                   ( (
                 A.AssetLevel = @ParentNode.GetLevel() + 1
                     AND A.AssetNode.IsDescendantOf(@ParentNode) = 1
               )
                   )
                   AND ( CP.Name = 'row' );
           END
   END
GO
