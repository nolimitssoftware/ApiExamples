CREATE PROCEDURE [visual].[SaveBackground]
   @BackgroundId UNIQUEIDENTIFIER = NULL
 , @Name NVARCHAR(256)
 , @ImageFormat TINYINT  
 , @Width REAL
 , @Height REAL
AS
   BEGIN
       
       IF ( @BackgroundId IS NULL )
           BEGIN
               INSERT  visual.Background
                       (
                         Name
                       , ImageFormat
                       , Width
                       , Height
                       , UserDefined
                       , DateCreated
                       , DateModified
                   
                       )
               OUTPUT  INSERTED.*
               VALUES
                       (
                         @Name
                       , @ImageFormat
                       , @Width
                       , @Height
                       , 1
                       , GETUTCDATE()
                       , GETUTCDATE()
                       ) ;
           END
       ELSE
           BEGIN
               UPDATE
                   visual.Background
               SET
                   Name = @Name
                 , ImageFormat = @ImageFormat                  
                 , Width = @Width
                 , Height = @Height
                 , DateModified = GETUTCDATE()
               OUTPUT
                   INSERTED.*
               WHERE
                   visual.Background.BackgroundId = @BackgroundId ;
           
           
           END
   END
GO
