SQL Script
CREATE PROC notification.RetrieveChangesForNotificationCollection
AS
   BEGIN
       SET NOCOUNT ON;

       DECLARE @ChangeMinTime DATETIME
       SELECT
           @ChangeMinTime = ISNULL(MIN(CN.LastNotified),
                             DATEADD(HOUR, -1, GETUTCDATE()))
       FROM
           notification.ChangeNotification AS CN
       LEFT JOIN notification.Notification AS N
           ON CN.NotificationId = N.Id;
      
       SELECT
           C.Id
         , C.ChangeMasterId
         , C.ContextId AS [AssetId]
         , C.StartTimestamp
         , NULL AS [EndTimestamp]
         , C.ChangedByUserName
         , C.ExtendedProperties
         , C.Context
         , C.ContextId
         , 0 AS [Severity]
         , CAST(1 AS BIT) AS [Change]
         , CM.Description
         , COALESCE(A.OverrideName, A.ShortName) AS [AssetName]
         , A.Type
       FROM
           log.Change AS C
       INNER JOIN log.ChangeMaster AS CM
           ON C.ChangeMasterId = CM.Id
       INNER JOIN core.Asset AS A
           ON C.ContextId = A.AssetId
       WHERE
           C.LogTypeId = 2 --only change
           AND A.[Type] <> 'unknown'
           AND CM.ChangeRecordType = 1 --only normal;
           AND CM.CanSendNotification = 1
           AND C.StartTimestamp > @ChangeMinTime;
   END
GO
Uses
  • [core].[Asset]
  • [log].[Change]
  • [log].[ChangeMaster]
  • [notification].[ChangeNotification]
  • [notification].[Notification]
  • [notification]
