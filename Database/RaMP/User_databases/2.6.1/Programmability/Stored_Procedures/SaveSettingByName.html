CREATE PROCEDURE [settings].[SaveSettingByName]
    @Group VARCHAR(64)
 , @Name VARCHAR(256)
 , @Value SQL_VARIANT = NULL
AS
    BEGIN
         DECLARE @Id UNIQUEIDENTIFIER = (
                                                     SELECT
                                           S.Id
                                                     FROM
                                                         settings.Setting AS S
                                                     WHERE
                                                         S.[Group] = @Group
                                                         AND S.Name = @Name
                                                   ) ;
   
         IF ( @Id IS NULL )
               BEGIN
                    INSERT  settings.Setting
                               ( Name, [Group], Value )
                    OUTPUT  INSERTED.Id
                    VALUES
                               ( @Name, @Group, @Value ) ;
               END
         ELSE
               BEGIN
                    UPDATE
                         settings.Setting
                    SET
                         Value = @Value
                    OUTPUT
                         INSERTED.Id
                    WHERE
                         settings.Setting.Id = @Id
               END
    END
GO
