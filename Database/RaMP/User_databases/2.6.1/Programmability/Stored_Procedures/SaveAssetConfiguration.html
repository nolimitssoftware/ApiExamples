CREATE PROCEDURE [core].[SaveAssetConfiguration]
   @Id UNIQUEIDENTIFIER
 , @AssetId UNIQUEIDENTIFIER
 , @Name NVARCHAR (256)
 , @PreviousVersion NVARCHAR(MAX)
 , @CurrentVersion NVARCHAR(MAX)
 , @PreviousVersionTimeStamp DATETIMEOFFSET = NULL
 , @CurrentVersionTimeStamp DATETIMEOFFSET  = NULL
 , @Source VARCHAR(32)
 , @ExtendedProperties XML = NULL
 , @AgentId UNIQUEIDENTIFIER = NULL
 , @PropertyXml XML = NULL
AS
BEGIN

   SET NOCOUNT ON;
   IF (@PreviousVersion IS NOT NULL AND @CurrentVersion IS NOT NULL)
       BEGIN
       
       IF (@Id IS NOT NULL)
           BEGIN
                   BEGIN
                       UPDATE core.AssetConfiguration
                       SET
                            PreviousVersion          = @PreviousVersion
                           ,CurrentVersion              = @CurrentVersion
                           ,PreviousVersionTimeStamp = @PreviousVersionTimeStamp
                           ,CurrentVersionTimeStamp  = @CurrentVersionTimeStamp
                           ,[Source]                      = @Source
                           ,PropertyXml              = @PropertyXml
                           ,AgentId                  = @AgentId
                           ,ExtendedProperties          = @ExtendedProperties
                       WHERE
                            Id = @Id
                   END
           END
       END
END
GO
