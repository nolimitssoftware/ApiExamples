CREATE PROCEDURE [visual].[RetrieveLayoutItem]
   @AssetId UNIQUEIDENTIFIER
AS
   BEGIN        
       SELECT
           COALESCE(A.Name, LI.[Name]) AS 'Name'
         , COALESCE(A.ShortName, LI.[Name]) AS 'ShortName'
         , COALESCE(A.OverrideName, LI.[Name]) AS 'OverrideName'
         , A.Type
         , ISNULL(core.GetStatus(A.AssetNode), A.[Status]) AS [DerivedStatus]
            ,    A.[Status]
         , CASE WHEN LI.Id IS NULL THEN CAST (0 AS BIT)
                ELSE CAST(1 AS BIT)
           END AS 'IsPlaced'
         , COALESCE(LI.Id, A.AssetId) AS 'Id'
         , A.AssetId
         , LI.LayoutId
         , LI.LayoutItemType
         , LI.PlacedId
         , LI.X
         , LI.Y
         , LI.XOffset
         , LI.YOffset
         , LI.Radius
         , LI.Width
         , LI.Height
         , LI.PhysicalWidth
         , LI.PhysicalHeight
         , LI.Color
         , LI.Direction
            , ( Select COUNT(*) from core.Asset Children Where Children.AssetNode.IsDescendantOf(A.AssetNode) = 1 and Children.Type != 'physicalGroup' ) as ChildCount
       FROM
           core.Asset AS A
       FULL JOIN visual.LayoutItem AS LI
           ON A.AssetId = LI.Id
       WHERE            
           LI.Id = @AssetId ;
   END
GO
