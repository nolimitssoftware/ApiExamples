CREATE PROCEDURE [settings].[RetrieveProtocolDesignCollection] @Protocol VARCHAR(32)
AS
   BEGIN
       SET NOCOUNT ON;
           
       SELECT
           PD.ProtocolDesignId
         , PD.DeviceTypeId
         , PD.Name
         , PD.Design
         , PD.UserDefined
         , PD.Active
       FROM
           settings.ProtocolDesign AS PD
       INNER JOIN settings.ProtocolIdentification AS PI
           ON PD.ProtocolDesignId = PI.ProtocolDesignId
       INNER JOIN settings.Protocol AS P
           ON PI.ProtocolId = P.ProtocolId
       WHERE
           P.Protocol = @Protocol;
   END
GO
