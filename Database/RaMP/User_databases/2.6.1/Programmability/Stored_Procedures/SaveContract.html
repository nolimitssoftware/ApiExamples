CREATE PROCEDURE [core].[SaveContract]
   @ContractId UNIQUEIDENTIFIER = NULL OUTPUT
 , @AssetId UNIQUEIDENTIFIER
 , @ContractType NVARCHAR(25)
 , @Enabled BIT
 , @StartDate DATETIME
 , @EndDate DATETIME
 , @TrackingNumber VARCHAR (64)
 , @HostedCompany VARCHAR (64)
 , @ContactName NVARCHAR (64)
 , @ContactPhone VARChAR (32)
 , @ContactEmail VARCHAR (64)
AS
   BEGIN
       DECLARE @ReturnId TABLE
           (
             ContractId UNIQUEIDENTIFIER
           );
       DECLARE @ContractTypeId UNIQUEIDENTIFIER;
       EXEC core.GetContractType @ContractType,
           @ContractTypeId OUTPUT;
   
       IF EXISTS ( SELECT
                       1
                   FROM
                       core.Contract AS C
                   WHERE
                       C.AssetId = @AssetId
                       AND C.ContractTypeId = @ContractTypeId )
           BEGIN
               UPDATE
                   core.Contract
               SET
                   StartDate = @StartDate
                 , EndDate = @EndDate
                 , Enabled = @Enabled
                 , TrackingNumber = @TrackingNumber
                 , HostedCompany = @HostedCompany
                 , ContactName = @ContactName
                 , ContactEmail = @ContactEmail
                 , ContactPhone = @ContactPhone
               OUTPUT
                   Inserted.ContractId
                   INTO @ReturnId
               WHERE
                   core.Contract.AssetId = @AssetId
                   AND core.Contract.ContractTypeId = @ContractTypeId;
           END
       ELSE
           BEGIN
               INSERT  INTO core.Contract
                       ( AssetId
                       , StartDate
                       , EndDate
                       , Enabled
                       , ContractTypeId
                       , TrackingNumber
                       , HostedCompany
                       , ContactName
                       , ContactPhone
                       , ContactEmail
                       )
               OUTPUT  Inserted.ContractId
                       INTO @ReturnId
               VALUES
                       ( @AssetId
                       , @StartDate
                       , @EndDate
                       , @Enabled
                       , @ContractTypeId
                       , @TrackingNumber
                       , @HostedCompany
                       , @ContactName
                       , @ContactPhone
                       , @ContactEmail
                       );
           END

       SELECT TOP 1
           RI.ContractId
         , AssetId
         , StartDate
         , EndDate
         , Enabled
         , ContractType.Name AS ContractTypeName
         , ExpiringMessageType
         , ExpiredMessageType
         , TrackingNumber
         , HostedCompany
         , ContactName
         , ContactPhone
         , ContactEmail
       FROM
           core.Contract
       INNER JOIN core.ContractType
           ON contract.ContractTypeId = contracttype.ContractTypeId
       INNER JOIN @ReturnId AS RI
           ON core.Contract.ContractId = RI.ContractId

   END
GO
