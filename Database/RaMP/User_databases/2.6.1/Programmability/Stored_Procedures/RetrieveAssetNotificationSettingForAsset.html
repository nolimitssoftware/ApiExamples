CREATE PROC notification.RetrieveAssetNotificationSettingForAsset
   @AssetId UNIQUEIDENTIFIER
 , @IsDebug BIT = NULL
AS
   BEGIN
     
       DECLARE @AssetType INT = ( SELECT
                                   core.GetAssetType(A.Type)
                                  FROM
                                   core.Asset AS A
                                  WHERE
                                   A.AssetId = @AssetId
                                );
       DECLARE @Parents TABLE
           (
             Id UNIQUEIDENTIFIER
           , [Row] INT IDENTITY(1, 1)
           );

       WITH    tree ( [Path], [PathName], AssetId, [Level], [Type] )
                 AS ( SELECT
                       A.AssetNode AS [Path]
                     , A.AssetNode.ToString() AS [PathName]
                     , A.AssetId
                     , A.AssetLevel AS [Level]
                     , A.Type
                      FROM
                       core.Asset AS A
                      WHERE
                       A.AssetId = @AssetId
                      UNION ALL
                      SELECT
                       A.AssetNode AS [Path]
                     , A.AssetNode.ToString() AS [PathName]
                     , A.AssetId
                     , CAST(( t.[Level] - 1 ) AS SMALLINT) AS [Level]
                     , A.Type
                      FROM
                       core.Asset AS A
                      INNER JOIN tree t
                       ON t.[Path].GetAncestor(1) = A.AssetNode
                      WHERE
                       A.[AssetLevel] >= 0
                    )
           INSERT  @Parents
                   ( Id                    
                   
                   )
                   SELECT
                       t.AssetId
                   FROM
                       tree AS t
                   WHERE
                       t.[Type] = 'physicalGroup'
                   ORDER BY
                       [Path];
           
       IF ( @IsDebug = 1 )
           BEGIN
               SELECT
                   *
               FROM
                   @Parents;
           END
   
       DECLARE @NotificationGroups TABLE
           (
             Id UNIQUEIDENTIFIER
           , Critical BIT
           , Warning BIT
           , Unreachable BIT
           , Administration BIT
           , Maintenance BIT
           , Change BIT
           , Type INT
           , OwnerId UNIQUEIDENTIFIER
           , OwnerType VARCHAR(32)
           , [Row] INT IDENTITY(1, 1)
           )

       INSERT  @NotificationGroups
               ( Id
               , Critical
               , Warning
               , Unreachable
               , Administration
               , Maintenance
               , Change
               , Type
               , OwnerId
               , OwnerType
               
               )
               SELECT
                   ANS.NotificationGroupId
                 , ANS.Critical
                 , ANS.Warning
                 , ANS.Unreachable
                 , ANS.Administration
                 , ANS.Maintenance
                 , ANS.Change
                 , ANS.Type
                 , ANS.AssetId
                 , A.Type
               FROM
                   notification.AssetNotificationSetting AS ANS
               INNER JOIN core.Asset AS A
                   ON ANS.AssetId = A.AssetId
               WHERE
                   ANS.AssetId = @AssetId;            
           
       DECLARE
           @Max INT = ( SELECT
                           MAX(P.Row)
                        FROM
                           @Parents AS P
                      )
         , @Current INT = 1;

       WHILE ( @Current <= @Max )
           BEGIN
               DECLARE @ParentId UNIQUEIDENTIFIER;
   
               SELECT
                   @ParentId = P.Id
               FROM
                   @Parents AS P
               WHERE
                   P.Row = @Current;
   
   
               INSERT  @NotificationGroups
                       ( Id
                       , Critical
                       , Warning
                       , Unreachable
                       , Administration
                       , Maintenance
                       , Change
                       , Type
                       , OwnerId
                       , OwnerType                  
                       
                       )
                       SELECT
                           ANS.NotificationGroupId
                         , ANS.Critical
                         , ANS.Warning
                         , ANS.Unreachable
                         , ANS.Administration
                         , ANS.Maintenance
                         , ANS.Change
                         , ANS.Type
                         , ANS.AssetId
                         , A.Type
                       FROM
                           notification.AssetNotificationSetting
                           AS ANS
                       INNER JOIN core.Asset AS A
                           ON ANS.AssetId = A.AssetId
                       WHERE
                           ANS.AssetId = @ParentId
                           AND ( ANS.Type IS NULL
                                 OR ( ANS.Type & @AssetType ) = @AssetType
                               );
                   
               SET @Current += 1;        
           END

       
       SELECT DISTINCT
           NG.Id AS [NotificationGroupId]
         , NG2.Name AS [NotificationGroupName]
         , NG2.Types AS [NotificationGroupTypes]
         , NG.Critical
         , NG.Warning
         , NG.Unreachable
         , NG.Administration
         , NG.Maintenance
         , NG.Change
         , NG.Type
         , NG.OwnerId AS 'AssetId'
         , NG.OwnerType
         , A.Name
         , A.ShortName
         , A.OverrideName
       FROM
           @NotificationGroups AS NG
       INNER JOIN settings.NotificationGroup AS NG2
           ON NG.Id = NG2.Id
       INNER JOIN core.Asset AS A
           ON A.AssetId = NG.OwnerId
       ORDER BY
           NG.Id;        
   END
GO
