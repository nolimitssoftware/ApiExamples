CREATE PROCEDURE [monitoring].[SaveSensorValue]
   @SensorId UNIQUEIDENTIFIER
 , @RollupType TINYINT
 , @Value DECIMAL(19, 2)
AS
   BEGIN        
   ;
       MERGE INTO monitoring.SensorValue AS target
           USING
               ( SELECT
                   @SensorId
                 , @RollupType
               ) AS source ( SensorId, RollupType )
           ON ( target.SensorId = source.SensorId
                AND target.RollupType = source.RollupType
              )
           WHEN MATCHED
               THEN
               UPDATE
                   SET
                       Value = @Value
                     , LastUpdate = GETUTCDATE()
           WHEN NOT MATCHED BY TARGET
               THEN            
               INSERT
                       (
                         SensorId
                       , RollupType
                       , Value
                       )
                   VALUES
                       ( @SensorId
                       , @RollupType
                       , @Value
                       );
   
   END
GO
