CREATE PROCEDURE [core].[FindAssetsByAssetTags]
   @AssetTags VARCHAR(MAX) = NULL
 , @Delim NCHAR(1) = ','
AS
   BEGIN
   
       DECLARE @Tags TABLE ( AssetTag NVARCHAR(64) )

       INSERT  @Tags
               (
                 AssetTag
               )
               SELECT
                   SD.str
               FROM
                   util.split_delimiter(@AssetTags, @Delim) AS SD ;                
       
       DECLARE @SearchResults TABLE ( Id UNIQUEIDENTIFIER ) ;
   
       INSERT  @SearchResults
               (
                 Id                        
               )
               SELECT
                   AP.AssetId
               FROM
                   core.AssetProperty AS AP
               INNER JOIN @Tags AS T
                   ON AP.SystemValue = T.AssetTag
               WHERE
                   AP.Name = 'assetTag'
               UNION
               SELECT
                   AP.AssetId
               FROM
                   core.AssetProperty AS AP
               INNER JOIN @Tags AS T
                   ON AP.UserValue = T.AssetTag
               WHERE
                   AP.Name = 'assetTag' ;                    
       
       SELECT
           SR.Id
         , A.Name
         , A.ShortName
         , A.OverrideName
         , A.Type
       FROM
           @SearchResults AS SR
       INNER JOIN core.Asset AS A
           ON SR.Id = A.AssetId ;
   END
GO
