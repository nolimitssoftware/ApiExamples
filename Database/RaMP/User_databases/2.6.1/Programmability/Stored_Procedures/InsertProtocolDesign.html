CREATE PROCEDURE [settings].[InsertProtocolDesign]
   @DeviceTypeId UNIQUEIDENTIFIER
 , @Name NVARCHAR(32)
 , @Design XML
 , @UserDefined BIT
 , @Active BIT
AS
   BEGIN
       INSERT  INTO [settings].[ProtocolDesign]
               (
                 DeviceTypeId
               , Name
               , Design
               , UserDefined
               , Active
               )
       OUTPUT  INSERTED.*
       VALUES
               (
                 @DeviceTypeId
               , @Name
               , @Design
               , @UserDefined
               , @Active
               )
   END
GO
