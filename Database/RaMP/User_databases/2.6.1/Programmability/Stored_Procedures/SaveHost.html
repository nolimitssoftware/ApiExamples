CREATE PROC [core].[SaveHost]
   @Id UNIQUEIDENTIFIER = NULL
 , @AssetId UNIQUEIDENTIFIER
 , @Address VARCHAR(57)
 , @PreviousAddress VARCHAR(57) = NULL
 , @Source VARCHAR(32) = NULL
 , @IsBmc BIT = NULL
 , @ExtendedProperties XML = NULL
 , @AgentId UNIQUEIDENTIFIER = NULL
 , @PropertyXml XML = NULL
AS
   BEGIN
       IF ( @Id IS NULL )
           BEGIN
               DECLARE @Temp TABLE ( Id UNIQUEIDENTIFIER ) ;
                                        
               INSERT  core.Host
                       (
                         AssetId
                       , [Address]
                       , [Source]
                       , IsBmc
                       , ExtendedProperties
                       , AgentId
                       , PropertyXml  
                               
                       )
               OUTPUT  INSERTED.Id
                       INTO @Temp
               VALUES
                       (
                         @AssetId
                       , @Address
                       , @Source
                       , @IsBmc
                       , @ExtendedProperties
                       , @AgentId
                       , @PropertyXml                      
                               
                       )
               SELECT TOP 1
                   @Id = T.Id
               FROM
                   @Temp AS T ;
              
           END
       ELSE
           BEGIN
               UPDATE
                   core.Host
               SET
                   Address = @Address
                 , PreviousAddress = @PreviousAddress
                 , ExtendedProperties = @ExtendedProperties
                 , PropertyXml = @PropertyXml
                 , AgentId = @AgentId
               OUTPUT
                   INSERTED.Id
               WHERE
                   core.Host.Id = @Id ;
                   
           END
 
       IF NOT EXISTS ( SELECT
                           1
                       FROM
                           monitoring.Host AS H
                       WHERE
                           H.HostId = @Id )
           BEGIN
               INSERT  monitoring.Host
                       ( HostId, Available, LastCheck )
               VALUES
                       ( @Id, 1, GETUTCDATE() ) ;
           END
       ELSE
           BEGIN
               UPDATE
                   monitoring.Host
               SET
                   Available = 1
                 , LastCheck = GETUTCDATE()
               WHERE
                   monitoring.Host.HostId = @Id ;
           END
   
       SELECT
           @Id ;
   END
GO
