CREATE PROC notification.RetrieveAssetChangeNotificationCollection @IsDebug BIT = NULL
AS
   BEGIN
       DECLARE @Parents TABLE
           (
             AssetId UNIQUEIDENTIFIER ,
             NotificationGroupId UNIQUEIDENTIFIER ,
             Change BIT ,
             [Type] INT ,
             [Row] INT IDENTITY(1, 1)
           );

       WITH    tree ( [Path], [PathName], AssetId, [Level], [Type], NotificationGroupId, Change )
                 AS ( SELECT   A.AssetNode AS [Path] ,
                               A.AssetNode.ToString() AS [PathName] ,
                               A.AssetId ,
                               A.AssetLevel AS [Level] ,
                               ANS.Type ,
                               ANS.NotificationGroupId ,
                               ANS.Change
                      FROM     core.Asset AS A
                               INNER JOIN notification.AssetNotificationSetting
                               AS ANS ON A.AssetId = ANS.AssetId
                      WHERE    A.Type = 'physicalGroup'
                               AND ANS.Change = 1
                               AND core.IsInExcludedAssetNode(A.AssetNode) = 0
                      UNION ALL
                      SELECT   A.AssetNode AS [Path] ,
                               A.AssetNode.ToString() AS [PathName] ,
                               A.AssetId ,
                               CAST(( t.[Level] - 1 ) AS SMALLINT) AS [Level] ,
                               t.[Type] ,
                               t.NotificationGroupId ,
                               t.Change
                      FROM     core.Asset AS A
                               INNER JOIN tree t ON A.AssetNode.IsDescendantOf(t.Path) = 1
                      WHERE    A.[AssetLevel] = t.[Path].GetLevel() + 1
                               AND core.IsInExcludedAssetNode(A.AssetNode) = 0
                    )
           INSERT  @Parents
                   ( AssetId ,
                     [Type] ,
                     NotificationGroupId ,
                     Change
                   )
                   SELECT DISTINCT
                           t.AssetId ,
                           t.[Type] ,
                           t.NotificationGroupId ,
                           t.Change
                   FROM    tree AS t;
               

       INSERT  @Parents
               ( AssetId ,
                 [Type] ,
                 NotificationGroupId ,
                 Change
               )
               SELECT  missing.AssetId ,
                       missing.[Type] ,
                       missing.NotificationGroupId ,
                       missing.Change
               FROM    ( SELECT    A.AssetId ,
                                   ANS.[Type] ,
                                   ANS.NotificationGroupId ,
                                   ANS.Change
                         FROM      notification.AssetNotificationSetting AS ANS
                                   INNER JOIN core.Asset AS A ON ANS.AssetId = A.AssetId
                                   WHERE core.IsInExcludedAssetNode(A.AssetNode) = 0
                       ) AS missing
                       LEFT JOIN @Parents AS P ON P.AssetId = missing.AssetId
                                                  AND P.NotificationGroupId = missing.NotificationGroupId
               WHERE   P.AssetId IS NULL
                       AND missing.Change = 1

       IF ( @IsDebug = 1 )
           BEGIN
               SELECT  *
               FROM    @Parents;
           END
   
       SELECT  CT.Id ,
               CT.AssetId ,
               CT.Name ,
               CT.NotificationGroupTypes ,
               CT.Change ,
               CT.[Type]
       FROM    ( SELECT    NG.Id ,
                           CN.AssetId ,
                           NG.Name ,
                           NG.Types AS 'NotificationGroupTypes' ,
                           CAST(0 AS BIT) AS [Change] ,
                           NULL AS [Type]
                 FROM      notification.ChangeNotification AS CN
                           INNER JOIN notification.Notification AS N ON CN.NotificationId = N.Id
                           INNER JOIN settings.NotificationGroup AS NG ON N.NotificationGroupId = NG.Id
                 WHERE     CN.AssetId NOT IN ( SELECT  NG2.AssetId
                                               FROM    @Parents AS NG2
                                               WHERE   NG2.Change = 1 )
                 UNION ALL
                 SELECT    NG.NotificationGroupId ,
                           NG.AssetId ,
                           NG2.Name ,
                           NG2.Types AS 'NotificationGroupTypes' ,
                           CASE WHEN SUM(CONVERT(INT, NG.Change)) > 0
                                THEN CAST(1 AS BIT)
                                ELSE CAST(0 AS BIT)
                           END AS [Change] ,
                           NG.[Type]
                 FROM      @Parents AS NG
                           INNER JOIN settings.NotificationGroup AS NG2 ON NG.NotificationGroupId = NG2.Id
                 GROUP BY  NG.NotificationGroupId ,
                           NG.AssetId ,
                           NG2.Name ,
                           NG2.Types ,
                           NG.[Type]
               ) AS CT
       ORDER BY CT.Change
   END
GO
