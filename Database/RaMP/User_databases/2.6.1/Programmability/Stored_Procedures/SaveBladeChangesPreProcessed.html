CREATE PROCEDURE [core].[SaveBladeChangesPreProcessed]
   @ChassisId UNIQUEIDENTIFIER
 , @BladeId UNIQUEIDENTIFIER
 , @BayLocation TINYINT = NULL
AS
   BEGIN
        
        --1. Update the Blade table with where the blade is.
       IF EXISTS ( SELECT
                       1
                   FROM
                       core.Blade
                   WHERE
                       AssetId = @ChassisId
                       AND BayLocation = @BayLocation )
           BEGIN

               UPDATE
                   core.Blade
               SET
                   BladeAssetId = @BladeId
               WHERE
                   AssetId = @ChassisId
                   AND BayLocation = @BayLocation;
                                                               
               --Possible that on a previous run we were only able to match by serial number but now we have a better match BUT we have an orphaned row that needs cleaning up.
               DELETE FROM
                   core.Blade
               WHERE
                   BladeAssetId = @BladeId
                   AND BayLocation IS NULL
           END
       ELSE
           BEGIN
               --Should only get here if would could only match by Serial number (ie. the bayLocation will be null here).
               INSERT  core.Blade
                       ( AssetId
                       , BladeAssetId
                       , Source
                       , BayLocation
                       )
               VALUES
                       ( @ChassisId
                       , @BladeId
                       , 'api'
                       , @BayLocation
                       );

           END

               
       --2. A blade can only be in one place at a time.  Clean up old places if needed
       UPDATE
           core.Blade
       SET
           BladeAssetId = NULL
       WHERE
           BladeAssetId = @BladeId
           AND ( BayLocation <> @BayLocation
                 OR AssetId <> @ChassisId
               );
             
       --3. Move the blade under the chassis
       EXEC core.MoveAsset @AssetId = @BladeId,
           @NewParentId = @ChassisId;

       --4. Set the chassis as the blades power source.
       IF NOT EXISTS ( SELECT
                           1
                       FROM
                           core.UpStreamPower
                       WHERE
                           AssetId = @BladeId
                           AND UpstreamAssetId = @ChassisId )
           BEGIN
               --For cleanness sake and that a blade cannot have more than one source
               DELETE FROM
                   [core].[UpstreamPower]
               WHERE
                   AssetId = @BladeId

               INSERT  INTO [core].[UpstreamPower]
                       ( [AssetId]
                       , [Source]
                       , [UpstreamAssetId]
                       )
               VALUES
                       ( @BladeId
                       , 'api'
                       , @ChassisId
                       )                            
           END
        
        
   END
GO
Uses
  • [core].[Blade]
  • [core].[UpstreamPower]
  • [core].[MoveAsset]
  • [core]
