CREATE PROCEDURE [core].[SaveLastModified]
   @Id uniqueidentifier,
   @LastModified DateTime
AS
   if Exists(Select 1 From core.LastModified Where Id = @Id)
   begin
       update core.LastModified
       set LastModifiedDate = @LastModified
       where Id = @Id
   end
   else
   begin
       insert into core.LastModified (Id, LastModifiedDate)
       values (@Id, @LastModified)
   end
GO
