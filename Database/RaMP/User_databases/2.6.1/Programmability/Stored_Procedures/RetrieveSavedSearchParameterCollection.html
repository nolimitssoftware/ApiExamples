CREATE PROCEDURE [search].[RetrieveSavedSearchParameterCollection]
   @SavedId UNIQUEIDENTIFIER
AS
   BEGIN
       SELECT
           SP.Id
         , SP.SavedSearchId
         , SP.SearchablePropertyId
         , SP.Condition
         , SP.Value
         , SQL_VARIANT_PROPERTY(SP.Value, 'baseType') AS [ValueType]
         , SP2.Name
         , COALESCE(SP2.PropertyName, SP2.Name) AS [PropertyName]
         , SP2.Type
         , SP2.[Values]
         , SP2.BaseUnit
         , SP2.AssetTypes
         , SP2.ExcludedAssetTypes
         , SP2.DisplayUnit
       FROM
           search.SearchParameter AS SP
       INNER JOIN search.SearchableProperty AS SP2
           ON SP.SearchablePropertyId = SP2.Id
       WHERE
           SP.SavedSearchId = @SavedId
       UNION
       SELECT
           SP.Id
         , SP.SavedSearchId
         , SP.SearchablePropertyId
         , SP.Condition
         , SP.Value
         , SQL_VARIANT_PROPERTY(SP.Value, 'baseType') AS [ValueType]
         , CP.Name
         , CP.[Name] AS [PropertyName]
         , CP.Type
         , CP.[Values]
         , CP.Unit as BaseUnit
         , CP.AssetTypes
         , 98304 --excluded physical and logical (no property support)
         , CP.Unit as DisplayUnit
       FROM
           search.SearchParameter AS SP
       INNER JOIN settings.CustomProperty AS CP
           ON SP.SearchablePropertyId = CP.Id            
       WHERE
           SP.SavedSearchId = @SavedId
       ORDER BY
           SP.Id;
   END
GO
