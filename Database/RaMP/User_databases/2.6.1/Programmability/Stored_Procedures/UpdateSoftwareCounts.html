[core].[UpdateSoftwareCounts]

PropertyValue

CREATE PROCEDURE [core].[UpdateSoftwareCounts]
AS
   BEGIN
       -- This will update the counts (for installations with the count greater than 0)
       MERGE settings.Software AS T
       USING
           ( SELECT    AP.[Name] ,
                       COUNT(AP.[Name]) AS 'Count'
             FROM      core.[Application] AS AP
                       JOIN core.[Asset] AS A ON AP.AssetId = A.AssetId
             WHERE     core.IsInExcludedAssetNode(A.AssetNode) = 0
             GROUP BY  AP.[Name]
           ) AS S
       ON ( T.[Name] = S.[Name] )
       WHEN MATCHED THEN
           UPDATE SET
                   T.NumberOfInstallations = S.[Count]
       WHEN NOT MATCHED BY TARGET THEN
           INSERT ( [Name] ,
                    NumberOfInstallations
                  )
           VALUES ( S.[Name], S.[Count] )
       WHEN NOT MATCHED BY SOURCE THEN
           UPDATE SET
                   T.NumberOfInstallations = 0;

   --It was decided to remove any software from the system that has an installation count of zero AND has no set licenses or un-approval.
       DELETE  [settings].[Software]
       WHERE   [NumberOfInstallations] = 0
               AND Approved = 1
               AND NumberOfLicenses IS NULL
   END
GO
Uses
  • [core].[Application]
  • [core].[Asset]
  • [settings].[Software]
  • [core].[IsInExcludedAssetNode]
  • [core]
