CREATE PROC [core].[BuildDiscoveredSensor]
    @DeviceId UNIQUEIDENTIFIER
 , @DiscoveryAgentId UNIQUEIDENTIFIER
 , @Source VARCHAR(32)
 , @Type VARCHAR(128)
 , @SystemName VARCHAR(128)
 , @UserName VARCHAR(128) = NULL
 , @Unit VARCHAR(16)
 , @Element VARCHAR(32) = NULL
 , @OriginType VARCHAR(32)
 , @SensorXml XML = NULL
 , @ExtendedProperties XML = NULL
AS
    BEGIN
    
         DECLARE @SensorId UNIQUEIDENTIFIER
         SELECT
               @SensorId = SensorId
         FROM
               [core].[Sensor]
         WHERE
               DeviceId = @DeviceId
               AND [Type] = @Type
               AND (@Element IS NULL OR [Element] = @Element OR (@Element = '1' AND [Element] IS NULL))

         --(@Element = 1 AND [Element] IS NULL) this cleans up a definition if the device was originally single phase but a definition change made it 3 phase

         IF ( @SensorId IS NOT NULL )
               BEGIN
                    UPDATE
                         [core].[Sensor]
                    SET
                       [DiscoveryAgentId] = @DiscoveryAgentId
                       , [Source] = @Source
                       , [Type] = @Type
                       , [SystemName] = @SystemName
                       , [Element] = @Element
                       , [Unit] = @Unit
                       , [OriginType] = @OriginType
                       , [ExtendedProperties] = @ExtendedProperties
                    WHERE
                       SensorId = @SensorId
               END
         ELSE
               BEGIN
                   --Determine default position.
                   DECLARE @Position INT = NULL
                   SELECT TOP 1 @Position = 1 FROM core.Asset WHERE @Type IN ('humidity', 'ambientTemperature') AND AssetId = @DeviceId AND [Type] IN ('server', 'bladeServer')

                   --First time this sensor has been found during a discovery
                   -- We will need to generate a new Id and make sure it is placed into the sensor xml for future use.
       
                    DECLARE @Temp TABLE ( Id UNIQUEIDENTIFIER ) ;
                                       
                    INSERT  INTO [core].[Sensor]
                               (
                                 [DeviceId]
                               , [DiscoveryAgentId]
                               , [Source]
                               , [Type]
                               , [SystemName]
                               , [UserName]
                               , [Unit]
                               , [Element]
                               , [OriginType]
                               , [LocationAssetId]
                               , [Position]
                               , [SensorXml]
                               , [ExtendedProperties]
                               )
                    OUTPUT  INSERTED.SensorId
                               INTO @Temp
                    VALUES
                               (
                                 @DeviceId
                               , @DiscoveryAgentId
                               , @Source
                               , @Type
                               , @SystemName
                               , @UserName
                               , @Unit
                               , @Element
                               , @OriginType
                               , @DeviceId        --Defaults to the asset's id
                               , @Position
                               , @SensorXml
                               , @ExtendedProperties
                               )
       
                    DECLARE @NewId UNIQUEIDENTIFIER = (
                                                                    SELECT TOP 1
                                                                         T.Id
                                                                    FROM
                                                                         @Temp AS T
                                                                 ) ;

                   SET @SensorId = @NewId ;    
               END
           
           IF ( @SensorXml IS NOT NULL )
               BEGIN
                   -- This needs to run for both updates and inserts.  It stuffs the sensor ID into the XML and in the case of the update will update the sensor XML with any design definition changes that we may made.
                   UPDATE
                           core.Sensor
                   SET
                           SensorXml = [core].[InsertSensorIdIntoSensorXml](@SensorId,    @SensorXml)
                   WHERE
                           SensorId = @SensorId ;
               END

           --Return sensor ID
           SELECT
               @SensorId
    END
GO
