CREATE PROCEDURE [search].[SaveSearch]
   @Id UNIQUEIDENTIFIER = NULL
 , @UserId UNIQUEIDENTIFIER
 , @Name NVARCHAR(64)
 , @AssetType INT
 , @AssetId UNIQUEIDENTIFIER = NULL
 , @IsGlobal BIT = 0
AS
   BEGIN
       DECLARE @Ids TABLE ( Id UNIQUEIDENTIFIER );
       IF ( @Id IS NULL )
           BEGIN
               INSERT  search.SavedSearch
                       ( UserId
                       , Name
                       , AssetType
                       , AssetId
                       , IsGlobal
                       )
               OUTPUT  INSERTED.Id
                       INTO @Ids
               VALUES
                       ( @UserId
                       , @Name
                       , @AssetType
                       , @AssetId
                       , @IsGlobal
                         );
           END    
       ELSE
           BEGIN
               UPDATE
                   search.SavedSearch
               SET
                   [Name] = @Name
                 , AssetType = @AssetType
                 , AssetId = @AssetId
                 , IsGlobal = @IsGlobal
               OUTPUT
                   INSERTED.Id
                   INTO @Ids
               WHERE
                   search.SavedSearch.Id = @Id;
           END

       SELECT
           SS.Id
         , SS.UserId
         , SS.Name
         , SS.AssetType
         , SS.AssetId
         , SS.IsGlobal
         , COALESCE(A.OverrideName, A.ShortName, A.Name) AS [Location]
         , A.AssetNode.ToString() AS [Path]
       FROM
           search.SavedSearch AS SS
       INNER JOIN @Ids AS I
           ON SS.Id = I.Id
       LEFT JOIN core.Asset AS A
           ON A.AssetId = SS.AssetId;
   END
GO
