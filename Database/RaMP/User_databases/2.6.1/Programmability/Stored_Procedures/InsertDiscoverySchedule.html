CREATE PROCEDURE [settings].[InsertDiscoverySchedule]
   @DiscoveryId UNIQUEIDENTIFIER
 , @ScheduleType TINYINT
 , @Description NVARCHAR(32)
 , @CronExpression VARCHAR(32)
AS
   BEGIN
 
       DECLARE    @DiscoveryScheduleId UNIQUEIDENTIFIER;
 
       --Currently only allow 1 truly scheduled discovery and run-nows.
       SELECT
           @DiscoveryScheduleId = DS.DiscoveryScheduleId
       FROM
           settings.DiscoverySchedule AS DS
       WHERE
           DS.DiscoveryId = @DiscoveryId
           AND
           @ScheduleType <> 8 AND @ScheduleType <> 16
   
   
       IF ( @DiscoveryScheduleId IS NULL )
           BEGIN
   
               INSERT    settings.DiscoverySchedule
                       (
                         DiscoveryId
                       , ScheduleType
                       , [Description]
                       , CronExpression
               
                       
                       )
               OUTPUT    INSERTED.*
                       SELECT
                           @DiscoveryId
                         , @ScheduleType
                         , @Description
                         , @CronExpression;
   
       
        
           END
       ELSE
           BEGIN
               UPDATE
                   settings.DiscoverySchedule
               SET    
                   ScheduleType = @ScheduleType
                 , [Description] = @Description
                 , CronExpression = @CronExpression
                 , LastRun = GETUTCDATE()
               OUTPUT
                   INSERTED.*
               WHERE
                   settings.DiscoverySchedule.DiscoveryScheduleId = @DiscoveryScheduleId
           END
   END
GO
