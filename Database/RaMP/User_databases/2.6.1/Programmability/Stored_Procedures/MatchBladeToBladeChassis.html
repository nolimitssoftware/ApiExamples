CREATE PROCEDURE [core].[MatchBladeToBladeChassis]
    @AssetId UNIQUEIDENTIFIER = NULL
AS
BEGIN
   DECLARE @SerialNumber NVARCHAR(256) = NULL
   DECLARE @ChassisId UNIQUEIDENTIFIER = NULL
   DECLARE @ReturnValue INT

   IF ( @AssetId IS NOT NULL ) BEGIN
       
       /* get the assets serial number */
       SELECT
           @SerialNumber = SerialNumber
       FROM
           [core].[SerialNumber]
       WHERE    
           [AssetId] = @AssetId
           
       /* Get Chassis assetid from the blade chassis with the same serial number */
       SELECT     
           @ChassisId = core.Asset.AssetId
       FROM         
           core.Asset
       INNER JOIN
           core.SerialNumber ON core.Asset.AssetId = core.SerialNumber.AssetId
       WHERE
           core.SerialNumber.SerialNumber = @SerialNumber AND
           core.Asset.Type = 'bladeChassis'

       /* if we found the chassis parent go ahead and move the asset */
       IF ( @ChassisId IS NOT NULL )BEGIN
           EXEC @ReturnValue = core.MoveAsset
           @AssetId = @AssetId,
           @NewParentId = @ChassisId ;
       END
       RETURN @ReturnValue
   END
   ELSE BEGIN
       RETURN 1    
   END
END
GO
