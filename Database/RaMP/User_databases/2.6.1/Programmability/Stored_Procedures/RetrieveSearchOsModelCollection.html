CREATE PROCEDURE settings.RetrieveSearchOsModelCollection
   @Search VARCHAR(128) = NULL
 , @MinimumInstallationCount INT = NULL
 , @SortField VARCHAR(512)
 , @IsApproved BIT = 0
 , @Start INT
 , @Limit INT
 , @IsAscending BIT = 1
 , @IsDebug BIT = 0
 , @TotalCount INT OUTPUT
AS
   BEGIN
   
       SET NOCOUNT ON ;
   
       DECLARE
           @sql NVARCHAR(MAX)
         , @params NVARCHAR(4000)
         , @sort VARCHAR(1024) ;     
       IF (
            @SortField IS NULL
            OR @SortField = 'Name'
          )
           BEGIN
               SET @SortField = 'ServerOperatingSystem.Name' ;
           END
   
       SET @sort = ', ROW_NUMBER() OVER ( ORDER BY [' + @SortField + ']' ;
   --      SET @sort = ', ROW_NUMBER() OVER ( ORDER BY [IsApproved] ASC, [' + @SortField + ']' ;
   
        IF ( @IsAscending = 1 )
           BEGIN
               SET @sort += ' ASC ' ;
           END
       ELSE
           BEGIN
               SET @sort += ' DESC ' ;
           END
       
       SET @sort += ' ) AS RowNum ' ;
       
       SET @sql ='
       
       SET NOCOUNT ON;
       
       DECLARE @End INT ;
       
       SELECT
           @TotalCount = COUNT(1)
       FROM
           core.ServerOperatingSystem AS SOS
       WHERE 1 = 1 '
       
       IF ( @Search IS NOT NULL )
           BEGIN
               SET @sql += 'AND SOS.Name LIKE @Search ';
           END     
       
       IF(@MinimumInstallationCount IS NOT NULL)
           BEGIN
               SET @sql += ' AND core.CountOSInstallations(SOS.Id, NULL, 0) >= @MinimumInstallationCount ';
           END                 
        
         IF(@IsApproved = 0)
           BEGIN
               SET @sql += ' AND SOS.IsApproved = 0 ';
               SET @sql += ' OR (SOS.LicenseCount > 0 AND core.CountOSInstallations(SOS.Id, NULL, 0) > SOS.LicenseCount) ';
           END
                        
       SET @sql += '
   
       IF ( @Limit IS NULL )
           BEGIN
               SET @Limit = @TotalCount ;
           END
       
       SET @End = @Start + @Limit - 1 ;
       '
       
       SET @sql += '
   SELECT
       *
   FROM
       (
       SELECT temp.* '
       SET @sql += @sort
       
       SET @sql += '
       FROM(
         SELECT
           M.Id AS ''ManufacturerId''
         , M.Name AS ''Manufacturer''
         , SUBSTRING(temp.Names, 0, LEN(temp.Names)) AS ''ManufacturerAlternateNames''
         , SOS.Id AS ''ServerOperatingSystem.Id''
         , NULL AS ''ServicePack.Id''
         , SOS.Name AS ''ServerOperatingSystem.Name''
         , NULL AS ''ServicePack.Name''
         , SOS.VersionString
         , SOS.SearchString
         , SOS.Major
         , SOS.Minor
         , SOS.Revision
         , SOS.Build
         , SOS.Other
         , SOS.Notes
         , SOS.ExtendedProperties
         , SOS.IsSystemDefined
         , SOS.SentToNLS
         , SOS.InternalId
         , SOS.LicenseCount
         , SOS.IsApproved          
         , SPI.HasUnapprovedServicePack
         , SPI.ServicePackCount
         , core.CountOSInstallations(SOS.Id, NULL, 0) AS [InstallationCount]                
         FROM
           core.Manufacturer AS M
         INNER JOIN core.ServerOperatingSystem AS SOS
           ON M.Id = SOS.ManufacturerId
         CROSS APPLY ( SELECT
               *
             FROM
               ( SELECT     
                    COUNT(SP1.ServerOperatingSystemId) AS [ServicePackCount]
                  , CAST(CASE WHEN ISNULL(SUM(CONVERT(INT, SP1.IsApproved)),
                                         0) = COUNT(SP1.ServerOperatingSystemId)
                             THEN 0
                             ELSE 1
                        END AS BIT) AS [HasUnapprovedServicePack]
                 FROM
                   ( SELECT
                       SP.Id
                     , SP.ServerOperatingSystemId
                     , SP.IsApproved
                     FROM
                       core.ServicePack AS SP
                     WHERE
                       SP.ServerOperatingSystemId = SOS.Id
                       AND
                           core.CountOSInstallations(SOS.Id, SP.Id, 1) > 0
                   ) AS SP1
               ) AS SP2
           ) AS SPI      
         CROSS APPLY (
                       SELECT
                           CONVERT(VARCHAR(256), ISNULL(MAN.[Name], ''''))
                           + '','' AS [text()]
                       FROM
                           core.ManufacturerAlternateName AS MAN
                       WHERE
                           MAN.ManufacturerId = M.Id
                     FOR
                       XML PATH('''')
                     ) AS temp ( [Names] )
                     
       WHERE 1=1 '
       IF ( @Search IS NOT NULL )
           BEGIN
               SET @sql += 'AND SOS.Name LIKE @Search ';
           END     
       
       IF(@MinimumInstallationCount IS NOT NULL)
           BEGIN
               SET @sql += ' AND core.CountOSInstallations(SOS.Id, NULL, 0) >= @MinimumInstallationCount ';                
           END                 
        
       IF(@IsApproved = 0)
           BEGIN
               SET @sql += ' AND SOS.IsApproved = 0 OR SPI.HasUnapprovedServicePack = 1';
               SET @sql += ' OR (SOS.LicenseCount > 0 AND core.CountOSInstallations(SOS.Id, NULL, 0) > SOS.LicenseCount) ';
           END
                                
       SET @sql += '
       ) AS temp ) AS OS
   WHERE
       OS.RowNum BETWEEN @Start AND @End
   ORDER BY
       OS.RowNum ;
   '
       
       SET @params = '
     @Search VARCHAR(128)
   , @MinimumInstallationCount INT      
   , @Start INT
   , @Limit INT
   , @TotalCount INT OUTPUT
'        

       IF ( @IsDebug = 1 )
           BEGIN
               PRINT @params ;
               PRINT @sql ;
           END

       EXEC sp_executesql
           @sql
         , @params
         , @Search
         , @MinimumInstallationCount
         , @Start
         , @Limit
         , @TotalCount = @TotalCount OUTPUT

       IF ( @IsDebug = 1 )
       BEGIN        
           PRINT 'Total Count: ' + CONVERT(VARCHAR(128), @TotalCount);        
       END
   END
GO
