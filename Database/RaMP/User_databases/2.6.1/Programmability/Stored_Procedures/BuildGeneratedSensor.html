CREATE PROC [core].[BuildGeneratedSensor]
    @DeviceId UNIQUEIDENTIFIER
 , @DiscoveryAgentId UNIQUEIDENTIFIER
 , @Source VARCHAR(32)
 , @Type VARCHAR(128)
 , @OriginType VARCHAR(32)
 , @Unit VARCHAR(16)
 , @SystemName VARCHAR(128)
 , @Element VARCHAR(64) = NULL
 , @UserName VARCHAR(128) = NULL
AS
    BEGIN

         DECLARE @SensorId UNIQUEIDENTIFIER
         SELECT TOP 1
               @SensorId = SensorId
         FROM
               [core].[Sensor]
         WHERE
               DeviceId = @DeviceId
               AND [Type] = @Type
               AND (@Element IS NULL OR [Element] = @Element OR (@Element = 1 AND [Element] IS NULL))
         ORDER BY SensorId ;
         --(@Element = 1 AND [Element] IS NULL) this cleans up a definition if the device was originally single phase but a definition change made it 3 phase

         IF ( @SensorId IS NOT NULL )
               BEGIN
                   --For generated sensors (sensors created by the analyzer for example) NEVER overwrite the agent Id!
                   --    If we did then the sensors that go back and forth from monitoring and generated will never go back to monitoring
                   --    if it doesn't have the monitoring agent ID.
               
                    UPDATE
                         [core].[Sensor]
                    SET
                         [OriginType] = @OriginType
                       , Element = @Element
                    OUTPUT  
                       INSERTED.SensorId
                    WHERE
                         SensorId = @SensorId
               END
         ELSE
               BEGIN
                   --First time this sensor has been written to.

                    INSERT  INTO [core].[Sensor]
                               (
                                 [DeviceId]
                               , [OriginType]
                               , [DiscoveryAgentId]
                               , [Source]
                               , [Type]
                               , [Element]
                               , [SystemName]
                               , [UserName]
                               , [Unit]
                               , [LocationAssetId]
                               )    
                    OUTPUT  
                       INSERTED.SensorId
                    VALUES
                               (
                                 @DeviceId
                               , @OriginType
                               , @DiscoveryAgentId
                               , @Source
                               , @Type
                               , @Element
                               , @SystemName
                               , @UserName
                               , @Unit
                               , @DeviceId        --Defaults to the asset's id
                               )
                               
               END
    END
GO
