CREATE PROC [log].[StartMessage]
   @AssetId UNIQUEIDENTIFIER
 , @Text nvarchar(512)
 , @Severity int
 , @MessageMasterId UNIQUEIDENTIFIER = NULL
 , @ExtendedProperties XML = NULL
 , @Context INT = NULL
 , @ContextId UNIQUEIDENTIFIER = NULL
 , @TrapId nvarchar(128) = NULL
AS
   BEGIN
    
       IF @ContextId IS NULL
           BEGIN
               SET @ContextId = @AssetId
           END
       -- If there is an ACTIVE message of that type for that asset then just update...else insert.
       IF EXISTS ( SELECT
                       1
                   FROM
                       [log].[Message]
                   WHERE
                       AssetId = @AssetId
                       AND MessageMasterId = @MessageMasterId
                       AND ContextId = @ContextId
                       AND (@TrapId is null OR TrapId = @TrapId)
                       AND EndTimestamp IS NULL )
           BEGIN
               UPDATE
                   [log].[Message]
               SET
                   ExtendedProperties = @ExtendedProperties,
                   Text = @Text,
                   Severity = @Severity
               OUTPUT
                   INSERTED.Id
               WHERE
                   AssetId = @AssetId
                   AND MessageMasterId = @MessageMasterId
                   AND ContextId = @ContextId
                   AND EndTimestamp is null
           END
       ELSE
           BEGIN    
               INSERT  INTO [log].[Message]
                       (
                         [MessageMasterId]
                       , [AssetId]
                       , [ExtendedProperties]
                       , Context
                       , ContextId
                       , Text
                       , Severity
                       , TrapId
                       )
               OUTPUT  INSERTED.Id
               VALUES
                       (
                         @MessageMasterId
                       , @AssetId
                       , @ExtendedProperties
                       , @Context
                       , @ContextId
                       , @Text
                       , @Severity
                       , @TrapId
                       )

               -- Update the asset's status
               EXEC [core].[UpdateAssetStatusFromActiveMessages]
                   @AssetId = @AssetId
                 , @CurrentMessageSeverity = @Severity ;
           END        
   END
GO
Uses
