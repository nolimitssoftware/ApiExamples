CREATE PROCEDURE [core].[RetrieveUpDownCollection]
AS
SELECT A.Name as AssetName, A.AssetId as AssetId, H.Address, H.Id as HostId, H2.Available, H.IsBmc FROM core.Host AS H
INNER JOIN monitoring.Host AS H2
ON H.Id = H2.HostId
INNER JOIN core.Asset AS A
ON A.AssetId = H.AssetId order by a.assetid
RETURN 0
GO
Uses

  • [core].[Host]
