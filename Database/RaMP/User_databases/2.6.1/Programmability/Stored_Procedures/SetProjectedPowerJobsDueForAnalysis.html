CREATE PROCEDURE [job].[SetProjectedPowerJobsDueForAnalysis]
   @AssetId UNIQUEIDENTIFIER
AS
   BEGIN

       --This will set every projected power job due that is UP in the given asset's power chain.

       WITH UpstreamPower_CTE (Id, AssetId, UpstreamAssetId)
       AS
           (
               SELECT Id, AssetId, UpstreamAssetId
               FROM core.UpstreamPower AS UP
               WHERE AssetId = @AssetId
               UNION ALL
               SELECT UP2.Id, UP2.AssetId, UP2.UpstreamAssetId
               FROM core.UpstreamPower AS UP2
               INNER JOIN UpstreamPower_CTE AS UPCTE ON UPCTE.UpstreamAssetId = UP2.AssetId
           )
               UPDATE
                   [job].[Processing]
               SET
                   [LastProcessed] = NULL,
                   LastJobStatus = 0
               WHERE
                   JobTypeId = 2
                   AND
                   ContextId IN
                       (
                           SELECT DISTINCT
                               UPCTE.UpstreamAssetId
                           FROM UpstreamPower_CTE AS UPCTE
                           INNER JOIN core.Asset AS A ON A.AssetId = UPCTE.UpstreamAssetId
                           UNION
                               SELECT @AssetId
                       )
   END
GO
