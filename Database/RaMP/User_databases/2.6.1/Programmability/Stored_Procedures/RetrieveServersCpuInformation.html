CREATE PROCEDURE [core].[RetrieveServersCpuInformation]
AS
BEGIN
   SELECT
       A.AssetId
     , COALESCE(A.OverrideName, A.ShortName, A.Name) AS 'DisplayName'
     , ( SELECT
           Value
         FROM
           [monitoring].SensorValue AS SV
         WHERE
           SV.SensorId = S.SensorId
           AND SV.RollupType = 8
       ) AS 'DailyAvg'
     , ( SELECT
           Value
         FROM
           [monitoring].SensorValue AS SV
         WHERE
           SV.SensorId = S.SensorId
           AND SV.RollupType = 16
       ) AS 'DailyMax'
     , ( SELECT
           1
         FROM
           [log].Message AS M
         WHERE
           M.AssetId = A.AssetId
           AND M.EndTimestamp IS NULL
           AND M.MessageMasterId IN ( '7f8b58a1-efa7-e011-8739-001c421c284c',
                                      '8e8b58a1-efa7-e011-8739-001c421c284c' )
       ) AS 'IsLowCpuMessageActive'
     , ( SELECT
           1
         FROM
           [log].Message AS M
         WHERE
           M.AssetId = A.AssetId
           AND M.EndTimestamp IS NULL
           AND M.MessageMasterId = '7e8b58a1-efa7-e011-8739-001c421c284c'
       ) AS 'IsVeryLowCpuMessageActive'
       , A.Type
   FROM
       [core].Asset AS A
   INNER JOIN [core].Sensor AS S
       ON A.AssetId = S.DeviceId
          AND S.Type = 'cpuLoad'
   WHERE
       A.Type IN ( 'server', 'virtual', 'bladeServer' )
       AND core.IsInExcludedAssetNode(A.AssetNode) = 0
END
GO
Uses
  • [core].[Asset]
  • [core].[Sensor]

  • [core].[IsInExcludedAssetNode]
  • [core]
