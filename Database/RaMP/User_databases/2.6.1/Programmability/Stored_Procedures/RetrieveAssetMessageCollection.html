CREATE PROCEDURE [log].[RetrieveAssetMessageCollection]
   @AssetId UNIQUEIDENTIFIER,
   @RetrieveOnlyActive BIT = 1,
   @Start SMALLINT = 1,
    @Limit INT = NULL,
   @TotalCount INT OUTPUT
AS
   DECLARE @End INT
    
   IF (@RetrieveOnlyActive = 1)
       BEGIN
           -- For just the active messages.
           
           SELECT
               @TotalCount = COUNT(1)
           FROM
               log.Message
           WHERE
               AssetId = @AssetId
               AND
               EndTimestamp IS NULL
   
           IF ( @Limit IS NULL )
           BEGIN
               SET @Limit = @TotalCount ;
           END
           
           SET @End = @Start + @Limit - 1 ;
           
           SELECT
                Id
               ,MessageMasterId
               ,Text
               ,Severity
               ,AssetId
               ,StartTimestamp
               ,EndTimestamp
               ,ClearedByUserName
               ,ExtendedProperties
               ,Ignore
           FROM (
                   SELECT
                       M.Id
                       , M.MessageMasterId
                       , M.Text
                       , M.Severity
                       , M.AssetId
                       , M.StartTimestamp
                       , M.EndTimestamp
                       , M.ClearedByUserName
                       , M.ExtendedProperties
                       , MessageMaster.Ignore
                       ,ROW_NUMBER () OVER (ORDER BY M.Severity ASC, M.StartTimestamp DESC) AS RowNum
                   FROM
                       [log].Message AS M
                       left join log.MessageMaster on M.MessageMasterId = MessageMaster.Id
                   WHERE
                       AssetId = @AssetId
                       AND
                       EndTimestamp IS NULL
                )  AS MyDerivedTable
           WHERE
               MyDerivedTable.RowNum BETWEEN @Start AND @End
           ORDER BY
               MyDerivedTable.RowNum
       END
   ELSE
       BEGIN
           --For all messages
           
           SELECT
               @TotalCount = COUNT(1)
           FROM
               log.Message
           WHERE
               AssetId = @AssetId
   
           IF ( @Limit IS NULL )
           BEGIN
               SET @Limit = @TotalCount ;
           END
           
           SET @End = @Start + @Limit - 1 ;
           
           SELECT
                Id
               ,MessageMasterId
               ,Text
               ,Severity
               ,AssetId
               ,StartTimestamp
               ,EndTimestamp
               ,ClearedByUserName
               ,ExtendedProperties
               ,Ignore
           FROM (
                   SELECT
                       M.Id
                       , M.MessageMasterId
                       , M.Text
                       , M.Severity
                       , M.AssetId
                       , M.StartTimestamp
                       , M.EndTimestamp
                       , M.ClearedByUserName
                       , M.ExtendedProperties
                       , MessageMaster.Ignore
                       ,ROW_NUMBER () OVER (ORDER BY M.Severity ASC, M.StartTimestamp DESC) AS RowNum
                   FROM
                       [log].Message AS M
                       left join log.MessageMaster on M.MessageMasterId = MessageMaster.Id
                   WHERE
                       AssetId = @AssetId
                )  AS MyDerivedTable
           WHERE
               MyDerivedTable.RowNum BETWEEN @Start AND @End
           ORDER BY
               MyDerivedTable.RowNum
       END
GO
