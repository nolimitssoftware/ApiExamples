CREATE PROCEDURE [core].MatchNicIpAddresses
   @AssetId UNIQUEIDENTIFIER
AS
BEGIN

   UPDATE core.IpAddress
   SET NetworkInterfaceId = NetworkInterface.Id
   FROM [core].IpAddress
   JOIN  core.NetworkInterface ON  
       (NetworkInterface.Mac = IpAddress.MacAddress or (
           NetworkInterface.AssetId = IpAddress.AssetId and NetworkInterface.IpAddressIndex = IpAddress.[Index] and NetworkInterface.Source = IpAddress.Source
       )) and NetworkInterface.IpEnabled = 1
   WHERE IpAddress.AssetId = @AssetId
   
END
GO
