CREATE PROC [core].[RetrieveAssetDirectNetworkConnectionCollection]
    @ParentId UNIQUEIDENTIFIER
   ,@IsUpDirection BIT
AS
    BEGIN
         SET NOCOUNT ON ;
         SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED ;
           
         IF ( @IsUpDirection = 0 )
               BEGIN
                   
                    SELECT
                   DISTINCT
                         GA.AssetId
                       , GA.Name
                       , GA.ShortName
                       , GA.OverrideName
                       , GA.Type
                       , GA.[Status]
                       , GA.[LifecycleStatus]
                       , GA.TypeAtDiscovery
                       , GA.DerivedStatus
                       , GA.AssetType
                       , GA.AssetNode
                       , GA.AssetLevel
                       , GA.Notes
                       , NULL AS 'AssetXml'
                       , GA.ModelId
                       , GA.ChildCount
                       , GA.ParentId
                       , CONVERT(VARCHAR(MAX), GA.ExtendedProperties) AS 'ExtendedProperties'
                       , GA.RackLocation
                       , GA.RaMPPort
                       , GA.[Location.Source]
                       , GA.[Parent.AssetId]
                       , GA.[Parent.Name]
                       , GA.[Parent.ShortName]
                       , GA.[Parent.OverrideName]
                       , GA.[Parent.Type]
                       , GA.[Parent.Status]
                       , GA.[Parent.DerivedStatus]
                       , GA.[Parent.AssetType]
                       , GA.[Parent.ParentId]
                       , GA.[Parent.ExtendedProperties]
                       , GA.[Layout.Direction]
                       , GA.[Layout.X]
                       , GA.[Layout.Y]
                       , GA.[Layout.RowFormat]
                       , GA.[Layout.ColumnFormat]
                       , GA.[Layout.Width]
                       , GA.[Layout.Height]
                       , GA.[Model.Manufacturer]
                       , GA.[Model.Id]
                       , GA.[Model.ManufacturerId]
                       , GA.[Model.Name]
                       , GA.[Model.UserName]
                       , GA.[Model.PartNumber]
                       , GA.[Model.UserPartNumber]
                       , GA.[Model.Type]
                       , GA.[Model.Power]
                       , GA.[Model.UserPower]                
                       , GA.[Model.Current]
                       , GA.[Model.Height]
                       , GA.[Model.Depth]
                       , GA.[Model.Width]
                       , GA.[Model.Weight]
                       , GA.[Model.ImagePath]
                       , GA.[Model.RackUnits]
                       , GA.[Model.Notes]
                       , GA.[Model.IsSystemDefined]
                       , CONVERT(VARCHAR(MAX), GA.[Model.ExtendedProperties]) AS 'Model.ExtendedProperties'
                       , GA.[Model.IsBasic]
                       , GA.[Model.SupportsRaMP]
                       , GA.AssetNode.ToString() AS 'AssetPath'
                       , DNI.PortName
                       , DNI.PortNumber
                       , '' AS 'NicAttachedMac'
                       , '' AS 'NicAttachedAddress'
                       , DNI.ManuallyConfigured
                    FROM
                         [core].[GetDownstreamNetworkInformation] (@ParentId) AS DNI
                    CROSS APPLY core.GetAsset(DNI.AssetId) AS GA
                    ORDER BY
                         DNI.PortNumber
               END
         ELSE
               BEGIN
                   
                    SELECT
                   DISTINCT
                         GA.AssetId
                       , GA.Name
                       , GA.ShortName
                       , GA.OverrideName
                       , GA.Type
                       , GA.[Status]
                       , GA.[LifecycleStatus]
                       , GA.TypeAtDiscovery
                       , GA.DerivedStatus
                       , GA.AssetType
                       , GA.AssetNode
                       , GA.AssetLevel
                       , GA.Notes
                       , NULL AS 'AssetXml'
                       , GA.ModelId
                       , GA.ChildCount
                       , GA.ParentId
                       , CONVERT(VARCHAR(MAX), GA.ExtendedProperties) AS 'ExtendedProperties'
                       , GA.RackLocation
                       , GA.RaMPPort
                       , GA.[Location.Source]
                       , GA.[Parent.AssetId]
                       , GA.[Parent.Name]
                       , GA.[Parent.ShortName]
                       , GA.[Parent.OverrideName]
                       , GA.[Parent.Type]
                       , GA.[Parent.Status]
                       , GA.[Parent.DerivedStatus]
                       , GA.[Parent.AssetType]
                       , GA.[Parent.ParentId]
                       , GA.[Parent.ExtendedProperties]
                       , GA.[Layout.Direction]
                       , GA.[Layout.X]
                       , GA.[Layout.Y]
                       , GA.[Layout.RowFormat]
                       , GA.[Layout.ColumnFormat]
                       , GA.[Layout.Width]
                       , GA.[Layout.Height]
                       , GA.[Model.Manufacturer]
                       , GA.[Model.Id]
                       , GA.[Model.ManufacturerId]
                       , GA.[Model.Name]
                       , GA.[Model.UserName]
                       , GA.[Model.PartNumber]
                       , GA.[Model.UserPartNumber]
                       , GA.[Model.Type]
                       , GA.[Model.Power]
                       , GA.[Model.UserPower]
                       , GA.[Model.Current]
                       , GA.[Model.Height]
                       , GA.[Model.Depth]
                       , GA.[Model.Width]
                       , GA.[Model.Weight]
                       , GA.[Model.ImagePath]
                       , GA.[Model.RackUnits]
                       , GA.[Model.Notes]
                       , GA.[Model.IsSystemDefined]
                       , CONVERT(VARCHAR(MAX), GA.[Model.ExtendedProperties]) AS 'Model.ExtendedProperties'
                       , GA.[Model.IsBasic]
                       , GA.[Model.SupportsRaMP]
                       , GA.AssetNode.ToString() AS 'AssetPath'
                       , UNI.PortName
                       , UNI.PortNumber
                       , UNI.NicAttachedMac
                       , UNI.NicAttachedAddress
                       , UNI.ManuallyConfigured
                    FROM
                         [core].[GetUpstreamNetworkInformation] (@ParentId) AS UNI
                    CROSS APPLY core.GetAsset(UNI.AssetId) AS GA
                    ORDER BY
                         GA.Name
               END
    END
GO
Uses
  • [core].[GetAsset]
  • [core].[GetDownstreamNetworkInformation]
  • [core].[GetUpstreamNetworkInformation]
  • [core]
