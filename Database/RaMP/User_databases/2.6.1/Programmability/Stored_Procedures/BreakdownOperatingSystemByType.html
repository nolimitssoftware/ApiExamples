CREATE PROC core.BreakdownOperatingSystemByType
   @Types VARCHAR(1024) = NULL
 , @ByServicePack BIT = NULL
 , @ParentId UNIQUEIDENTIFIER = NULL  
 , @IsDebug BIT = 0
AS
   BEGIN
       SET NOCOUNT ON ;
       
       DECLARE @ParentNode HIERARCHYID ;
       
       IF ( @ParentId IS NOT NULL )
           BEGIN
               SELECT
                   @ParentNode = A.AssetNode
               FROM
                   core.Asset AS A
               WHERE
                   A.AssetId = @ParentId ;
           END

       IF ( @ParentNode IS NULL )
           BEGIN
               SET @ParentNode = HIERARCHYID::GetRoot() ;
           END
           
       DECLARE
           @sql NVARCHAR(MAX)
         , @params NVARCHAR(4000) ;
       
       SET @sql = ' SET NOCOUNT ON;
               SELECT ' ;
       IF ( @ByServicePack = 1 )
           BEGIN
               SET @sql += '
                   CASE WHEN SOS.Name IS NULL THEN ''Unknown'' ELSE SOS.Name + ISNULL('' '' + SP.Name, '''') END AS ''OperatingSystem.Name''
                 , COUNT(CASE WHEN SOS.Name IS NULL THEN ''Unknown'' ELSE SOS.Name + ISNULL('' '' + SP.Name, '''') END) AS ''Count''
                 ' ;
           END
       ELSE
           BEGIN
               SET @sql += '
                   COALESCE(SOS.[Name], ''Unknown'') AS ''OperatingSystem.Name''
                 , COUNT(COALESCE(SOS.[Name], ''Unknown'')) AS ''Count''
                 ' ;
           END
       SET @sql += '
               FROM
                   core.Asset AS A ' ;
       IF ( @Types IS NOT NULL )
           BEGIN
               SET @sql += '
               INNER JOIN util.split_delimiter(@Types, '','') AS SD
                   ON A.Type = SD.str ' ;
           END
       SET @sql += '
               LEFT JOIN core.OperatingSystem AS OS
                   ON A.AssetId = OS.AssetId
               LEFT JOIN core.ServerOperatingSystem AS SOS
                   ON OS.ServerOperatingSystemId = SOS.Id ' ;
       IF ( @ByServicePack = 1 )
           BEGIN
               SET @sql += '
               LEFT JOIN core.ServicePack AS SP
                   ON OS.ServicePackId = SP.Id ' ;
           END
       SET @sql += '
                    WHERE
                   A.AssetNode.IsDescendantOf(@ParentNode) = 1
                   AND core.IsInExcludedAssetNode(A.AssetNode) = 0
               GROUP BY
                   COALESCE(SOS.[Name], ''Unknown'') ' ;
       IF ( @ByServicePack = 1 )
           BEGIN
               SET @sql += ' , SP.[Name] ' ;
           END
       SET @sql += '                
               ORDER BY
                   ''OperatingSystem.Name'';
       ' ;
       
       SET @params = '
           @Types VARCHAR(1024),
           @ParentNode HIERARCHYID
       ' ;    
       
       IF ( @IsDebug = 1 )
           BEGIN
               PRINT @params ;
               PRINT @sql ;            
           END
       
       EXEC sp_executesql
           @sql
         , @params
         , @Types
         , @ParentNode;
       
   END
GO
