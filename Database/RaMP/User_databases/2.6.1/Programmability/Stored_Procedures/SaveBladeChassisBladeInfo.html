CREATE PROC [core].[SaveBladeChassisBladeInfo]
   @Id UNIQUEIDENTIFIER = NULL
 , @AssetId UNIQUEIDENTIFIER
 , @Name NVARCHAR(256) = NULL
 , @BayLocation TINYINT
 , @BladeType INT
 , @BmcMac CHAR(17) = NULL
 , @BmcAddress VARCHAR(56) = NULL
 , @Mac1 CHAR(17) = NULL
 , @Mac2 CHAR(17) = NULL
 , @Manufacturer NVARCHAR(256) = NULL
 , @Model NVARCHAR(256) = NULL
 , @PartNumber NVARCHAR(256) = NULL
 , @SerialNumber1 NVARCHAR(256) = NULL
 , @SerialNumber2 NVARCHAR(256) = NULL
 , @SerialNumber3 NVARCHAR(256) = NULL
 , @Guid1 UNIQUEIDENTIFIER = NULL
 , @Guid2 UNIQUEIDENTIFIER = NULL
 , @Source VARCHAR(32)
 , @ExtendedProperties XML = NULL
 , @AgentId UNIQUEIDENTIFIER = NULL
 , @BladeAssetId UNIQUEIDENTIFIER = NULL
 , @PropertyXml XML = NULL
AS
   BEGIN
       IF ( @Id IS NULL )
           BEGIN
               SELECT
                   @Id = B.Id
               FROM
                   core.Blade AS B
               WHERE
                   B.AssetId = @AssetId
                   AND B.BayLocation = @BayLocation
           END
           
       IF ( @Id IS NULL )
           BEGIN            
               
               INSERT  core.Blade
                       ( AssetId
                       , BladeAssetId
                       , BmcMac
                       , BmcAddress
                       , [Name]
                       , BladeType
                       , Mac1
                       , Mac2
                       , Manufacturer
                       , Model
                       , PartNumber
                       , SerialNumber1
                       , SerialNumber2
                       , SerialNumber3
                       , Guid1
                       , Guid2
                       , BayLocation
                       , [Source]
                       , ExtendedProperties
                       , AgentId
                       , PropertyXml                      
                       )
               OUTPUT  INSERTED.Id
               VALUES
                       ( @AssetId
                       , @BladeAssetId
                       , @BmcMac
                       , @BmcAddress
                       , @Name
                       , @BladeType
                       , @Mac1
                       , @Mac2
                       , @Manufacturer
                       , @Model
                       , @PartNumber
                       , @SerialNumber1
                       , @SerialNumber2
                       , @SerialNumber3
                       , @Guid1
                       , @Guid2
                       , @BayLocation
                       , @Source
                       , @ExtendedProperties
                       , @AgentId
                       , @PropertyXml                      
                       )
           END
       ELSE
           BEGIN
               UPDATE
                   [core].[Blade]
               SET
                   [BmcMac] = @BmcMac
                 , BladeAssetId = @BladeAssetId
                 , [Name] = @Name
                 , BladeType = @BladeType
                 , [Mac1] = @Mac1
                 , [Mac2] = @Mac2
                 , Manufacturer = @Manufacturer
                 , Model = @Model
                 , PartNumber = @PartNumber
                 , SerialNumber1 = @SerialNumber1
                 , SerialNumber2 = @SerialNumber2
                 , SerialNumber3 = @SerialNumber3
                 , Guid1 = @Guid1
                 , Guid2 = @Guid2
                 , [BayLocation] = @BayLocation
                 , [Source] = @Source
                 , [ExtendedProperties] = @ExtendedProperties
                 , [PropertyXml] = @PropertyXml
                 , [AgentId] = @AgentId
               WHERE
                   core.[Blade].Id = @Id
           END
   
       SELECT
           @Id;
   END
GO
