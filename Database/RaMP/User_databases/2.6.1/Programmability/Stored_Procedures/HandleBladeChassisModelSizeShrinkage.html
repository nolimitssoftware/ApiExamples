CREATE PROC [core].[HandleBladeChassisModelSizeShrinkage]
   @ModelId UNIQUEIDENTIFIER
 , @TotalBays INT
AS
   BEGIN            
       -- Places devices in a "unplaced" state that were in a slot that has since disappeared with the user shrinking a chassis size.
       --    This join also makes sure not to touch chassis where we read both the dimensions for a chassis from the devices or it has been set by the user.
       
       UPDATE
           core.Blade
       SET
           BayLocation = NULL
       FROM
           core.Asset AS A
           LEFT JOIN core.AssetProperty AS AP ON A.AssetId = AP.AssetId AND AP.Name = 'rows'
           LEFT JOIN core.AssetProperty AS AP2 ON A.AssetId = AP2.AssetId AND AP2.Name = 'columns'
           INNER JOIN core.Blade AS B ON B.AssetId = A.AssetId        
       WHERE
           A.[Type] = 'bladeChassis'
           AND
           A.ModelId = @ModelId
           AND
           (
               (AP.SystemValue IS NULL AND AP.UserValue IS NULL)
               OR
               (AP2.SystemValue IS NULL AND AP2.UserValue IS NULL)
           )
           AND
           B.BayLocation > @TotalBays
   END
GO
