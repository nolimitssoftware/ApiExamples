CREATE PROCEDURE [visual].[SaveLayout]
   @Id UNIQUEIDENTIFIER
 , @LayoutType TINYINT
 , @BackgroundId UNIQUEIDENTIFIER = NULL
 , @Width REAL = NULL
 , @Height REAL = NULL
 , @ColumnFormat NVARCHAR(4) = NULL
 , @RowFormat NVARCHAR(4) = NULL
AS
   BEGIN
       IF EXISTS ( SELECT
                       1
                   FROM
                       visual.Layout AS L
                   WHERE
                       L.Id = @Id )
           BEGIN
               UPDATE
                   visual.Layout
               SET
                   LayoutType = @LayoutType
                 , BackgroundId = @BackgroundId
                 , Width = @Width
                 , Height = @Height
                 , ColumnFormat = @ColumnFormat
                 , RowFormat = @RowFormat
               WHERE
                   visual.Layout.Id = @Id ;
           END
       ELSE
           BEGIN
               INSERT  visual.Layout
                       (
                         Id
                       , LayoutType
                       , BackgroundId
                       , Width
                       , Height
                       , ColumnFormat
                       , RowFormat
                       )
               VALUES
                       (
                         @Id
                       , @LayoutType
                       , @BackgroundId
                       , @Width
                       , @Height
                       , @ColumnFormat
                       , @RowFormat
                       ) ;
           END
   END
GO
