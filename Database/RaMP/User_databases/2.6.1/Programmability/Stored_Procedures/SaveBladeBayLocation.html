CREATE PROCEDURE [core].[SaveBladeBayLocation]
    @BladeAssetId UNIQUEIDENTIFIER
 , @NewChassisAssetId UNIQUEIDENTIFIER
 , @OldParentId UNIQUEIDENTIFIER
 , @NewBayLocation TINYINT = NULL
 , @OldBayLocation TINYINT = NULL
 , @LocationChangeSource VARCHAR(32)
 , @BladeType int
AS
    BEGIN
    
       --Clean up
       IF EXISTS ( SELECT 1 FROM core.Blade WHERE AssetId = @OldParentId AND BayLocation = @OldBayLocation AND @LocationChangeSource = 'api')
           BEGIN
               -- Only do this if it was a manual location change (source will be api)
               UPDATE
                   core.Blade
               SET
                   Name = NULL
                   , BladeAssetId = NULL
                   , BmcMac = NULL
                   , Mac1 = NULL
                   , Mac2 = NULL
                   , Status = NULL
                   , Guid1 = NULL
                   , Guid2 = NULL
                   , BladeType = @BladeType
                   , Source = @LocationChangeSource
               WHERE
                   AssetId = @OldParentId
                   AND
                   BayLocation = @OldBayLocation
           
           END
    
           IF EXISTS ( SELECT
                               1
                         FROM
                               core.Blade
                         WHERE
                               AssetId = @NewChassisAssetId
                               AND
                               BayLocation = @NewBayLocation )
               BEGIN
                    UPDATE
                         core.Blade
                    SET
                         BladeAssetId = @BladeAssetId
                    OUTPUT
                         INSERTED.Id
                    WHERE
                           AssetId = @NewChassisAssetId
                           AND
                           BayLocation = @NewBayLocation
                   
                   --It is possible if we matched by Serial first that we will need to do some clean up.
                   DELETE FROM
                       core.Blade
                   WHERE
                       BladeAssetId = @BladeAssetId AND BayLocation IS NULL
                   
                       
               END
           ELSE IF EXISTS ( SELECT
                               1
                         FROM
                               core.Blade
                         WHERE
                               BladeAssetId = @BladeAssetId
                               AND
                               BayLocation IS NULL )
               BEGIN
                   -- We will get here if we matchd by only Serial Number before and now the user is manually changing the bay location
                    UPDATE
                         core.Blade
                    SET
                         AssetId = @NewChassisAssetId
                       , Source = @LocationChangeSource
                       , BayLocation = @NewBayLocation
                    OUTPUT
                         INSERTED.Id
                    WHERE
                       BladeAssetId = @BladeAssetId
                       AND
                       BayLocation IS NULL
               END
               
           ELSE IF EXISTS ( SELECT
                               1
                         FROM
                               core.Asset
                         WHERE
                               AssetId = @NewChassisAssetId
                               AND
                               [Type] = 'bladeChassis')
               BEGIN            
                   --Only do insert if the new location is a chassis!
                    INSERT  core.Blade
                               (
                                 AssetId
                               , BladeAssetId
                               , BayLocation
                               , BladeType
                               , Source
                               )
                    OUTPUT  
                       INSERTED.Id
                    VALUES
                               (
                                 @NewChassisAssetId
                               , @BladeAssetId
                               , @NewBayLocation            
                               , @BladeType        
                               , @LocationChangeSource
                               ) ;
                   
               END
               
           -- A blade can only be in one place at a time.  Clean up old places if needed
           UPDATE
               core.Blade
           SET
               BladeAssetId = NULL
           WHERE
               BladeAssetId = @BladeAssetId
               AND ((BayLocation <> @NewBayLocation OR AssetId <> @NewChassisAssetId)  
                   OR (@NewBayLocation IS NULL AND BayLocation IS NOT NULL));
    END
GO
