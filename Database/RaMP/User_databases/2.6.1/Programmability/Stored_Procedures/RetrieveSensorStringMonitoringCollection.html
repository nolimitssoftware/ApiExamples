CREATE PROCEDURE [monitoring].[RetrieveSensorStringMonitoringCollection]
   @AgentId UNIQUEIDENTIFIER
 , @Source VARCHAR(32)
AS
   BEGIN
       SET NOCOUNT ON ;

       SELECT
           SSM.Id
         , SSM.AssetId
         , SSM.AgentId
         , SSM.Source
         , SSM.MonitorXml
       FROM
           monitoring.SensorStringMonitoring AS SSM
       INNER JOIN core.Asset AS A ON SSM.AssetId = A.AssetId
       WHERE
           SSM.AgentId = @AgentId
           AND SSM.Source = @Source
           AND core.IsInExcludedAssetNode(A.AssetNode) = 0;
   END
GO
