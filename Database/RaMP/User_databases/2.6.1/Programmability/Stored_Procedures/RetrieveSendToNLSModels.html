SQL Script
CREATE PROC core.RetrieveSendToNLSModels
AS
   BEGIN
       SELECT
           M2.Name AS 'Manufacturer'
         , M.Id
         , M.ManufacturerId
         , M.Name
            , M.UserName
            , M.PartNumber
            , M.UserPartNumber
         , M.Type
         , M.Power
            , M.UserPower
         , M.[Current]
         , M.Height
         , M.Depth
         , M.Width
         , M.Weight
         , M.RackUnits
         , M.Notes
         , M.IsSystemDefined
         , M.ExtendedProperties
         , M.IsBasic
         , M.SupportsRaMP
         , A.AssetXml
         , A.RampXml
         , '' AS ImagePath
       FROM
           core.Model AS M
       INNER JOIN core.Manufacturer AS M2
           ON M.ManufacturerId = M2.Id
       INNER JOIN core.Asset AS A
           ON M.Id = A.ModelId
       WHERE
           M.IsSystemDefined = 0
           AND M.SentToNLS = 0
       ORDER BY
           M2.Name
         , M.Name;
   END
GO
Uses

  • [core].[Manufacturer]
