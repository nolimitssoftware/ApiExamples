CREATE PROC search.RetrieveSearchablePropertyCollection
   @AssetType INT = 0
 , @Status TINYINT = 0 --only get active
AS
   BEGIN
       SELECT
           *
       FROM
           ( SELECT
               SP.Id
             , SP.[Name]
             , COALESCE(SP.PropertyName, SP.[Name]) AS [PropertyName]
             , SP.Type
             , SP.[Values]
             , SP.BaseUnit
             , SP.AssetTypes
             , SP.ExcludedAssetTypes
             , '0' AS [PropertyType]
             , SP.Status AS [Status]
             , SP.DisplayUnit
             FROM
               search.SearchableProperty AS SP
             WHERE
               ( SP.AssetTypes & @AssetType ) = @AssetType
               OR SP.AssetTypes = 0
             UNION ALL
             SELECT
               CP.Id
             , CP.[Name]
             , CP.[Name] AS [PropertyName]
             , CP.Type
             , CP.[Values]
             , CP.Unit
             , CP.AssetTypes
             , 98304 --excluded physical and logical (no property support)
             , '1' AS [PropertyType]
             , 0 AS [Status]
             , CP.Unit
             FROM
               settings.CustomProperty AS CP
             WHERE
               ( CP.AssetTypes & @AssetType ) = @AssetType
               OR CP.AssetTypes = 0
           ) AS P
       WHERE
           ( @Status IS NULL )
           OR ( @Status IS NOT NULL
                AND P.Status = @Status
              )
       ORDER BY
           P.Name;
   END
GO
Uses
