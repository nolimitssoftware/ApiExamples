CREATE PROC [core].[RetrievePduBreakerCollection]
   @AssetId UNIQUEIDENTIFIER,
   @PanelId UNIQUEIDENTIFIER = NULL
AS
   BEGIN
       SELECT
           P.Id
         , P.AssetId
         , P.PanelId
         , P.Name
         , P.Number
         , P.LabelNumber
         , P.TieIndicator
         , P.LastTieIndicator
         , P.BreakerRating
         , P.BreakerRatingSource
         , P.Source
         , P.ExtendedProperties
         , P.AgentId
         , P.AmpSensorId
         , PP.Name AS 'PanelName'
         , PP.OverrideName AS 'PanelOverrideName'
         , PP.Number AS 'PanelNumber'
       FROM
           core.PduBreaker AS P
           INNER JOIN core.Asset AS A ON P.AssetId = A.AssetId
           LEFT JOIN core.PduPanel AS PP ON P.PanelId = PP.Id
       WHERE
           P.AssetId = @AssetId
           AND core.IsInExcludedAssetNode(A.AssetNode) = 0
           AND ((@PanelId IS NULL)
               OR (@PanelId IS NOT NULL AND P.PanelId = @PanelId))
       ORDER BY
           P.[Number] ASC;
   END
GO
