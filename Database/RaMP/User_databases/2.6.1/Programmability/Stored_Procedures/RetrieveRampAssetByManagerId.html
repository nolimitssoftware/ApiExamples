CREATE PROC core.RetrieveRampAssetByManagerId
   @ManagerId UNIQUEIDENTIFIER
AS
   BEGIN        
       SELECT
           GA.*
         , GA.AssetNode.ToString() AS 'AssetPath'
       FROM
           core.RampManager AS RM
       CROSS APPLY core.GetAsset(RM.AssetId) AS GA
       WHERE
           RM.ManagerId = @ManagerId
   END
GO
