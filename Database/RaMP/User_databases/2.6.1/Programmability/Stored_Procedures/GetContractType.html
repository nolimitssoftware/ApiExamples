CREATE PROC [core].[GetContractType]
(
   @ContractTypeName nvarchar(25),
   @ContractTypeId uniqueidentifier = null output
)
AS
BEGIN
   set @ContractTypeId = (Select ContractTypeId from Core.ContractType Where Name = @ContractTypeName);
   IF @ContractTypeId is null
   BEGIN
       RAISERROR('%s is an invalid contract type.', 10, 1, @ContractTypeName);
       Return -1;
   END
END
GO
