CREATE PROC core.RetrieveUndefinedAssetCollection
   @IsDebug BIT = NULL ,
   @SortField VARCHAR(32) = NULL ,
   @Start INT = 1 ,
   @Limit INT = NULL ,
   @IsAscending BIT = 1 ,
   @TotalCount INT OUTPUT
AS
   BEGIN
       DECLARE @OutParentId UNIQUEIDENTIFIER = NULL;
       DECLARE @End INT = @Start + @Limit - 1;
       DECLARE @OrderBy VARCHAR(128);        
       DECLARE @SortDir CHAR(6);
           
       IF ( @IsAscending = 1 )
           SET @SortDir = ' ASC  ';
       ELSE
           SET @SortDir = ' DESC ';
               
       IF ( @SortField IS NULL
            OR @SortField = 'DisplayName'
          )
           BEGIN
               SET @OrderBy = ' COALESCE(A.OverrideName,A.ShortName,A.Name) '
                   + @SortDir + ' ';
           END
       ELSE
           BEGIN            
               SET @OrderBy = 'A.' + @SortField + @SortDir;
           END
               
       DECLARE @sql NVARCHAR(MAX);
       DECLARE @params NVARCHAR(4000);
   
       SET @sql = '
           SET NOCOUNT ON;
           DECLARE @Assets TABLE(AssetId UNIQUEIDENTIFIER) ;
         
           INSERT  @Assets
                   ( AssetId            
                   )
                   SELECT DISTINCT
                           A.AssetId
                   FROM    core.Asset AS A
                           INNER JOIN core.Host AS H ON A.AssetId = H.AssetId
                   WHERE   A.Type = ''device''
                           AND core.IsInExcludedAssetNode(A.AssetNode) = 0;
         '
       SET @sql += ' SELECT @TotalCount = COUNT(*) FROM @Assets; IF(@TotalCount IS NULL) BEGIN SET @TotalCount = 0; END; IF(@End IS NULL) BEGIN SET @End = @TotalCount; END; '
         
       SET @sql += '
       SELECT  A.* ,
               A.AssetNode.ToString() AS ''AssetPath''
       FROM    ( SELECT DISTINCT
                           A.AssetId ,
                           A.ModelId ,
                           ( SELECT    COUNT(*)
                             FROM      core.Asset A2
                             WHERE     1 = 1
                                       AND A2.Type = ''device''
                                       AND A2.AssetNode.IsDescendantOf(A.AssetNode) = 1
                                       AND A2.AssetLevel = A.AssetLevel + 1
                           ) AS ''ChildCount'' ,
                           '''' AS ''AssetXml'' ,
                           COALESCE(@OutParentId,
                                    ( SELECT   A2.AssetId
                                      FROM     core.Asset AS A2
                                      WHERE    A2.AssetNode = A.AssetNode.GetAncestor(1)
                                    )) AS ''ParentId'' ,
                           ROW_NUMBER() OVER ( ORDER BY ' + @OrderBy + ' ) AS ''row_num''
                 FROM      @Assets AS A2
                           INNER JOIN core.Asset AS A ON A2.AssetId = A.AssetId
               ) AS A2
               CROSS APPLY core.GetAsset(A2.AssetId) AS A
       WHERE   A2.row_num BETWEEN @Start AND @End
       ORDER BY A2.Row_Num ASC;'
           
       SET @params = '
       @OutParentId UNIQUEIDENTIFIER
       ,@Start INT
       ,@End INT
       ,@TotalCount INT OUTPUT';
         
       IF ( @IsDebug = 1 )
           BEGIN
               PRINT @params;
               PRINT @Sql;
           END
       EXEC sp_executesql @sql, @params, @OutParentId, @Start, @End,
           @TotalCount = @TotalCount OUTPUT;
   
   END
GO
