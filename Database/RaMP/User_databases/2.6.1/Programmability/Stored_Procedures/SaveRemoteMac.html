CREATE PROCEDURE [core].[SaveRemoteMac]
   @Id UNIQUEIDENTIFIER = NULL
 , @AssetId UNIQUEIDENTIFIER
 , @Mac NVARCHAR(256)
 , @Source VARCHAR(32)
 , @Port INT = NULL
 , @DesignatedRootMac CHAR(17) = NULL
 , @DesignatedBridgeMac CHAR(17) = NULL
 , @DesignatedPort INT = NULL
 , @BridgeStatus VARCHAR(32) = NULL
 , @Vlan INT = NULL
 , @ExtendedProperties XML = NULL
 , @AgentId UNIQUEIDENTIFIER = NULL
 , @PropertyXml XML = NULL
AS
   BEGIN
       IF ( @Id IS NULL )
           BEGIN
                            
               INSERT  core.RemoteMac
                       ( AssetId
                       , Mac
                       , [Source]
                       , [Port]
                       , DesignatedRootMac
                       , DesignatedBridgeMac
                       , DesignatedPort
                       , BridgeStatus
                       , Vlan
                       , ExtendedProperties
                       , AgentId
                       , PropertyXml                      
                       )
               OUTPUT  INSERTED.Id
               VALUES
                       ( @AssetId
                       , @Mac
                       , @Source
                       , @Port
                       , @DesignatedRootMac
                       , @DesignatedBridgeMac
                       , @DesignatedPort
                       , @BridgeStatus
                       , @Vlan
                       , @ExtendedProperties
                       , @AgentId
                       , @PropertyXml                      
                       )
           END
       ELSE
           BEGIN
               UPDATE
                   core.RemoteMac
               SET
                   Port = @Port
                 , DesignatedRootMac = @DesignatedRootMac
                 , DesignatedBridgeMac = @DesignatedBridgeMac
                 , DesignatedPort = @DesignatedPort
                 , BridgeStatus = @BridgeStatus
                 , Vlan = @Vlan
                 , ExtendedProperties = @ExtendedProperties
                 , AgentId = @AgentId
                 , PropertyXml = @PropertyXml
               OUTPUT
                   INSERTED.Id
               WHERE
                   core.RemoteMac.Id = @Id;                
           END  
   END
GO
