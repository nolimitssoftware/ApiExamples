CREATE PROC core.FindAssetsByMacs
   @Macs VARCHAR(MAX) = NULL
 , @Source VARCHAR(32) = NULL
 , @Delim NCHAR(1) = ','
AS
   BEGIN
   
       DECLARE @MacAddresses TABLE ( Mac CHAR(17) )

       INSERT  @MacAddresses
               (
                 Mac                 
               )
               SELECT
                   SD.str
               FROM
                   util.split_delimiter(@Macs, @Delim) AS SD
               INNER JOIN core.MacOUI AS MO
                   ON MO.Prefix = LEFT(SD.str, 8)        
       DECLARE @SearchResults TABLE
           (
             Id UNIQUEIDENTIFIER
           ) ;
   
       IF ( @Source IS NULL )
           BEGIN
               INSERT  @SearchResults
                       (
                         Id
                       )
                       SELECT DISTINCT
                           NI.AssetId
                       FROM
                           core.NetworkInterface AS NI
                       INNER JOIN @MacAddresses AS MA
                           ON NI.Mac = MA.Mac
                       UNION
                       SELECT DISTINCT
                           AP.AssetId
                       FROM
                           core.AssetProperty AS AP
                       INNER JOIN @MacAddresses AS MA
                           ON AP.SystemValue = MA.Mac
                       WHERE
                           AP.Name IN ('mac', 'mac1', 'mac2') ;            
                           
               --NOTE: 'mac1' and 'mac2' come from ICU devices.         
           
           END
       ELSE
           BEGIN
               INSERT  @SearchResults
                       (
                         Id
                       )
                       SELECT DISTINCT
                           NI.AssetId
                       FROM
                           core.NetworkInterface AS NI
                       INNER JOIN @MacAddresses AS MA
                           ON NI.Mac = MA.Mac
                       WHERE
                           NI.Source = @Source;
           END
       
       SELECT
           SR.Id
         , A.Name
         , A.ShortName
         , A.OverrideName
         , A.Type
       FROM
           @SearchResults AS SR
       INNER JOIN core.Asset AS A
           ON SR.Id = A.AssetId ;
   END
GO
