CREATE PROC [core].[CascadeAssetExtendedProperty]
   @AssetId UNIQUEIDENTIFIER = NULL
 , @PropertyName NVARCHAR(32)
 , @NewValue SQL_VARIANT = NULL
 , @AssetTypesList VARCHAR(MAX) = NULL
 , @IsDebug BIT = NULL
AS
   BEGIN
           
   --Assumes that the save for the given asset has already been done.
   
       DECLARE
           @sql NVARCHAR(MAX)
         , @params NVARCHAR(4000)
         , @Display BIT
         , @Editable BIT
   
       SELECT
           @Display = AP.Display
         , @Editable = AP.Editable
       FROM
           core.AssetProperty AS AP
       WHERE
           AP.AssetId = @AssetId
           AND AP.Name = @PropertyName ;
   
           IF (@Display IS NULL)
               BEGIN
                   SET @Display = 1
               END
       
           IF (@Editable IS NULL)
               BEGIN
                   SET @Editable = 1
               END

       SET @sql = '
DECLARE @AssetNode HIERARCHYID ;
IF ( @AssetId IS NOT NULL )
   BEGIN    
       SELECT
           @AssetNode = A.AssetNode                          
       FROM
           [core].Asset AS A
       WHERE
           A.AssetId = @AssetId ;
   END
ELSE
   BEGIN
       SET @AssetNode = HIERARCHYID::GetRoot() ;
         SET @Display = 1 ;
   END    

IF ( @AssetNode IS NULL )
   BEGIN
       RAISERROR (N''Not a valid asset node.'',
         16,
         1) ;
   END
ELSE  
    BEGIN
       SELECT @NewValue = COALESCE(@NewValue, AP.SystemValue)
       FROM core.AssetProperty AS AP
       INNER JOIN core.Asset AS A
       ON AP.AssetId = A.AssetId
       WHERE A.AssetNode = @AssetNode
       AND AP.Name = @PropertyName;
    END
   '   
       IF ( @IsDebug = 1 )
           BEGIN
               SET @sql += '
SELECT
   A.ShortName
 , AP.Name
 , AP.SystemValue
 , AP.UserValue
 , AP.SystemValueCheckSum
 , AP.Display
 , AP.Editable
FROM
   core.AssetProperty AS AP
INNER JOIN core.Asset AS A
   ON AP.AssetId = A.AssetId
WHERE
   A.AssetNode.IsDescendantOf(@AssetNode) = 1
   AND AP.[Name] = @PropertyName '
           END
       
       SET @sql += '
DECLARE @AssetChildren TABLE
   (
     AssetId UNIQUEIDENTIFIER
   , [AssetNode] HIERARCHYID
   , [Row] INT IDENTITY(1, 1)
   , Id UNIQUEIDENTIFIER
   , [UserValue] SQL_VARIANT
   , [SysValue] SQL_VARIANT
   , [Ignore] BIT
) '

   --The order by AssetNode will ensure that parents have a higher row number than any of their descendants
       SET @sql += '
INSERT  INTO @AssetChildren
(
 AssetId
, AssetNode
, Id
, [SysValue]
, UserValue
, [Ignore]
)
SELECT DISTINCT
   A.AssetId
 , A.AssetNode
 , AP.Id
 , AP.SystemValue
 , AP.UserValue
 , 0
FROM
   [core].[Asset] AS A
LEFT JOIN core.AssetProperty AS AP
   ON A.AssetId = AP.AssetId
WHERE
   AssetNode.IsDescendantOf(@AssetNode) = 1
   AND AssetNode <> @AssetNode
   AND AP.Name = @PropertyName '
               
       IF ( @AssetTypesList IS NOT NULL )
           SET @sql += '
   AND [Type] IN (
   SELECT
       *
   FROM
       [util].[split_delimiter](@AssetTypesList, '','') ) '
       SET @sql += '
UNION
SELECT DISTINCT
   A.AssetId
 , A.AssetNode
 , NULL
 , @NewValue
 , NULL
 , 0
FROM
   [core].[Asset] AS A
WHERE
   AssetId NOT IN (SELECT AP.AssetId FROM core.AssetProperty AS AP WHERE AP.Name = @PropertyName)
   AND AssetNode.IsDescendantOf(@AssetNode) = 1
   AND AssetNode <> @AssetNode '

       IF ( @AssetTypesList IS NOT NULL )
           SET @sql += '
   AND [Type] IN (
   SELECT
       *
   FROM
       [util].[split_delimiter](@AssetTypesList, '','') ) '
   
       SET @sql += '
ORDER BY
   A.AssetNode
   
DECLARE
   @Current INT = (SELECT MIN(Row) FROM @AssetChildren)
 , @Max INT = (SELECT MAX(Row) FROM @AssetChildren );
WHILE ( @Current <= @Max )
   BEGIN
       DECLARE
           @ChildAssetId UNIQUEIDENTIFIER
         , @ChildAssetNode HIERARCHYID
         , @Id UNIQUEIDENTIFIER
         , @UserValue SQL_VARIANT
         , @SysValue SQL_VARIANT
         , @Ignore BIT
           
       SELECT
           @ChildAssetId = AC.AssetId
         , @ChildAssetNode = AC.AssetNode
         , @Id = AC.Id
         , @UserValue = AC.UserValue
         , @SysValue = AC.SysValue
         , @Ignore = AC.Ignore
       FROM
           @AssetChildren AS AC
       WHERE
           [Row] = @Current
           
       IF ( @Ignore = 0 )
           BEGIN
           -- Set System Value
           
               IF ( @Id IS NOT NULL)
                   BEGIN
                       UPDATE
                           core.AssetProperty
                       SET
                            SystemValue = @NewValue
                           ,LastUpdated = GETUTCDATE()
                       WHERE
                           core.AssetProperty.Id = @Id ;                    
                   END
               ELSE
                   BEGIN
                                                                    
                       EXEC core.InsertAssetProperty
                           @AssetId = @ChildAssetId
                         , @Name = @PropertyName
                         , @SystemValue = @NewValue
                         , @UserValue = NULL
                         , @Display = @Display
                         , @Editable = @Editable '                             
       SET @sql +=  
                 ' END
                       
                       IF (
                       @UserValue IS NOT NULL
                       AND @UserValue <> ''''
                    )
                       BEGIN '            
               --If User Value is not empty then then all of its descendants should be set to be ignored.
               
           SET @sql +=
                       ' UPDATE
                             @AssetChildren
                          SET
                             [Ignore] = 1
                          WHERE
                             AssetNode.IsDescendantOf(@ChildAssetNode) = 1
                             AND AssetNode <> @ChildAssetNode
               
                       END        
        
               SET @Current += 1
           END ;
   END ' ;
       IF ( @IsDebug = 1 )
           BEGIN
               SET @sql += 'SELECT
   A.ShortName
 , AP.Name
 , AP.SystemValue
 , AP.UserValue
 , AP.SystemValueCheckSum
 , AP.Display
 , AP.Editable
FROM
   core.AssetProperty AS AP
INNER JOIN core.Asset AS A
   ON AP.AssetId = A.AssetId
WHERE
   A.AssetNode.IsDescendantOf(@AssetNode) = 1
   AND AP.[Name] = @PropertyName ' ;
           END
       
       SET @params = '
       @AssetId UNIQUEIDENTIFIER
       , @PropertyName VARCHAR(256)
       , @NewValue SQL_VARIANT
       , @Display BIT
       , @Editable BIT
       , @AssetTypesList VARCHAR(MAX) '
       
       IF ( @IsDebug = 1 )
           BEGIN
               PRINT @params ;
               PRINT @sql ;
           END
        ELSE
           BEGIN
       
             EXEC sp_executesql
                   @sql
                , @params
                , @AssetId
                , @PropertyName
                , @NewValue
                , @Display
                , @Editable
                , @AssetTypesList
           END
   END
GO
