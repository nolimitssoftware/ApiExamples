CREATE PROC [core].[BuildIcuRack]
    @RackId UNIQUEIDENTIFIER
   ,@DiscoveryAgentId UNIQUEIDENTIFIER
   ,@InternalIcuId BIGINT
   ,@SourceAddress VARCHAR(32)
   ,@SourceMethod VARCHAR(16)
   ,@SourcePort INT
AS
BEGIN

   IF EXISTS (SELECT 1 FROM core.Icu WHERE InternalIcuId = @InternalIcuId)
       BEGIN
           UPDATE [core].Icu
              SET
                  [DiscoveryAgentId] = @DiscoveryAgentId,
                  [AssetId] = @RackId
            WHERE
               InternalIcuId = @InternalIcuId
       END
   ELSE
       BEGIN
           INSERT INTO [core].Icu
               ([AssetId]
               ,[DiscoveryAgentId]
               ,[InternalIcuId])
           VALUES
               (@RackId
               ,@DiscoveryAgentId
               ,@InternalIcuId)            
       END

   --Create/update settings used for monitoring.
   IF EXISTS (SELECT 1 FROM monitoring.AssetMonitorInformation WHERE AssetId = @RackId)
       BEGIN
           UPDATE [monitoring].AssetMonitorInformation
              SET
                  [DiscoveryAgentId] = @DiscoveryAgentId,
                  SourceAddress = @SourceAddress,
                  SourceMethod = @SourceMethod,
                  SourcePort = @SourcePort,
                  Protocol = 'icu',
                  LastUpdate = GETUTCDATE()
            WHERE
               AssetId = @RackId
       END
   ELSE
       BEGIN
           INSERT INTO [monitoring].AssetMonitorInformation
               ([AssetId]
               ,[DiscoveryAgentId]
               ,SourceAddress
               ,SourceMethod
               ,SourcePort
               ,Protocol
               ,LastUpdate)
           VALUES
               (@RackId
               ,@DiscoveryAgentId
               ,@SourceAddress
               ,@SourceMethod
               ,@SourcePort
               ,'icu'
               ,GETUTCDATE())            
       END
END
GO
