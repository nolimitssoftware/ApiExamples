CREATE PROC [core].[RetrieveAssetExtendedProperties]
   @AssetId UNIQUEIDENTIFIER = NULL
AS
   BEGIN        
    
       --Use root if assetId is null
       IF ( @AssetId IS NULL )
           BEGIN
               SELECT
                   @AssetId = A.AssetId
               FROM
                   core.Asset A
               WHERE
                   A.AssetNode = HIERARCHYID::GetRoot()
           END
    
       SELECT
           (
             SELECT
               property.Name AS 'name'
             , property.SystemValue AS 'systemValue'
             , SQL_VARIANT_PROPERTY(property.SystemValue, 'basetype') AS 'systemValueType'
                 , property.SystemValueSource AS 'systemValueSource'
             , property.UserValue AS 'userValue'
             , SQL_VARIANT_PROPERTY(property.UserValue, 'basetype') AS 'userValueType'
                 , property.UserValueSource AS 'userValueSource'
             , property.Display AS 'display'
             , property.Editable AS 'editable'
             FROM
               core.AssetProperty AS property
             WHERE
               property.AssetId = @AssetId
           FOR
             XML PATH('property')
               , ROOT('properties')
           ) AS ExtendedProperties
   END
GO
