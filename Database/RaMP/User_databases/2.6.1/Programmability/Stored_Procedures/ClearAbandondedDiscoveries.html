CREATE PROCEDURE settings.[ClearAbandondedDiscoveries]
   @DiscoveryAgentId UNIQUEIDENTIFIER
AS
   BEGIN
       SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;
       
       UPDATE
           settings.Discovery
       SET
           [Status] = 204
       WHERE
           DiscoveryAgentId = @DiscoveryAgentId
           AND settings.Discovery.Status < 101;
           
       UPDATE
           [log].DiscoveryLog
       SET
           DiscoveryStatusId = 204
       FROM
           log.DiscoveryLog AS DL
       INNER JOIN settings.Discovery AS D
           ON DL.DiscoveryId = D.DiscoveryId
       WHERE
           DL.DiscoveryStatusId < 101
           AND D.DiscoveryAgentId = @DiscoveryAgentId;
   END
GO
