CREATE PROCEDURE [log].[RetrieveChangeCollection]
   @ContextIds VARCHAR(MAX) = NULL
 , @ChangeRecordType INT = 0
 , @Contexts INT = 0
 , @Components VARCHAR(MAX) = NULL
 , @LogTypes INT = NULL
 , @Since DATETIME2 = NULL
 , @MasterId UNIQUEIDENTIFIER = NULL
 , @SortField VARCHAR(32) = NULL
 , @Start INT
 , @Limit INT
 , @IsAscending BIT = 0
 , @IsDebug BIT = NULL
 , @TotalCount INT OUTPUT
AS
   BEGIN
  
       DECLARE @End INT = @Start + @Limit - 1 ;
       DECLARE @SortDir CHAR(6) ;
       DECLARE @OrderBy VARCHAR(1024) ;        
       DECLARE @SinceClause VARCHAR(1024) = '' ;
       DECLARE @MasterSort VARCHAR(128) = '' ;
   
       IF ( @Since IS NOT NULL )
           BEGIN
               SET @SinceClause = ' AND C.StartTimeStamp > @Since '
           END
       
       IF ( @IsAscending = 1 )
           SET @SortDir = ' ASC  ' ;
       ELSE
           SET @SortDir = ' DESC ' ;
   
       IF ( @MasterId IS NOT NULL )
           BEGIN
               IF ( ( @Contexts & 1 ) = @Contexts ) --Asset
                   BEGIN
                       SET @MasterSort += ' A.Sort ASC, ' ;        
                   END
               IF ( ( @Contexts & 2 ) = @Contexts ) --Logical Groups
                   BEGIN            
                       SET @MasterSort += ' LG.Sort ASC, ' ;
                   END
           END
   
       IF ( @SortField IS NULL )
           BEGIN
               SET @SortField = ' GCC.StartTimestamp ' ;
               SET @OrderBy = ', ROW_NUMBER() OVER ( ORDER BY ' + @MasterSort
                   + @SortField + @SortDir + ' ) AS ''row_num'' ' ;
           END
       ELSE
           BEGIN            
               IF ( @SortField = 'displayName' )
                   BEGIN                    
                       SET @OrderBy = ', CASE
           GCC.Context WHEN 0 THEN NULL '

                       IF ( ( @Contexts & 1 ) = @Contexts ) --Asset
                           BEGIN
                               SET @OrderBy += ' WHEN 1 THEN ROW_NUMBER() OVER( ORDER BY '
                                   + @MasterSort
                                   + ' COALESCE(A.OverrideName, A.ShortName) '
                                   + @SortDir + ', GCC.StartTimeStamp DESC )' ;
                           END
                               
                       IF ( ( @Contexts & 2 ) = @Contexts ) --Logical Groups
                           BEGIN
                               SET @OrderBy += ' WHEN 2 THEN ROW_NUMBER() OVER( ORDER BY '
                                   + @MasterSort
                                   + ' COALESCE(LG.OverrideName, LG.ShortName) '
                                   + @SortDir + ', GCC.StartTimeStamp DESC) ' ;
                           END                        
       
                       SET @OrderBy += ' ELSE '''' END AS ''row_num''' ;    
                       
                   
                   END
               ELSE
                   BEGIN                
                       SET @OrderBy = ', ROW_NUMBER() OVER ( ORDER BY '
                           + @MasterSort + @SortField + @SortDir
                           + ' ) AS ''row_num'' ' ;
                   END
           END            
   
       
       DECLARE @sql NVARCHAR(MAX) = '';
       DECLARE @params NVARCHAR(4000) ;
                   
       SET @sql += '
   SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED ;
   SET NOCOUNT ON ;
   DECLARE @NoChanges UNIQUEIDENTIFIER =''0A5A5C9D-9B6B-E011-A3A5-001C421C284C'';
   
   DECLARE @Changes TABLE
       (
         ChangeId UNIQUEIDENTIFIER
, Context INT
         ,ContextId UNIQUEIDENTIFIER
     , ChangeMasterId UNIQUEIDENTIFIER
       ) ;
         
   INSERT  @Changes
           (
               ChangeId
             , Context
             , ContextId
             , ChangeMasterId
           )
           SELECT
               C.Id, C.Context '
       IF ( @ContextIds IS NOT NULL )
           SET @sql += ' , CAST(SD_C.str AS UNIQUEIDENTIFIER) '
       ELSE
           SET @sql += ' , NULL '
       
       SET @sql += ', C.ChangeMasterId
           FROM
               log.Change AS C
           INNER JOIN log.ChangeMaster AS CM
               ON C.ChangeMasterId = CM.Id '
       IF ( @ContextIds IS NOT NULL )
           SET @sql += ' INNER JOIN util.split_delimiter(@ContextIds, '','') AS SD_C ON C.ContextId = CAST(SD_C.str AS UNIQUEIDENTIFIER) '

       IF ( @Components IS NOT NULL )
           SET @sql += ' INNER JOIN util.split_delimiter(@Components, '','') AS SD_C ON C.Context = SD_C.str '
   
       SET @sql += ' WHERE
               1 = 1 '
       SET @sql += @SinceClause ;
   
       IF ( @LogTypes IS NOT NULL )
           SET @sql += ' AND ( C.LogTypeId & @LogTypes ) = @LogTypes '

       IF ( @ChangeRecordType > 0 )
           SET @sql += ' AND ( CM.ChangeRecordType & @ChangeRecordType ) = CM.ChangeRecordType '
   
       IF ( @Contexts <> 0 )
           SET @sql += ' AND (@Contexts & C.Context) = @Contexts '
       
       IF ( @ContextIds IS NOT NULL )
           SET @sql += '
               INSERT @Changes
                   ( ChangeId, Context, ContextId, ChangeMasterId )
               SELECT NULL, NULL, CAST(SD_C.str AS UNIQUEIDENTIFIER), @NoChanges
                   FROM util.split_delimiter(@ContextIds, '','') AS SD_C                     
               WHERE
                   CAST(SD_C.str AS UNIQUEIDENTIFIER) NOT IN (SELECT C.ContextId FROM @Changes AS C);

       UPDATE @Changes
       SET Context = 1
       FROM @Changes AS C
       INNER JOIN core.Asset AS A ON C.ContextId = A.AssetId
       WHERE C.Context IS NULL;

       UPDATE @Changes
       SET Context = 2
       FROM @Changes AS C
       INNER JOIN core.LogicalAsset AS LA ON C.ContextId = LA.AssetId
       WHERE C.Context IS NULL;
       '
       
       SET @sql += '            
   SELECT
       @TotalCount = COUNT(*)
   FROM
       @Changes AS C ;
   IF ( @TotalCount IS NULL )
       BEGIN
           SET @TotalCount = 0 ;
       END ;
   SELECT
       temp.Id
     , temp.ChangeMasterId
     , temp.LogTypeId
     , temp.Context
     , temp.ContextId
     , temp.ChangeAction
     , temp.Component
     , temp.StartTimestamp
     , temp.ChangedByUserName
     , temp.FromAddress
     , temp.Authorized
     , temp.AuthorizedByUserName
     , temp.ExtendedProperties
     , temp.Severity
     , temp.ChangeRecordType
     , temp.Description
     , temp.UserOverrideType
     , temp.UserOverrideHelp
     , temp.ContextDisplayName
     , temp.ContextType
     , temp.row_num
   FROM
       (
         SELECT
           GCC.Id
         , GCC.ChangeMasterId
         , GCC.LogTypeId
         , GCC.Context
         , GCC.ContextId
         , GCC.ChangeAction
         , GCC.Component
         , GCC.StartTimestamp
         , GCC.ChangedByUserName
         , GCC.FromAddress
         , GCC.Authorized
         , GCC.AuthorizedByUserName
         , GCC.ExtendedProperties
         , GCC.Severity
         , GCC.ChangeRecordType
         , GCC.Description
         , GCC.UserOverrideType
         , GCC.UserOverrideHelp '
       
       SET @sql += ', CASE
                   WHEN GCC.Context = 0 THEN NULL '
           
       IF ( ( @Contexts & 1 ) = @Contexts ) --Asset
           BEGIN
               SET @sql += ' WHEN GCC.Context = 1 THEN COALESCE(A.OverrideName, A.ShortName) '
           END
       
       IF ( ( @Contexts & 2 ) = @Contexts ) --Logical Groups
           BEGIN
               SET @sql += ' WHEN GCC.Context = 2 THEN COALESCE(LG.OverrideName, LG.ShortName) '
           END    
         

       SET @sql += ' ELSE ''''  
         END AS ''ContextDisplayName''
         '
       SET @sql += ', CASE
                    WHEN GCC.Context = 0 THEN NULL '
           
       IF ( ( @Contexts & 1 ) = @Contexts ) --Asset
           BEGIN
               SET @sql += ' WHEN GCC.Context = 1 THEN A.Type '
           END
       
       IF ( ( @Contexts & 2 ) = @Contexts ) --Logical Groups
           BEGIN
               SET @sql += ' WHEN GCC.Context = 2 THEN ''logicalGroup'''
           END
                     

       SET @sql += ' ELSE ''''  
         END AS ''ContextType'''   
         
       SET @sql += @OrderBy ;
             
       SET @sql += '          
         FROM
           @Changes AS C2
         CROSS APPLY log.GetChangeCollection(C2.ChangeId, C2.ChangeMasterId, C2.Context, C2.ContextId) AS GCC
            ' ;
           
       IF ( ( @Contexts & 1 ) = @Contexts ) --Asset
           BEGIN
               SET @sql += ' LEFT JOIN (SELECT A.*, CASE A.AssetId WHEN @MasterId THEN 1 ELSE 2 END AS ''Sort'' FROM core.Asset AS A) AS A ON GCC.ContextId = A.AssetId '
           --IF(@MasterId IS NOT NULL)
           --BEGIN
               --SET @sql += '
           --END
           --SET @sql += '
           END
       
       IF ( ( @Contexts & 2 ) = @Contexts ) --Logical Groups
           BEGIN
               SET @sql += ' LEFT JOIN (SELECT LG.*, CASE LG.AssetId WHEN @MasterId THEN 1 ELSE 2 END AS ''Sort'' FROM core.LogicalAsset AS LG) AS LG ON GCC.ContextId = LG.AssetId '
           END
       
       SET @sql += ' WHERE 1=1 ' ;        
       
       SET @sql += '    
       ) AS temp
   WHERE
       temp.row_num BETWEEN @Start AND @End
   ORDER BY
       temp.row_num ASC; '
       
    
       
       SET @params = '    
       @ContextIds VARCHAR(MAX)
     , @MasterId UNIQUEIDENTIFIER        
     , @ChangeRecordType INT
     , @Contexts INT
     , @Components VARCHAR(MAX)
     , @LogTypes INT
     , @Since DATETIME2
     , @Start INT
     , @End INT             
     , @TotalCount INT OUTPUT
   ';
   
      IF ( @IsDebug = 1 )
           BEGIN
               PRINT @params;
               PRINT @Sql ;    
           END
   
       EXEC sp_executesql
           @sql
         , @params
         , @ContextIds
         , @MasterId
         , @ChangeRecordType
         , @Contexts
         , @Components
         , @LogTypes
         , @Since
         , @Start
         , @End
         , @TotalCount = @TotalCount OUTPUT ;                
   END
GO
