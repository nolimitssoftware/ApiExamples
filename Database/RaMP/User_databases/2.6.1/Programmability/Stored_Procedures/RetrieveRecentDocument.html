CREATE PROCEDURE [doc].[RetrieveRecentDocument]
AS
       SELECT top 10
           [Id]
           ,[Name]
           ,[Extension]
           ,[MimeType]
           ,[FileType]
           ,[AssociationType]
           ,[Size]
           ,[Address]
       FROM
           [doc].[Document]
       Order By CreationDate desc
RETURN 0
GO
