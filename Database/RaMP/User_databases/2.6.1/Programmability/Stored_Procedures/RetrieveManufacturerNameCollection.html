CREATE PROCEDURE [settings].[RetrieveManufacturerNameCollection]
   @Manufacturer NVARCHAR (256)
AS
   BEGIN
       DECLARE @ManufacturerId UNIQUEIDENTIFIER
       
       SELECT
           @ManufacturerId = M.Id
       FROM
           core.Manufacturer AS M
       LEFT JOIN
           core.ManufacturerAlternateName AS MAN
           ON M.Id = MAN.ManufacturerId
       WHERE
           (
             M.[Name] = @Manufacturer
             OR MAN.[Name] = @Manufacturer
           )
           
       SELECT
           M.Name
       FROM
           core.Manufacturer AS M
       WHERE
           M.Id = @ManufacturerId
       UNION
       SELECT
           MA.Name
       FROM
           core.ManufacturerAlternateName AS MA
       WHERE
           MA.ManufacturerId = @ManufacturerId
               
   END
GO
