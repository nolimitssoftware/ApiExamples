CREATE PROC log.ClearNonApprovedServicePackMessages
AS
   BEGIN
       DECLARE @MasterMessageId UNIQUEIDENTIFIER = (
                                                     SELECT
                                                       MM.Id
                                                     FROM
                                                       log.MessageMaster AS MM
                                                       join log.InternalMessageToMessageMaster on  mm.id = InternalMessageToMessageMaster.Id
                                                     WHERE
                                                       InternalMessageToMessageMaster.Description = 'nonApprovedServicePack'
                                                   ) ;
   
   
       DECLARE @Assets TABLE
           (
             AssetId UNIQUEIDENTIFIER
           , [Row] INT IDENTITY(1, 1)
           ) ;
       INSERT  @Assets
               SELECT
                   M.AssetId
               FROM
                   log.Message AS M
               WHERE
                   M.MessageMasterId = @MasterMessageId
                   AND M.EndTimestamp IS NULL ;
       DELETE
           @Assets
       FROM
           @Assets AS A2
       WHERE
           AssetId IN (
           SELECT DISTINCT
               OS.Assetid
           FROM
               core.OperatingSystem AS OS
           INNER JOIN core.ServicePack AS SP
               ON OS.ServicePackId = SP.Id
                  AND OS.ServerOperatingSystemId = SP.ServerOperatingSystemId
           INNER JOIN core.ServerOperatingSystem AS SOS
               ON SP.ServerOperatingSystemId = SOS.Id
           WHERE
               SP.IsApproved = 0
               AND SOS.IsApproved <> 0 ) ;
                       
       DECLARE
           @Current INT = 1
         , @Max INT = (
                        SELECT
                           MAX(A.Row)
                        FROM
                           @Assets AS A
                      ) ;

       WHILE ( @Current <= @Max )
           BEGIN
               DECLARE @AssetId UNIQUEIDENTIFIER ;
               SET @AssetId = NULL ;
   
               SELECT
                   @AssetId = A.AssetId
               FROM
                   @Assets AS A
               WHERE
                   A.Row = @Current ;
   
               IF ( @AssetId IS NOT NULL )
                   BEGIN
                       EXEC log.ClearMessage
                           @AssetId = @AssetId
                         , @MessageMasterId = @MasterMessageId
                         , @ClearedByUserName = 'RaMP'
                         , @ContextId = @AssetId --only 1 message per asset
       
                   END
   
               SET @Current += 1 ;
           END
   END
GO
