CREATE PROCEDURE [core].[RetrieveSensor]
   @SensorId UNIQUEIDENTIFIER
AS
   SELECT
        S.[SensorId]
       ,S.[DeviceId]
       ,A2.ShortName AS 'ParentShortName'
       ,A2.OverrideName AS 'ParentOverrideName'
       ,S.[DiscoveryAgentId]
       ,S.[Source]
       ,S.[Type]
       ,S.[SystemName]
       ,S.[UserName]
       ,S.[Unit]
       ,S.[Element]
       ,S.[OriginType]
       ,S.[SensorXml]
       ,S.[ExtendedProperties]
       ,S.LocationAssetId
       ,A.Name AS 'LocationAssetName'
       ,A.ShortName AS 'LocationAssetShortName'
       ,A.OverrideName AS 'LocationAssetOverrideName'
       ,A.[Type] AS 'LocationAssetType'
       ,S.RackLocation
       ,S.Position
       ,SV.Value AS 'CurrentValue'
       ,SV.LastUpdate AS 'CurrentValueLastUpdate'
       ,SC.LocationEditable
       ,SC.DisplayInAssetSensorList
   FROM
       [core].[Sensor] AS S
   INNER JOIN
       [core].Asset AS A ON S.LocationAssetId = A.AssetId
   INNER JOIN
       [core].Asset AS A2 ON S.DeviceId = A2.AssetId
   LEFT JOIN monitoring.SensorValue AS SV
       ON S.[SensorId] = SV.SensorId AND SV.RollupType = 1
   LEFT JOIN settings.SensorConfiguration AS SC
       ON S.[Type] = SC.SensorType
   WHERE
       S.SensorId = @SensorId
GO
Uses
  • [core].[Asset]
  • [core].[Sensor]
  • [monitoring].[SensorValue]
  • [settings].[SensorConfiguration]
