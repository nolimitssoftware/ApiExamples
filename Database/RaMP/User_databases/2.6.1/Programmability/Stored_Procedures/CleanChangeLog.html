CREATE PROC log.CleanChangeLog
   @Context INT
 , @LogTypes INT = 0
 , @ContextId UNIQUEIDENTIFIER
 , @ContextDisplayName VARCHAR(256)
 , @ContextType VARCHAR(32)
AS
   BEGIN
       IF ( @LogTypes = 0 )
           BEGIN
           
               --Delete all change that is only this log type
               
             
               DELETE
                   log.Change
               WHERE
                   ContextId = @ContextId ;
                   
           -- Update all change log entries that are not also entries for other logs.
                                   
               UPDATE
                   log.Change
               SET
                   ExtendedProperties = core.AddExtendedProperty(C.ExtendedProperties,
                                                             'contextType', 1,
                                                             0, @ContextType,
                                                             'string',
                                                             DEFAULT, DEFAULT,
                                                             DEFAULT, DEFAULT)
               FROM
                   log.Change AS C
               WHERE
                   C.ContextId = @ContextId
                   AND C.Context = @Context ;
                   
               UPDATE
                   log.Change
               SET
                   ExtendedProperties = core.AddExtendedProperty(C.ExtendedProperties,
                                                             'contextDisplayName',
                                                             1, 0,
                                                             @ContextDisplayName,
                                                             'string',
                                                             DEFAULT, DEFAULT,
                                                             DEFAULT, DEFAULT)
               FROM
                   log.Change AS C
               WHERE
                   C.ContextId = @ContextId
                   AND C.Context = @Context ;
           END
       ELSE
           BEGIN
           
           --Delete all change that is only this log type
               
             
               DELETE
                   log.Change
               WHERE
                   ContextId = @ContextId
                   AND LogTypeId = @LogTypes ;
                   
           -- Update all change log entries that are not also entries for other logs.
                                   
               UPDATE
                   log.Change
               SET
                   ExtendedProperties = core.AddExtendedProperty(C.ExtendedProperties,
                                                             'contextType', 1,
                                                             0, @ContextType,
                                                             'string',
                                                             DEFAULT, DEFAULT,
                                                             DEFAULT, DEFAULT)
               FROM
                   log.Change AS C
               WHERE
                   C.ContextId = @ContextId
                   AND C.Context = @Context
                   AND ( C.LogTypeId & @LogTypes ) = @LogTypes ;
                   
               UPDATE
                   log.Change
               SET
                   ExtendedProperties = core.AddExtendedProperty(C.ExtendedProperties,
                                                             'contextDisplayName',
                                                             1, 0,
                                                             @ContextDisplayName,
                                                             'string',
                                                             DEFAULT, DEFAULT,
                                                             DEFAULT, DEFAULT)
                 , LogTypeId = C.LogTypeId ^ @LogTypes
               FROM
                   log.Change AS C
               WHERE
                   C.ContextId = @ContextId
                   AND C.Context = @Context
                   AND ( C.LogTypeId & @LogTypes ) = @LogTypes ;
           END
   END
GO
