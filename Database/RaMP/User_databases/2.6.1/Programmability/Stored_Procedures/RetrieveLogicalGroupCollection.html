CREATE PROCEDURE [core].[RetrieveLogicalGroupCollection]
   @IsDebug BIT = 0
 , @SortField VARCHAR(32) = NULL
 , @Start INT = 1
 , @Limit INT = NULL
 , @IsAscending BIT = 1
 , @TotalCount INT OUTPUT
AS
   BEGIN
       DECLARE @End INT = @Start + @Limit - 1 ;
       DECLARE @SortDir CHAR(6) ;
       DECLARE @OrderBy VARCHAR(128) ;        
       
       IF ( @IsAscending = 1 )
           SET @SortDir = ' ASC  ' ;
       ELSE
           SET @SortDir = ' DESC ' ;
               
       IF (
            @SortField IS NULL
            OR @SortField = 'DisplayName'
          )
           BEGIN
               SET @OrderBy = ' COALESCE(A.OverrideName,A.ShortName,A.Name) '
                   + @SortDir + ' ' ;
           END
       ELSE
           BEGIN            
               SET @OrderBy = 'A.' + @SortField + @SortDir ;
           END            
       
       DECLARE @sql NVARCHAR(MAX) ;
       DECLARE @params NVARCHAR(4000) ;
   
       SET @sql = '
         SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED ;SET NOCOUNT ON;
         
       DECLARE @OutParentId UNIQUEIDENTIFIER
       SELECT
           @OutParentId = LA.AssetId
       FROM
           core.LogicalAsset AS LA
       WHERE
           LA.AssetNode = HIERARCHYID::GetRoot() ;
       
       SELECT @TotalCount = COUNT(1)
       FROM core.LogicalAsset AS LA
       WHERE LA.AssetLevel > 0; ' ;
           
       SET @sql += '
SELECT * FROM (
       SELECT
           AssetId
         , Name
         , ShortName
         , OverrideName
         , Type
         , Status
         , Status AS [DerivedStatus]
         , AssetType
         , AssetNode.ToString() AS ''AssetPath''
         , AssetLevel
         , '''' AS Notes
         , NULL AS AssetXml
         , 0 AS ChildCount
         , CONVERT(VARCHAR(MAX), ExtendedProperties) AS ExtendedProperties
         , @OutParentId AS ParentId
         ,    ROW_NUMBER() OVER(ORDER BY ' + @OrderBy + ' '
                       
       SET @sql += ') AS ''row_num''        
       FROM
           core.LogicalAsset AS A
       WHERE
           AssetLevel > 0 ) AS LA
       WHERE LA.row_num BETWEEN @Start AND @End;'
       
       SET @params = '
         @Start INT,
         @End INT,
         @TotalCount INT OUTPUT' ;
         
       IF ( @IsDebug = 1 )
           BEGIN
               PRINT @params ;
               PRINT @Sql ;
           END
       EXEC sp_executesql
           @sql
         , @params
         , @Start
         , @End
         , @TotalCount = @TotalCount OUTPUT ;
   END
GO
