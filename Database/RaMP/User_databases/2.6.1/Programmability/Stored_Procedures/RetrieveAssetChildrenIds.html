CREATE PROC [core].RetrieveAssetChildrenIds
   @ParentAssetId UNIQUEIDENTIFIER
AS
   BEGIN
       DECLARE @ParentAssetNode HIERARCHYID
       
       SELECT TOP 1 @ParentAssetNode = AssetNode
       FROM    core.Asset
       WHERE   AssetId = @ParentAssetId
       
       SELECT  A.AssetId
       FROM    core.Asset AS A
       WHERE   A.AssetNode.GetAncestor(1) = @ParentAssetNode
       AND        core.IsInExcludedAssetNode(A.AssetNode) = 0

   END
GO
