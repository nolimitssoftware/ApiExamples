PropertyValue

SQL Script
CREATE PROCEDURE settings.RetrieveOsModelCollection
AS
   BEGIN
       SELECT
           M.Id AS 'ManufacturerId'
         , M.Name AS 'Manufacturer'
         , SUBSTRING(temp.Names, 0, LEN(temp.Names)) AS 'ManufacturerAlternateNames'
         , SOS.Id AS 'ServerOperatingSystem.Id'
         , NULL AS 'ServicePack.Id'
         , SOS.Name AS 'ServerOperatingSystem.Name'
         , NULL AS 'ServicePack.Name'
         , SOS.VersionString
         , SOS.SearchString
         , SOS.Major
         , SOS.Minor
         , SOS.Revision
         , SOS.Build
         , SOS.Other
         , SOS.Notes
         , SOS.ExtendedProperties
         , SOS.IsSystemDefined
         , SOS.SentToNLS
         , SOS.InternalId
         , SOS.LicenseCount
         , SOS.IsApproved
         , SPI.HasUnapprovedServicePack          
         , core.CountOSInstallations(SOS.Id, NULL, 0) AS 'InstallationCount'
         , SPI.ServicePackCount
       FROM
           core.Manufacturer AS M
       INNER JOIN core.ServerOperatingSystem AS SOS
           ON M.Id = SOS.ManufacturerId
       CROSS APPLY ( SELECT
               *
             FROM
               ( SELECT     
                    COUNT(SP1.ServerOperatingSystemId) AS [ServicePackCount]
                  , CAST(CASE WHEN ISNULL(SUM(CONVERT(INT, SP1.IsApproved)),
                                         0) = COUNT(SP1.ServerOperatingSystemId)
                             THEN 0
                             ELSE 1
                        END AS BIT) AS [HasUnapprovedServicePack]
                 FROM
                   ( SELECT
                       SP.Id
                     , SP.ServerOperatingSystemId
                     , SP.IsApproved
                     FROM
                       core.ServicePack AS SP
                     WHERE
                       SP.ServerOperatingSystemId = SOS.Id
                       AND
                           core.CountOSInstallations(SOS.Id, SP.Id, 1) > 0
                   ) AS SP1
               ) AS SP2
           ) AS SPI       
       CROSS APPLY (
                     SELECT
                       CONVERT(VARCHAR(256), ISNULL(MAN.[Name], '')) + ',' AS [text()]
                     FROM
                       core.ManufacturerAlternateName AS MAN
                     WHERE
                       MAN.ManufacturerId = M.Id
                   FOR
                     XML PATH('')
                   ) AS temp ( [Names] )
       UNION ALL
       SELECT
           M.Id AS 'ManufacturerId'
         , M.Name AS 'Manufacturer'
         , SUBSTRING(temp.Names, 0, LEN(temp.Names)) AS 'ManufacturerAlternateNames'
         , SP.ServerOperatingSystemId AS 'ServerOperatingSystem.Id'
         , SP.Id AS 'ServicePack.Id'
         , SOS.Name AS 'ServerOperatingSystem.Name'
         , SP.Name AS 'ServicePack.Name'
         , SP.VersionString
         , SP.SearchString
         , SP.Major
         , SP.Minor
         , SP.Revision
         , SP.Build
         , SP.Other
         , SP.Notes
         , SP.ExtendedProperties
         , SP.IsSystemDefined
         , SP.SentToNLS
         , SP.InternalId
         , 0 AS 'LicenseCount'
         , CASE SOS.IsApproved
             WHEN CAST(1 AS BIT) THEN SP.IsApproved
             ELSE CAST(0 AS BIT)
           END AS 'IsApproved'
           , SP.IsApproved AS [HasUnapprovedServicePack]
         , core.CountOSInstallations(SOS.Id, SP.Id, 1) AS 'InstallationCount'
            , 0 as ServicePackCount
       FROM
           core.Manufacturer AS M
       INNER JOIN core.ServerOperatingSystem AS SOS
           ON M.Id = SOS.ManufacturerId
       INNER JOIN core.ServicePack AS SP
           ON SOS.Id = SP.ServerOperatingSystemId
       CROSS APPLY (
                     SELECT
                       CONVERT(VARCHAR(256), ISNULL(MAN.[Name], '')) + ',' AS [text()]
                     FROM
                       core.ManufacturerAlternateName AS MAN
                     WHERE
                       MAN.ManufacturerId = M.Id
                   FOR
                     XML PATH('')
                   ) AS temp ( [Names] )
       ORDER BY
           SOS.VersionString
   
   END
GO
Uses
