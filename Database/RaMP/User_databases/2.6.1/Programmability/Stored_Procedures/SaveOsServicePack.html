CREATE PROCEDURE [settings].[SaveOsServicePack]
   @Id UNIQUEIDENTIFIER = NULL
 , @OperatingSystemId UNIQUEIDENTIFIER = NULL
 , @OperatingSystem VARCHAR(256) = NULL
 , @Name NVARCHAR(256) = ''
 , @VersionString VARCHAR(256) = NULL
 , @SearchString VARCHAR(128) = NULL
 , @Major VARCHAR(16) = NULL
 , @Minor VARCHAR(16) = NULL
 , @Revision VARCHAR(16) = NULL
 , @Build VARCHAR(16) = NULL
 , @Other VARCHAR(16) = NULL
 , @Notes VARCHAR(MAX) = NULL
 , @ExtendedProperties XML = NULL
 , @IsApproved BIT = 1
AS
   BEGIN
       
       IF ( @OperatingSystemId IS NULL )
           BEGIN
               IF ( @OperatingSystem IS NULL )
                   BEGIN
                       SET @OperatingSystem = 'Unknown' ;
                   END
               SELECT TOP 1
                   @OperatingSystemId = SOS.Id
               FROM
                   core.ServerOperatingSystem AS SOS
               WHERE
                   SOS.Name = @OperatingSystem ;
           END
           
       IF ( @Id IS NULL )
           BEGIN
               SELECT TOP 1
                   @Id = SP.Id
               FROM
                   core.ServicePack AS SP
               WHERE
                   SP.ServerOperatingSystemId = @OperatingSystemId
                   AND SP.Name = @Name ;
           END
               
       IF ( @Id IS NOT NULL )
           BEGIN
               
               UPDATE
                   core.ServicePack
               SET
                   ServerOperatingSystemId = @OperatingSystemId
                 , [Name] = @Name
                 , Major = @Major
                 , Minor = @Minor
                 , Revision = @Revision
                 , Build = @Build
                 , Other = @Other
                 , Notes = @Notes
                 , ExtendedProperties = @ExtendedProperties
                 , IsApproved = @IsApproved
               OUTPUT
                   INSERTED.Id
               WHERE
                   Id = @Id ;                
           END
                    
       ELSE
           BEGIN
               
               INSERT  core.ServicePack
                       (
                         ServerOperatingSystemId
                       , Name
                       , VersionString
                       , SearchString
                       , Major
                       , Minor
                       , Revision
                       , Build
                       , Other
                       , Notes
                       , ExtendedProperties
                       , IsSystemDefined
                       , IsApproved
                       )
               OUTPUT  INSERTED.Id
               VALUES
                       (
                         @OperatingSystemId
                       , @Name
                       , @VersionString
                       , @SearchString
                       , @Major
                       , @Minor
                       , @Revision
                       , @Build
                       , @Other
                       , @Notes
                       , @ExtendedProperties
                       , 0
                       , @IsApproved        
                       )
           END
       
   END
GO
