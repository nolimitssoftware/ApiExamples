CREATE PROC settings.RetrieveSoftwareIdCollectionWithExceededLicenseCount
AS
   BEGIN
       DECLARE @MasterMessageId UNIQUEIDENTIFIER = (
                                                     SELECT
                                                       MM.Id
                                                     FROM
                                                       log.MessageMaster AS MM
                                                       join log.InternalMessageToMessageMaster on mm.id = InternalMessageToMessageMaster.Id
                                                     WHERE
                                                       InternalMessageToMessageMaster.Description = 'installationsExceededSoftwareLicence'
                                                   ) ;
       DECLARE @Root UNIQUEIDENTIFIER = (
                                          SELECT
                                           A.AssetId
                                          FROM
                                           core.Asset AS A
                                          WHERE
                                           A.AssetNode = HIERARCHYID::GetRoot()
                                        ) ;
                                        
       SELECT DISTINCT
           S.SoftwareId
         , S.Name
         , S.Approved
         , S.NumberOfLicenses
         , S.NumberOfInstallations
       FROM
           core.Application AS A
       INNER JOIN settings.Software AS S
           ON A.Name = S.Name            
       WHERE            
           S.SoftwareId NOT IN (
           SELECT
               M.ContextId
           FROM
               log.Message AS M
           WHERE
               M.AssetId = @Root
               AND M.Context = 64 --Software
               AND M.MessageMasterId = @MasterMessageId
               AND M.EndTimestamp IS NULL )
               AND S.NumberOfLicenses > 0
               AND S.NumberOfLicenses < S.NumberOfInstallations
       
   END
GO
