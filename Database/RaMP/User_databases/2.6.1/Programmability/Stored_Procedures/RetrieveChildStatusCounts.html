CREATE PROCEDURE [core].[RetrieveChildStatusCounts]
   @ParentId UNIQUEIDENTIFIER = NULL
 , @Level INT = 0
AS
   BEGIN
       DECLARE @ParentNode HIERARCHYID ;

       IF ( @ParentId IS NOT NULL )
           BEGIN
               SELECT
                   @ParentNode = A.AssetNode
               FROM
                   core.Asset AS A
               WHERE
                   A.AssetId = @ParentId ;
           END
       ELSE
           BEGIN
               SELECT
                   @ParentNode = HIERARCHYID::GetRoot() ;
           END
   
       SELECT
           COALESCE(S.Status, 65536) AS 'Status'
         , COUNT(COALESCE(S.Status, 65536)) AS 'Count'
       FROM
           (
             SELECT
               A.[Status] AS Status
             FROM
               core.Asset AS A
             WHERE
               A.AssetNode.IsDescendantOf(@ParentNode) = 1
               AND core.IsInExcludedAssetNode(A.AssetNode) = 0
               AND (
                     @Level = 0
                     OR (
                          @Level > 0
                          AND A.AssetLevel = @ParentNode.GetLevel() + @Level
                        )
                   )
               AND A.Type NOT IN ('unknown', 'physicalGroup')
           ) AS S
       GROUP BY
           S.Status ;
   END
GO
