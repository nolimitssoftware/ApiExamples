CREATE PROCEDURE [search].[RetrieveSavedSearchCollection]
   @UserId UNIQUEIDENTIFIER
AS
   BEGIN
       SELECT
           SS.Id
         , SS.UserId
         , SS.Name
         , SS.AssetType
         , SS.AssetId
         , COALESCE(A.OverrideName, A.ShortName, A.Name ) AS [Location]
         , A.AssetNode.ToString() AS [Path]
         , SS.IsGlobal
       FROM
           search.SavedSearch AS SS
       LEFT JOIN core.Asset AS A
           ON SS.AssetId = A.AssetId
       WHERE
           SS.UserId = @UserId;
   END
GO
