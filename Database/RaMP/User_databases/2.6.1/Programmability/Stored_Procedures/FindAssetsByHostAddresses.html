CREATE PROCEDURE [core].[FindAssetsByHostAddresses]
   @HostAddresses VARCHAR(MAX) = NULL
 , @Delim NCHAR(1) = ','
AS
   BEGIN
       DECLARE @SearchResults TABLE ( Id UNIQUEIDENTIFIER ) ;
                   INSERT  @SearchResults
               (
                 Id
               )
               SELECT DISTINCT
                   H.AssetId
               FROM
                   core.Host AS H
               INNER JOIN util.split_delimiter(@HostAddresses, @Delim) AS SD
                   ON H.[Address] = SD.STR;
               --UNION
               --SELECT
               --    VM.DeviceId
               --FROM
               --    core.VirtualMachine AS VM
               --INNER JOIN util.split_delimiter(@HostAddresses, @Delim) AS SD
               --    ON VM.[Address] = SD.str
               --WHERE
               --    VM.DeviceId IS NOT NULL ;   
       
       SELECT
           SR.Id
         , A.Name
         , A.ShortName
         , A.OverrideName
         , A.Type
       FROM
           @SearchResults AS SR
       INNER JOIN core.Asset AS A
           ON SR.Id = A.AssetId ;       
   END
GO
