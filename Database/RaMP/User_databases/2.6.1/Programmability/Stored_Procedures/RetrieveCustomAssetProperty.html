CREATE PROC core.RetrieveCustomAssetProperty
   @AssetId UNIQUEIDENTIFIER
 , @CustomPropertyId UNIQUEIDENTIFIER
AS
   BEGIN

       SELECT DISTINCT
           CP.Id
         , CP.Name
         , CP.Type
         , CP.[Values]
         , CP.Unit
         , ISNULL(CP.AssetTypes, -1) AS AssetTypes
         , CP.IsActive
         , CP.IsSearchField
         , CP.DefaultValue
         , CP.DefaultValueType
         , CAP.UserValue
         , SQL_VARIANT_PROPERTY(CAP.UserValue, 'basetype') AS 'UserValueType'
       FROM
           ( SELECT DISTINCT
               CP.Id
             , CP.Name
             , CP.Type
             , CP.[Values]
             , CP.Unit
             , CP.AssetTypes
             , CP.IsActive
             , CP.IsSearchField
             , CP.DefaultValue
             , SQL_VARIANT_PROPERTY(CP.DefaultValue, 'basetype') AS 'DefaultValueType'
             FROM
               settings.CustomProperty AS CP
             LEFT JOIN core.CustomAssetProperty AS CAP
               ON CP.Id = CAP.CustomPropertyId
             WHERE
               CP.Id = @CustomPropertyId
           ) AS CP
       LEFT JOIN core.CustomAssetProperty AS CAP
           ON CP.Id = CAP.CustomPropertyId
              AND CAP.AssetId = @AssetId
       ORDER BY
           CP.Name;
   END
GO
