CREATE PROC core.SaveProcessor
   @Id UNIQUEIDENTIFIER = NULL
 , @AssetId UNIQUEIDENTIFIER
 , @Name NVARCHAR(256)
 , @Model NVARCHAR(256)
 , @Family NVARCHAR(256) = NULL
 , @MaxClockSpeed INT = NULL
 , @Cores TINYINT
 , @PowerManagementSupported BIT = NULL
 , @Source VARCHAR(32)
 , @ExtendedProperties XML
 , @AgentId UNIQUEIDENTIFIER = NULL
 , @PropertyXml XML = NULL
AS
   BEGIN
       IF ( @Id IS NULL )
           BEGIN
     
               INSERT  core.Processor
                       (
                         AssetId
                       , Name
                       , Model
                       , Family
                       , MaxClockSpeed
                       , Cores
                       , PowerManagementSupported
                       , [Source]
                       , ExtendedProperties
                       , AgentId
                       , PropertyXml                      
                         )
               OUTPUT  INSERTED.Id
               VALUES
                       (
                         @AssetId
                       , ISNULL(@Name, @Model)
                       , @Model
                       , @Family
                       , @MaxClockSpeed
                       , @Cores
                       , @PowerManagementSupported
                       , @Source
                       , @ExtendedProperties
                       , @AgentId
                       , @PropertyXml                      
                         )
           END
       ELSE
           BEGIN
       
               UPDATE
                   core.Processor
               SET
                   Name = @Name
                 , Model = @Model
                 , Family = @Family
                 , MaxClockSpeed = @MaxClockSpeed
                 , Cores = @Cores
                 , PowerManagementSupported = @PowerManagementSupported
                 , ExtendedProperties = @ExtendedProperties
                 , AgentId = @AgentId
                 , PropertyXml = @PropertyXml
               OUTPUT
                   INSERTED.Id
               WHERE
                   core.Processor.Id = @Id ;
       
           END
           
       SELECT
           @Id ;
   END
GO
