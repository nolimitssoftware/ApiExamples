CREATE PROCEDURE [settings].[RetrieveDiscoverySchedule]
   @DiscoveryScheduleId UNIQUEIDENTIFIER
AS
   BEGIN
       SELECT
           DS.DiscoveryScheduleId
         , DS.DiscoveryId
         , DS.Description
         , DS.ScheduleType
         , DS.CronExpression
         , DS.LastRun
       FROM
           settings.DiscoverySchedule AS DS
       WHERE
           DS.DiscoveryScheduleId = @DiscoveryScheduleId
   END
GO
