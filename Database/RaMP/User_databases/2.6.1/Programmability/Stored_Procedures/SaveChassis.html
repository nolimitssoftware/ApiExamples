CREATE PROC core.SaveChassis
   @Id UNIQUEIDENTIFIER = NULL
 , @AssetId UNIQUEIDENTIFIER
 , @HasIntrusionDetection BIT
 , @HasFrontPanelLockout BIT
 , @HasDiagnosticInterrupt BIT
 , @HasPowerInterlock BIT
 , @Source VARCHAR(32)
 , @ExtendedProperties XML
 , @AgentId UNIQUEIDENTIFIER = NULL
 , @PropertyXml XML = NULL
AS
   BEGIN
       IF ( @Id IS NULL )
           BEGIN                
               INSERT  core.Chassis
                       (
                         AssetId
                       , HasIntrusionDetection
                       , HasFrontPanelLockout
                       , HasDiagnosticInterrupt
                       , HasPowerInterlock
                       , [Source]
                       , ExtendedProperties
                       , AgentId
                       , PropertyXml                      
                         )
               OUTPUT  INSERTED.Id
               VALUES
                       (
                         @AssetId
                       , @HasIntrusionDetection
                       , @HasFrontPanelLockout
                       , @HasDiagnosticInterrupt
                       , @HasPowerInterlock
                       , @Source
                       , @ExtendedProperties
                       , @AgentId
                       , @PropertyXml                      
                         )
           END    
       ELSE
           BEGIN
               UPDATE
                   core.Chassis
               SET
                   HasIntrusionDetection = @HasIntrusionDetection
                 , HasFrontPanelLockout = @HasFrontPanelLockout
                 , HasDiagnosticInterrupt = @HasDiagnosticInterrupt
                 , HasPowerInterlock = @HasPowerInterlock
                 , ExtendedProperties = @ExtendedProperties
                 , AgentId = @AgentId
                 , PropertyXml = @PropertyXml
               OUTPUT
                   INSERTED.Id
               WHERE
                   core.Chassis.Id = @Id ;
           END
           
       SELECT
           @Id ;
           
   END
GO
