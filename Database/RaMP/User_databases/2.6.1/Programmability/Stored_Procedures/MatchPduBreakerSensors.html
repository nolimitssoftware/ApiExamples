CREATE PROCEDURE [core].[MatchPduBreakerSensors]
   @PduId UNIQUEIDENTIFIER
AS
   BEGIN
       UPDATE
           core.[PduBreaker]
       SET
           AmpSensorId = S.SensorId
       FROM
           [core].[PduBreaker] AS PB
       LEFT JOIN
           core.Sensor AS S ON S.DeviceId = PB.AssetId AND PB.Number = S.Element AND S.[Type] = 'breakerCurrent'
       WHERE
           PB.AssetId = @PduId
   END
GO
