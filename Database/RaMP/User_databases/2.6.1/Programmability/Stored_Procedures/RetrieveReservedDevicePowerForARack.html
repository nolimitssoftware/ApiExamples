CREATE PROCEDURE core.RetrieveReservedDevicePowerForARack
   @RackId UNIQUEIDENTIFIER
AS
   BEGIN
       DECLARE @AssetNode HIERARCHYID
       DECLARE @AssetLevel SMALLINT

       SELECT
           @AssetNode = A.AssetNode,
           @AssetLevel = A.AssetLevel
       FROM
           core.Asset AS A
       WHERE
           A.AssetId = @RackId
       
       SELECT
           SUM(COALESCE(M.UserPower, M.[Power], 0))
       FROM
           core.Asset AS A
           INNER JOIN core.Model AS M ON A.ModelId = M.Id
       WHERE
           A.LifecycleStatus = 1
           AND
           A.AssetNode.IsDescendantOf(@AssetNode) = 1
           AND
           A.AssetLevel > 1
   END
GO
