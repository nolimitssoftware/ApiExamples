CREATE PROC core.RetrieveAvailablePowerSourceCollection
    @AssetId UNIQUEIDENTIFIER
AS
   BEGIN
       --In ticket:1276, it was decided that this will only return power devices that are in the same rack as the given asset.
       --    It will return nothing if the given asset is not in a rack.
    
       DECLARE @CurrentNode HIERARCHYID ;
       DECLARE @ParentNode HIERARCHYID ;
       DECLARE @IsInRack BIT = 0 ;
       DECLARE @GivenAssetType VARCHAR(32) ;
       DECLARE @RackNode HIERARCHYID ;
   
       SELECT
           @CurrentNode = A.AssetNode
           , @GivenAssetType = A.[Type]
       FROM
           core.Asset AS A
       WHERE
           A.AssetId = @AssetId ;
   
       IF ( @CurrentNode IS NOT NULL )
           BEGIN
               SET @ParentNode = @CurrentNode.GetAncestor(1) ;
               DECLARE @ParentType VARCHAR(32) = (
                                                           SELECT TOP 1
                                                               A.Type
                                                           FROM
                                                               core.Asset AS A
                                                           WHERE
                                                               A.AssetNode = @ParentNode
                                                       ) ;
               IF ( @ParentType = 'rack' )
                   BEGIN
                       SELECT
                           @IsInRack = 1
                           , @RackNode = @ParentNode ;
           
                   END
           END
   
       IF (
           @ParentNode IS NOT NULL
           AND @IsInRack = 1
       )
       BEGIN
           DECLARE @Assets TABLE ( Id UNIQUEIDENTIFIER ) ;
           INSERT  @Assets
               (
               Id
                    
               )
               SELECT
                   A.AssetId
               FROM
                   core.Asset AS A
               WHERE
                   A.AssetId <> @AssetId
                   AND A.AssetNode.IsDescendantOf(@RackNode) = 1
                   AND core.IsInExcludedAssetNode(A.AssetNode) = 0
                   AND A.AssetLevel = @RackNode.GetLevel() + 1
                   AND A.Type IN ('rpdu', 'ups', 'pdu', 'rackOrTowerUps', 'powerMeter')
               
           SELECT
               GA.*
               , GA.AssetNode.ToString() AS 'AssetPath'
           FROM
               @Assets AS A
           CROSS APPLY core.GetAsset(A.Id) AS GA
           ORDER BY
               GA.OverrideName
               , GA.ShortName
               , GA.Name ;
       END
             
   END
GO
