CREATE PROC settings.RetrieveModelByManufacturer
    @ManufacturerId UNIQUEIDENTIFIER = NULL
 , @Manufacturer VARCHAR(256) = NULL
 , @Type VARCHAR(32) = NULL
 , @Limit INT = NULL
 , @Start INT
 , @IsAscending BIT = 1
 , @SortField VARCHAR(36) = NULL
 , @IsDebug BIT = 1
 , @TotalCount INT OUTPUT  
AS
BEGIN
         
    SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED ;
         
    IF (
           @ManufacturerId IS NULL
           AND @Manufacturer IS NOT NULL
        )
         BEGIN
               SELECT
                    @ManufacturerId = M.Id
               FROM
                    core.Manufacturer AS M
               LEFT JOIN core.ManufacturerAlternateName AS MAN
                    ON M.Id = MAN.ManufacturerId
               WHERE
                    M.Name = @Manufacturer
                    OR MAN.Name = @Manufacturer ;
         END
         
 
    IF (
           @ManufacturerId IS NULL
           AND @Manufacturer IS NOT NULL
        )
         BEGIN
           --If the manufacturer string was provided but we could not determine an existing manufacture ID then return nothing for models...since there isn't any.
           RETURN;
         END

    DECLARE @End INT;
   
IF(@Limit IS NOT NULL)
   SET @End = @Start + @Limit - 1 ;

    DECLARE @SortDir CHAR(6) ;
    DECLARE @OrderBy VARCHAR(128) ;        
       
    IF ( @IsAscending = 1 )
         SET @SortDir = ' ASC  ' ;
    ELSE
         SET @SortDir = ' DESC ' ;
               
       
    SET @OrderBy = ' M2.Name ' + @SortDir + ' , COALESCE(M.UserName, M.Name) ' + @SortDir + ' ' ;
         
    DECLARE @sql NVARCHAR(MAX) ;
    DECLARE @params NVARCHAR(4000) ;
         
    SET @sql = ' DECLARE @Models TABLE ( Id UNIQUEIDENTIFIER ) ; '
         
    SET @sql += '
    INSERT  @Models
               (
                 Id
               )
               SELECT DISTINCT
                    M.Id
               FROM
                    core.Model AS M
               INNER JOIN core.Manufacturer AS M2
                    ON M.ManufacturerId = M2.Id
               WHERE 1 = 1 ' ;
    IF ( @ManufacturerId IS NOT NULL )
         SET @Sql += ' AND M.ManufacturerId = @ManufacturerId ' ;
   
    IF ( @Type IS NOT NULL )
         SET @sql += ' AND M.[Type] = @Type '
         
    SET @sql += ' AND M.Name <> ''Unknown'';' ;
    
    SET @sql += ' SELECT @TotalCount = COUNT(M.Id) FROM @Models AS M;'
    
    SET @sql += '
    SELECT
       Manufacturer
       , Id
       , ManufacturerId
       , Name
       , UserName
       , PartNumber
       , UserPartNumber
       , Type
       , Power
       , UserPower
       , [Current]
       , Height
       , Depth
       , Width
       , Weight
       , RackUnits
       , Notes
       , ImagePath
       , ExtendedProperties
       , IsSystemDefined
       , SupportsRaMP
   FROM (SELECT
               M2.Name AS ''Manufacturer''
            , M.Id
            , M.ManufacturerId
            , M.Name
            , M.UserName
            , M.PartNumber
            , M.UserPartNumber
            , M.Type
            , M.Power
            , M.UserPower
            , M.[Current]
            , M.Height
            , M.Depth
            , M.Width
            , M.Weight
            , M.RackUnits
            , M.Notes
            , M.ImagePath
            , M.ExtendedProperties
            , M.IsSystemDefined
            , M.SupportsRaMP
            , ROW_NUMBER() OVER(ORDER BY ' + @OrderBy
         + ' ) AS ''row_num''        
               FROM
               @Models AS M3
           INNER JOIN
                    core.Model AS M
                    ON M.Id = M3.Id
               INNER JOIN core.Manufacturer AS M2
                    ON M.ManufacturerId = M2.Id
               ) AS temp '
       IF(@End IS NOT NULL)
           SET @sql += ' WHERE row_num BETWEEN @Start AND @End '

       SET @sql += '
       ORDER BY Row_Num ASC;' ;
                   
    IF ( @IsDebug = 1 )
         BEGIN
               PRINT @Sql ;
         END
         
    
    SET @params = '
         @ManufacturerId UNIQUEIDENTIFIER,           
         @Type VARCHAR(32),           
         @Start INT,
         @End INT,
         @TotalCount INT OUTPUT' ;
    EXEC sp_executesql
         @sql
       , @params
       , @ManufacturerId
       , @Type
       , @Start
       , @End
       , @TotalCount = @TotalCount OUTPUT ;
END
GO
