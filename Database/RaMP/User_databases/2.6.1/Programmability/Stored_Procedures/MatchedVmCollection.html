CREATE PROCEDURE [core].[MatchedVmCollection]
   @AssetId UNIQUEIDENTIFIER
AS
   BEGIN
       SET NOCOUNT ON;

       DECLARE @MatchedAssets TABLE
           (
             MatchedAssetId UNIQUEIDENTIFIER
           , VirtualId UNIQUEIDENTIFIER
           , CurrentAssetId UNIQUEIDENTIFIER
           )
 

       INSERT  @MatchedAssets
               SELECT DISTINCT
                   SN.AssetId
                 , VM.Id
                 , VM.DeviceId
               FROM
                   core.VirtualMachine AS VM
               INNER JOIN ( SELECT
                               core.RetrieveGuidFromVWSerial(SN.SerialNumber) AS 'UUID'
                             , SN.AssetId
                            FROM
                               core.SerialNumber AS SN
                            WHERE
                               LOWER(SUBSTRING(SN.SerialNumber, 0, 7)) = 'vmware'
                          ) AS SN
                   ON VM.[Guid] = SN.UUID
               WHERE
                   SN.AssetId = @AssetId
               UNION
               SELECT
                   A.AssetId
                 , G.Id
                 , G.DeviceId
               FROM
                   core.Asset AS A
               INNER JOIN ( SELECT
                               VM.Id
                             , VM.DeviceId
                             , G.AssetId
                            FROM
                               core.VirtualMachine AS VM
                            INNER JOIN core.Guid AS G
                               ON G.Id = VM.Guid
                            WHERE
                               VM.AssetId = @AssetId
                          ) AS G
                   ON A.AssetId = G.AssetId
               UNION
               SELECT --SOME WMI Servers report a different GUID, so use the serial number
                   A.AssetId
                 , MSN.Id
                 , MSN.DeviceId
               FROM
                   core.Asset AS A
               INNER JOIN ( SELECT
                               VM.Id
                             , VM.DeviceId
                             , SN.AssetId
                            FROM
                               core.VirtualMachine AS VM
                            INNER JOIN core.SerialNumber AS SN
                               ON VM.SerialNumber = SN.SerialNumber
                            WHERE
                               VM.AssetId = @AssetId
                          ) AS MSN
                   ON A.AssetId = MSN.AssetId
       INSERT  @MatchedAssets
               SELECT
                   A.AssetId
                 , HN.Id
                 , HN.DeviceId
               FROM
                   core.Asset AS A
               INNER JOIN ( SELECT
                               VM.Id
                             , VM.DeviceId
                             , A2.AssetId
                            FROM
                               core.VirtualMachine AS VM
                            INNER JOIN core.Asset AS A2
                               ON A2.Name = VM.Hostname
                            WHERE
                               VM.AssetId = @AssetId
                               AND core.IsNameTooGeneric(VM.Hostname) = 0
                          ) AS HN
                   ON A.AssetId = HN.AssetId

       INSERT  @MatchedAssets
               SELECT
                   A.AssetId
                 , N.Id
                 , N.DeviceId
               FROM
                   core.Asset AS A
               INNER JOIN ( SELECT
                               VM.Id
                             , VM.DeviceId
                             , A2.AssetId
                             , VM.Name
                            FROM
                               core.VirtualMachine AS VM
                            INNER JOIN core.Asset AS A2
                               ON A2.Name = VM.Name
                            WHERE
                               VM.AssetId = @AssetId
                               AND core.IsNameTooGeneric(VM.Name) = 0
                               AND A2.AssetId = @AssetId
                          ) AS N
                   ON A.AssetId = N.AssetId

     
       SELECT DISTINCT
           MA.MatchedAssetId AS [MatchedAssetId]
         , MA.VirtualId AS [VmId]
         , MA.CurrentAssetId AS [VmAssetId]
         , A2.AssetId AS [VmAssetParentId]
       FROM
           @MatchedAssets AS MA
       INNER JOIN core.Asset AS A
           ON A.AssetId = MA.MatchedAssetId
       INNER JOIN core.Asset AS A2
           ON A2.AssetNode = A.AssetNode.GetAncestor(1);

   END
GO
