CREATE PROC [notification].[InsertMessageNotification]
   @Id UNIQUEIDENTIFIER
 , @NotificationGroupId UNIQUEIDENTIFIER
 , @NotificationGroupType TINYINT
AS
   BEGIN
       IF ( @NotificationGroupType = 1 )
           BEGIN            
               DECLARE @Result TABLE ( Id UNIQUEIDENTIFIER )
               INSERT  @Result
                       SELECT
                           N.Id
                       FROM
                           notification.Notification AS N
                       INNER JOIN settings.NotificationGroupRecipient
                           AS NGR
                           ON N.NotificationTypeId = NGR.Id
                       WHERE
                           N.NotificationGroupId = @NotificationGroupId
                           AND N.NotificationType = @NotificationGroupType
                           AND NGR.Id NOT IN (
                           SELECT
                               N.NotificationTypeId
                           FROM
                               notification.MessageNotification
                               AS MN
                           INNER JOIN [notification].[Notification]
                               AS N
                               ON MN.NotificationId = N.Id
                           WHERE
                               MN.MessageId = @Id
                               AND N.NotificationGroupId = @NotificationGroupId
                               AND N.NotificationType = @NotificationGroupType );
               BEGIN TRANSACTION
               BEGIN TRY
               
                   INSERT  notification.Notification
                           ( NotificationGroupId
                           , NotificationType
                           , NotificationTypeId
                           , NewMessagesSetting                                         
                           )
                   OUTPUT  INSERTED.Id
                           INTO @Result
                           SELECT
                               @NotificationGroupId
                             , @NotificationGroupType
                             , NGR.Id
                             , NG.NewMessages
                           FROM
                               settings.NotificationGroupRecipient
                               AS NGR
                           INNER JOIN settings.NotificationGroup
                               AS NG
                               ON NGR.NotificationGroupId = NG.Id
                           WHERE
                               NG.Id = @NotificationGroupId
                               AND ( NG.Types
                                     & @NotificationGroupType ) = @NotificationGroupType
                               AND NGR.Id NOT IN (
                               SELECT
                                   N.NotificationTypeId
                               FROM
                                   notification.MessageNotification
                                   AS MN
                               INNER JOIN [notification].[Notification]
                                   AS N
                                   ON MN.NotificationId = N.Id
                               WHERE
                                   MN.MessageId = @Id
                                   AND N.NotificationGroupId = @NotificationGroupId
                                   AND N.NotificationType = @NotificationGroupType );
      
                   INSERT  [notification].MessageNotification
                           ( NotificationId
                           , MessageId
                           , LastNotifiedStatus
                           , LastNotified
                           , LastNotifiedMessage
                           )
                           SELECT
                               R.Id
                             , @Id
                             , NULL
                             , NULL
                             , NULL
                             --, 2
                             --, ISNULL(( SELECT
                             --              MAX(MN.LastNotified)
                             --           FROM
                             --              notification.MessageNotification
                             --              AS MN
                             --           WHERE
                             --              MN.NotificationId = R.Id
                             --         ), GETUTCDATE())
                             --, 'Initial Watch'
                           FROM
                               @Result AS R
               END TRY
               BEGIN CATCH
                   SELECT
                       ERROR_NUMBER() AS ErrorNumber
                     , ERROR_SEVERITY() AS ErrorSeverity
                     , ERROR_STATE() AS ErrorState
                     , ERROR_PROCEDURE() AS ErrorProcedure
                     , ERROR_LINE() AS ErrorLine
                     , ERROR_MESSAGE() AS ErrorMessage;

                   IF ( @@TRANCOUNT > 0 )
                       ROLLBACK TRANSACTION;

               END CATCH;
 
               IF ( @@TRANCOUNT > 0 )
                   COMMIT TRANSACTION;
 
           END
   END
GO
