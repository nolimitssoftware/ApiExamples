CREATE PROCEDURE core.RetrieveUnknownGroupAssets
 @Start INT = 1
 , @Limit INT = NULL
 , @TotalCount INT OUTPUT
AS
BEGIN
   
   DECLARE @UnknownGroupHierarchyid hierarchyid = 0x58

   IF ( @Limit IS NULL )
       BEGIN
           SET @Limit = 2500 ;
       END
   DECLARE @End INT = @Start + @Limit - 1 ;
  
   SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED ;
   SET NOCOUNT ON ;
  
   SELECT
       @TotalCount = COUNT(A.AssetId)
   FROM
       core.Asset AS A
   WHERE
       A.AssetNode.IsDescendantOf(@UnknownGroupHierarchyid) = 1
       AND
       A.AssetLevel = 2 ;  
               
   SELECT
       Name
       ,ShortName
       ,OverrideName
       ,Type
       ,Status
       ,AssetId
       ,LastLocation
       ,ManufacturerName
       ,ModelName
   FROM
       (
           SELECT
               A.Name
               ,A.ShortName
               ,A.OverrideName
               ,A.Type
               ,A.Status
               ,A.AssetId
               ,LL.LastLocation
               ,MAN.Name AS 'ManufacturerName'
               ,M.Name AS 'ModelName'
               , ROW_NUMBER() OVER ( ORDER BY A.OverrideName, A.ShortName ) AS 'row_num'
           FROM
               core.Asset AS A
               LEFT JOIN core.Model AS M ON A.ModelId = M.Id
               LEFT JOIN core.Manufacturer AS MAN ON M.ManufacturerId = MAN.Id
               LEFT JOIN core.LastLocation AS LL ON A.AssetId = LL.AssetId
           WHERE
               A.AssetNode.IsDescendantOf(@UnknownGroupHierarchyid) = 1
               AND
               A.AssetLevel = 2
       ) AS temp
   WHERE
       temp.row_num BETWEEN @Start AND @End
   ORDER BY
       temp.row_num

END
GO
