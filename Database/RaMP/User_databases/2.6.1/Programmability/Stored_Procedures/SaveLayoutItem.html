CREATE PROCEDURE [visual].[SaveLayoutItem]
   @Id UNIQUEIDENTIFIER = NULL
 , @LayoutId UNIQUEIDENTIFIER
 , @LayoutItemType SMALLINT
 , @PlacedId VARCHAR(128) = NULL
 , @Name VARCHAR(128) = NULL
 , @X REAL = NULL
 , @Y REAL = NULL
 , @XOffset REAL = NULL
 , @YOffset REAL = NULL
 , @Radius REAL = NULL
 , @Width REAL = NULL
 , @Height REAL = NULL
 , @PhysicalWidth REAL = NULL
 , @PhysicalHeight REAL = NULL
 , @Color CHAR(7) = NULL
 , @Direction TINYINT = 0
AS
   BEGIN
       IF ( @Id IS NULL )
           BEGIN
               INSERT  visual.LayoutItem
                       (
                         LayoutId
                       , LayoutItemType
                       , PlacedId
                       , Name
                       , X
                       , Y
                       , XOffset
                       , YOffset
                       , Radius
                       , Width
                       , Height
                       , PhysicalWidth
                       , PhysicalHeight
                       , Color
                       , Direction
                     
                               
                       )
               VALUES
                       (
                         @LayoutId
                       , @LayoutItemType
                       , @PlacedId
                       , @Name
                       , @X
                       , @Y
                       , @XOffset
                       , @YOffset
                       , @Radius
                       , @Width
                       , @Height
                       , @PhysicalWidth
                       , @PhysicalHeight
                       , @Color
                       , @Direction                    
                               
                       ) ;
           END
       ELSE
           BEGIN
               IF EXISTS ( SELECT
                               1
                           FROM
                               visual.LayoutItem AS LI
                           WHERE
                               LI.Id = @Id )
                   BEGIN
                       UPDATE
                           visual.LayoutItem
                       SET
                           LayoutItemType = @LayoutItemType
                         , PlacedId = @PlacedId
                         , Name = @Name
                         , X = @X
                         , Y = @Y
                         , XOffset = @XOffset
                         , YOffset = @YOffset
                         , Radius = @Radius
                         , Width = @Width
                         , Height = @Height
                         , Color = @Color
                         , Direction = @Direction
                       WHERE
                           visual.LayoutItem.Id = @Id ;
                   END
               ELSE
                   BEGIN
                       INSERT  visual.LayoutItem
                               (
                                 Id
                               , LayoutId
                               , LayoutItemType
                               , PlacedId
                               , Name
                               , X
                               , Y
                               , XOffset
                               , YOffset
                               , Radius
                               , Width
                               , Height
                               , PhysicalWidth
                               , PhysicalHeight
                               , Color
                               , Direction
                     
                                         
                               )
                       VALUES
                               (
                                 @Id
                               , @LayoutId
                               , @LayoutItemType
                               , @PlacedId
                               , @Name
                               , @X
                               , @Y
                               , @XOffset
                               , @YOffset
                               , @Radius
                               , @Width
                               , @Height
                               , @PhysicalWidth
                               , @PhysicalHeight
                               , @Color
                               , @Direction                    
                                         
                               ) ;
                   END
           END
   END
GO
