CREATE PROC core.MoveAsset
--DECLARE
    @AssetId UNIQUEIDENTIFIER ,
    @NewParentId UNIQUEIDENTIFIER ,
    @IsDebug BIT = 0
AS --SELECT
--@AssetId = 'BDF4D5CE-624B-4914-9F20-42E1AA903CBB',
----@AssetId = '80FAE519-0D60-4099-A1FD-1412347B714D',
--    @NewParentId = '00000010-1B16-6786-583A-00076E0228C9',
--    --@NewParentId = 'BDF4D5CE-624B-4914-9F20-42E1AA903CBB',
--    @IsDebug = 1;
    BEGIN
    
--Declare some variables for the positions
         DECLARE @NewParentNode HIERARCHYID ;
         DECLARE @AssetNode HIERARCHYID ;
         DECLARE @NewAssetNode HIERARCHYID ;
         DECLARE @AssetLastLocationString VARCHAR(MAX) ;

--Get the new parent node
         SELECT
               @NewParentNode = A.AssetNode
         FROM
               core.Asset AS A
         WHERE
               A.AssetId = @NewParentId

/*            
Work out the new position.  
Pretty easy to do this anyway for a single node, it just works out what the position WOULD be in the new place if a child node was added.  
*/
         SET @NewAssetNode = @NewParentNode.GetDescendant((
                                                                             SELECT
                                                                               MAX(A.AssetNode)
                                                                             FROM
                                                                               core.Asset AS A
                                                                             WHERE
                                                                               A.AssetNode.GetAncestor(1) = @NewParentNode
                                                                           ), NULL) ;

--Get current location first.  We need this later.   
         SELECT
               @AssetNode = A.AssetNode
         FROM
               core.Asset AS A
         WHERE
               A.AssetId = @AssetId ;
       
         --Get current location (in word form "Group1 >> Group2").  We need this later.   
         SELECT
               @AssetLastLocationString = A.Location
         FROM
               core.AssetView AS A
         WHERE
               A.AssetId = @AssetId ;

         IF ( @IsDebug = 1 )
               BEGIN
                    SELECT
                         @NewParentNode.ToString() AS '@NewParentNode' ;
                    SELECT
                         @AssetNode.ToString() AS '@AssetNode'
                    SELECT
                         AssetNode.GetReparentedValue(@AssetNode,
                                                                @NewParentNode.GetDescendant((
                                                                                 SELECT
                                                                                 MAX(A.AssetNode)
                                                                                 FROM
                                                                                 core.Asset AS A
                                                                                 WHERE
                                                                                 A.AssetNode.GetAncestor(1) = @NewParentNode
                                                                                 ), NULL)).ToString() AS 'ReparentedValue'
                    FROM
                         core.Asset AS A
                    WHERE
                         A.AssetNode.IsDescendantOf(@AssetNode) = 1 AND
                         AssetNode != @AssetNode ;
                    SELECT
                         @NewAssetNode.ToString() AS '@NewAssetNode'
               END
         ELSE
               BEGIN
/*
This is the magic bit that updates all of the child nodes under the current position (notice the first part of the outer where clause).
The clever bit is how GetReparentedValue is used on a set based operation rather than just dumbly ran with a cursor or while loop or other such nasty things.  
The last bit of the outer where clause stops this being applied to the top level node.
*/
                    UPDATE
                         core.Asset
                    SET
                         AssetNode = AssetNode.GetReparentedValue(@AssetNode,
                                                                                @NewParentNode.GetDescendant((
                                                                                 SELECT
                                                                                 MAX(A.AssetNode)
                                                                                 FROM
                                                                                 core.Asset AS A
                                                                                 WHERE
                                                                                 A.AssetNode.GetAncestor(1) = @NewParentNode
                                                                                 ), NULL))
                    WHERE
                         AssetNode.IsDescendantOf(@AssetNode) = 1 AND
                         AssetNode != @AssetNode ;

--Last trivial step is to set the position of the top node to the new position, if you were also making other changes to the row you'd do that here.
                    UPDATE
                         core.Asset
                    SET
                         AssetNode = @NewAssetNode
                    WHERE
                         core.Asset.AssetId = @AssetId ;
                         
                   --Clean up core.Location
                   DELETE FROM [core].[Location]
                       WHERE
                           [Id] = @AssetId
                           AND
                           [AssetId] <> @NewParentId

                   --This will make sure the ramp/rack relationship is correct.  (used with ramp monitoring)
                   EXECUTE [core].[UpdateRampManagerRackRelationship] @AssetId, @NewParentId
                   
                   --Update the Last Location table
                   IF EXISTS(SELECT 1 FROM core.LastLocation WITH (NOLOCK) WHERE AssetId = @AssetId)
                       BEGIN
                           UPDATE [core].[LastLocation]
                           SET
                               LastLocation = @AssetLastLocationString
                           WHERE
                               AssetId = @AssetId ;
                       END
                   ELSE
                       BEGIN
                           INSERT INTO [core].[LastLocation]
                               (AssetId, [LastLocation])
                           VALUES
                               (@AssetId, @AssetLastLocationString)
                       END
               END
    END
GO
