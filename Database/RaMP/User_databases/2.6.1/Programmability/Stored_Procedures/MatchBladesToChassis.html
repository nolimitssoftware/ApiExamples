CREATE PROCEDURE [core].[MatchBladesToChassis]
   @AssetId UNIQUEIDENTIFIER = NULL  -- AssetId of the chassis.
AS
   BEGIN
       DECLARE @MatchedBlades TABLE
           (
             BladeAssetId UNIQUEIDENTIFIER ,
             [Name] NVARCHAR(256) ,
             UUID NVARCHAR(256) NULL ,
             ChassisAssetId UNIQUEIDENTIFIER ,
             BayLocation TINYINT NULL ,
             [Row] INT IDENTITY(1, 1)
           )
         
       INSERT  INTO @MatchedBlades
               ( BladeAssetId ,
                 Name ,
                 ChassisAssetId ,
                 BayLocation
               )
               SELECT DISTINCT
                       BA.AssetId ,
                       BA.Name ,
                       CA.AssetId ,
                       BI.BayLocation
               FROM    core.Asset AS BA
                       INNER JOIN core.NetworkInterface AS BH ON BA.AssetId = BH.AssetId
                       LEFT JOIN core.Blade AS BI ON BH.Mac = BI.BmcMac
                                                     OR BH.Mac = BI.Mac1
                                                     OR BH.Mac = BI.Mac2
                       INNER JOIN core.Asset AS CA ON BI.AssetId = CA.AssetId
               WHERE   ( @AssetId IS NULL OR CA.AssetId = @AssetId )
                       AND BA.[Type] IN ( 'bladeServer', 'bladeNetwork' )
                       AND CA.[Type] = 'bladeChassis'
                       AND BI.BmcMac IS NOT NULL
                       AND core.IsInExcludedAssetNode(BA.AssetNode) = 0
                   
   
       --Match by serial number (don't get bay info with this one)
       INSERT  INTO @MatchedBlades
               ( BladeAssetId ,
                 Name ,
                 UUID ,
                 ChassisAssetId
                   
               )
               SELECT  BA.AssetId ,
                       BA.Name ,
                       SN.SerialNumber ,
                       CA.AssetId
               FROM    core.Asset AS BA
                       INNER JOIN core.SerialNumber AS SN ON BA.AssetId = SN.AssetId
                       INNER JOIN core.SerialNumber AS SN2 ON SN.SerialNumber = SN2.SerialNumber
                       INNER JOIN core.Asset AS CA ON SN2.AssetId = CA.AssetId
                       LEFT JOIN @MatchedBlades AS MB -- This is to not insert duplicates
                       ON MB.BladeAssetId = BA.AssetID
                          AND MB.ChassisAssetId = CA.AssetId
               WHERE   ( @AssetId IS NULL
                         OR CA.AssetId = @AssetId
                       )
                       AND BA.[Type] IN ( 'bladeServer', 'bladeNetwork' )
                       AND CA.[Type] = 'bladeChassis'
                       AND MB.BladeAssetId IS NULL
                       AND core.IsInExcludedAssetNode(BA.AssetNode) = 0
           
           
       SELECT  BladeAssetId ,
               ChassisAssetId ,
               BayLocation
       FROM    @MatchedBlades
   END
GO
