CREATE PROCEDURE log.RetrieveApplicationLog
   @Start SMALLINT,
   @Limit SMALLINT,
   @TotalCount INT OUTPUT
AS
DECLARE @End INT = @Start + @Limit - 1;

SELECT
   @TotalCount = COUNT(1)
FROM
   log.Change
WHERE LogTypeId = 1

SELECT
    Id
   ,ChangeMasterId
   ,LogTypeId
   ,Context
   ,ContextId
   ,ChangeAction
   ,Component
   ,StartTimestamp
   ,ChangedByUserName
   ,Authorized
   ,AuthorizedByUserName
   ,ExtendedProperties
FROM (
       SELECT
           Id
           ,ChangeMasterId
           ,LogTypeId
           ,Context
           ,ContextId
           ,ChangeAction
           ,Component
           ,StartTimestamp
           ,ChangedByUserName
           ,Authorized
           ,AuthorizedByUserName
           ,ExtendedProperties
           ,ROW_NUMBER () OVER (ORDER BY StartTimestamp DESC) AS RowNum
       FROM
           log.change
       WHERE
           LogTypeId = 1
    )  AS MyDerivedTable
WHERE
   MyDerivedTable.RowNum BETWEEN @Start AND @End
ORDER BY
   StartTimestamp DESC
GO
