CREATE PROC [core].[HandleRpduModelOutletCountShrinkage]
   @ModelId UNIQUEIDENTIFIER
 , @NewOutletCount INT
AS
   BEGIN    
       -- Deletes what is plugged into a RPDU on outlets that don't exist anymore do to an outletCount change at the model.
       --    This join also makes sure not to touch devices where we read the outlet count from the device or it has been set by the user.
       
       DELETE core.UpstreamPower
       FROM
           core.Asset AS A
           LEFT JOIN core.AssetProperty AS AP ON A.AssetId = AP.AssetId AND AP.Name = 'outletCount'
           INNER JOIN core.UpstreamPower AS UP ON UP.UpstreamAssetId = A.AssetId
       WHERE
           A.[Type] = 'rpdu'
           AND
           A.ModelId = @ModelId
           AND
           AP.SystemValue IS NULL
           AND
           AP.UserValue IS NULL
           AND
           UP.Outlet > @NewOutletCount
   END
GO
