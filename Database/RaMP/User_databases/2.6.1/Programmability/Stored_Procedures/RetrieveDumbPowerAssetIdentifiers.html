CREATE PROCEDURE [core].[RetrieveDumbPowerAssetIdentifiers]
AS
   BEGIN
       SELECT
           A.AssetId
       FROM
           core.Asset AS A
           LEFT JOIN core.AssetProperty AS AP ON A.AssetId = AP.AssetId
           LEFT JOIN core.Sensor AS S ON S.DeviceId = A.AssetId
       WHERE
           A.Type IN ('ups', 'rackOrTowerUps', 'pdu', 'rpdu', 'powerMeter')
           AND
           AP.Name = 'manuallyAdded'
           AND
           AP.SystemValue = 1
           AND
               (NOT EXISTS (SELECT 1 FROM core.Sensor WHERE DeviceId = A.AssetId AND Type IN ('power', 'totalPower', 'outputTotalPower'))
               OR
               (
                   S.Type IN ('power', 'totalPower', 'outputTotalPower')
                   AND
                   S.OriginType NOT IN ('realTimeMeasured', 'manual'))
               )
           AND core.IsInExcludedAssetNode(A.AssetNode) = 0
       ORDER BY CASE
           WHEN A.Type = 'rpdu' THEN 1
           WHEN A.Type = 'rackOrTowerUps' THEN 2
           WHEN A.Type = 'powerMeter' THEN 3
           WHEN A.Type = 'pdu' THEN 4
           WHEN A.Type = 'ups' THEN 5
           ELSE 6
        END ;
   END
GO
