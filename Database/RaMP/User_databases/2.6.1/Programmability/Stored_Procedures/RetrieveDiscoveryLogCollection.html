CREATE PROCEDURE [log].[RetrieveDiscoveryLogCollection]
   @DiscoveryId UNIQUEIDENTIFIER = NULL,
   @Start SMALLINT,
   @Limit SMALLINT,
   @TotalCount INT OUTPUT
AS
   BEGIN
    
       DECLARE @End INT = @Start + @Limit - 1;
       
      SELECT
           @TotalCount = COUNT(1)
       FROM
           log.DiscoveryLog AS DL
       WHERE DL.DiscoveryId = CASE WHEN @DiscoveryId IS NULL THEN DL.DiscoveryId ELSE @DiscoveryId END

       SELECT
           DiscoveryLogId ,
           DiscoveryId ,
           Name,
           StartTime ,
           EndTime ,
           DiscoveryStatusId ,
           DiscoveredDevices,
           DiscoveredDeviceTotal,
           ExtendedProperties
       FROM (
           SELECT
               DL.DiscoveryLogId ,
               DL.DiscoveryId ,
               D.Name,
               DL.StartTime ,
               DL.EndTime ,
               DL.DiscoveryStatusId ,
               DL.DiscoveredDevices,
               DL.DiscoveredDeviceTotal,
               DL.ExtendedProperties,
               ROW_NUMBER () OVER (ORDER BY StartTime DESC) AS RowNum
           FROM
               log.DiscoveryLog AS DL
           LEFT JOIN
               settings.Discovery D ON D.DiscoveryId = DL.DiscoveryId
           WHERE
               DL.DiscoveryId = CASE WHEN @DiscoveryId IS NULL THEN DL.DiscoveryId ELSE @DiscoveryId END
       ) AS MyDerivedTable
       WHERE
           MyDerivedTable.RowNum BETWEEN @Start AND @End
       ORDER BY StartTime DESC
   END
GO
