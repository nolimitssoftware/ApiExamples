CREATE PROC [core].[RetrieveDownlinkRouterHostMacs]
    @DesignatedPort INT,
    @DesignatedRootMac CHAR(17),
    @CurrentNetworkDeviceId UNIQUEIDENTIFIER
AS
   BEGIN
       SELECT
           H.Mac AS 'HostMac'
       FROM
           core.NetworkInterface AS N
       INNER JOIN
           core.NetworkInterface AS H
       ON
           N.AssetId = H.AssetId
           and H.DesignatedBridgeMac is null
       WHERE
           N.ConnectionFlag = 'downlink'
           AND
           N.AssetId <> @CurrentNetworkDeviceId
           AND
           N.DesignatedRootMac = @DesignatedRootMac
           AND
           N.DesignatedPort = @DesignatedPort
   END
GO
