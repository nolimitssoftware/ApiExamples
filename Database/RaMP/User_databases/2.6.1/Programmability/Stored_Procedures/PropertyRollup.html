CREATE PROC core.PropertyRollup
   @Properties core.AssetProperty READONLY ,
   @ParentId UNIQUEIDENTIFIER = NULL ,
   @ParentNode HIERARCHYID = NULL ,
   @AssetType VARCHAR(32) = NULL ,
   @AssetStatus VARCHAR(32) = NULL ,
   @IncludeUnknown BIT = 1 ,
   @IsDebug BIT = 1
AS
   BEGIN
       SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED ;
       DECLARE @ChildLevel INT ;
       DECLARE @OutParentId UNIQUEIDENTIFIER = @ParentId ;
       
       DECLARE
           @sql NVARCHAR(MAX) ,
           @params NVARCHAR(4000) ;
               
       IF (
            @ParentId IS NULL AND
            @ParentNode IS NULL
          )
           BEGIN
               SELECT
                   @ParentId = A.AssetId ,
                   @ChildLevel = A.AssetLevel + 1
               FROM
                   core.Asset AS A
               WHERE
                   A.AssetNode = HIERARCHYID::GetRoot()
           END
       ELSE
           IF ( @ParentId IS NOT NULL )
               BEGIN
                   SELECT
                       @ParentNode = A.AssetNode ,
                       @ChildLevel = A.AssetLevel + 1
                   FROM
                       core.Asset AS A
                   WHERE
                       A.AssetId = @ParentId ;    
               END            

       SET @sql = '
       SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED ;   
   WITH Ranked ( AssetId, rnk, PropertyValue )
                AS ( SELECT AP.AssetId,
                                ROW_NUMBER() OVER( PARTITION BY AP.AssetId ORDER BY P.PropertyOrder ),
                                CAST( AP.PropertyValue AS VARCHAR(8000) )
                         FROM core.AssetProperty AS AP
                         INNER JOIN core.Asset AS A
                         ON AP.AssetId = A.AssetId
                         INNER JOIN @Properties AS P
                         ON AP.PropertyName = P.PropertyName
                         AND AP.PropertyLevel = P.PropertyLevel '
       IF (
            @AssetStatus IS NOT NULL OR
            @AssetType IS NOT NULL
          )
           SET @sql += '
                   INNER JOIN core.AssetInformation AS AI
                       ON AP.AssetId = AI.AssetId ' ;
       SET @sql += '
           WHERE 1 = 1 ' ;
           
       IF ( @ParentId IS NOT NULL )
           SET @sql += ' AND
               A.AssetId <> @ParentId ' ;
       IF ( @ParentNode IS NOT NULL )
           SET @sql += ' AND A.AssetNode.IsDescendantOf(@ParentNode) = 1 ' ;
       
       IF (
            @ParentNode IS NOT NULL AND
            @ParentId IS NOT NULL
          )
           SET @sql += ' AND A.AssetLevel = @ChildLevel ' ;
               
       IF ( @AssetStatus IS NOT NULL )
           SET @sql += ' AND AI.Status = @Status ' ;
               
       IF ( @AssetType IS NOT NULL )
           SET @sql += ' AND AI.Type = @Type ' ;
               
       SET @sql += '    ),
   AnchorRanked ( AssetId, rnk, PropertyValue )
                AS ( SELECT AssetId, rnk, Ranked.PropertyValue
                         FROM Ranked
                        WHERE rnk = 1 ),
   RecurRanked ( AssetId, rnk, PropertyValue )
                AS ( SELECT AssetId, rnk, PropertyValue
                         FROM AnchorRanked
                        UNION ALL '
       IF ( @IncludeUnknown = 1 )
           BEGIN
               SET @sql += '
                        SELECT A.AssetId, 1, ''Unknown''
                  FROM core.Asset AS A  '
               IF (
                    @AssetStatus IS NOT NULL OR
                    @AssetType IS NOT NULL
                  )
                   SET @sql += '
                   INNER JOIN core.AssetInformation AS AI
                       ON A.AssetId = AI.AssetId ' ;
                                        
               SET @sql += ' WHERE A.AssetId NOT IN (SELECT Ranked.AssetId FROM Ranked) '
               IF ( @ParentId IS NOT NULL )
                   SET @sql += ' AND
               A.AssetId <> @ParentId ' ;
               IF ( @ParentNode IS NOT NULL )
                   SET @sql += ' AND A.AssetNode.IsDescendantOf(@ParentNode) = 1 ' ;
       
               IF (
                    @ParentNode IS NOT NULL AND
                    @ParentId IS NOT NULL
                  )
                   SET @sql += ' AND A.AssetLevel = @ChildLevel ' ;
                   
               IF ( @AssetStatus IS NOT NULL )
                   SET @sql += ' AND AI.Status = @Status ' ;
               
               IF ( @AssetType IS NOT NULL )
                   SET @sql += ' AND AI.Type = @Type '
                  
               SET @sql += ' UNION ALL '
               
           END
                       
       SET @sql += '
                       SELECT Ranked.AssetId, Ranked.rnk,
                                RecurRanked.PropertyValue + '' '' + Ranked.PropertyValue
                         FROM Ranked
                        INNER JOIN RecurRanked
                            ON Ranked.AssetId = RecurRanked.AssetId
                           AND Ranked.rnk = RecurRanked.rnk + 1 )
   SELECT
       PropertyValue,
       COUNT( PropertyValue ) AS ''Count''        
    FROM
       RecurRanked      
   GROUP BY
       PropertyValue
   ORDER BY
       PropertyValue;'
 
       SET @params = '@ParentId UNIQUEIDENTIFIER, @ParentNode HIERARCHYID, @ChildLevel INT, @Properties core.AssetProperty READONLY, @Status VARCHAR(32), @Type VARCHAR(32)' ;
 
       IF ( @IsDebug = 1 )
           PRINT @sql ;
    
       EXEC sp_executesql @sql, @params, @ParentId, @ParentNode, @ChildLevel,
           @Properties, @AssetStatus, @AssetType ;
   END
GO
