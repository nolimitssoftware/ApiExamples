CREATE PROC [core].[InsertUpstreamPower]
   @AssetId UNIQUEIDENTIFIER
 , @UpstreamAssetId UNIQUEIDENTIFIER
 , @Source VARCHAR(32)
 , @AgentId UNIQUEIDENTIFIER = NULL
 , @PduBreakerId UNIQUEIDENTIFIER = NULL
 , @Outlet TINYINT = NULL
 , @ExtendedProperties XML = NULL
AS
   BEGIN
       --The WHERE should take into account RPDUs, UPSs, and PDUs
       IF NOT EXISTS ( SELECT
                           Id
                       FROM
                           [core].[UpstreamPower]
                       WHERE
                           Assetid = @AssetId
                           AND Upstreamassetid = @UpstreamAssetId
                           AND ( ( @Outlet IS NOT NULL
                                   AND Outlet = @Outlet
                                 )  -- RPDU
                                 OR ( @PduBreakerId IS NOT NULL
                                      AND PduBreakerId = @PduBreakerId
                                    )  -- PDU
                                 OR ( @PduBreakerId IS NULL
                                      AND @Outlet IS NULL
                                    )  -- UPS
                                 
                               ) )
           BEGIN
               INSERT  core.UpstreamPower
                       ( AssetId
                       , UpstreamAssetId
                       , [Source]
                       , AgentId
                       , PduBreakerId
                       , Outlet
                       , ExtendedProperties
                       )
               VALUES
                       ( @AssetId
                       , @UpstreamAssetId
                       , @Source
                       , @AgentId
                       , @PduBreakerId
                       , @Outlet
                       , @ExtendedProperties
                       )
           END
   END
GO
