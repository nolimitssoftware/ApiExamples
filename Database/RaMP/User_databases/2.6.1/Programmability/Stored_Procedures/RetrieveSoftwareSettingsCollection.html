CREATE PROCEDURE [settings].[RetrieveSoftwareSettingsCollection]
   @WarningsOnly BIT = 0,
   @Start INT = 1
 , @Limit INT = NULL
 , @TotalCount INT OUTPUT
AS
   BEGIN
       SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED ;
       SET NOCOUNT ON ;
  
       SELECT
           @TotalCount = COUNT(1)
       FROM
           settings.Software AS S
       WHERE @WarningsOnly = 0
           OR (S.Approved = 0 OR (S.NumberOfLicenses > 0 AND S.NumberOfLicenses < S.NumberOfInstallations)) ;        
    
       IF ( @Limit IS NULL )
           BEGIN
               SET @Limit = @TotalCount ;
           END
 
       DECLARE @End INT = @Start + @Limit - 1 ;
             
       SELECT
           SoftwareId
         , Name
         , Approved
         , NumberOfLicenses
         , NumberOfInstallations
         , IsWarning
       FROM
           (
             SELECT
               S.SoftwareId
             , S.Name
             , S.Approved
             , S.NumberOfLicenses
             , S.NumberOfInstallations
             , CAST(S.IsWarning AS BIT) AS [IsWarning]
   --          , ROW_NUMBER() OVER ( ORDER BY S.[IsWarning] DESC, S.Name ) AS 'row_num'
             , ROW_NUMBER() OVER ( ORDER BY S.Name ) AS 'row_num'
             FROM
               (
                 SELECT
                   S2.SoftwareId
                 , S2.Name
                 , S2.Approved
                 , S2.NumberOfLicenses
                 , S2.NumberOfInstallations
                 , CASE WHEN S2.Approved = 0 THEN 1
                        WHEN S2.Approved = 1
                             AND (
                                   S2.NumberOfLicenses IS NULL
                                   OR S2.NumberOfLicenses = 0
                                 ) THEN 0
                        WHEN S2.NumberOfLicenses < S2.NumberOfInstallations
                        THEN 1
                        ELSE 0
                   END AS [IsWarning]
                 FROM
                   settings.Software AS S2
                   WHERE @WarningsOnly = 0
                       OR (S2.Approved = 0 OR ( S2.NumberOfLicenses > 0 AND S2.NumberOfLicenses < S2.NumberOfInstallations))
               ) AS S
           ) AS temp
       WHERE
           temp.row_num BETWEEN @Start AND @End
       ORDER BY
           temp.row_num ;
                       
   END
GO
