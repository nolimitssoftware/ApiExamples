CREATE PROCEDURE [Core].[RetrieveContractsByAsset]
   @assetId UNIQUEIDENTIFIER
AS
   SELECT    
       ContractId
       , @assetId as AssetId
       , StartDate, EndDate
       , Enabled
       , TrackingNumber
       , HostedCompany
       , ContactName
       , ContactPhone
       , ContactEmail
       , ContractType.Name as ContractTypeName
       , ContractType.ExpiringMessageType as ExpiringMessageType
       , ContractType.ExpiredMessageType as ExpiredMessageType
   FROM
       core.Contract
   RIGHT JOIN
       core.ContractType ON Contract.ContractTypeId = ContractType.ContractTypeId AND Contract.AssetId = @AssetId
GO
