CREATE FUNCTION util.ConvertSqlVariant
   (
     @Property SQL_VARIANT
   )
RETURNS TABLE
   WITH SCHEMABINDING
   --(
   --  [Value] SQL_VARIANT
   --, BaseType VARCHAR(32)
   --, DateValue DATETIME2
   --, NumericValue NUMERIC
   --, BooleanValue BIT
   --, StringValue BIT
   --)
AS
   RETURN
   (
     SELECT
       @Property AS 'Value'
     , SQL_VARIANT_PROPERTY(@Property, 'basetype') AS 'BaseType'
     , CASE SQL_VARIANT_PROPERTY(@Property, 'basetype')
         WHEN 'datetime2' THEN CAST(@Property AS DATETIME2)
         WHEN 'datetime' THEN CAST(@Property AS DATETIME2)
         WHEN 'datetimeoffset' THEN CAST(@Property AS DATETIME2)
         WHEN 'smalldatetime' THEN CAST(@Property AS DATETIME2)
         WHEN 'date' THEN CAST(@Property AS DATETIME2)
         WHEN 'time' THEN CAST(@Property AS DATETIME2)
         ELSE NULL
       END AS 'DateValue'
     , CASE SQL_VARIANT_PROPERTY(@Property, 'basetype')
         WHEN 'int' THEN CAST(@Property AS NUMERIC)
         WHEN 'bigint' THEN CAST(@Property AS NUMERIC)
         WHEN 'tinyint' THEN CAST(@Property AS NUMERIC)
         WHEN 'smallint' THEN CAST(@Property AS NUMERIC)
         WHEN 'money' THEN CAST(@Property AS NUMERIC(38, 2))
         WHEN 'smallmoney' THEN CAST(@Property AS NUMERIC(38, 2))
         WHEN 'real' THEN CAST(@Property AS NUMERIC(38, 2))
         WHEN 'float' THEN CAST(@Property AS NUMERIC(38, 2))
         WHEN 'decimal' THEN CAST(@Property AS NUMERIC(38, 2))
         WHEN 'numeric' THEN CAST(@Property AS NUMERIC(38, 2))
         ELSE NULL
       END AS 'NumericValue'
     , CASE SQL_VARIANT_PROPERTY(@Property, 'basetype')
         WHEN 'bit' THEN CAST(@Property AS BIT)
         ELSE NULL
       END AS 'BooleanValue'
     , CASE SQL_VARIANT_PROPERTY(@Property, 'basetype')
         WHEN 'nvarchar' THEN CAST(@Property AS NVARCHAR(MAX))
         WHEN 'nchar' THEN CAST(@Property AS NVARCHAR(MAX))
         WHEN 'varchar' THEN CAST(@Property AS NVARCHAR(MAX))
         WHEN 'char' THEN CAST(@Property AS NVARCHAR(MAX))
         WHEN 'datetime2' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'datetime' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'datetimeoffset' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'smalldatetime' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'date' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'time' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'int' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'bigint' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'tinyint' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'smallint' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'money' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'smallmoney' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'real' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'float' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'decimal' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'numeric' THEN CONVERT(VARCHAR(38), @Property)
         WHEN 'bit' THEN CASE WHEN CAST(@Property AS BIT) = CAST(1 AS BIT) THEN 'true'
                              ELSE 'false'
                         END
         ELSE NULL
       END AS 'StringValue'
   )
GO
