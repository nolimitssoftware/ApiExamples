CREATE FUNCTION log.GetChangeCollection
   (
     @ChangeId UNIQUEIDENTIFIER
   , @ChangeMasterId UNIQUEIDENTIFIER
   , @Context INT
   , @ContextId UNIQUEIDENTIFIER
   )
RETURNS @ChangeCollection TABLE
   (
     Id UNIQUEIDENTIFIER
   , ChangeMasterId UNIQUEIDENTIFIER NOT NULL
   , LogTypeId INT
   , Context INT
   , ContextId UNIQUEIDENTIFIER
   , ChangeAction TINYINT
   , [Component] VARCHAR(32)
   , StartTimestamp DATETIME
   , ChangedByUserName VARCHAR(256)
   , FromAddress VARCHAR(57)
   , Authorized BIT
   , AuthorizedByUserName VARCHAR(256)
   , ExtendedProperties XML
   , Severity INT NOT NULL
   , ChangeRecordType INT NOT NULL
   , Description VARCHAR(128) NOT NULL
   , UserOverrideType VARCHAR(16) NULL
   ,  -- "url" or "custom"
     UserOverrideHelp VARCHAR(MAX) NULL
   )
   WITH SCHEMABINDING
AS
   BEGIN
       IF ( @ChangeId IS NOT NULL )
           BEGIN
               INSERT  @ChangeCollection
                       SELECT
                           C.Id
                         , C.ChangeMasterId
                         , C.LogTypeId
                         , C.Context
                         , C.ContextId
                         , C.ChangeAction
                         , C.Component
                         , C.StartTimestamp
                         , C.ChangedByUserName
                         , C.FromAddress
                         , C.Authorized
                         , C.AuthorizedByUserName
                         , C.ExtendedProperties
                         , CM.Severity
                         , CM.ChangeRecordType
                         , CM.Description
                         , CM.UserOverrideType
                         , CM.UserOverrideHelp
                       FROM
                           log.Change AS C
                       INNER JOIN log.ChangeMaster AS CM
                           ON C.ChangeMasterId = CM.Id
                       WHERE
                           C.Id = @ChangeId
           END
       ELSE
           BEGIN
               INSERT  @ChangeCollection
                       SELECT
                           C.Id
                         , CM.Id
                         , C.LogTypeId
                         , @Context
                         , @ContextId
                         , C.ChangeAction
                         , C.Component
                         , GETUTCDATE()
                         , C.ChangedByUserName
                         , C.FromAddress
                         , C.Authorized
                         , C.AuthorizedByUserName
                         , C.ExtendedProperties
                         , CM.Severity
                         , CM.ChangeRecordType
                         , CM.Description
                         , CM.UserOverrideType
                         , CM.UserOverrideHelp
                       FROM
                           log.ChangeMaster AS CM
                       LEFT JOIN log.Change AS C
                           ON C.ChangeMasterId = CM.Id
                       WHERE
                           CM.Id = @ChangeMasterId
           END
       RETURN
   END
GO
