SQL Script
CREATE FUNCTION [core].[GetDownstreamNetworkInformation]
    (
       @StartAssetId UNIQUEIDENTIFIER
    )
RETURNS TABLE
    WITH SCHEMABINDING
AS      
RETURN

       SELECT DISTINCT
           A.AssetId
           , NI.Name AS 'PortName'
           , NI.Port AS 'PortNumber'
           , '' AS 'NicAttachedMac'
           , '' AS 'NicAttachedAddress'
           , CAST(1 AS BIT) AS 'ManuallyConfigured'
       FROM
           core.Asset AS A
       INNER JOIN core.ManualDownstreamNetworkConnection AS MDNC
           ON MDNC.DownstreamAssetId = A.AssetId
       INNER JOIN core.NetworkInterface AS NI
           ON NI.Id = MDNC.NetworkInterfaceId
       WHERE
           NI.AssetId = @StartAssetId
   UNION                    
       SELECT DISTINCT
           A.AssetId
           , NI.Name AS 'PortName'
           , NI.Port AS 'PortNumber'
           , '' AS 'NicAttachedMac'
           , '' AS 'NicAttachedAddress'
           , CAST(0 AS BIT) AS 'ManuallyConfigured'
       FROM
           core.Asset AS A
       CROSS APPLY core.GetAsset(A.AssetId) AS GA
       INNER JOIN core.RemoteMac AS RM
           ON A.AssetId = RM.RemoteAssetId
       INNER JOIN core.NetworkInterface AS NI
           ON NI.AssetId = @StartAssetId
               AND (NI.Port = RM.Port OR NI.IpAddressIndex = RM.Port)
       WHERE
           RM.AssetId = @StartAssetId
           AND RM.ConnectionFlag = 'direct'
GO
