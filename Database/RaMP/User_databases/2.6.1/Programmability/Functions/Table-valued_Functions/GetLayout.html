SQL Script
CREATE FUNCTION visual.GetLayout
   (
     @AssetId UNIQUEIDENTIFIER
   , @LayoutType TINYINT
   )
RETURNS TABLE
   WITH SCHEMABINDING
AS RETURN
   Select top 1            
           x.[Layout.Id]
         , x.[Layout.Direction]
         , x.[Layout.X]
         , x.[Layout.Y]
         , x.[Layout.ColumnFormat]
         , x.[Layout.RowFormat]
         , x.[Layout.Width]
         , x.[Layout.Height]
   from (
       SELECT
           LI.Id AS 'Layout.Id'
         , LI.Direction AS 'Layout.Direction'
         , LI.X AS 'Layout.X'
         , LI.Y AS 'Layout.Y'
         , L.ColumnFormat AS 'Layout.ColumnFormat'
         , L.RowFormat AS 'Layout.RowFormat'
         , L.Width AS 'Layout.Width'
         , L.Height AS 'Layout.Height'
       FROM
           visual.LayoutItem AS LI
       INNER JOIN visual.Layout AS L
           ON LI.LayoutId = L.Id
       WHERE
           LI.Id = @AssetId
           AND L.LayoutType = @LayoutType

       Union

       SELECT
           NULL AS 'Layout.Id'
         , NULL AS 'Layout.Direction'
         , NULL AS 'Layout.X'
         , NULL AS 'Layout.Y'
         , L.ColumnFormat AS 'Layout.ColumnFormat'
         , L.RowFormat AS 'Layout.RowFormat'
         , L.Width AS 'Layout.Width'
         , L.Height AS 'Layout.Height'
       FROM
           visual.Layout AS L
           join core.Asset parent on l.id = parent.AssetId
           join core.asset child on Child.AssetNode.GetAncestor(1) = parent.AssetNode
       WHERE
           child.assetID = @AssetId
           AND L.LayoutType = @LayoutType
   ) x
GO
