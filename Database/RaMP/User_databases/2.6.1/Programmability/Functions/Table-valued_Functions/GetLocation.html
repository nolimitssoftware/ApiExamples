CREATE FUNCTION [core].[GetLocation] ( @Id UNIQUEIDENTIFIER )
RETURNS @returntable TABLE
   (
     Id UNIQUEIDENTIFIER
   , AssetId UNIQUEIDENTIFIER
   , RackLocation TINYINT
   , RaMPPort CHAR(4)
   , [Source] VARCHAR(32)
   , ExtendedProperties XML
   )
AS
   BEGIN
       INSERT  @returntable
               SELECT
                   L.Id
                 , L.AssetId
                 , L.RackLocation
                 , L.RaMPPort
                 , L.Source
                 , L.ExtendedProperties
               FROM
                   core.Location AS L
               WHERE
                   L.Id = @Id
       RETURN
   END
GO
