CREATE FUNCTION [core].[GetParentAsset] ( @Id UNIQUEIDENTIFIER )
RETURNS TABLE
   WITH SCHEMABINDING
AS RETURN
   SELECT
       P.[Parent.AssetId]
     , P.[Parent.Name]
     , P.[Parent.ShortName]
     , P.[Parent.OverrideName]
     , P.[Parent.Type]
     , P.[Parent.Status]
       , P.[Parent.DerivedStatus]
     , P.[Parent.AssetType]
     , P.[Parent.ParentId]
     , P.[Parent.ExtendedProperties]
   FROM
       (
         SELECT
           A.AssetId AS 'Parent.AssetId'
         , A.[Name] AS 'Parent.Name'
         , A.ShortName AS 'Parent.ShortName'
         , A.OverrideName AS 'Parent.OverrideName'
         , A.[Type] AS 'Parent.Type'
         , A.[Status] AS 'Parent.Status'
            , ISNULL(core.GetStatus(A.AssetNode), A.[Status]) AS [Parent.DerivedStatus]
         , A.AssetType AS 'Parent.AssetType'
         , (
             SELECT TOP 1
               A2.AssetId
             FROM
               core.Asset AS A2
             WHERE
               A2.AssetNode = A.AssetNode.GetAncestor(1)
           ) AS 'Parent.ParentId'
         , (
             SELECT
               property.Name AS 'name'
             , property.SystemValue AS 'systemValue'
             , SQL_VARIANT_PROPERTY(property.SystemValue, 'basetype') AS 'systemValueType'
             , property.SystemValueSource AS 'systemValueSource'
             , property.UserValue AS 'userValue'
             , SQL_VARIANT_PROPERTY(property.UserValue, 'basetype') AS 'userValueType'
             , property.UserValueSource AS 'userValueSource'
             , property.Display AS 'display'
             , property.Editable AS 'editable'
             FROM
               core.AssetProperty AS property
             WHERE
               property.AssetId = A.AssetId
           FOR
             XML PATH('property')
               , ROOT('properties')
           ) AS 'Parent.ExtendedProperties'
         FROM
           core.Asset AS A
         WHERE
           A.AssetId = @Id
       ) AS P ;
GO
