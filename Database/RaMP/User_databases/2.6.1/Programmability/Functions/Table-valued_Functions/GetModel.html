CREATE FUNCTION settings.GetModel ( @Id UNIQUEIDENTIFIER )
RETURNS TABLE
   WITH SCHEMABINDING
AS      
RETURN
   SELECT
       M2.Name AS 'Model.Manufacturer'
     , M.Id AS 'Model.Id'
     , M.ManufacturerId AS 'Model.ManufacturerId'
     , M.[Name] AS 'Model.Name'
     , M.UserName AS 'Model.UserName'
     , M.PartNumber AS 'Model.PartNumber'
     , M.[Type] AS 'Model.Type'
     , M.[Power] AS 'Model.Power'
     , M.[UserPower] AS 'Model.UserPower'
     , M.[UserPartNumber] AS 'Model.UserPartNumber'
     , M.[Current] AS 'Model.Current'
     , M.Height AS 'Model.Height'
     , M.Depth AS 'Model.Depth'
     , M.Width AS 'Model.Width'
     , M.Weight AS 'Model.Weight'
     , M.RackUnits AS 'Model.RackUnits'
     , M.Notes AS 'Model.Notes'
     , M.ImagePath AS 'Model.ImagePath'
     , M.IsSystemDefined AS 'Model.IsSystemDefined'
     , M.ExtendedProperties AS 'Model.ExtendedProperties'
     , M.IsBasic AS 'Model.IsBasic'
     , M.SupportsRaMP AS 'Model.SupportsRaMP'
   FROM
       core.Model AS M
   INNER JOIN core.Manufacturer AS M2
       ON M.ManufacturerId = M2.Id
   OUTER APPLY ( SELECT
                   CONVERT(VARCHAR(256), ISNULL(MAN.[Name],
                                           '')) + ',' AS [text()]
                 FROM
                   core.ManufacturerAlternateName AS MAN
                 WHERE
                   MAN.ManufacturerId = M.Id
               FOR
                 XML PATH('')
               ) AS temp ( [Names] )
   WHERE
       M.Id = @Id;
GO
