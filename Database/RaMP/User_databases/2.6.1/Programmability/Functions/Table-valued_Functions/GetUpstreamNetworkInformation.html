SQL Script
CREATE FUNCTION [core].[GetUpstreamNetworkInformation]
   (
     @StartAssetId UNIQUEIDENTIFIER
    
   )
RETURNS TABLE
   WITH SCHEMABINDING
AS      
RETURN
   --Union of manually set and read network connections

       SELECT DISTINCT
           A.AssetId
         , NI.Name AS 'PortName'
         , NI.Port AS 'PortNumber'
         , '' AS 'NicAttachedMac'
         , '' AS 'NicAttachedAddress'
         , CAST(1 AS BIT) AS 'ManuallyConfigured'
       FROM
           core.Asset AS A
       INNER JOIN core.NetworkInterface AS NI
           ON NI.AssetId = A.AssetId
       INNER JOIN core.ManualDownstreamNetworkConnection AS MDNC
           ON MDNC.NetworkInterfaceId = NI.Id
       WHERE
           MDNC.DownstreamAssetId = @StartAssetId
       UNION
       SELECT DISTINCT
           A.AssetId
         , NI.Name AS 'PortName'
         , NI.Port AS 'PortNumber'
         , RM.Mac AS 'NicAttachedMac'
         , IP.[Address] AS 'NicAttachedAddress'
         , CAST(0 AS BIT) AS 'ManuallyConfigured'
       FROM
           core.Asset AS A
       INNER JOIN core.RemoteMac AS RM
           ON A.AssetId = RM.AssetId
       INNER JOIN core.NetworkInterface AS NI
           ON NI.AssetId = RM.AssetId
              AND ( NI.Port = RM.Port
                    OR NI.IpAddressIndex = RM.Port
                  )
       LEFT JOIN core.NetworkInterface AS NI2
           ON NI2.AssetId = RM.RemoteAssetId
              AND NI2.Mac = RM.Mac
       LEFT JOIN core.IpAddress AS IP
           ON IP.NetworkInterfaceId = NI2.Id
       WHERE
           RM.RemoteAssetId = @StartAssetId
           AND RM.ConnectionFlag = 'direct'
           --Do not include IP address b/c of vlans and nexus mapping
           --AND IP.Address IS NOT NULL
GO
