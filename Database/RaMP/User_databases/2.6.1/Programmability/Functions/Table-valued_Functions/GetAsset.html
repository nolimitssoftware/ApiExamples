CREATE FUNCTION core.GetAsset
    (
       @AssetId UNIQUEIDENTIFIER
    )
RETURNS TABLE
    WITH SCHEMABINDING
AS      
RETURN
    SELECT
         A2.[AssetId]
       , A2.Name
       , A2.ShortName
       , A2.OverrideName
       , A2.[Type]
       , A2.TypeAtDiscovery
       , A2.[Status]
       , A2.[LifecycleStatus]
       , A2.DerivedStatus
       , A2.AssetType
       , A2.[AssetNode]
       , A2.AssetLevel
       , A2.Notes
       , A2.AssetXml
       , A2.ModelId
       , A2.ChildCount
       , A2.ParentId
       , A2.ExtendedProperties
       , PA.[Parent.AssetId]
       , PA.[Parent.Name]
       , PA.[Parent.ShortName]
       , PA.[Parent.OverrideName]
       , PA.[Parent.Type]
       , PA.[Parent.Status]
       , PA.[Parent.DerivedStatus]
       , PA.[Parent.AssetType]
       , PA.[Parent.ExtendedProperties]
       , PA.[Parent.ParentId]    
       , GL.[Layout.Id]
       , GL.[Layout.Direction]
       , GL.[Layout.X]
       , GL.[Layout.Y]
       , GL.[Layout.ColumnFormat]
       , GL.[Layout.RowFormat]
       , GL.[Layout.Width]
       , GL.[Layout.Height]
       , M.[Model.Manufacturer]
       , M.[Model.Id]
       , M.[Model.ManufacturerId]
       , M.[Model.Name]
       , M.[Model.UserName]
       , M.[Model.PartNumber]
       , M.[Model.UserPartNumber]
       , M.[Model.Type]
       , M.[Model.Power]
       , M.[Model.UserPower]
       , M.[Model.Current]
       , M.[Model.Height]
       , M.[Model.Depth]
       , M.[Model.Width]
       , M.[Model.Weight]
       , M.[Model.RackUnits]
       , M.[Model.Notes]
       , M.[Model.ImagePath]
       , M.[Model.IsSystemDefined]
       , M.[Model.ExtendedProperties]
       , M.[Model.IsBasic]
       , M.[Model.SupportsRaMP]
       , L.RackLocation
       , L.RaMPPort
       , L.Source AS 'Location.Source'
    FROM
         (
            SELECT
               A.AssetId
            , A.Name
            , A.ShortName
            , A.OverrideName
            , A.[Type]
            , A.TypeAtDiscovery
            , ISNULL(core.GetStatus(A.AssetNode), A.[Status]) AS [DerivedStatus]
            , A.[Status]
            , A.[LifecycleStatus]
            , A.AssetType
            , A.[AssetNode]
            , A.AssetLevel
            , A.Notes
            , A.AssetXml
            , A.ModelId
            , (
                 SELECT
                    COUNT(*)
                 FROM
                    core.Asset AS ACC
                 WHERE
                    ACC.AssetNode.IsDescendantOf(A.AssetNode) = 1
                    AND ACC.AssetLevel = A.AssetLevel + 1
               ) AS 'ChildCount'
            , (
                 SELECT TOP 1
                    PAID.AssetId
                 FROM
                    core.Asset AS PAID
                 WHERE
                    PAID.AssetNode = A.AssetNode.GetAncestor(1)
               ) AS 'ParentId'
            , (
                 SELECT
                    property.Name AS 'name'
                 , property.SystemValue AS 'systemValue'
                 , SQL_VARIANT_PROPERTY(property.SystemValue, 'basetype') AS 'systemValueType'
                 , property.SystemValueSource AS 'systemValueSource'
                 , property.UserValue AS 'userValue'
                 , SQL_VARIANT_PROPERTY(property.UserValue, 'basetype') AS 'userValueType'
                 , property.UserValueSource AS 'userValueSource'
                 , property.Display AS 'display'
                 , property.Editable AS 'editable'
                 FROM
                    core.AssetProperty AS property
                 WHERE
                    property.AssetId = @AssetId
               FOR
                 XML PATH('property')
                    , ROOT('properties')
               ) AS ExtendedProperties
            FROM
               core.Asset AS A
         ) AS A2
    LEFT JOIN core.Location AS L
         ON A2.AssetId = L.Id
    OUTER APPLY core.GetParentAsset(A2.ParentId) AS PA
    OUTER APPLY visual.GetLayout(A2.AssetId, 1) AS GL
    OUTER APPLY settings.GetModel(A2.ModelId) AS M
    WHERE
         A2.AssetId = @AssetId
GO
