CREATE FUNCTION core.GetAssetType ( @Type VARCHAR(32) )
RETURNS INT
WITH SCHEMABINDING
   BEGIN
       DECLARE @AssetType INT
   
       SELECT
           @AssetType = CASE @Type
                          WHEN 'device' THEN 1
                          WHEN 'server' THEN 2
                          WHEN 'ups' THEN 4
                          WHEN 'rpdu' THEN 8
                          WHEN 'net' THEN 16
                          WHEN 'esmon' THEN 32
                          WHEN 'inRowCooling' THEN 64
                          WHEN 'conversion' THEN 128
                          WHEN 'virtual' THEN 256
                          WHEN 'bladeChassis' THEN 512
                          WHEN 'bladeServer' THEN 1024
                          WHEN 'san' THEN 2048
                          WHEN 'pdu' THEN 4096
                          WHEN 'ramp' THEN 8192
                          WHEN 'rack' THEN 16384
                          WHEN 'physicalGroup' THEN 32768
                          WHEN 'logicalGroup' THEN 65536
                          WHEN 'pduPanel' THEN 131072
                          WHEN 'rackOrTowerUps' THEN 262144
                          WHEN 'bladeNetwork' THEN 524288
                          WHEN 'chiller' THEN 1048576
                          WHEN 'crac' THEN 2097152
                          WHEN 'crah' THEN 4194304
                          WHEN 'unknownGroup' THEN 8388608
                          when 'camera' then 16777216
                          when 'powerMeter' then 33554432
                          ELSE 0
                        END ;
   
       RETURN @AssetType ;
   END
GO
