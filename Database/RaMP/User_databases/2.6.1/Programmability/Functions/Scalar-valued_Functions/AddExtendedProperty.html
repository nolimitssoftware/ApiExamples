CREATE FUNCTION [core].[AddExtendedProperty]
   (
     @ExistingXml XML
   , @Name VARCHAR(256)
   , @Display BIT = 1
   , @Editable BIT = 1
   , @SystemValue VARCHAR(256) = ''
   , @SystemValueType VARCHAR(16) = ''
   , @SystemValueSource VARCHAR(32) = ''
   , @UserValue VARCHAR(256) = ''
   , @UserValueType VARCHAR(16) = ''
   , @UserValueSource VARCHAR(32) = ''
   )
RETURNS XML
AS
   BEGIN

       DECLARE @IsDisplay VARCHAR(5) ;
       DECLARE @IsEditable VARCHAR(5) ;
   
       IF ( @Display = 1 )
           BEGIN
               SET @IsDisplay = 'true' ;
           END
       ELSE
           BEGIN
               SET @IsDisplay = 'false' ;
           END
   
       IF ( @Editable = 1 )
           BEGIN
               SET @IsEditable = 'true' ;
           END
       ELSE
           BEGIN
               SET @IsEditable = 'false' ;
           END
   
       DECLARE @xp XML = '
   <' + @Name + '>
   <display>' + @IsDisplay + '</display>
   <editable>' + @IsDisplay + '</editable>
   <systemValue>' + @SystemValue + '</systemValue>
   <systemValueType>' + @SystemValueType + '</systemValueType>
   <systemValueSource>' + @SystemValueSource + '</systemValueSource>
   <userValue>' + @UserValue + '</userValue>
   <userValueType>' + @UserValueType + '</userValueType>
   <userValueSource>' + @UserValueSource + '</userValueSource>
 </' + @Name + '>' ;

       SET @ExistingXml.modify('
       insert sql:variable("@xp")
       as first into (/properties)[1]
   ')
       
       RETURN @ExistingXml
   END
GO
