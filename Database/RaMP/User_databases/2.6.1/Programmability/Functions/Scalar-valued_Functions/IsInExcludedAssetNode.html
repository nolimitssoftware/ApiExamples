SQL Script
CREATE FUNCTION [core].[IsInExcludedAssetNode]
   (
     @AssetNode HIERARCHYID
   )
RETURNS BIT
AS
   BEGIN    
       DECLARE @ExcludedGroupNode HIERARCHYID;
       DECLARE @IsInNode BIT;
       IF ( @AssetNode IS NULL )
           BEGIN
               SET @IsInNode = 0;
           END
       ELSE
           BEGIN
               SELECT
                   @ExcludedGroupNode = A.AssetNode
               FROM
                   core.Asset A
               WHERE
                   A.AssetId = '164C4794-F0CA-454D-97C7-5456C3C6E301';
               SELECT
                   @IsInNode = @AssetNode.IsDescendantOf(@ExcludedGroupNode);
           END
       RETURN @IsInNode;
   END
GO
