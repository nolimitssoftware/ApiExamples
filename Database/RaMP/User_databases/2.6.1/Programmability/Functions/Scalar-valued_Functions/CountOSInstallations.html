SQL Script
CREATE FUNCTION core.CountOSInstallations
   (
     @OperatingSystemId UNIQUEIDENTIFIER ,
     @ServicePackId UNIQUEIDENTIFIER = NULL ,
     @IsServicePack BIT = 0
   )
RETURNS INT
AS
   BEGIN
       DECLARE @Count INT;
       
       IF ( @IsServicePack = 0 )
           BEGIN
               SELECT  @Count = COUNT(OS.AssetId)
               FROM    core.OperatingSystem AS OS
                       JOIN core.Asset AS A ON OS.AssetId = A.AssetId
               WHERE   OS.ServerOperatingSystemId = @OperatingSystemId
                       AND core.IsInExcludedAssetNode(A.AssetNode) = 0;
           END
       ELSE
           BEGIN
               SELECT  @Count = COUNT(OS.AssetId)
               FROM    core.OperatingSystem AS OS
                       JOIN core.Asset AS A ON OS.AssetId = A.AssetId
               WHERE   OS.ServerOperatingSystemId = @OperatingSystemId
                       AND ( OS.ServicePackId = @ServicePackId
                             OR ( OS.ServicePackId IS NULL
                                  AND @ServicePackId IS NULL
                                )
                           )
                       AND core.IsInExcludedAssetNode(A.AssetNode) = 0;
           END
       IF ( @Count IS NULL )
           BEGIN
               SET @Count = 0;
           END
   
       RETURN @Count;
   
   END
GO
