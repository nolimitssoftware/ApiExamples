-- Description:    Function to take a given id and place into the given sensor xml as an attribute.
-- =============================================
CREATE FUNCTION [core].[InsertSensorIdIntoSensorXml]
(
    @SensorId UNIQUEIDENTIFIER
   ,@SensorXml XML
)
RETURNS XML
AS
BEGIN
   SET @SensorXml.modify('           
       insert attribute sensorId {sql:variable("@SensorId") }      
       into (/sensor)[1] ');     

   RETURN @SensorXml
END
GO
