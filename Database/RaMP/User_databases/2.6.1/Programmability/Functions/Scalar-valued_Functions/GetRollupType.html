CREATE FUNCTION core.GetRollupType
   (
     @RollupTypeId TINYINT
   )
RETURNS VARCHAR(16)
   WITH SCHEMABINDING
   BEGIN
       DECLARE    @RollupType VARCHAR(16)
   
       SELECT
           @RollupType = CASE @RollupTypeId
                           WHEN 1 THEN 'current'
                           WHEN 2 THEN 'hourlyAvg'
                           WHEN 4 THEN 'hourlyMax'
                           WHEN 8 THEN 'dailyAvg'
                           WHEN 16 THEN 'dailyMax'
                           ELSE 'none'
                         END
   
       RETURN @RollupType;
   END
GO
