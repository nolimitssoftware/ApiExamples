CREATE FUNCTION [core].[GetStatusAsNumber] (
   @Status VARCHAR(128)
)
RETURNS INT
WITH SCHEMABINDING
   BEGIN
       DECLARE @StatusNumber INT
   
       SELECT
           @StatusNumber = CASE @Status
                          WHEN 'unknown' THEN 0
                          WHEN 'maintenance' THEN 4
                          WHEN 'unreachable' THEN 16
                          WHEN 'critical' THEN 32
                          WHEN 'warning' THEN 64
                          WHEN 'administration' THEN 2048
                          WHEN 'information' THEN 32768
                          WHEN 'normal' THEN 65536
                          WHEN 'clear' THEN 131072
                          ELSE 0
                        END ;
   
       RETURN @StatusNumber ;
   END
GO
