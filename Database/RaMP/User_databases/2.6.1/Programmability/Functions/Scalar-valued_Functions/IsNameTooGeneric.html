CREATE FUNCTION core.IsNameTooGeneric ( @Name NVARCHAR(256) )
RETURNS BIT
AS
   BEGIN
       /************************
       *
       *
       * THIS SHOULD BE THE SAME AS THE GENERICNAMECHEKCER CLASS  
       *
       *
       ************************/
       DECLARE @IsNameToGeneric BIT = 0;
       IF ( @Name IN ( 'unknown', '[unknown]', 'sysName', 'Default PDU Name',
                       'localhost', 'localhost.localdomain' ) )
           BEGIN
               SET @IsNameToGeneric = 1;
           END       

       RETURN @IsNameToGeneric;
   END
GO
