CREATE FUNCTION core.GetProtocolDiscoveryState
   (
     @AssetId UNIQUEIDENTIFIER
   , @Protocol VARCHAR(32)
   )
RETURNS VARCHAR(32)
AS
   BEGIN
       DECLARE @HasValues VARCHAR(32)
       IF EXISTS ( SELECT
                       1
                   FROM
                       core.AssetDiscoveryLog AS ADL
                   INNER JOIN settings.Protocol AS P
                       ON ADL.ProtocolId = P.ProtocolId
                   WHERE
                       ADL.AssetId = @AssetId
                       AND P.Protocol = @Protocol
                       AND ADL.ResultType = 2 )
           BEGIN
               SET @HasValues = 'Success';
           END
       ELSE
           IF EXISTS ( SELECT
                           1
                       FROM
                           core.AssetDiscoveryLog AS ADL
                       INNER JOIN settings.Protocol AS P
                           ON ADL.ProtocolId = P.ProtocolId
                       WHERE
                           ADL.AssetId = @AssetId
                           AND P.Protocol = @Protocol
                           AND ADL.ResultType = 4 )
               BEGIN
                   SET @HasValues = 'Port Closed';
               END
           ELSE
               BEGIN
                   SET @HasValues = 'Wrong Credentials'
               END

       RETURN @HasValues;
   END
GO
Uses
