CREATE FUNCTION [core].[GetIpAddressListByAsset]
   (
     @AssetId UNIQUEIDENTIFIER    
   )
RETURNS VARCHAR(MAX)
AS
   BEGIN
       DECLARE @t util.stringList
       INSERT  @t
               SELECT
                   IP.Address
               FROM
                   core.IpAddress AS IP
               WHERE
                   IP.AssetId = @AssetId ;
       DECLARE @retval VARCHAR(MAX) = NULL
       SELECT
           @retval = util.join_delimeter(@t, ',') ;
       RETURN @retval
   END
GO
