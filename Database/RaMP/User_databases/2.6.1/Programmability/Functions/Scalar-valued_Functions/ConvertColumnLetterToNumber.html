CREATE FUNCTION util.ConvertColumnLetterToNumber
   (
     @ColumnName VARCHAR(8000)
   )
RETURNS INT
AS
   BEGIN
   
       DECLARE @Sum INT = 0 ;
       IF ( ISNUMERIC(@ColumnName) = 1 )
           BEGIN
               SET @Sum = CONVERT(INT, @ColumnName) ;
           END
       ELSE
           BEGIN
               DECLARE @Position INT = 1
               WHILE ( @Position <= DATALENGTH(@ColumnName) )
                   BEGIN
                       SET @Sum *= 26 ;
                       DECLARE @t INT = ASCII(SUBSTRING(@ColumnName,
                                                        @Position, 1)) ;                
                       SET @Sum += ( @t - ASCII('A') + 1 ) ;
                       SET @Position += 1 ;
                   END
           END
       RETURN @Sum
   END
GO
