SQL Script
CREATE FUNCTION [util].[join_delimeter]
   (
     @stringList AS util.stringList READONLY
   , @delim VARCHAR(4) = ','
   )
RETURNS VARCHAR(MAX)
   WITH SCHEMABINDING
AS
   BEGIN

       DECLARE @retval VARCHAR(MAX) = NULL
       DECLARE @length INT = LEN(CONVERT(varchar(36), CONVERT(XML, @delim)))
       SET @retval = STUFF(( SELECT
                               @delim
                               + CAST(string AS VARCHAR(8000))
                             FROM
                               @stringList
                             ORDER BY
                               [order]
                           FOR
                             XML PATH('')
                           ), 1, @length, '')

       RETURN @retval
   END
GO
