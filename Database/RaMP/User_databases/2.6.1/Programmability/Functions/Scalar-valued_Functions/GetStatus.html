SQL Script
CREATE FUNCTION core.GetStatus
   (
     @AssetNode HIERARCHYID
   )
RETURNS INT
   WITH SCHEMABINDING
AS
   BEGIN
       
       --It was decided to NOT include virtuals in derived status; the idea is that VM's will usually be in a
       --warning/critical state and we don't want to clutter the status'. We do need to allow a virtual's status
       --to be taken into account when it is the devices the GET is being done on.
       DECLARE @IncludeVirtual BIT = 0
       SELECT @IncludeVirtual = 1 FROM core.Asset WHERE AssetNode = @AssetNode AND [TYPE] = 'virtual';

       DECLARE @Status INT
       SELECT
           @Status = SUM(DISTINCT A.[Status])
       FROM
           core.Asset AS A
       WHERE
           A.AssetNode.IsDescendantOf(@AssetNode) = 1
           AND A.[Status] <> 65536
           AND (@IncludeVirtual = 1 OR A.[Type] <> 'virtual');
           
         --FIlter normal off for the asset collection bitwise check
       IF ( @Status IS NULL )
           BEGIN
               SET @Status = 65536 ;
           END
            
       RETURN @Status ;
   END
GO
